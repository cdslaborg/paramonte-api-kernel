\doxysubsection{Detailed Description}
Generate an equally-\/weighted (flattened) array of the input weighted {\ttfamily Array} of rank 1 or 2. 


\begin{DoxyParams}[1]{Parameters}
\mbox{\texttt{ in}}  & {\em Array} & \+: The input contiguous array of shape {\ttfamily (\+:)} or {\ttfamily (\+:,\+:)} of either type {\ttfamily real} of kind \mbox{\hyperlink{namespaceConstants__mod_aaf5c6c840ed0befe5a37da626c35716f}{RK32}}, \mbox{\hyperlink{namespaceConstants__mod_a855643f887b4c787e6542e9e7849680f}{RK64}}, or \mbox{\hyperlink{namespaceConstants__mod_aeef7c44b90b75aeeebdef1da504886cb}{RK128}} or type {\ttfamily integer} of kind \mbox{\hyperlink{namespaceConstants__mod_ae3573d46486b5f1a0a8d5033a9d12879}{IK8}}, \mbox{\hyperlink{namespaceConstants__mod_acd3c33e94d12f6e7d7c0ee79b2d6280b}{IK16}}, \mbox{\hyperlink{namespaceConstants__mod_ab12a86d74008f0d7f287c3564c3defd9}{IK32}}, or \mbox{\hyperlink{namespaceConstants__mod_ad6aaf2f477491dc84e7ebbab3b64694e}{IK64}}, containing the weighted array to be flattened. \\
\hline
\mbox{\texttt{ in}}  & {\em Weight} & \+: The input contiguous array of shape {\ttfamily (\+:)} of type {\ttfamily integer} of default kind \mbox{\hyperlink{namespaceConstants__mod_a2bf4d6767c9954d3a495d10ab6b538d6}{IK}} of the same size as the {\ttfamily Array} containing the weights of points in the sample. The values of elements of Weight are allowed to be negative or zero, in which case, the corresponding elements will be excluded from the output {\ttfamily Flat\+Array}. \\
\hline
\mbox{\texttt{ in}}  & {\em sum\+Weight} & \+: The input scalar of type {\ttfamily integer} of default kind \mbox{\hyperlink{namespaceConstants__mod_a2bf4d6767c9954d3a495d10ab6b538d6}{IK}} representing the sum of all {\bfseries{positive non-\/zero}} elements of {\ttfamily Weight} ({\bfseries{optional}}, default = {\ttfamily sum(Weight, mask = Weight \texorpdfstring{$>$}{>} 0)}. If a precomputed value available and present as argument, it can lead to a faster runtime).\\
\hline
\end{DoxyParams}
\begin{DoxyReturn}{Returns}
{\ttfamily Flat\+Array} \+: The output array of shape {\ttfamily (1\+:count)} of the same type and kind as the input {\ttfamily x1} containing the evenly-\/spaced sequence within the interval specified by {\ttfamily x1} and {\ttfamily x2}.
\end{DoxyReturn}
\begin{DoxyWarning}{Warning}
This procedure is {\ttfamily impure} when the library is compiled with macro {\ttfamily CHECK\+\_\+\+ENABLED=1}. By default, it is {\ttfamily pure} in {\ttfamily release} build and {\ttfamily impure} in {\ttfamily debug} and {\ttfamily testing} builds.
\end{DoxyWarning}
\begin{DoxyRemark}{Remarks}
Setting both {\ttfamily fopen = .true.} and {\ttfamily lopen = .true.} will lead to an output {\ttfamily Lin\+Space} whose points are centers of the {\ttfamily count + 1} equally-\/sized bins in the interval {\ttfamily \mbox{[}x1,x2\mbox{]}}.
\end{DoxyRemark}
