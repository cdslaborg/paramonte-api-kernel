\begin{DoxyRemark}{Remarks}

\begin{DoxyCode}{0}

\DoxyCodeLine{variance = sum( (weight*(point-\/mean))**2 ) / (sum(weight)-\/1)}
\end{DoxyCode}

\end{DoxyRemark}
