\doxysubsection{Detailed Description}
Generate {\ttfamily count} evenly-\/logarithmically-\/spaced points over the interval {\ttfamily \mbox{[}base$\ast$$\ast$logx1, base$\ast$$\ast$logx2\mbox{]}} if {\ttfamily logx1 \texorpdfstring{$<$}{<} logx2}, or over the interval {\ttfamily \mbox{[}logx2, logx1\mbox{]}} if {\ttfamily logx2 \texorpdfstring{$<$}{<} logx1}. 


\begin{DoxyParams}[1]{Parameters}
\mbox{\texttt{ in}}  & {\em logx1} & \+: The input scalar of type {\ttfamily real} of kind \mbox{\hyperlink{namespaceConstants__mod_aaf5c6c840ed0befe5a37da626c35716f}{RK32}}, \mbox{\hyperlink{namespaceConstants__mod_a855643f887b4c787e6542e9e7849680f}{RK64}}, or \mbox{\hyperlink{namespaceConstants__mod_aeef7c44b90b75aeeebdef1da504886cb}{RK128}} representing the starting exponent such that {\ttfamily base$\ast$$\ast$logx1} is the starting value of the sequence. \\
\hline
\mbox{\texttt{ in}}  & {\em logx2} & \+: The input scalar of the same type and kind as {\ttfamily logx1} representing the ending exponent such that {\ttfamily base$\ast$$\ast$logx1} is the last value of the sequence. \\
\hline
\mbox{\texttt{ in}}  & {\em count} & \+: The input scalar of type {\ttfamily integer} of default kind \mbox{\hyperlink{namespaceConstants__mod_a2bf4d6767c9954d3a495d10ab6b538d6}{IK}} representing the length of {\ttfamily Log\+Space} to generate. \\
\hline
\mbox{\texttt{ in}}  & {\em fopen} & \+: The input scalar of type {\ttfamily logical} of default kind. If {\ttfamily .true.}, the {\ttfamily Log\+Space} will be {\bfseries{f}}irst-\/open, meaning that {\ttfamily logx1} will NOT be in the output {\ttfamily Log\+Space} sequence ({\bfseries{optional}}, default = {\ttfamily .false.}). \\
\hline
\mbox{\texttt{ in}}  & {\em lopen} & \+: The input scalar of type {\ttfamily logical} of default kind. If {\ttfamily .true.}, the {\ttfamily Log\+Space} will be {\bfseries{l}}ast-\/open, meaning that {\ttfamily logx2} will NOT be in the output {\ttfamily Log\+Space} sequence ({\bfseries{optional}}, default = {\ttfamily .false.}). \\
\hline
\mbox{\texttt{ in}}  & {\em base} & \+: The input scalar of the same type and kind as {\ttfamily logx1} representing the the base of the logarithm ({\bfseries{oprional}}, default = {\ttfamily exp(1)}, that is, Euler\`{}s number).\\
\hline
\end{DoxyParams}
\begin{DoxyReturn}{Returns}
{\ttfamily Log\+Space} \+: The output array of shape {\ttfamily (1\+:count)} of the same type and kind as the input {\ttfamily logx1} containing the log-\/evenly-\/spaced sequence within the interval specified by {\ttfamily logx1} and {\ttfamily logx2}.
\end{DoxyReturn}
\begin{DoxyWarning}{Warning}
This procedure is {\ttfamily impure} when the library is compiled with macro {\ttfamily CHECK\+\_\+\+ENABLED=1}. By default, it is {\ttfamily pure} in {\ttfamily release} build and {\ttfamily impure} in {\ttfamily debug} and {\ttfamily testing} builds.
\end{DoxyWarning}
\begin{DoxyRemark}{Remarks}
Setting both {\ttfamily fopen = .true.} and {\ttfamily lopen = .true.} will lead to an output Log\+Space whose points are centers of the {\ttfamily count + 1} equally-\/sized bins in the interval {\ttfamily \mbox{[}logx1,logx2\mbox{]}}.
\end{DoxyRemark}
