\doxysubsection{Detailed Description}
Return an array whose elements are the values in the range {\ttfamily lower\+:upper} that are not in {\ttfamily Set}. 


\begin{DoxyParams}[1]{Parameters}
\mbox{\texttt{ in}}  & {\em Set} & \+: The input set whose elements will be subtracted from the superset range {\ttfamily lower\+:upper}. \\
\hline
\mbox{\texttt{ in}}  & {\em lower} & \+: The lower limit of the input range that represents the superset. \\
\hline
\mbox{\texttt{ in}}  & {\em lower} & \+: The upper limit of the input range that represents the superset.\\
\hline
\end{DoxyParams}
\begin{DoxyReturn}{Returns}
{\ttfamily Complement} \+: The array of elements of {\ttfamily Set} that are not in the input superset range {\ttfamily lower\+:upper}. By definition, {\ttfamily size(\+Complement) = (upper -\/ lower + 1) -\/ size(\+Set)}.
\end{DoxyReturn}
\begin{DoxyWarning}{Warning}
All elements of {\ttfamily Set} must be {\bfseries{unique}}. The unique elements of any arbitrary vector can be obtained by passing the array to \mbox{\hyperlink{interfaceUnique__mod_1_1genUnique}{Unique\+\_\+mod\+::get()}}.

It is assumed that {\ttfamily Set} is subset of the range {\ttfamily lower\+:upper}. This condition will NOT be checked in this procedure. If this condition does not hold, the results will not be accurate.
\end{DoxyWarning}
