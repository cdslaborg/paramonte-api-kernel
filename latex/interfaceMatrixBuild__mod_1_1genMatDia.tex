\doxysubsection{Detailed Description}
Generate a diagonal matrix of shape {\ttfamily (nrow,ncol)} with its diagonal elements set to the input {\ttfamily Diag}. All off-\/diagonal elements are set to zero. 


\begin{DoxyParams}[1]{Parameters}
\mbox{\texttt{ in}}  & {\em Diag} & \+: The input scalar or contiguous array of rank {\ttfamily 1} of either ~\newline
 ~ type {\ttfamily real} of kind \mbox{\hyperlink{namespaceConstants__mod_aaf5c6c840ed0befe5a37da626c35716f}{RK32}}, \mbox{\hyperlink{namespaceConstants__mod_a855643f887b4c787e6542e9e7849680f}{RK64}}, or \mbox{\hyperlink{namespaceConstants__mod_aeef7c44b90b75aeeebdef1da504886cb}{RK128}} or ~\newline
 ~ type {\ttfamily complex} of kind \mbox{\hyperlink{namespaceConstants__mod_ac8b456bd9511bc9a52b4a03bda4b72df}{CK32}}, \mbox{\hyperlink{namespaceConstants__mod_aa6ea61256fca0b20db0ec2b90755b9f3}{CK64}}, or \mbox{\hyperlink{namespaceConstants__mod_ad3873e1ecc5eb3cf3651f72630cae087}{CK128}} ~\newline
 containing the diagonal elements of the output matrix. \\
\hline
\mbox{\texttt{ in}}  & {\em nrow} & \+: The input positive {\ttfamily integer} of default kind \mbox{\hyperlink{namespaceConstants__mod_a2bf4d6767c9954d3a495d10ab6b538d6}{IK}} representing the number of rows of the output matrix. \\

\mbox{\texttt{ in}}  & {\em ncol} & \+: The input positive {\ttfamily integer} of default kind \mbox{\hyperlink{namespaceConstants__mod_a2bf4d6767c9954d3a495d10ab6b538d6}{IK}} representing the number of columns of the output matrix. \\

\hline
\end{DoxyParams}
\begin{DoxyReturn}{Returns}
{\ttfamily Diag\+Mat} \+: The output diagonal matrix of the same type and kind as the input {\ttfamily Diag} and shape {\ttfamily (nrow,ncol)} whose (shifted) diagonal elements are set to the input values. All other matrix elements will be set to zero.
\end{DoxyReturn}
\begin{DoxyWarning}{Warning}
The size of input {\ttfamily Diag} must be consistent with the input values for {\ttfamily nrow}, {\ttfamily ncol}, and {\ttfamily shift} (if it is present). This condition is checked for only when the preprocessor flag {\ttfamily CHECK\+\_\+\+ENABLED=1} is activated at build time.

This procedure is {\ttfamily impure} when the library is compiled with macro {\ttfamily CHECK\+\_\+\+ENABLED=1}. By default, it is {\ttfamily pure} in {\ttfamily release} build and {\ttfamily impure} in {\ttfamily debug} and {\ttfamily testing} builds.
\end{DoxyWarning}
