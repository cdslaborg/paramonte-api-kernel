\doxysubsection{Detailed Description}
Return an array of elements of {\ttfamily Super\+Set} that are not in {\ttfamily Set}. 


\begin{DoxyParams}[1]{Parameters}
\mbox{\texttt{ in}}  & {\em Set} & \+: The input set whose elements will be subtracted from the {\ttfamily Super\+Set}. \\
\hline
\mbox{\texttt{ in}}  & {\em Super\+Set} & \+: The input {\ttfamily Super\+Set} whose missing elements in {\ttfamily Set} will be returned.\\
\hline
\end{DoxyParams}
\begin{DoxyReturn}{Returns}
{\ttfamily Complement} \+: The array of elements of {\ttfamily Set} that are not in {\ttfamily Super\+Set}. By definition, {\ttfamily size(\+Complement) = size(\+Super\+Set) -\/ size(\+Set)}.
\end{DoxyReturn}
\begin{DoxyWarning}{Warning}
All elements of {\ttfamily Set} and {\ttfamily Super\+Set} must be {\bfseries{unique}}. The unique elements of any arbitrary vector can be obtained by passing the array to \mbox{\hyperlink{interfaceUnique__mod_1_1genUnique}{Unique\+\_\+mod\+::get()}}.

It is assumed that {\ttfamily Set} is subset of {\ttfamily Super\+Set}. This condition will NOT be checked in this procedure. If this condition does not hold, the results will not be accurate.
\end{DoxyWarning}
