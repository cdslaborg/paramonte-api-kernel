\doxysubsection{Detailed Description}
Convert the input value to string, with the requested format, if provided. 


\begin{DoxyParams}[1]{Parameters}
\mbox{\texttt{ in}}  & {\em value} & \+: The input value. It can be {\ttfamily integer}, {\ttfamily real}, {\ttfamily complex} of any kind supported by the compiler and of rank {\ttfamily 0}, {\ttfamily 1}, or {\ttfamily 2}. \\
\hline
\mbox{\texttt{ in}}  & {\em format} & \+: The Fortran IO format to be used when writing the value to the string ({\bfseries{optional}}, default = {\ttfamily \char`\"{}(g0,\+:,\textquotesingle{},\textquotesingle{})\char`\"{}}). \\
\hline
\mbox{\texttt{ in}}  & {\em len} & \+: The length of the output string. If provided, the output string will be preallocated with the requested input length. Then, the value will be written to it. If missing, the blank characters on both ends of the output string will be trimmed. ({\bfseries{optional}}, default = automatically determined such that all leading and trailing blank characters are removed)\\
\hline
\end{DoxyParams}
\begin{DoxyReturn}{Returns}
{\ttfamily string} \+: The output string.
\end{DoxyReturn}
\begin{DoxyRemark}{Remarks}
This procedure is a static method of the class \mbox{\hyperlink{structString__mod_1_1String__type}{String\+\_\+type}}.
\end{DoxyRemark}
\begin{DoxyWarning}{Warning}
Note that {\ttfamily len}, if provided, must be large enough to contain the full input value.
\end{DoxyWarning}
