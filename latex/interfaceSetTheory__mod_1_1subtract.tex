\doxysubsection{Detailed Description}
Return an array of elements of {\ttfamily SetA} that are not in {\ttfamily SetB}. 


\begin{DoxyParams}[1]{Parameters}
\mbox{\texttt{ in}}  & {\em SetA} & \+: The input {\ttfamily SetA} whose missing elements in {\ttfamily SetB} will be returned. \\
\hline
\mbox{\texttt{ in}}  & {\em SetB} & \+: The input {\ttfamily SetB} whose elements will be subtracted from {\ttfamily SetB}. \\
\hline
\mbox{\texttt{ in}}  & {\em ordered} & \+: A logical input. If {\ttfamily .true.} the input set members are assumed to be in ascending order. This leads to significantly faster search.\\
\hline
\end{DoxyParams}
\begin{DoxyReturn}{Returns}
{\ttfamily Set\+Diff} \+: The array of elements of {\ttfamily SetA} that are not in {\ttfamily SetB}.
\end{DoxyReturn}
\begin{DoxyWarning}{Warning}
All elements of {\ttfamily SetA} and {\ttfamily SetB} must be unique. The unique elements of any arbitrary vector can be obtained by passing the array to \mbox{\hyperlink{interfaceUnique__mod_1_1genUnique}{Unique\+\_\+mod\+::get()}}.
