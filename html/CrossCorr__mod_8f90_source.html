75 function getBatchMeansIAC(np,Point,Weight,batchSize) result(iac)
76#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
77 !DEC$ ATTRIBUTES DLLEXPORT :: getBatchMeansIAC
78#endif
79 use Constants_mod, only: IK, RK
80 use Sample_mod, only: getVariance
81 use Array_mod, only: genCumSum
82 !use Stirng_mod, only: num2char
83 !use Err_mod, only: Err_type
84
85 implicit none
86
87 character(len=*), parameter :: PROCEDURE_NAME = MODULE_NAME // "@getBatchMeansIAC()"
88
89 integer(IK), intent(in) :: np
90 real(RK), intent(in) :: Point(np)
91 integer(IK), intent(in), optional :: Weight(np), batchSize
92 !type(Err_type), intent(out) :: Err
93 real(RK) :: iac
94
95 integer(IK) :: CumSumWeight(np), currentSampleEndLoc, batchSizeDefault, sampleCount, isample
96 integer(IK) :: ip, ipVerbose, ipStart, ipEnd, npEffective
97 real(RK) :: avgPoint, varPoint, avgBatchMean, varBatchMean
98 real(RK) :: diffSquared, batchSizeDefaultInverse
99 real(RK), allocatable :: BatchMean(:)
100
101 !Err%occurred = .false.
102
103 if (present(Weight)) then
104 CumSumWeight = genCumSum(np,Weight)
105 else
106 CumSumWeight(np) = np
107 end if
108
109 ! compute batch size and count
110 if (present(batchSize)) then
111 batchSizeDefault = batchSize
112 else
113 batchSizeDefault = int( real(CumSumWeight(np),kind=RK)**(0.666666666666666_RK) )
114 end if
115 batchSizeDefaultInverse = 1._RK / real(batchSizeDefault,kind=RK)
116 sampleCount = CumSumWeight(np) / batchSizeDefault
117 npEffective = sampleCount * batchSizeDefault
118
119 if (sampleCount<2) then
120 !Err%occurred = .true.
121 !Err%msg = PROCEDURE_NAME // ": sampleCount<10: " // num2char(sampleCount)
122 !iac = -huge(iac)
123 iac = 1
124 return
125 end if
126
127 ! xxx: here goes another GFortran 7.3 bug: EndOfLineLoc is assumed already allocated, despite the first appearance here.
128 if (allocated(BatchMean)) deallocate(BatchMean)
129 allocate(BatchMean(sampleCount))
130
131 ! improvement: iterate from the end to the beginning of the chain to ignore initial points instead of the last points.
132 ! this would be beneficial for MCMC samples
133
134 ! compute the Batch-Means avergage and variance, also average of Point
135 avgPoint = 0._RK
136 ipStart = 1
137 ipEnd = 0
138 if (present(Weight)) then
139 ip = 1
140 isample = 1
141 ipVerbose = 0
142 currentSampleEndLoc = batchSizeDefault
143 BatchMean(isample) = 0._RK
144 loopOverWeight: do
145 ipVerbose = ipVerbose + 1
146 if (ipVerbose>CumSumWeight(ip)) ip = ip + 1
147 if (ipVerbose>currentSampleEndLoc) then ! we are done with the current batch
148 avgPoint = avgPoint + BatchMean(isample)
149 BatchMean(isample) = BatchMean(isample) * batchSizeDefaultInverse
150 if (ipVerbose>npEffective) exit loopOverWeight ! condition equivalent to currentSampleEndLoc==npEffective
151 currentSampleEndLoc = currentSampleEndLoc + batchSizeDefault
152 isample = isample + 1
153 BatchMean(isample) = 0._RK
154 end if
155 BatchMean(isample) = BatchMean(isample) + Point(ip)
156 end do loopOverWeight
157 else ! there is no weight
158 do isample = 1, sampleCount
159 BatchMean(isample) = 0._RK
160 ipEnd = ipEnd + batchSizeDefault
161 do ip = ipStart, ipEnd
162 BatchMean(isample) = BatchMean(isample) + Point(ip)
163 end do
164 ipStart = ipEnd + 1_IK
165 avgPoint = avgPoint + BatchMean(isample)
166 BatchMean(isample) = BatchMean(isample) * batchSizeDefaultInverse
167 end do
168 end if
169 avgBatchMean = sum( BatchMean ) / real(sampleCount,kind=RK)
170 varBatchMean = sum( (BatchMean - avgBatchMean)**2 ) / real(sampleCount-1,kind=RK)
171 avgPoint = avgPoint / real(npEffective,kind=RK)
172
173 ! compute the variance of Point
174
175 varPoint = 0._RK
176 if (present(Weight)) then
177 ip = 1
178 ipVerbose = 0
179 diffSquared = ( Point(ip) - avgPoint )**2
180 loopComputeVarPoint: do
181 ipVerbose = ipVerbose + 1
182 if (ipVerbose>npEffective) exit loopComputeVarPoint
183 if (ipVerbose>CumSumWeight(ip)) then
184 ip = ip + 1 ! by definition, ip never become > np, otherwise it leads to disastrous errors
185 diffSquared = ( Point(ip) - avgPoint )**2
186 end if
187 varPoint = varPoint + diffSquared
188 end do loopComputeVarPoint
189 else
190 do ip = 1, npEffective
191 varPoint = varPoint + ( Point(ip) - avgPoint )**2
192 end do
193 end if
194 varPoint = varPoint / real(npEffective-1,kind=RK)
195
196 ! compute the IAC
197
198 iac = batchSizeDefault * varBatchMean / varPoint
199
200 end function getBatchMeansIAC

215 function genCumSumIAC(np,Point,Weight,significance) result(iac)
216#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
217 !DEC$ ATTRIBUTES DLLEXPORT :: genCumSumIAC
218#endif
219 use Constants_mod, only: IK, RK
220 use Array_mod, only: genCumSum
221 implicit none
222 integer(IK) , intent(in) :: np
223 real(RK) , intent(in) :: Point(np)
224 integer(IK) , intent(in), optional :: Weight(np)
225 real(RK) , intent(in), optional :: significance ! in units of sigma
226 real(RK) :: iac, meanPoint, normFac, NormedData(np), cutoff, significanceDefault
227 real(RK) , allocatable :: AutoCorr(:)
228 integer(IK) :: i, paddedLen, sumWeight, cutoffIndex
229 significanceDefault = 2._RK
230 if (present(significance)) significanceDefault = significance
231 if (present(Weight)) then
232 sumWeight = sum(Weight)
233 meanPoint = sum(Point*Weight) / real(sumWeight,kind=RK)
234 else
235 sumWeight = np
236 meanPoint = sum(Point) / real(np,kind=RK)
237 end if
238 NormedData = Point - meanPoint
239 paddedLen = getPaddedLen(sumWeight)
240 AutoCorr = getCrossCorrWeightedFFT( lenCompactData1 = np &
241 , lenCompactData2 = np &
242 , paddedLen = paddedLen &
243 , CompactData1 = NormedData &
244 , CompactData2 = NormedData &
245 , Weight1 = Weight &
246 , Weight2 = Weight &
247 )
248 normFac = 1._RK / AutoCorr(1)
249 AutoCorr = AutoCorr * normFac
250 ! For autocorrelation, under the assumption of a completely random series, the ACF standard error reduces to sqrt(1/ndata)
251 cutoff = significanceDefault * sqrt(1._RK/sumWeight) ! standardErrorAutoCorr
252 cutoffIndex = 1_IK
253 do i = 1, paddedLen
254 if (AutoCorr(i)<cutoff) then
255 cutoffIndex = i
256 exit
257 end if
258 end do
259 iac = 2_IK * sum(AutoCorr(1:cutoffIndex)) - 1_IK
260 end function genCumSumIAC

496 function getCrossCorrFFT(paddedLen, PaddedData1, PaddedData2) result(CrossCorrFFT)
497#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
498 !DEC$ ATTRIBUTES DLLEXPORT :: getCrossCorrFFT
499#endif
500 use iso_fortran_env, only: output_unit
501 use Constants_mod, only: IK, RK, CK
502
503 implicit none
504
505 integer(IK), intent(in) :: paddedLen
506 real(RK), intent(inout) :: PaddedData1(paddedLen), PaddedData2(paddedLen)
507 real(RK) :: CrossCorrFFT(paddedLen)
508
509 character(len=*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@getCrossCorrFFT()"
510 complex(CK), dimension(paddedLen/2) :: Cdat1, Cdat2
511 integer(IK) :: paddedLenHalf, paddedLenQuarter
512
513 if (iand(paddedLen,paddedLen-1) /= 0) then
514 write(output_unit,"(A)") PROCEDURE_NAME//": paddedLen must be a power of 2."
515 error stop
516 end if
517
518 paddedLenHalf = paddedLen / 2
519 paddedLenQuarter = paddedLenHalf / 2
520 call realft(paddedLen, paddedLenHalf, paddedLenQuarter, PaddedData1, 1_IK, Cdat1)
521 call realft(paddedLen, paddedLenHalf, paddedLenQuarter, PaddedData2, 1_IK, Cdat2)
522 Cdat1(1) = cmplx( real(Cdat1(1)) * real(Cdat2(1)) / paddedLenHalf, aimag(Cdat1(1)) * aimag(Cdat2(1)) / paddedLenHalf , kind = CK )
523 Cdat1(2:paddedLenHalf) = Cdat1(2:paddedLenHalf) * conjg(Cdat2(2:paddedLenHalf)) / paddedLenHalf
524 call realft(paddedLen, paddedLenHalf, paddedLenQuarter, CrossCorrFFT, -1_IK, Cdat1)
525
526 end function getCrossCorrFFT

586 ! Here, `paddedLenQuarter` is 1/4 of the length of the input array, which is already padded with zeros.
587 subroutine realft(paddedLen, paddedLenHalf, paddedLenQuarter, PaddedData, isign, Zdata)
588#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
589 !DEC$ ATTRIBUTES DLLEXPORT :: realft
590#endif
591 use Constants_mod, only: IK, RK, CK
592 use Misc_mod, only: zroots_unity
593
594 implicit none
595
596 integer(IK), intent(in) :: paddedLen, paddedLenHalf, paddedLenQuarter, isign
597 real(RK), intent(inout) :: PaddedData(paddedLen)
598 complex(CK), optional, target :: Zdata(paddedLenHalf)
599
600 complex(CK), dimension(paddedLenQuarter-1) :: h1, h2
601 complex(CK), pointer :: Cdata(:)
602 complex(CK) :: w(paddedLenQuarter)
603 complex(CK) :: z
604 real(RK) :: c1, c2
605
606 c1 = 0.5_RK
607
608 if (present(Zdata)) then
609 Cdata => Zdata
610 if (isign == 1_IK) Cdata = cmplx(PaddedData(1:paddedLen-1:2), PaddedData(2:paddedLen:2), kind=CK)
611 else
612 allocate(Cdata(paddedLenHalf))
613 Cdata = cmplx(PaddedData(1:paddedLen-1:2), PaddedData(2:paddedLen:2), kind=CK)
614 end if
615
616 if (isign == 1_IK) then
617 c2 = -0.5_RK
618 call four1(paddedLenHalf, Cdata, +1_IK)
619 else
620 c2 = 0.5_RK
621 end if
622
623 w = zroots_unity(sign(paddedLen,isign), paddedLenQuarter)
624 w = cmplx(-aimag(w), real(w), kind=CK)
625 h1 = c1 * ( Cdata(2:paddedLenQuarter) + conjg(Cdata(paddedLenHalf:paddedLenQuarter+2:-1)) )
626 h2 = c2 * ( Cdata(2:paddedLenQuarter) - conjg(Cdata(paddedLenHalf:paddedLenQuarter+2:-1)) )
627 Cdata(2:paddedLenQuarter) = h1 + w(2:paddedLenQuarter) * h2
628 Cdata(paddedLenHalf:paddedLenQuarter+2:-1) = conjg(h1 - w(2:paddedLenQuarter) * h2)
629 z = Cdata(1)
630
631 if (isign == 1_IK) then
632 Cdata(1) = cmplx(real(z)+aimag(z), real(z)-aimag(z), kind=CK)
633 else
634 Cdata(1) = cmplx(c1*(real(z)+aimag(z)), c1*(real(z)-aimag(z)), kind=CK)
635 call four1(paddedLenHalf, Cdata, -1)
636 end if
637
638 if (present(Zdata)) then
639 if (isign /= 1_IK) then
640 PaddedData(1:paddedLen-1:2) = real(Cdata)
641 PaddedData(2:paddedLen:2) = aimag(Cdata)
642 end if
643 else
644 PaddedData(1:paddedLen-1:2) = real(Cdata)
645 PaddedData(2:paddedLen:2) = aimag(Cdata)
646 nullify(Cdata)
647 end if
648
649 end subroutine realft

653 ! The `lenCompactData` is the length of the compact input array `CompactData`, which is NOT padded by default.
654 subroutine realftWeighted(lenCompactData, paddedLen, paddedLenHalf, paddedLenQuarter, CompactData, Zdata, Weight)
655#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
656 !DEC$ ATTRIBUTES DLLEXPORT :: realftWeighted
657#endif
658 use Constants_mod, only: IK, RK, CK
659 use Misc_mod, only: zroots_unity
660
661 implicit none
662
663 integer(IK) , intent(in) :: lenCompactData, paddedLen, paddedLenHalf, paddedLenQuarter
664 real(RK) , intent(in) :: CompactData(lenCompactData)
665 integer(IK) , intent(in), optional :: Weight(lenCompactData)
666 complex(CK) , intent(out) :: Zdata(paddedLenHalf)
667
668 integer(IK) :: id, idEnd, iw, counter
669 complex(CK) :: w(paddedLenQuarter)
670 complex(CK) :: h1(paddedLenQuarter-1)
671 complex(CK) :: h2(paddedLenQuarter-1)
672 real(RK) :: zReal, zImag
673 real(RK) :: c1, c2
674
675 c1 = 0.5_RK
676
677 if (present(Weight)) then
678
679 iw = 1
680 counter = 0
681 loopOverCompactData: do id = 1, lenCompactData
682 loopOverWeight: do
683 if (iw>Weight(id)) then
684 iw = 1
685 cycle loopOverCompactData
686 elseif (iw==Weight(id)) then
687 counter = counter + 1
688 if (id==lenCompactData) then
689 Zdata(counter) = cmplx( CompactData(id) , 0._RK , kind=CK )
690 exit loopOverCompactData
691 else
692 Zdata(counter) = cmplx( CompactData(id) , CompactData(id+1) , kind=CK )
693 iw = 2
694 cycle loopOverCompactData
695 end if
696 else
697 counter = counter + 1
698 Zdata(counter) = cmplx( CompactData(id) , CompactData(id) , kind=CK )
699 iw = iw + 2
700 cycle loopOverWeight
701 end if
702 end do loopOverWeight
703 end do loopOverCompactData
704 Zdata(counter+1:paddedLenHalf) = cmplx( 0._RK , 0._RK , kind=CK )
705
706 else
707
708 idEnd = lenCompactData / 2_IK
709 do concurrent(id=1:idEnd)
710 Zdata(id) = cmplx( CompactData(2*id-1) , CompactData(2*id) , kind=CK )
711 end do
712 if (2*idEnd<lenCompactData) then
713 idEnd = idEnd + 1
714 Zdata(idEnd) = cmplx( CompactData(lenCompactData) , 0._RK , kind=CK )
715 end if
716 Zdata(idEnd+1:paddedLenHalf) = cmplx(0._RK, 0._RK, kind=CK)
717
718 end if
719
720 c2 = -0.5_RK
721 call four1(paddedLenHalf,Zdata,1_IK)
722
723 w = zroots_unity(paddedLen, paddedLenQuarter)
724 w = cmplx(-aimag(w), real(w,kind=RK), kind=CK)
725 h1 = c1 * ( Zdata(2:paddedLenQuarter) + conjg(Zdata(paddedLenHalf:paddedLenQuarter+2:-1)) )
726 h2 = c2 * ( Zdata(2:paddedLenQuarter) - conjg(Zdata(paddedLenHalf:paddedLenQuarter+2:-1)) )
727 Zdata(2:paddedLenQuarter) = h1 + w(2:paddedLenQuarter) * h2
728 Zdata(paddedLenHalf:paddedLenQuarter+2:-1) = conjg(h1 - w(2:paddedLenQuarter) * h2)
729 zReal = real(Zdata(1)); zImag = aimag(Zdata(1))
730 Zdata(1) = cmplx(zReal+zImag, zReal-zImag, kind=CK)

732 end subroutine realftWeighted

736 subroutine four1(ndata,Data,isign)
737#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
738 !DEC$ ATTRIBUTES DLLEXPORT :: four1
739#endif
740 use Constants_mod, only: IK, RK, CK, TWO_PI
741 use Misc_mod, only: arth
742 implicit none
743 integer(IK), intent(in) :: ndata, isign
744 complex(CK), intent(inout) :: Data(ndata)
745 complex(CK), allocatable :: dat(:,:), temp(:,:)
746 complex(CK), allocatable :: w(:),wp(:)
747 real(RK) , allocatable :: theta(:)
748 integer(IK) :: m1,m2,j
749 m1=2**ceiling(0.5_RK*log(real(ndata,RK))/0.693147_RK)
750 m2=ndata/m1
751 allocate(dat(m1,m2),theta(m1),w(m1),wp(m1),temp(m2,m1))
752 dat=reshape(Data,shape(dat))
753 call fourrow(dat,isign)
754 theta=arth(0,isign,m1)*TWO_PI/ndata
755 wp=cmplx(-2.0_RK*sin(0.5_RK*theta)**2,sin(theta),kind=CK)
756 w=cmplx(1.0_RK,0.0_RK,kind=CK)
757 do j=2,m2
758 w=w*wp+w
759 dat(:,j)=dat(:,j)*w
760 end do
761 temp=transpose(dat)
762 call fourrow(temp,isign)
763 Data=reshape(temp,shape(Data))
764 deallocate(dat,w,wp,theta,temp)

769 subroutine fourrow(Data,isign)
770#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
771 !DEC$ ATTRIBUTES DLLEXPORT :: fourrow
772#endif
773 use Constants_mod, only: IK, RK, CK, PI
774 use Misc_mod, only: swap
775 implicit none
776 complex(CK), dimension(:,:), intent(inout) :: Data
777 integer(IK), intent(in) :: isign
778 integer(IK) :: n,i,istep,j,m,mmax,n2
779 real(RK) :: theta
780 complex(CK), dimension(size(Data,1)) :: temp
781 complex(CK) :: w,wp
782 complex(CK) :: ws
783 n=size(Data,2)
784 n2=n/2
785 j=n2
786 do i=1,n-2
787 if (j > i) call swap(Data(:,j+1),Data(:,i+1))
788 m=n2
789 do
790 if (m < 2 .or. j < m) exit
791 j=j-m
792 m=m/2
793 end do
794 j=j+m
795 end do
796 mmax=1
797 do
798 if (n <= mmax) exit
799 istep=2*mmax
800 theta=PI/(isign*mmax)
801 wp=cmplx(-2.0_RK*sin(0.5_RK*theta)**2,sin(theta),kind=CK)
802 w=cmplx(1.0_RK,0.0_RK,kind=CK)
803 do m=1,mmax
804 ws=w
805 do i=m,n,istep
806 j=i+mmax
807 temp=ws*Data(:,j)
808 Data(:,j)=Data(:,i)-temp
809 Data(:,i)=Data(:,i)+temp
810 end do
811 w=w*wp+w
812 end do
813 mmax=istep
814 end do
815 end subroutine fourrow

893 end do

895 end if
