75  function getbatchmeansiac(np,Point,Weight,batchSize) result(iac)
76 #if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
77  !DEC$ ATTRIBUTES DLLEXPORT :: getBatchMeansIAC
78 #endif
79  use constants_mod, only: ik, rk
80  use statistics_mod, only: getvariance
81  use math_mod, only: getcumsum
82  !use Stirng_mod, only: num2str
83  !use Err_mod, only: Err_type
84 
85  implicit none
86 
87  character(len=*), parameter :: procedure_name = module_name // "@getBatchMeansIAC()"
88 
89  integer(IK), intent(in) :: np
90  real(rk), intent(in) :: point(np)
91  integer(IK), intent(in), optional :: weight(np), batchsize
92  !type(Err_type), intent(out) :: Err
93  real(rk) :: iac
94 
95  integer(IK) :: cumsumweight(np), currentsampleendloc, batchsizedefault, samplecount, isample
96  integer(IK) :: ip, ipverbose, ipstart, ipend, npeffective
97  real(rk) :: avgpoint, varpoint, avgbatchmean, varbatchmean
98  real(rk) :: diffsquared, batchsizedefaultinverse
99  real(rk), allocatable :: batchmean(:)
100 
101  !Err%occurred = .false.
102 
103  if (present(weight)) then
104  cumsumweight = getcumsum(np,weight)
105  else
106  cumsumweight(np) = np
107  end if
108 
109  ! compute batch size and count
110  if (present(batchsize)) then
111  batchsizedefault = batchsize
112  else
113  batchsizedefault = int( real(cumsumweight(np),kind=rk)**(0.666666666666666_rk) )
114  end if
115  batchsizedefaultinverse = 1._rk / real(batchsizedefault,kind=rk)
116  samplecount = cumsumweight(np) / batchsizedefault
117  npeffective = samplecount * batchsizedefault
118 
119  if (samplecount<2) then
120  !Err%occurred = .true.
121  !Err%msg = PROCEDURE_NAME // ": sampleCount<10: " // num2str(sampleCount)
122  !iac = -huge(iac)
123  iac = 1
124  return
125  end if
126 
127  ! xxx: here goes another GFortran 7.3 bug: EndOfLineLoc is assumed already allocated, despite the first appearance here.
128  if (allocated(batchmean)) deallocate(batchmean)
129  allocate(batchmean(samplecount))
130 
131  ! improvement: iterate from the end to the beginning of the chain to ignore initial points instead of the last points.
132  ! this would be beneficial for MCMC samples
133 
134  ! compute the Batch-Means avergage and variance, also average of Point
135  avgpoint = 0._rk
136  ipstart = 1
137  ipend = 0
138  if (present(weight)) then
139  ip = 1
140  isample = 1
141  ipverbose = 0
142  currentsampleendloc = batchsizedefault
143  batchmean(isample) = 0._rk
144  loopoverweight: do
145  ipverbose = ipverbose + 1
146  if (ipverbose>cumsumweight(ip)) ip = ip + 1
147  if (ipverbose>currentsampleendloc) then ! we are done with the current batch
148  avgpoint = avgpoint + batchmean(isample)
149  batchmean(isample) = batchmean(isample) * batchsizedefaultinverse
150  if (ipverbose>npeffective) exit loopoverweight ! condition equivalent to currentSampleEndLoc==npEffective
151  currentsampleendloc = currentsampleendloc + batchsizedefault
152  isample = isample + 1
153  batchmean(isample) = 0._rk
154  end if
155  batchmean(isample) = batchmean(isample) + point(ip)
156  end do loopoverweight
157  else ! there is no weight
158  do isample = 1, samplecount
159  batchmean(isample) = 0._rk
160  ipend = ipend + batchsizedefault
161  do ip = ipstart, ipend
162  batchmean(isample) = batchmean(isample) + point(ip)
163  end do
164  ipstart = ipend + 1_ik
165  avgpoint = avgpoint + batchmean(isample)
166  batchmean(isample) = batchmean(isample) * batchsizedefaultinverse
167  end do
168  end if
169  avgbatchmean = sum( batchmean ) / real(samplecount,kind=rk)
170  varbatchmean = sum( (batchmean - avgbatchmean)**2 ) / real(samplecount-1,kind=rk)
171  avgpoint = avgpoint / real(npeffective,kind=rk)
172 
173  ! compute the variance of Point
174 
175  varpoint = 0._rk
176  if (present(weight)) then
177  ip = 1
178  ipverbose = 0
179  diffsquared = ( point(ip) - avgpoint )**2
180  loopcomputevarpoint: do
181  ipverbose = ipverbose + 1
182  if (ipverbose>npeffective) exit loopcomputevarpoint
183  if (ipverbose>cumsumweight(ip)) then
184  ip = ip + 1 ! by definition, ip never become > np, otherwise it leads to disastrous errors
185  diffsquared = ( point(ip) - avgpoint )**2
186  end if
187  varpoint = varpoint + diffsquared
188  end do loopcomputevarpoint
189  else
190  do ip = 1, npeffective
191  varpoint = varpoint + ( point(ip) - avgpoint )**2
192  end do
193  end if
194  varpoint = varpoint / real(npeffective-1,kind=rk)
195 
196  ! compute the IAC
197 
198  iac = batchsizedefault * varbatchmean / varpoint
199 
200  end function getbatchmeansiac

215  function getcumsumiac(np,Point,Weight,significance) result(iac)
216 #if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
217  !DEC$ ATTRIBUTES DLLEXPORT :: getCumSumIAC
218 #endif
219  use constants_mod, only: ik, rk
220  use math_mod, only: getcumsum
221  implicit none
222  integer(IK) , intent(in) :: np
223  real(rk) , intent(in) :: point(np)
224  integer(IK) , intent(in), optional :: weight(np)
225  real(rk) , intent(in), optional :: significance ! in units of sigma
226  real(rk) :: iac, meanpoint, normfac, normeddata(np), cutoff, significancedefault
227  real(rk) , allocatable :: autocorr(:)
228  integer(IK) :: i, paddedlen, sumweight, cutoffindex
229  significancedefault = 2._rk
230  if (present(significance)) significancedefault = significance
231  if (present(weight)) then
232  sumweight = sum(weight)
233  meanpoint = sum(point*weight) / real(sumweight,kind=rk)
234  else
235  sumweight = np
236  meanpoint = sum(point) / real(np,kind=rk)
237  end if
238  normeddata = point - meanpoint
239  paddedlen = getpaddedlen(sumweight)
240  autocorr = getcrosscorrweightedfft( lencompactdata1 = np &
241  , lencompactdata2 = np &
242  , paddedlen = paddedlen &
243  , compactdata1 = normeddata &
244  , compactdata2 = normeddata &
245  , weight1 = weight &
246  , weight2 = weight &
247  )
248  normfac = 1._rk / autocorr(1)
249  autocorr = autocorr * normfac
250  ! For autocorrelation, under the assumption of a completely random series, the ACF standard error reduces to sqrt(1/ndata)
251  cutoff = significancedefault * sqrt(1._rk/sumweight) ! standardErrorAutoCorr
252  cutoffindex = 1_ik
253  do i = 1, paddedlen
254  if (autocorr(i)<cutoff) then
255  cutoffindex = i
256  exit
257  end if
258  end do
259  iac = 2_ik * sum(autocorr(1:cutoffindex)) - 1_ik
260  end function getcumsumiac

586  ! Here, `paddedLenQuarter` is 1/4 of the length of the input array, which is already padded with zeros.
587  subroutine realft(paddedLen, paddedLenHalf, paddedLenQuarter, PaddedData, isign, Zdata)
588 #if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
589  !DEC$ ATTRIBUTES DLLEXPORT :: realft
590 #endif
591  use constants_mod, only: ik, rk, ck
592  use misc_mod, only: zroots_unity
593 
594  implicit none
595 
596  integer(IK), intent(in) :: paddedLen, paddedLenHalf, paddedLenQuarter, isign
597  real(RK), intent(inout) :: PaddedData(paddedLen)
598  complex(CK), optional, target :: Zdata(paddedLenHalf)
599 
600  complex(CK), dimension(paddedLenQuarter-1) :: h1, h2
601  complex(CK), pointer :: Cdata(:)
602  complex(CK) :: w(paddedLenQuarter)
603  complex(CK) :: z
604  real(RK) :: c1, c2
605 
606  c1 = 0.5_rk
607 
608  if (present(zdata)) then
609  cdata => zdata
610  if (isign == 1_ik) cdata = cmplx(paddeddata(1:paddedlen-1:2), paddeddata(2:paddedlen:2), kind=ck)
611  else
612  allocate(cdata(paddedlenhalf))
613  cdata = cmplx(paddeddata(1:paddedlen-1:2), paddeddata(2:paddedlen:2), kind=ck)
614  end if
615 
616  if (isign == 1_ik) then
617  c2 = -0.5_rk
618  call four1(paddedlenhalf, cdata, +1_ik)
619  else
620  c2 = 0.5_rk
621  end if
622 
623  w = zroots_unity(sign(paddedlen,isign), paddedlenquarter)
624  w = cmplx(-aimag(w), real(w), kind=ck)
625  h1 = c1 * ( cdata(2:paddedlenquarter) + conjg(cdata(paddedlenhalf:paddedlenquarter+2:-1)) )
626  h2 = c2 * ( cdata(2:paddedlenquarter) - conjg(cdata(paddedlenhalf:paddedlenquarter+2:-1)) )
627  cdata(2:paddedlenquarter) = h1 + w(2:paddedlenquarter) * h2
628  cdata(paddedlenhalf:paddedlenquarter+2:-1) = conjg(h1 - w(2:paddedlenquarter) * h2)
629  z = cdata(1)
630 
631  if (isign == 1_ik) then
632  cdata(1) = cmplx(real(z)+aimag(z), real(z)-aimag(z), kind=ck)
633  else
634  cdata(1) = cmplx(c1*(real(z)+aimag(z)), c1*(real(z)-aimag(z)), kind=ck)
635  call four1(paddedlenhalf, cdata, -1)
636  end if
637 
638  if (present(zdata)) then
639  if (isign /= 1_ik) then
640  paddeddata(1:paddedlen-1:2) = real(cdata)
641  paddeddata(2:paddedlen:2) = aimag(cdata)
642  end if
643  else
644  paddeddata(1:paddedlen-1:2) = real(cdata)
645  paddeddata(2:paddedlen:2) = aimag(cdata)
646  nullify(cdata)
647  end if
648 
649  end subroutine realft
