91 subroutine getRefinedChain( RefinedChain &
92 , CFC &
93 , Err &
94 , refinedChainSize &
95 )
96#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
97 !DEC$ ATTRIBUTES DLLEXPORT :: getRefinedChain
98#endif
99 use, intrinsic :: iso_fortran_env, only: output_unit
100 use Constants_mod, only: NEGLOGINF_RK
101 use MathCumPropExp_mod, only: genCumPropExp

105 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@getRefinedChain()"
106
107 class(RefinedChain_type) , intent(inout) :: RefinedChain
108 type(ChainFileContents_type), intent(inout) :: CFC
109 type(Err_type) , intent(out) :: Err
110 integer(IK) , intent(in) :: refinedChainSize
111
112 real(RK) :: unifrnd
113 real(RK) , allocatable :: WeightCDF(:)
114 integer(IK) , allocatable :: SampleWeight(:)
115 integer(IK) :: i, ip
116
117 Err%occurred = .false.
118
119 ! This is to avoid memory overflow due to extremely large maxRefinementCount requested by the user.
120
121 if (CFC%ndim == 0_IK) then
122 RefinedChain%ndim = size(CFC%State(:,1))
123 CFC%ndim = RefinedChain%ndim
124 else
125 RefinedChain%ndim = CFC%ndim
126 end if
127
128 ! Allocate components.
129
130 RefinedChain%Size(1)%verbose = refinedChainSize
131 if (CFC%size == 0_IK) CFC%size = size(CFC%State(1,:)) ! LCOV_EXCL_LINE
132
133 ! Assign the column headers.
134
135 if (allocated(CFC%Header)) then
136 if (allocated(RefinedChain%Header)) deallocate(RefinedChain%Header)
137 allocate(RefinedChain%Header(0:RefinedChain%ndim))
138 do i = 0, RefinedChain%ndim
139 RefinedChain%Header(i)%value = CFC%Header(i+NUM_DEF_COL)%value
140 end do
141 else
142 Err%occurred = .true. ! LCOV_EXCL_LINE
143 Err%msg = PROCEDURE_NAME // ": Fatal error occurred. CFC%Header is not allocated." ! LCOV_EXCL_LINE
144 error stop ! LCOV_EXCL_LINE
145 end if
146
147 if (allocated(SampleWeight)) deallocate(SampleWeight) ! LCOV_EXCL_LINE
148 allocate(SampleWeight(CFC%size), source = 0_IK)
149
150 !RefinedChain%LogFuncState(0 ,1:RefinedChain%Size(1)%compact) = CFC%LogFunc(1:CFC%size)
151 !RefinedChain%LogFuncState(1:RefinedChain%ndim,1:RefinedChain%Size(1)%compact) = CFC%State(1:RefinedChain%ndim, 1:CFC%size)
152
153 ! Refine the weighted chain.
154 ! \todo: This refinement strategy should be improved. The burnin part must be excluded.
155
156 !WeightCDF = genCumPropExp(CFC%size, LogValue = CFC%LogWeight(1:CFC%size), maxLogValue = CFC%maxLogWeight)
157 WeightCDF = genCumPropExp(CFC%LogWeight(1:CFC%size), CFC%maxLogWeight)
158
159 RefinedChain%Size(1)%compact = 0_IK
160 do i = 1, refinedChainSize
161 call random_number(unifrnd)
162 ip = minloc(WeightCDF, dim = 1, mask = WeightCDF > unifrnd)
163 if (SampleWeight(ip) == 0_IK) RefinedChain%Size(1)%compact = RefinedChain%Size(1)%compact + 1_IK
164 SampleWeight(ip) = SampleWeight(ip) + 1_IK
165 end do
166
167 if (allocated(RefinedChain%LogFuncState)) deallocate(RefinedChain%LogFuncState) ! LCOV_EXCL_LINE
168 allocate(RefinedChain%LogFuncState(0:RefinedChain%ndim, RefinedChain%Size(1)%compact))
169 if (allocated(RefinedChain%Weight)) deallocate(RefinedChain%Weight) ! LCOV_EXCL_LINE
170 allocate(RefinedChain%Weight(RefinedChain%Size(1)%compact))
171 !write(*,*) "RefinedChain%Size(1)%compact", RefinedChain%Size(1)%compact
172 !write(*,*) "RefinedChain%Size(1)%verbose", RefinedChain%Size(1)%verbose
173 ip = 0_IK
174 do i = 1, CFC%size
175 if (SampleWeight(i) > 0_IK) then
176 ip = ip + 1_IK
177 RefinedChain%Weight(ip) = SampleWeight(i)
178 RefinedChain%LogFuncState(0,ip) = CFC%LogFunc(i)
179 RefinedChain%LogFuncState(1:RefinedChain%ndim,ip) = CFC%State(1:RefinedChain%ndim,i)
180 end if
181 end do
182
183 end subroutine getRefinedChain

232 function readRefinedChain(sampleFilePath,delimiter,ndim,tenabled) result(RefinedChain)
233#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
234 !DEC$ ATTRIBUTES DLLEXPORT :: readRefinedChain
235#endif
236 use FileContents_mod, only: getNumRecordInFile
237 use String_mod, only: StrSplit_type, val2char
238 implicit none
239 integer(IK) , intent(in) :: ndim
240 character(*), intent(in) :: sampleFilePath, delimiter
241 logical , intent(in), optional :: tenabled
242 type(RefinedChain_type) :: RefinedChain
243 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@readRefinedChain()"
244 integer(IK) :: sampleFileUnit, isample, i
245 logical :: tenabledDefault
246 type(StrSplit_type) :: Line
247
248 if (allocated(Line%value)) deallocate(Line%value) ! LCOV_EXCL_LINE
249 allocate( character(99999) :: Line%value )
250
251 RefinedChain%ndim = ndim
252
253 ! find the number of lines in the sample file
254
255 call getNumRecordInFile(filePath=sampleFilePath, numRecord=RefinedChain%Size(1)%verbose, Err=RefinedChain%Err)
256 if (RefinedChain%Err%occurred) then
257 ! LCOV_EXCL_START
258 RefinedChain%Err%msg = PROCEDURE_NAME // RefinedChain%Err%msg
259 return
260 ! LCOV_EXCL_STOP
261 end if
262
263 RefinedChain%Size(1)%verbose = RefinedChain%Size(1)%verbose - 1_IK ! remove header from the count
264
265 open( newunit = sampleFileUnit &
266 , file = sampleFilePath &
267 , status = "old" &
268#if INTEL_COMPILER_ENABLED && WINDOWS_ENABLED
269 , SHARED &
270#endif
271 )
272
273 ! read header
274
275 read(sampleFileUnit,"(A)") Line%value
276 Line%Parts = Line%split(Line%value,delimiter,Line%nPart)
277 if (Line%nPart/=ndim+1_IK) then
278 ! LCOV_EXCL_START
279 RefinedChain%Err%occurred = .true.
280 RefinedChain%Err%msg = PROCEDURE_NAME // ": The number of header columns ("//val2char(Line%nPart)//") is not equal to ndim + 1: "//val2char(ndim+1_IK)
281 return
282 ! LCOV_EXCL_STOP
283 end if
284
285 allocate(RefinedChain%Header(0:ndim))
286 do i = 0, ndim
287 RefinedChain%Header(i)%value = Line%Parts(i+1)%value
288 end do
289
290 ! read contents
291
292 tenabledDefault = .false.
293 if (present(tenabled)) tenabledDefault = tenabled
294
295 if (tenabledDefault) then
296
297 allocate( RefinedChain%LogFuncState(RefinedChain%Size(1)%verbose, 0:ndim) )
298 do isample = 1, RefinedChain%Size(1)%verbose
299 read(sampleFileUnit, "(A)") Line%value
300 Line%Parts = Line%split(trim(adjustl(Line%value)),delimiter)
301 do i = 0, ndim
302 read(Line%Parts(i+1)%value,*) RefinedChain%LogFuncState(isample,i)
303 end do
304 end do
305
306 else
307
308 allocate( RefinedChain%LogFuncState(0:ndim, RefinedChain%Size(1)%verbose) )
309 do isample = 1, RefinedChain%Size(1)%verbose
310 read(sampleFileUnit, "(A)") Line%value
311 Line%Parts = Line%split(trim(adjustl(Line%value)),delimiter)
312 do i = 0, ndim
313 read(Line%Parts(i+1)%value,*) RefinedChain%LogFuncState(i,isample)
314 end do
315 end do
316
317 end if
318
319 close(sampleFileUnit)
320
321 end function readRefinedChain
