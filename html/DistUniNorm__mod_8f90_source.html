123 function construct(mean,std) result(self)
124#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
125 !DEC$ ATTRIBUTES DLLEXPORT :: construct
126#endif
127 real(RK) , intent(in), optional :: mean, std
128 type(DistUniNorm_type) :: self
129 if (present(mean)) then
130 self%mean = mean
131 else
132 self%mean = 0._RK
133 end if
134 if (present(std)) then
135 self%std = std
136 self%var = std**2
137 self%stdInverse = 1._RK / self%std
138 self%stdInverseLog = log(self%stdInverse)
139 else
140 self%std = 1._RK
141 self%var = 1._RK
142 self%stdInverse = 1._RK
143 self%stdInverseLog = 0._RK
144 end if
145 self%mad = 0._RK
146 self%mode = self%mean
147 self%median = self%mean
148 self%skewness = 0._RK
149 self%kurtosis = 0._RK
150 self%mad = getMad(std = self%std)
151 self%entropy = getEntropy(logVar = 2 * log(self%std))
152 self%FisherInfo = getFisherInfo(varInverse = self%stdInverse**2)
153 self%logNormFacPDF = getLogNormFacPDF(stdInverseLog = self%stdInverseLog)
154 end function construct

561 function genRandStan() result(randev)
562#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
563 !DEC$ ATTRIBUTES DLLEXPORT :: genRandStan
564#endif
565 integer(IK), save :: iset = 0_IK
566 real(RK) , save :: gset
567 real(RK) :: fac,rsq,vec(2)
568 real(RK) :: randev
569 if (iset == 0_IK) then
570 do
571 !block
572 !integer :: i, n
573 !real(RK) :: unifrnd(30)
574 !!integer, dimension(:), allocatable :: seed
575 !if (paradramPrintEnabled .or. paradisePrintEnabled) then
576 ! !do i = 1, 22
577 ! call random_number(unifrnd)
578 ! write(*,"(*(g0,:,'"//new_line("a")//"'))") unifrnd
579 ! !end do
580 ! !call random_seed(size = n); allocate(seed(n))
581 ! !call random_seed(get = seed)
582 ! !write(*,"(*(g0,:,' '))") seed
583 ! !write(*,"(*(g0,:,' '))") StateOld
584 ! !write(*,"(*(g0,:,' '))") StateNew
585 ! !write(*,"(*(g0,:,' '))") ChoLow
586 ! !write(*,"(*(g0,:,' '))") domainCheckCounter
587 ! paradisePrintEnabled = .false.
588 ! paradramPrintEnabled = .false.
589 !end if
590 !end block
591 call random_number(vec)
592 vec = 2._RK * vec - 1._RK
593 rsq = vec(1)**2 + vec(2)**2
594 if (rsq > 0._RK .and. rsq < 1._RK) exit
595 end do
596 fac = sqrt(-2._RK * log(rsq) / rsq)
597 randev = vec(2) * fac
598 gset = vec(1) * fac
599 iset = 1_IK
600 else
601 randev = gset
602 iset = 0_IK
603 endif
604 end function genRandStan
