241 function construct(mean,std) result(self)
242#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
243 !DEC$ ATTRIBUTES DLLEXPORT :: construct
244#endif
245 real(RK) , intent(in), optional :: mean, std
246 type(DistUniNorm_type) :: self
247 if (present(mean)) then
248 self%mean = mean
249 else
250 self%mean = 0._RK
251 end if
252 if (present(std)) then
253 self%std = std
254 self%var = std**2
255 self%stdInverse = 1._RK / self%std
256 self%stdInverseLog = log(self%stdInverse)
257 else
258 self%std = 1._RK
259 self%var = 1._RK
260 self%stdInverse = 1._RK
261 self%stdInverseLog = 0._RK
262 end if
263 self%mad = 0._RK
264 self%mode = self%mean
265 self%median = self%mean
266 self%skewness = 0._RK
267 self%kurtosis = 0._RK
268 self%mad = getMad(std = self%std)
269 self%entropy = getEntropy(logVar = 2 * log(self%std))
270 self%FisherInfo = getFisherInfo(varInverse = self%stdInverse**2)
271 self%logNormFacPDF = getLogNormFacPDF(stdInverseLog = self%stdInverseLog)
272 end function construct

437#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)

440 real(RK) , intent(in) :: meanRef, varRef, mean, var
