134 subroutine setTic(Timer)
135#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
136 !DEC$ ATTRIBUTES DLLEXPORT :: setTic
137#endif
138 use Constants_mod, only: RK
139 implicit none
140 class(Timer_type), intent(inout) :: Timer
141 call system_clock(count = Timer%Count%start, count_rate = Timer%Count%rate, count_max = Timer%Count%max)
142 Timer%period = 1._RK / Timer%Count%rate
143 Timer%Count%stop = Timer%Count%start
144 Timer%Count%total = 0
145 Timer%Count%delta = 0
146 Timer%Time%start = real(Timer%Count%start,kind=RK) * Timer%period

171 subroutine setToc(Timer)
172#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
173 !DEC$ ATTRIBUTES DLLEXPORT :: setToc
174#endif
175 implicit none
176 class(Timer_type), intent(inout) :: Timer
177 integer(IK64) :: stopCount
178 real(RK) :: stopTime
179 call system_clock(count = stopCount)
180 Timer%Count%delta = stopCount - Timer%Count%stop
181 Timer%Count%total = stopCount - Timer%Count%start
182 Timer%Count%stop = stopCount
183 stopTime = real(stopCount,kind=RK) * Timer%period
184 Timer%Time%delta = stopTime - Timer%Time%stop
185 Timer%Time%total = stopTime - Timer%Time%start
186 Timer%Time%stop = stopTime
187 end subroutine setToc
