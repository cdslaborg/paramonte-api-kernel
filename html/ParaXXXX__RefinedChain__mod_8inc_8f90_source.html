123 subroutine getrefinedchain ( RefinedChain &
124 , CFC &
125 , Err &
126 , burninLoc &
127 , refinedChainSize &
128 , sampleRefinementCount &
129 , sampleRefinementMethod &
130 )
131#if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
132 !DEC$ ATTRIBUTES DLLEXPORT :: getRefinedChain
133#endif
134
135 use, intrinsic :: iso_fortran_env, only: output_unit
136 use paraxxxx_chainfilecontents_mod, only: chainfilecontents_type, num_def_col
137 use crosscorr_mod, only: getbatchmeansiac, getcumsumiac, getmaxcumsumiac
138 use string_mod, only: getlowercase, replacestr
139 use sort_mod, only: getmedian
140
141 implicit none
142
143 character(*), parameter :: PROCEDURE_NAME = module_name//"@getRefinedChain()"
144
145 class(RefinedChain_type) , intent(inout) :: RefinedChain
146 type(ChainFileContents_type), intent(inout) :: CFC
147 type(Err_type) , intent(out) :: Err
148 character(*) , intent(in), optional :: sampleRefinementMethod
149 integer(IK) , intent(in), optional :: burninLoc, refinedChainSize, sampleRefinementCount
150 integer(IK) :: burninLocDefault, i, countCompact, ndimPlusOne
151 integer(IK) :: maxRefinementCurrent, maxRefinementCount
152 real(RK) :: integratedAutoCorrTime, ndimPlusOneInverse
153 logical :: refinedChainSizeIsPresent
154 logical :: maxRefinementCountIsReached
155 character(:) , allocatable :: sampleRefinementMethodDefault, sampleRefinementMethodDefaultLowerCase, srmethod
156 type(Count_type), allocatable :: DumCount(:)
157 real(RK) , allocatable :: DumIAC(:,:), DumVec(:)
158 integer(IK) , allocatable :: SampleWeight(:)
159 type(Method_type) :: Method
160 type(ChainFileContents_type) :: DumCFC
161
162 err%occurred = .false.
163
164 ! if the size of the refined sample is given as input, then generate the requested sample straight
165
166 refinedchainsizeispresent = present(refinedchainsize)
167 if (refinedchainsizeispresent) then ! ignore sampleRefinementCount, even if it is given by the user
168 maxrefinementcount = 1_ik
169 else
170 maxrefinementcount = 20_ik ! this is a temporary maximum value, to be increased later if needed
171 if (present(samplerefinementcount)) maxrefinementcount = samplerefinementcount
172 end if
173
174 ! this is to avoid memory overflow due to extremely large maxRefinementCount requested by the user
175
176 maxrefinementcurrent = min(2_ik, maxrefinementcount)
177
178 ! compute ndim and the initial chain size
179
180 refinedchain%numRefinement = 0_ik
181
182 if (cfc%ndim == 0_ik) then
183 refinedchain%ndim = size(cfc%State(:,1))
184 cfc%ndim = refinedchain%ndim
185 else
186 refinedchain%ndim = cfc%ndim
187 end if
188
189 ! allocate components
190
191 if (allocated(refinedchain%IAC)) deallocate(refinedchain%IAC)
192 if (allocated(refinedchain%Count)) deallocate(refinedchain%Count)
193 allocate(refinedchain%IAC(0:refinedchain%ndim,0:maxrefinementcurrent))
194 allocate(refinedchain%Count(0:maxrefinementcurrent))
195 if (cfc%Count%compact == 0_ik) then
196 refinedchain%Count(refinedchain%numRefinement)%compact = size(cfc%State(1,:))
197 cfc%Count%compact = refinedchain%Count(refinedchain%numRefinement)%compact
198 else
199 refinedchain%Count(refinedchain%numRefinement)%compact = cfc%Count%compact
200 end if
201 if (cfc%Count%verbose == 0_ik) cfc%Count%verbose = sum(cfc%Weight(1:cfc%Count%compact))
202 refinedchain%Count(refinedchain%numRefinement)%verbose = cfc%Count%verbose
203
204 ! define the AIC computation method
205
206 samplerefinementmethoddefault = batch_means_method_name; if (present(samplerefinementmethod)) samplerefinementmethoddefault = samplerefinementmethod
207 samplerefinementmethoddefaultlowercase = getlowercase(samplerefinementmethoddefault)
208
209 if (index(samplerefinementmethoddefaultlowercase,getlowercase(batch_means_method_name))>0) then
210 method%isBatchMeans = .true.
211 elseif (index(samplerefinementmethoddefaultlowercase,getlowercase(cutoff_autocorr_method_name))>0 .or. index(samplerefinementmethoddefaultlowercase,"cutoff")>0) then
212 method%isCutoffAutoCorr = .true.
213 elseif (index(samplerefinementmethoddefaultlowercase,getlowercase(max_cumsum_autocorr_method_name))>0 .or. index(samplerefinementmethoddefaultlowercase,"cumsum")>0) then
214 method%isMaxCumSumAutoCorr = .true.
215 else
216 ! LCOV_EXCL_START
217 err%occurred = .true.
218 err%msg = procedure_name // ": Unknown unsupported IAC computation method name: " // samplerefinementmethoddefault
219 return
220 ! LCOV_EXCL_STOP
221 end if
222
223 ! define the chain types to use for the AIC computation
224
225 method%isViaCompactChain = index(samplerefinementmethoddefaultlowercase,"compact") > 0
226 method%isViaVerboseChain = index(samplerefinementmethoddefaultlowercase,"verbose") > 0
227 if (.not.(method%isViaCompactChain .or. method%isViaVerboseChain)) then
228 method%isViaCompactChain = .true.
229 method%isViaVerboseChain = .true.
230 end if
231
232 ! define the statistic to use for the AIC computation
233
234 srmethod = replacestr(string=samplerefinementmethoddefaultlowercase,search=" ",substitute="-")
235 method%isAvg = index(srmethod,"-avg") > 0 .or. index(srmethod,"-average") > 0
236 method%isMed = index(srmethod,"-med") > 0 .or. index(srmethod,"-median") > 0
237 method%isMin = index(srmethod,"-min") > 0 .or. index(srmethod,"-minimum") > 0
238 method%isMax = index(srmethod,"-max") > 0 .or. index(srmethod,"-maximum") > 0
239 if ( method%isAvg .and. (method%isMed .or. method%isMax .or. method%isMin) ) err%occurred = .true.
240 if ( method%isMed .and. (method%isMax .or. method%isMin) ) err%occurred = .true.
241 if ( method%isMax .and. method%isMin ) err%occurred = .true.
242 if (.not.(method%isAvg .or. method%isMed .or. method%isMin .or. method%isMax)) method%isAvg = .true. ! default method of AIC summarization.
243
244 if (method%isAvg) ndimplusoneinverse = 1._rk / (refinedchain%ndim + 1_ik)
245 if (method%isMed) ndimplusone = refinedchain%ndim + 1_ik
246
247 ! assign the column headers
248
249 if (allocated(cfc%ColHeader)) then
250 if (allocated(refinedchain%ColHeader)) deallocate(refinedchain%ColHeader)
251 allocate(refinedchain%ColHeader(0:refinedchain%ndim))
252 do i = 0, refinedchain%ndim
253 refinedchain%ColHeader(i)%record = cfc%ColHeader(i+num_def_col)%record
254 end do
255 else
256 ! LCOV_EXCL_START
257 err%occurred = .true.
258 err%msg = procedure_name // ": Internal error occurred. CFC%ColHeader is NULL."
259 return
260 ! LCOV_EXCL_STOP
261 end if
262
263 if (present(burninloc)) then
264 burninlocdefault = burninloc
265 else
266 burninlocdefault = cfc%BurninLoc(cfc%Count%compact)
267 end if
268
269 refinedchain%Count(refinedchain%numRefinement)%compact = cfc%Count%compact - burninlocdefault + 1
270 if (allocated(refinedchain%LogFuncState)) deallocate(refinedchain%LogFuncState)
271 allocate(refinedchain%LogFuncState(0:refinedchain%ndim,refinedchain%Count(refinedchain%numRefinement)%compact))
272 refinedchain%LogFuncState(0 ,1:refinedchain%Count(refinedchain%numRefinement)%compact) = cfc%LogFunc(burninlocdefault:cfc%Count%compact)
273 refinedchain%LogFuncState(1:refinedchain%ndim,1:refinedchain%Count(refinedchain%numRefinement)%compact) = cfc%State(1:refinedchain%ndim, burninlocdefault:cfc%Count%compact)
274
275 ! check if there are more than 1 sample points in the burnin-subtracted CFC
276
277 if (refinedchain%Count(refinedchain%numRefinement)%compact==0_ik) then
278 ! LCOV_EXCL_START
279 err%occurred = .true.
280 err%msg = procedure_name // ": The size of the refined sample is zero."
281 return
282 ! LCOV_EXCL_STOP
283 elseif (refinedchain%Count(refinedchain%numRefinement)%compact==1_ik) then
284 if (allocated(refinedchain%Weight)) deallocate(refinedchain%Weight)
285 allocate(refinedchain%Weight(refinedchain%Count(refinedchain%numRefinement)%compact))
286 if (refinedchainsizeispresent) then
287 refinedchain%Weight = refinedchainsize
288 else
289 refinedchain%Weight = 1
290 end if
291 if (allocated(refinedchain%IAC)) deallocate(refinedchain%IAC)
292 if (allocated(refinedchain%Count)) deallocate(refinedchain%Count)
293 allocate(refinedchain%IAC(0:refinedchain%ndim,0:0))
294 allocate(refinedchain%Count(0:0))
295 refinedchain%IAC = 0._rk
296 refinedchain%Count(refinedchain%numRefinement)%verbose = sum(refinedchain%Weight)
297 return
298 end if
299
300 refinedchain%Weight = cfc%Weight(burninlocdefault:cfc%Count%compact) ! Weight is intentionally separately assigned from State here
301
302 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
303
304 looprefinement: do
305
306 countcompact = refinedchain%Count(refinedchain%numRefinement)%compact
307
308 ! set the sample weight
309
310 if (method%isViaCompactChain) then
311 if (allocated(sampleweight)) deallocate(sampleweight)
312 elseif (method%isViaVerboseChain) then
313 sampleweight = refinedchain%Weight
314 endif
315
316 ! obtain the IAC for each individual variable
317
318 do i = 0, refinedchain%ndim
319 dumvec = refinedchain%LogFuncState(i,1:refinedchain%Count(refinedchain%numRefinement)%compact)
320 if (method%isBatchMeans) then
321 refinedchain%IAC(i,refinedchain%numRefinement) = getbatchmeansiac( np = countcompact &
322 , point = dumvec & ! LCOV_EXCL_LINE
323 , weight = sampleweight & ! LCOV_EXCL_LINE
324 )
325 elseif (method%isCutoffAutoCorr) then
326 refinedchain%IAC(i,refinedchain%numRefinement) = getcumsumiac( np = countcompact &
327 , point = dumvec & ! LCOV_EXCL_LINE
328 , weight = sampleweight & ! LCOV_EXCL_LINE
329 )
330 elseif (method%isMaxCumSumAutoCorr) then
331 refinedchain%IAC(i,refinedchain%numRefinement) = getmaxcumsumiac( np = countcompact &
332 , point = dumvec & ! LCOV_EXCL_LINE
333 , weight = sampleweight & ! LCOV_EXCL_LINE
334 )
335 end if
336 end do
337
338 if (refinedchainsizeispresent) then
339 integratedautocorrtime = real(sum(refinedchain%Weight),kind=rk) / real(refinedchainsize,kind=rk)
340 else
341 if (method%isAvg) then
342 integratedautocorrtime = sum( refinedchain%IAC(0:refinedchain%ndim,refinedchain%numRefinement) ) * ndimplusoneinverse
343 elseif (method%isMed) then
344 call getmedian(lenarray=ndimplusone,array=refinedchain%IAC(0:refinedchain%ndim,refinedchain%numRefinement),median=integratedautocorrtime,err=err)
345 if (err%occurred) then
346 ! LCOV_EXCL_START
347 err%msg = procedure_name//err%msg
348 return
349 ! LCOV_EXCL_STOP
350 end if
351 elseif (method%isMax) then
352 integratedautocorrtime = maxval( refinedchain%IAC(0:refinedchain%ndim,refinedchain%numRefinement) )
353 elseif (method%isMin) then
354 integratedautocorrtime = minval( refinedchain%IAC(0:refinedchain%ndim,refinedchain%numRefinement) )
355 end if
356 end if
357
358 ! so far, we have computed the max IAC of the sample, but no refinement. Refine the sample only if needed.
359
360 maxrefinementcountisreached = refinedchain%numRefinement==maxrefinementcount
361 if (integratedautocorrtime<2._rk .or. maxrefinementcountisreached) then
362 if (method%isViaCompactChain .and. method%isViaVerboseChain) then
363 !if (maxRefinementCountIsReached) maxRefinementCount = maxRefinementCount * 2_IK
364 maxrefinementcount = maxrefinementcount * 2_ik
365 method%isViaCompactChain = .false.
366 cycle looprefinement
367 end if
368 exit looprefinement
369 end if
370
371 ! generate the refined sample, dump it in CFC, then put it back into RefinedChain to start over again
372
373 refinedchain%numRefinement = refinedchain%numRefinement + 1_ik
374
375 ! reallocate to bigger array if nedded
376
377 if (refinedchain%numRefinement>maxrefinementcurrent) then
378 call move_alloc( from = refinedchain%IAC, to = dumiac )
379 call move_alloc( from = refinedchain%Count, to = dumcount )
380 maxrefinementcurrent = min( maxrefinementcurrent*2 , maxrefinementcount )
381 allocate( refinedchain%IAC( 0:refinedchain%ndim , 0:maxrefinementcurrent ) )
382 allocate( refinedchain%Count( 0:maxrefinementcurrent ) )
383 refinedchain%IAC(0:refinedchain%ndim,0:refinedchain%numRefinement-1) = dumiac
384 refinedchain%Count(0:refinedchain%numRefinement-1) = dumcount
385 end if
386
387 if (integratedautocorrtime<2._rk) cycle looprefinement ! no need for refinement. should happen only when transitioning from compact to verbose
388
389 call refineweightedsample ( nd = refinedchain%ndim & ! LCOV_EXCL_LINE
390 , np = countcompact & ! LCOV_EXCL_LINE
391 , skip = integratedautocorrtime & ! LCOV_EXCL_LINE
392 , sample = refinedchain%LogFuncState & ! LCOV_EXCL_LINE
393 , weight = refinedchain%Weight & ! LCOV_EXCL_LINE
394 , refinedchain = dumcfc%State & ! LCOV_EXCL_LINE
395 , refinedweight = dumcfc%Weight & ! LCOV_EXCL_LINE
396 , pointcount = refinedchain%Count(refinedchain%numRefinement) &
397 , refinedchainsize = refinedchainsize & ! LCOV_EXCL_LINE
398 )
399 refinedchain%Weight = dumcfc%Weight
400 refinedchain%LogFuncState = dumcfc%State
401
402 end do looprefinement
403
404 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
405
406 end subroutine getrefinedchain

432 pure function getrefinedweight(np,Weight,skip,refinedChainSize) result(RefinedWeight)
433#if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
434 !DEC$ ATTRIBUTES DLLEXPORT :: getRefinedWeight
435#endif
436 use math_mod, only: getcumsum
437 use constants_mod, only: ik, rk
438 implicit none
439 real(RK), intent(in) :: skip
440 integer(IK), intent(in) :: np, Weight(np)
441 integer(IK) , intent(in) , optional :: refinedChainSize
442 integer(IK) :: RefinedWeight(np)
443 integer(IK) :: ip, refinedChainSizeCounter, offset
444 real(RK) :: sumSkips, CumSumWeight(np)
445 logical :: refinedChainSizeIsPresent
446 refinedchainsizeispresent = present(refinedchainsize)
447 if (refinedchainsizeispresent) refinedchainsizecounter = 0_ik
448 cumsumweight = real(getcumsum(np,weight),kind=rk)
449 sumskips = skip
450 offset = 1_ik
451 ip = offset
452 refinedweight = 0_ik
453 loopoverallsample: do
454 loopovercurrentsample: do
455 if (sumskips>cumsumweight(ip)) then
456 exit loopovercurrentsample
457 elseif (refinedchainsizeispresent) then
458 if (refinedchainsizecounter==refinedchainsize) exit loopoverallsample
459 refinedchainsizecounter = refinedchainsizecounter + 1_ik
460 end if
461 refinedweight(ip) = refinedweight(ip) + 1_ik
462 sumskips = sumskips + skip
463 cycle loopovercurrentsample
464 end do loopovercurrentsample
465 if (ip==np) then
466 if (refinedchainsizeispresent) then
467 if (refinedchainsizecounter<refinedchainsize) then
468 offset = offset + 1_ik
469 if (offset==np) offset = 1_ik
470 ip = offset
471 sumskips = skip
472 if (offset/=1_ik) sumskips = sumskips + cumsumweight(ip-1)
473 cycle loopoverallsample ! LCOV_EXCL_LINE
474 end if
475 end if
476 exit loopoverallsample
477 end if
478 ip = ip + 1_ik
479 end do loopoverallsample
480 end function getrefinedweight

497 pure subroutine refineweightedsample(nd,np,skip,Sample,Weight,RefinedChain,RefinedWeight,PointCount,refinedChainSize)
498#if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
499 !DEC$ ATTRIBUTES DLLEXPORT :: refineWeightedSample
500#endif
501 use constants_mod, only: ik, rk
502 implicit none
503 integer(IK) , intent(in) :: nd, np, Weight(np) !, skip
504 real(RK) , intent(in) :: Sample(0:nd,np), skip
505 real(RK) , intent(out), allocatable :: RefinedChain(:,:)
506 integer(IK) , intent(out), allocatable :: RefinedWeight(:)
507 integer(IK) , intent(in) , optional :: refinedChainSize
508 type(Count_type), intent(out) :: PointCount
509 integer(IK) :: ip, ipRefined, npRefined, UpdatedWeight(np) ! LCOV_EXCL_LINE
510 updatedweight = getrefinedweight(np,weight,skip,refinedchainsize)
511 nprefined = count(updatedweight>0)
512 allocate( refinedchain(0:nd,nprefined) , refinedweight(nprefined) )
513 iprefined = 0
514 pointcount%verbose = 0
515 do ip = 1, np
516 if (updatedweight(ip)>0) then
517 iprefined = iprefined + 1
518 refinedchain(0:nd,iprefined) = sample(0:nd,ip)
519 refinedweight(iprefined) = updatedweight(ip)
520 pointcount%verbose = pointcount%verbose + refinedweight(iprefined)
521 end if
522 end do
523 pointcount%compact = nprefined
524 end subroutine refineweightedsample

602 function readrefinedchain(sampleFilePath,delimiter,ndim,tenabled) result(RefinedChain)
603#if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
604 !DEC$ ATTRIBUTES DLLEXPORT :: readRefinedChain
605#endif
606 use filecontents_mod, only: getnumrecordinfile
607 use string_mod, only: string_type, num2str
608 implicit none
609 integer(IK) , intent(in) :: ndim
610 character(*), intent(in) :: sampleFilePath, delimiter
611 logical , intent(in), optional :: tenabled
612 type(RefinedChain_type) :: RefinedChain
613 character(*), parameter :: PROCEDURE_NAME = module_name//"@readRefinedChain()"
614 integer(IK) :: sampleFileUnit, isample, i
615 logical :: tenabledDefault
616 type(String_type) :: Record
617
618 if (allocated(record%value)) deallocate(record%value) ! LCOV_EXCL_LINE
619 allocate( character(99999) :: Record%value )
620
621 refinedchain%ndim = ndim
622 refinedchain%numRefinement = 0_ik
623 allocate(refinedchain%Count(refinedchain%numRefinement:refinedchain%numRefinement)) ! allocate just the zeroth level of `RefinedChain`.
624
625 ! find the number of lines in the sample file
626
627 call getnumrecordinfile(filepath=samplefilepath, numrecord=refinedchain%Count(refinedchain%numRefinement)%verbose, err=refinedchain%Err)
628 if (refinedchain%Err%occurred) then
629 ! LCOV_EXCL_START
630 refinedchain%Err%msg = procedure_name // refinedchain%Err%msg
631 return
632 ! LCOV_EXCL_STOP
633 end if
634
