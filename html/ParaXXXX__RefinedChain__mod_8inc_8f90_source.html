118 subroutine getRefinedChain( RefinedChain &
119 , CFC &
120 , Err &
121 , burninLoc &
122 , refinedChainSize &
123 , outputSampleRefinementCount &
124 , outputSampleRefinementMethod &
125 )
126#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
127 !DEC$ ATTRIBUTES DLLEXPORT :: getRefinedChain
128#endif
129
130 use, intrinsic :: iso_fortran_env, only: output_unit
131 use CrossCorr_mod, only: getBatchMeansIAC, genCumSumIAC, getMaxCumSumIAC
132 use String_mod, only: getLowerCase, replace
133 use Quantile_mod, only: getMedian
134
135 implicit none
136
137 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@getRefinedChain()"
138
139 class(RefinedChain_type) , intent(inout) :: RefinedChain
140 type(ChainFileContents_type), intent(inout) :: CFC
141 type(Err_type) , intent(out) :: Err
142 character(*) , intent(in), optional :: outputSampleRefinementMethod
143 integer(IK) , intent(in), optional :: burninLoc, refinedChainSize, outputSampleRefinementCount
144 integer(IK) :: burninLocDefault, i, countCompact, ndimPlusOne
145 integer(IK) :: maxRefinementCurrent, maxRefinementCount
146 real(RK) :: integratedAutoCorrTime, ndimPlusOneInverse
147 logical :: refinedChainSizeIsPresent
148 logical :: maxRefinementCountIsReached
149 character(:) , allocatable :: outputSampleRefinementMethodDefault, outputSampleRefinementMethodDefaultLowerCase, srmethod
150 type(Size_type) , allocatable :: DumSize(:)
151 real(RK) , allocatable :: DumIAC(:,:), DumVec(:)
152 integer(IK) , allocatable :: SampleWeight(:)
153 type(Method_type) :: Method
154 type(ChainFileContents_type) :: DumCFC
155
156 Err%occurred = .false.
157
158 ! if the size of the refined sample is given as input, then generate the requested sample straight
159
160 refinedChainSizeIsPresent = present(refinedChainSize)
161 if (refinedChainSizeIsPresent) then ! ignore outputSampleRefinementCount, even if it is given by the user
162 maxRefinementCount = 1_IK
163 else
164 maxRefinementCount = 20_IK ! this is a temporary maximum value, to be increased later if needed
165 if (present(outputSampleRefinementCount)) maxRefinementCount = outputSampleRefinementCount
166 end if
167
168 ! this is to avoid memory overflow due to extremely large maxRefinementCount requested by the user
169
170 maxRefinementCurrent = min(2_IK, maxRefinementCount)
171
172 ! compute ndim and the initial chain size
173
174 RefinedChain%nref = 0_IK
175
176 if (CFC%ndim == 0_IK) then
177 RefinedChain%ndim = size(CFC%State(:,1))
178 CFC%ndim = RefinedChain%ndim
179 else
180 RefinedChain%ndim = CFC%ndim
181 end if
182
183 ! allocate components
184
185 if (allocated(RefinedChain%IAC)) deallocate(RefinedChain%IAC)
186 if (allocated(RefinedChain%Size)) deallocate(RefinedChain%Size)
187 allocate(RefinedChain%IAC(0:RefinedChain%ndim,0:maxRefinementCurrent))
188 allocate(RefinedChain%Size(0:maxRefinementCurrent))
189 if (CFC%Size%compact == 0_IK) then
190 RefinedChain%Size(RefinedChain%nref)%compact = size(CFC%State(1,:))
191 CFC%Size%compact = RefinedChain%Size(RefinedChain%nref)%compact
192 else
193 RefinedChain%Size(RefinedChain%nref)%compact = CFC%Size%compact
194 end if
195
196 if (CFC%Size%verbose == 0_IK) CFC%Size%verbose = sum(CFC%Weight(1:CFC%Size%compact))
197
198 if (refinedChainSizeIsPresent) then
199 RefinedChain%Size(RefinedChain%nref)%verbose = refinedChainSize
200 else
201 RefinedChain%Size(RefinedChain%nref)%verbose = CFC%Size%verbose
202 end if
203
204 ! define the AIC computation method
205
206 outputSampleRefinementMethodDefault = BATCH_MEANS_METHOD_NAME; if (present(outputSampleRefinementMethod)) outputSampleRefinementMethodDefault = outputSampleRefinementMethod
207 outputSampleRefinementMethodDefaultLowerCase = getLowerCase(outputSampleRefinementMethodDefault)
208
209 if (index(outputSampleRefinementMethodDefaultLowerCase,getLowerCase(BATCH_MEANS_METHOD_NAME))>0) then
210 Method%isBatchMeans = .true.
211 elseif (index(outputSampleRefinementMethodDefaultLowerCase,getLowerCase(CUTOFF_AUTOCORR_METHOD_NAME))>0 .or. index(outputSampleRefinementMethodDefaultLowerCase,"cutoff")>0) then
212 Method%isCutoffAutoCorr = .true.
213 elseif (index(outputSampleRefinementMethodDefaultLowerCase,getLowerCase(MAX_CUMSUM_AUTOCORR_METHOD_NAME))>0 .or. index(outputSampleRefinementMethodDefaultLowerCase,"cumsum")>0) then
214 Method%isMaxCumSumAutoCorr = .true.
215 else
216 ! LCOV_EXCL_START
217 Err%occurred = .true.
218 Err%msg = PROCEDURE_NAME // ": Unknown unsupported IAC computation method name: " // outputSampleRefinementMethodDefault
219 return
220 ! LCOV_EXCL_STOP
221 end if
222
223 ! define the chain types to use for the AIC computation
224
225 Method%isViaCompactChain = index(outputSampleRefinementMethodDefaultLowerCase,"compact") > 0
226 Method%isViaVerboseChain = index(outputSampleRefinementMethodDefaultLowerCase,"verbose") > 0
227 if (.not.(Method%isViaCompactChain .or. Method%isViaVerboseChain)) then
228 Method%isViaCompactChain = .true.
229 Method%isViaVerboseChain = .true.
230 end if
231
232 ! define the statistic to use for the AIC computation
233
234 srmethod = replace(value=outputSampleRefinementMethodDefaultLowerCase,search=" ",substitute="-")
235 Method%isAvg = index(srmethod,"-avg") > 0 .or. index(srmethod,"-average") > 0
236 Method%isMed = index(srmethod,"-med") > 0 .or. index(srmethod,"-median") > 0
237 Method%isMin = index(srmethod,"-min") > 0 .or. index(srmethod,"-minimum") > 0
238 Method%isMax = index(srmethod,"-max") > 0 .or. index(srmethod,"-maximum") > 0
239 if ( Method%isAvg .and. (Method%isMed .or. Method%isMax .or. Method%isMin) ) Err%occurred = .true.
240 if ( Method%isMed .and. (Method%isMax .or. Method%isMin) ) Err%occurred = .true.
241 if ( Method%isMax .and. Method%isMin ) Err%occurred = .true.
242 if (.not.(Method%isAvg .or. Method%isMed .or. Method%isMin .or. Method%isMax)) Method%isAvg = .true. ! default method of AIC summarization.
243
244 if (Method%isAvg) ndimPlusOneInverse = 1._RK / (RefinedChain%ndim + 1_IK)
245 if (Method%isMed) ndimPlusOne = RefinedChain%ndim + 1_IK
246
247 ! assign the column headers
248
249 if (allocated(CFC%Header)) then
250 if (allocated(RefinedChain%Header)) deallocate(RefinedChain%Header)
251 allocate(RefinedChain%Header(0:RefinedChain%ndim))
252 do i = 0, RefinedChain%ndim
253 RefinedChain%Header(i)%value = CFC%Header(i+NUM_DEF_COL)%value
254 end do
255 else
256 ! LCOV_EXCL_START
257 Err%occurred = .true.
258 Err%msg = PROCEDURE_NAME // ": Internal error occurred. CFC%Header is NULL."
259 return
260 ! LCOV_EXCL_STOP
261 end if
262
263 if (present(burninLoc)) then
264 burninLocDefault = burninLoc
265 else
266 burninLocDefault = CFC%BurninLoc(CFC%Size%compact)
267 end if
268
269 RefinedChain%Size(RefinedChain%nref)%compact = CFC%Size%compact - burninLocDefault + 1
270 if (allocated(RefinedChain%LogFuncState)) deallocate(RefinedChain%LogFuncState)
271 allocate(RefinedChain%LogFuncState(0:RefinedChain%ndim,RefinedChain%Size(RefinedChain%nref)%compact))
272 RefinedChain%LogFuncState(0 ,1:RefinedChain%Size(RefinedChain%nref)%compact) = CFC%LogFunc(burninLocDefault:CFC%Size%compact)
273 RefinedChain%LogFuncState(1:RefinedChain%ndim,1:RefinedChain%Size(RefinedChain%nref)%compact) = CFC%State(1:RefinedChain%ndim, burninLocDefault:CFC%Size%compact)
274
275 ! check if there are more than 1 sample points in the burnin-subtracted CFC
276
277 if (RefinedChain%Size(RefinedChain%nref)%compact==0_IK) then
278 ! LCOV_EXCL_START
279 Err%occurred = .true.
280 Err%msg = PROCEDURE_NAME // ": The size of the refined sample is zero."
281 return
282 ! LCOV_EXCL_STOP
283 elseif (RefinedChain%Size(RefinedChain%nref)%compact==1_IK) then
284 if (allocated(RefinedChain%Weight)) deallocate(RefinedChain%Weight)
285 allocate(RefinedChain%Weight(RefinedChain%Size(RefinedChain%nref)%compact))
286 if (refinedChainSizeIsPresent) then
287 RefinedChain%Weight = refinedChainSize
288 else
289 RefinedChain%Weight = 1
290 end if
291 if (allocated(RefinedChain%IAC)) deallocate(RefinedChain%IAC)
292 if (allocated(RefinedChain%Size)) deallocate(RefinedChain%Size)
293 allocate(RefinedChain%IAC(0:RefinedChain%ndim,0:0))
294 allocate(RefinedChain%Size(0:0))
295 RefinedChain%IAC = 0._RK
296 RefinedChain%Size(RefinedChain%nref)%verbose = sum(RefinedChain%Weight)
297 return
298 end if
299
300 RefinedChain%Weight = CFC%Weight(burninLocDefault:CFC%Size%compact) ! Weight is intentionally separately assigned from State here
301
302 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
303
304 loopRefinement: do
305
306 countCompact = RefinedChain%Size(RefinedChain%nref)%compact
307
308 ! set the sample weight
309
310 if (Method%isViaCompactChain) then
311 if (allocated(SampleWeight)) deallocate(SampleWeight)
312 elseif (Method%isViaVerboseChain) then
313 SampleWeight = RefinedChain%Weight
314 endif
315
316 ! obtain the IAC for each individual variable
317
318 do i = 0, RefinedChain%ndim
319 DumVec = RefinedChain%LogFuncState(i,1:RefinedChain%Size(RefinedChain%nref)%compact)
320 if (Method%isBatchMeans) then
321 RefinedChain%IAC(i,RefinedChain%nref) = getBatchMeansIAC( np = countCompact &
322 , Point = DumVec & ! LCOV_EXCL_LINE
323 , Weight = SampleWeight & ! LCOV_EXCL_LINE
324 )
325 elseif (Method%isCutoffAutoCorr) then
326 RefinedChain%IAC(i,RefinedChain%nref) = genCumSumIAC( np = countCompact &
327 , Point = DumVec & ! LCOV_EXCL_LINE
328 , Weight = SampleWeight & ! LCOV_EXCL_LINE
329 )
330 elseif (Method%isMaxCumSumAutoCorr) then
331 RefinedChain%IAC(i,RefinedChain%nref) = getMaxCumSumIAC( np = countCompact &
332 , Point = DumVec & ! LCOV_EXCL_LINE
333 , Weight = SampleWeight & ! LCOV_EXCL_LINE
334 )
335 end if
336 end do
337
338 if (refinedChainSizeIsPresent) then
339 integratedAutoCorrTime = real(sum(RefinedChain%Weight),kind=RK) / real(refinedChainSize,kind=RK)
340 else
341 if (Method%isAvg) then
342 integratedAutoCorrTime = sum( RefinedChain%IAC(0:RefinedChain%ndim,RefinedChain%nref) ) * ndimPlusOneInverse
343 elseif (Method%isMed) then
344 call getMedian(lenArray=ndimPlusOne,Array=RefinedChain%IAC(0:RefinedChain%ndim,RefinedChain%nref),median=integratedAutoCorrTime)
345 elseif (Method%isMax) then
346 integratedAutoCorrTime = maxval( RefinedChain%IAC(0:RefinedChain%ndim,RefinedChain%nref) )
347 elseif (Method%isMin) then
348 integratedAutoCorrTime = minval( RefinedChain%IAC(0:RefinedChain%ndim,RefinedChain%nref) )
349 end if
350 end if
351
352 ! so far, we have computed the max IAC of the sample, but no refinement. Refine the sample only if needed.
353
354 maxRefinementCountIsReached = RefinedChain%nref==maxRefinementCount
355 if (integratedAutoCorrTime<2._RK .or. maxRefinementCountIsReached) then
356 if (Method%isViaCompactChain .and. Method%isViaVerboseChain) then
357 !if (maxRefinementCountIsReached) maxRefinementCount = maxRefinementCount * 2_IK
358 maxRefinementCount = maxRefinementCount * 2_IK
359 Method%isViaCompactChain = .false.
360 cycle loopRefinement
361 end if
362 exit loopRefinement
363 end if
364
365 ! generate the refined sample, dump it in CFC, then put it back into RefinedChain to start over again
366
367 RefinedChain%nref = RefinedChain%nref + 1_IK
368
369 ! reallocate to bigger array if nedded
370
371 if (RefinedChain%nref>maxRefinementCurrent) then
372 call move_alloc( from = RefinedChain%IAC, to = DumIAC )
373 call move_alloc( from = RefinedChain%Size, to = DumSize )
374 maxRefinementCurrent = min( maxRefinementCurrent*2 , maxRefinementCount )
375 allocate( RefinedChain%IAC( 0:RefinedChain%ndim , 0:maxRefinementCurrent ) )
376 allocate( RefinedChain%Size( 0:maxRefinementCurrent ) )
377 RefinedChain%IAC(0:RefinedChain%ndim,0:RefinedChain%nref-1) = DumIAC
378 RefinedChain%Size(0:RefinedChain%nref-1) = DumSize
379 end if
380
381 if (integratedAutoCorrTime<2._RK) cycle loopRefinement ! no need for refinement. should happen only when transitioning from compact to verbose
382
383 call refineWeightedSample( nd = RefinedChain%ndim & ! LCOV_EXCL_LINE
384 , np = countCompact & ! LCOV_EXCL_LINE
385 , skip = integratedAutoCorrTime & ! LCOV_EXCL_LINE
386 , Sample = RefinedChain%LogFuncState & ! LCOV_EXCL_LINE
387 , Weight = RefinedChain%Weight & ! LCOV_EXCL_LINE
388 , RefinedChain = DumCFC%State & ! LCOV_EXCL_LINE
389 , RefinedWeight = DumCFC%Weight & ! LCOV_EXCL_LINE
390 , PointCount = RefinedChain%Size(RefinedChain%nref) & ! LCOV_EXCL_LINE
391 , refinedChainSize = refinedChainSize & ! LCOV_EXCL_LINE
392 )
393 RefinedChain%Weight = DumCFC%Weight
394 RefinedChain%LogFuncState = DumCFC%State
395
396 end do loopRefinement
397
398 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
399
400 end subroutine getRefinedChain

426 pure function genRefinedWeight(np,Weight,skip,refinedChainSize) result(RefinedWeight)
427#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
428 !DEC$ ATTRIBUTES DLLEXPORT :: genRefinedWeight
429#endif
430 use Math_mod, only: genCumSum
431 use Constants_mod, only: IK, RK
432 implicit none
433 real(RK), intent(in) :: skip
434 integer(IK), intent(in) :: np, Weight(np)
435 integer(IK) , intent(in) , optional :: refinedChainSize
436 integer(IK) :: RefinedWeight(np)
437 integer(IK) :: ip, refinedChainSizeCounter, offset
438 real(RK) :: sumSkips, CumSumWeight(np)
439 logical :: refinedChainSizeIsPresent
440 refinedChainSizeIsPresent = present(refinedChainSize)
441 if (refinedChainSizeIsPresent) refinedChainSizeCounter = 0_IK
442 CumSumWeight = real(genCumSum(Weight),kind=RK)
443 sumSkips = skip
444 offset = 1_IK
445 ip = offset
446 RefinedWeight = 0_IK
447 loopOverAllSample: do
448 loopOverCurrentSample: do
449 if (sumSkips>CumSumWeight(ip)) then
450 exit loopOverCurrentSample
451 elseif (refinedChainSizeIsPresent) then
452 if (refinedChainSizeCounter==refinedChainSize) exit loopOverAllSample
453 refinedChainSizeCounter = refinedChainSizeCounter + 1_IK
454 end if
455 RefinedWeight(ip) = RefinedWeight(ip) + 1_IK
456 sumSkips = sumSkips + skip
457 cycle loopOverCurrentSample
458 end do loopOverCurrentSample
459 if (ip==np) then
460 if (refinedChainSizeIsPresent) then
461 if (refinedChainSizeCounter<refinedChainSize) then
462 offset = offset + 1_IK
463 if (offset==np) offset = 1_IK
464 ip = offset
465 sumSkips = skip
466 if (offset/=1_IK) sumSkips = sumSkips + CumSumWeight(ip-1)
467 cycle loopOverAllSample ! LCOV_EXCL_LINE
468 end if
469 end if
470 exit loopOverAllSample
471 end if
472 ip = ip + 1_IK
473 end do loopOverAllSample
474 end function genRefinedWeight

491 pure subroutine refineWeightedSample(nd,np,skip,Sample,Weight,RefinedChain,RefinedWeight,PointCount,refinedChainSize)
492#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
493 !DEC$ ATTRIBUTES DLLEXPORT :: refineWeightedSample
494#endif
495 use Constants_mod, only: IK, RK
496 implicit none
497 integer(IK) , intent(in) :: nd, np, Weight(np) !, skip
498 real(RK) , intent(in) :: Sample(0:nd,np), skip
499 real(RK) , intent(out), allocatable :: RefinedChain(:,:)
500 integer(IK) , intent(out), allocatable :: RefinedWeight(:)
501 integer(IK) , intent(in) , optional :: refinedChainSize
502 type(Size_type) , intent(out) :: PointCount
503 integer(IK) :: ip, ipRefined, npRefined, UpdatedWeight(np) ! LCOV_EXCL_LINE
504 UpdatedWeight = genRefinedWeight(np,Weight,skip,refinedChainSize)
505 npRefined = count(UpdatedWeight>0)
506 allocate( RefinedChain(0:nd,npRefined) , RefinedWeight(npRefined) )
507 ipRefined = 0
508 PointCount%verbose = 0
509 do ip = 1, np
510 if (UpdatedWeight(ip)>0) then
511 ipRefined = ipRefined + 1
512 RefinedChain(0:nd,ipRefined) = Sample(0:nd,ip)
513 RefinedWeight(ipRefined) = UpdatedWeight(ip)
514 PointCount%verbose = PointCount%verbose + RefinedWeight(ipRefined)
515 end if
516 end do
517 PointCount%compact = npRefined
518 end subroutine refineWeightedSample

533 pure function getSkip4NewSampleSize(oldSampleSize,newSampleSize) result(skip4NewSampleSize)
534#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
535 !DEC$ ATTRIBUTES DLLEXPORT :: getSkip4NewSampleSize
536#endif
537 use Constants_mod, only: IK, RK
538 implicit none
539 integer(IK) , intent(in) :: oldSampleSize,newSampleSize
540 integer(IK) :: skip4NewSampleSize, addition, quotient
541 if (oldSampleSize < newSampleSize) then
542 skip4NewSampleSize = -1_IK
543 return
544 end if
545 addition = 1
546 quotient = oldSampleSize / newSampleSize
547 if (mod(oldSampleSize,newSampleSize)==0) addition = 0
548 skip4NewSampleSize = quotient + addition
549 end function getSkip4NewSampleSize

560 subroutine writeRefinedChain(RefinedChain,sampleFileUnit,sampleFileHeaderFormat,sampleFileContentsFormat)
561#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
562 !DEC$ ATTRIBUTES DLLEXPORT :: writeRefinedChain
563#endif
564 implicit none
565 class(RefinedChain_type), intent(in) :: RefinedChain
566 integer(IK) , intent(in) :: sampleFileUnit
567 character(*), intent(in) :: sampleFileHeaderFormat, sampleFileContentsFormat
568 integer(IK) :: isample, iweight, i
569 write(sampleFileUnit, sampleFileHeaderFormat) (trim(adjustl(RefinedChain%Header(i)%value)),i=0,RefinedChain%ndim)
570 do isample = 1, RefinedChain%Size(RefinedChain%nref)%compact

573 end do
574 end do
575 flush(sampleFileUnit)
576 end subroutine writeRefinedChain ! LCOV_EXCL_LINE

596 function readRefinedChain(sampleFilePath,delimiter,ndim,tenabled) result(RefinedChain)
597#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
598 !DEC$ ATTRIBUTES DLLEXPORT :: readRefinedChain
599#endif
600 use FileContents_mod, only: getNumRecordInFile
601 use String_mod, only: StrSplit_type, val2char
602 implicit none
603 integer(IK) , intent(in) :: ndim
604 character(*), intent(in) :: sampleFilePath, delimiter
605 logical , intent(in), optional :: tenabled
606 type(RefinedChain_type) :: RefinedChain
607 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@readRefinedChain()"
608 integer(IK) :: sampleFileUnit, isample, i
609 logical :: tenabledDefault
610 type(StrSplit_type) :: Line
611
612 if (allocated(Line%value)) deallocate(Line%value) ! LCOV_EXCL_LINE
613 allocate( character(99999) :: Line%value )
614
615 RefinedChain%ndim = ndim
616 RefinedChain%nref = 0_IK
617 allocate(RefinedChain%Size(RefinedChain%nref:RefinedChain%nref)) ! allocate just the zeroth level of `RefinedChain`.
618
619 ! find the number of lines in the sample file
620
621 call getNumRecordInFile(filePath=sampleFilePath, numRecord=RefinedChain%Size(RefinedChain%nref)%verbose, Err=RefinedChain%Err)
622 if (RefinedChain%Err%occurred) then
623 ! LCOV_EXCL_START
624 RefinedChain%Err%msg = PROCEDURE_NAME // RefinedChain%Err%msg
625 return
626 ! LCOV_EXCL_STOP
627 end if
628
629 RefinedChain%Size(RefinedChain%nref)%verbose = RefinedChain%Size(RefinedChain%nref)%verbose - 1_IK ! remove header from the count
630
631 open( newunit = sampleFileUnit &
632 , file = sampleFilePath &
633 , status = "old" &
634#if INTEL_COMPILER_ENABLED && WINDOWS_ENABLED
