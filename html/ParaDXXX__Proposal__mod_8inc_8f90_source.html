184 function constructproposalsymmetric ( ndim &
185 , SpecBase &
186 , SpecMCMC &
187 , SpecDRAM &
188 , Image &
189 , name &
190 , brand &
191 , LogFile &
192 , RestartFile &
193 , isFreshRun &
194 ) result(self)
195#if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
196 !DEC$ ATTRIBUTES DLLEXPORT :: constructProposalSymmetric
197#endif
198 use constants_mod, only: ik, rk, null_rk
199 use paramonte_mod, only: image_type
200 use paramonte_mod, only: logfile_type
201 use paramonte_mod, only: restartfile_type
202 use specbase_mod, only: specbase_type
203 use specmcmc_mod, only: specmcmc_type
204 use specdram_mod, only: specdram_type

206 use string_mod, only: num2str
207 use err_mod, only: abort
208#if defined UNIFORM

210#endif
211 implicit none
212
213 integer(IK) , intent(in) :: ndim
214 type(SpecBase_type) , intent(in) :: SpecBase
215 type(SpecMCMC_type) , intent(in) :: SpecMCMC
216 type(SpecDRAM_type) , intent(in) :: SpecDRAM
217 type(Image_type) , intent(in) :: Image
218 character(*) , intent(in) :: name
219 character(*) , intent(in) :: brand
220 type(LogFile_type) , intent(in) :: LogFile
221 type(RestartFile_type) , intent(in) :: RestartFile
222 logical :: isFreshRun
223
224 type(Proposal_type) :: self
225
226 character(*), parameter :: PROCEDURE_NAME = module_name // "@constructProposalSymmetric()"
227 integer :: i, j
228
229 self%Err%occurred = .false.
230
231 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
232 ! setup sampler update global save variables
233 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
234
235 if (allocated(self%MeanOld)) deallocate(self%MeanOld); allocate(self%MeanOld(ndim))
236 self%MeanOld(1:ndim) = specmcmc%StartPointVec%Val
237 self%logSqrtDetOld = null_rk
238 self%sampleSizeOld = 1_ik
239 self%adaptiveScaleFactorSq = 1._rk
240
241 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
242 ! setup general proposal specifications
243 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
244
245#if defined MPI_ENABLED
246 mc_ndimsqplusndim = ndim*(ndim+1_ik)
247#endif
248 mc_ndim = ndim
249 mc_domainlowerlimitvec = specbase%DomainLowerLimitVec%Val
250 mc_domainupperlimitvec = specbase%DomainUpperLimitVec%Val
251 mc_delayedrejectionscalefactorvec = specdram%delayedRejectionScaleFactorVec%Val
252 !mc_isNormal = SpecMCMC%ProposalModel%isNormal
253 mc_image = image
254 mc_methodname = name
255 mc_methodbrand = brand
256 self%logFileUnit = logfile%unit
257 self%restartFileUnit = restartfile%unit
258 mc_restartfileformat = restartfile%format
259 mc_isbinaryrestartfileformat = specbase%RestartFileFormat%isBinary
260 mc_isasciirestartfileformat = specbase%RestartFileFormat%isAscii
261 mc_defaultscalefactorsq = specmcmc%ScaleFactor%val**2
262 !self%AccRate%sumUpToLastUpdate = 0._RK
263 mc_maxnumdomainchecktowarn = specbase%MaxNumDomainCheckToWarn%val
264 mc_maxnumdomainchecktostop = specbase%MaxNumDomainCheckToStop%val
265 mc_delayedrejectioncount = specdram%DelayedRejectionCount%val
266 mc_delayedrejectionrequested = mc_delayedrejectioncount > 0_ik
267 mc_scalingrequested = specbase%TargetAcceptanceRate%scalingRequested
268 mc_ndiminverse = 1._rk/real(ndim,kind=rk)
269 mc_targetacceptanceratelimit = specbase%TargetAcceptanceRate%Val
270 mc_targetacceptancerate = 0.234_rk
271 if (mc_scalingrequested) mc_targetacceptancerate = sum(mc_targetacceptanceratelimit) / 2._rk
272 !mc_maxScaleFactorSq = 4._RK**mc_ndimInverse
273 !mc_maxScaleFactor = sqrt(mc_maxScaleFactorSq)
274
275 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
276 ! setup ProposalSymmetric specifications
277 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
278
279 ! setup covariance matrix
280
281#if defined PARADISE
282 if (allocated(self%InvCovMat)) deallocate(self%InvCovMat)
283 allocate( self%InvCovMat(ndim,0:ndim,0:mc_delayedrejectioncount) )
284 if (allocated(self%logSqrtDetInvCovMat)) deallocate(self%logSqrtDetInvCovMat)
285 allocate( self%logSqrtDetInvCovMat(0:mc_delayedrejectioncount) )
286#endif
287
288 ! On the second dimension, the zeroth index refers to the Diagonal elements of the Cholesky lower triangular matrix
289 ! This rearrangement was done for more efficient communication of the matrix across processes.
290 if (allocated(self%CholDiagLower)) deallocate(self%CholDiagLower)
291 allocate( self%CholDiagLower(ndim,0:ndim,0:mc_delayedrejectioncount) )
292#if defined CAF_ENABLED
293 if (allocated(comv_choldiaglower)) deallocate(comv_choldiaglower)
294 allocate( comv_choldiaglower(ndim,0:ndim)[*] )
295#endif
296
297 self%CholDiagLower(1:ndim,1:ndim,0) = specmcmc%ProposalStartCovMat%Val
298
299 ! Now scale the covariance matrix
300
301 do j = 1, ndim
302 do i = 1, j
303 self%CholDiagLower(i,j,0) = self%CholDiagLower(i,j,0) * mc_defaultscalefactorsq
304 end do
305 end do
306
307 ! Now get the Cholesky Factor of the Covariance Matrix. Lower self%CholDiagLower will be the CholFac
308
309 call getcholeskyfactor( ndim, self%CholDiagLower(:,1:ndim,0), self%CholDiagLower(1:ndim,0,0) ) ! The `:` instead of `1:ndim` avoids temporary array creation.
310 if (self%CholDiagLower(1,0,0)<0._rk) then
311 ! LCOV_EXCL_START
312 self%Err%msg = mc_image%name // procedure_name // ": Singular input covariance matrix by user was detected. This is strange.\nCovariance matrix lower triangle:"
313 do j = 1, ndim
314 do i = 1, j
315 self%Err%msg = self%Err%msg // "\n" // num2str(self%CholDiagLower(1:i,j,0))
316 end do
317 end do
318 self%Err%occurred = .true.
319 call abort( err = self%Err, prefix = mc_methodbrand, newline = "\n", outputunit = self%logFileUnit )
320 return
321 ! LCOV_EXCL_STOP
322 end if
323
324 if (mc_delayedrejectionrequested) call updatedelrejcholdiaglower(self%CholDiagLower)
325#if defined PARADISE
326 call getinvcovmat(choldiaglower = self%CholDiagLower, invcovmat = self%InvCovMat, logsqrtdetinvcovmat = self%LogSqrtDetInvCovMat)
327#endif
328 self%logSqrtDetOld = sum(log( self%CholDiagLower(1:ndim,0,0) ))
329
330 ! Scale the higher-stage delayed-rejection Cholesky Lower matrices
331
332 if (mc_delayedrejectionrequested) call updatedelrejcholdiaglower(self%CholDiagLower)
333
334 ! This will be used for Domain boundary checking during the simulation
335
336 mc_negativetotalvariationmsg = module_name//"@doAdaptation(): Non-positive adaptation measure detected, possibly due to round-off error: "
337 mc_maxnumdomainchecktowarnmsg = module_name//"@getNew(): "//num2str(mc_maxnumdomainchecktowarn)//&
338 " proposals were drawn out of the objective function's Domain without any acceptance."
339 mc_maxnumdomainchecktostopmsg = module_name//"@getNew(): "//num2str(mc_maxnumdomainchecktostop)//&
340 " proposals were drawn out of the objective function's Domain. As per the value set for the&
341 & simulation specification variable 'maxNumDomainCheckToStop', "//mc_methodname//" will abort now."
342
343#if defined UNIFORM
344 if (ndim>1_ik) then
345 mc_neglogvolunitball = -getlogvolunitball(ndim)
346 else
347 mc_neglogvolunitball = 0._rk
348 end if
349#endif
350
351 ! read/write the first entry of the restart file
352
353 if (mc_image%isLeader) then
354 block
355 real(RK) :: meanAccRateSinceStart
356 if (isfreshrun) then
357 if (mc_isbinaryrestartfileformat) then
358 call writerestartfilebinary(restartfileunit = self%restartFileUnit, meanaccratesincestart = 1._rk)
359 else
360 call writerestartfileascii ( restartfileunit = self%restartFileUnit & ! LCOV_EXCL_LINE
361 , meanaccratesincestart = 1._rk & ! LCOV_EXCL_LINE
362 , samplesizeold = self%sampleSizeOld & ! LCOV_EXCL_LINE
363 , logsqrtdetold = self%logSqrtDetOld & ! LCOV_EXCL_LINE
364 , adaptivescalefactorsq = self%adaptiveScaleFactorSq & ! LCOV_EXCL_LINE
365 , meanold = self%MeanOld & ! LCOV_EXCL_LINE
366 , covmatupper = self%CholDiagLower(1:ndim,1:ndim,0) & ! LCOV_EXCL_LINE
367 )
368 end if
369 else
370 if (mc_isbinaryrestartfileformat) then
371 call readrestartfilebinary(restartfileunit = self%restartFileUnit, meanaccratesincestart = meanaccratesincestart)
372 else
373 call readrestartfileascii(restartfileunit = self%restartFileUnit, meanaccratesincestart = meanaccratesincestart)
374 end if
375 end if
376 end block
377 end if
378
379 end function constructproposalsymmetric

445 end if
446 exit loopboundarycheck
447 end do loopboundarycheck

531 subroutine doadaptation ( self &
532 , nd &
533 , chainSize &
534 , Chain &
535 , ChainWeight &
536 , isFreshRun &
537 , samplerUpdateIsGreedy &
538 , meanAccRateSinceStart &
539 , samplerUpdateSucceeded &
540 , adaptationMeasure &
541 )
542#if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
543 !DEC$ ATTRIBUTES DLLEXPORT :: doAdaptation
544#endif
545
546 use statistics_mod, only: getsamcovuppermeantrans, getweisamcovuppmeantrans, mergemeancovupper ! LCOV_EXCL_LINE
547 use matrix_mod, only: getcholeskyfactor, getlogsqrtdetposdefmat
548 use constants_mod, only: rk, ik ! , EPS_RK
549 use string_mod, only: num2str
550 use err_mod, only: err_type, abort, warn
551 implicit none
552
553 character(*), parameter :: PROCEDURE_NAME = module_name // "@doAdaptation()"
554
555 class(Proposal_type), intent(inout) :: self
556 integer(IK) , intent(in) :: nd
557 integer(IK) , intent(in) :: chainSize
558 real(RK) , intent(in) :: Chain(nd,chainSize)
559 integer(IK) , intent(in) :: ChainWeight(chainSize)
560 logical , intent(in) :: isFreshRun
561 logical , intent(in) :: samplerUpdateIsGreedy
562 real(RK) , intent(inout) :: meanAccRateSinceStart ! is intent(out) in restart mode, intent(in) in fresh mode.
563 logical , intent(out) :: samplerUpdateSucceeded
564 real(RK) , intent(out) :: adaptationMeasure
565
566 integer(IK) :: i, j
567 real(RK) :: MeanNew(nd)
568 real(RK) :: MeanCurrent(nd)
569 real(RK) :: CovMatUpperOld(nd,nd)
570 real(RK) :: CovMatUpperCurrent(nd,nd)
571 real(RK) :: logSqrtDetNew
572 real(RK) :: logSqrtDetSum
573 real(RK) :: adaptiveScaleFactor
574 logical :: adaptationMeasureComputationNeeded ! only used to avoid redundant affinity computation, if no update occurs
575 logical :: singularityOccurred, scalingNeeded
576 integer(IK) :: sampleSizeOld, sampleSizeCurrent
577
578 scalingneeded = .false.
579 samplesizeold = self%sampleSizeOld ! this is kept only for restoration of self%sampleSizeOld, if needed.
580
581 ! read/write meanAccRateSinceStart from/to restart file
582
583 if (mc_image%isLeader) then
584 if (.not. isfreshrun) then
585 if (mc_isbinaryrestartfileformat) then
586 call readrestartfilebinary(restartfileunit = self%restartFileUnit, meanaccratesincestart = meanaccratesincestart)
587 else
588 call readrestartfileascii(restartfileunit = self%restartFileUnit, meanaccratesincestart = meanaccratesincestart)
589 end if
590 end if
591 end if
592
593 ! First if there are less than nd+1 points for new covariance computation, then just scale the covariance and return
594
595 blocksufficientsamplesizecheck: if (chainsize>nd) then
596
597 ! get the upper covariance matrix and Mean of the new sample
598
599 if (samplerupdateisgreedy) then
600 samplesizecurrent = chainsize
601 call getsamcovuppermeantrans(samplesizecurrent,nd,chain,covmatuppercurrent,meancurrent)
602 else
603 samplesizecurrent = sum(chainweight)
604 call getweisamcovuppmeantrans(chainsize,samplesizecurrent,nd,chain,chainweight,covmatuppercurrent,meancurrent)
605 end if
606
607 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
608
609 ! combine old and new covariance matrices if both exist
610
611 blockmergecovmat: if (self%sampleSizeOld==1_ik) then
612
613 ! There is no prior old Covariance matrix to combine with the new one from the new chain
614
615 self%MeanOld(1:nd) = meancurrent
616 self%sampleSizeOld = samplesizecurrent
617
618 ! copy and then scale the new covariance matrix by the default scale factor, which will be then used to get the Cholesky Factor
619
620 do j = 1, nd
621 do i = 1, j
622 covmatupperold(i,j) = self%CholDiagLower(i,j,0) ! This will be used to recover the old covariance in case of update failure, and to compute the adaptation measure
623 self%CholDiagLower(i,j,0) = covmatuppercurrent(i,j) * mc_defaultscalefactorsq
624 end do
625 end do
626
627 else blockmergecovmat
628
629 ! first scale the new covariance matrix by the default scale factor, which will be then used to get the Cholesky Factor
630
631 do j = 1, nd
632 do i = 1, j
633 covmatupperold(i,j) = self%CholDiagLower(i,j,0) ! This will be used to recover the old covariance in case of update failure, and to compute the adaptation measure
634 covmatuppercurrent(i,j) = covmatuppercurrent(i,j) * mc_defaultscalefactorsq
635 end do
636 end do
637
638 ! now combine it with the old covariance matrix.
639 ! Do not set the full boundaries' range `(1:nd)` for `self%CholDiagLower` in the following subroutine call.
640 ! Setting the boundaries forces the compiler to generate a temporary array.
641
642 call mergemeancovupper ( nd = nd & ! LCOV_EXCL_LINE
643 , npa = self%sampleSizeOld & ! LCOV_EXCL_LINE
644 , meanveca = self%MeanOld & ! LCOV_EXCL_LINE
645 , covmatuppera = covmatupperold & ! LCOV_EXCL_LINE
646 , npb = samplesizecurrent & ! LCOV_EXCL_LINE
647 , meanvecb = meancurrent & ! LCOV_EXCL_LINE
648 , covmatupperb = covmatuppercurrent & ! LCOV_EXCL_LINE
649 , meanvecab = meannew & ! LCOV_EXCL_LINE

842 , outputunit = self%logFileUnit &

850!integer, save :: counter = 0
851!counter = counter + 1
