150  function minimizebrent(getFunc, x0, x1, x2, xtol) result(BrentMinimum)
151 #if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
152  !DEC$ ATTRIBUTES DLLEXPORT :: minimizeBrent
153 #endif
154  use constants_mod, only: ik, rk
155  procedure(getfunc1d_proc) :: getfunc
156  real(rk) , intent(in), optional :: x0, x1, x2
157  real(rk) , intent(in), optional :: xtol
158  type(brentminimum_type) :: brentminimum
159 
160  character(*), parameter :: procedure_name = module_name//"@minimizeBrent"
161  integer(IK) , parameter :: itmax = 1000_ik ! the maximum number of iterations
162  real(rk) , parameter :: cgold = 0.3819660_rk ! Golden Section switch criterion
163  real(rk) , parameter :: zeps = sqrt(epsilon(1._rk))**3 ! tiny nonzero value
164 
165  integer(IK) :: iter
166  logical :: ispresentx0, ispresentx1, ispresentx2
167  real(rk) :: a, b, d, e, etemp, fu, fv, fw, fx, p, q, r, tol1, tol2, u, v, w, x, xm
168  real(rk) :: ax, bx, cx, fa, fb, fc
169 
170  brentminimum%Err%occurred = .false.
171 
172  if (present(xtol)) brentminimum%xtol = xtol
173 
174  ispresentx0 = present(x0)
175  ispresentx1 = present(x1)
176  ispresentx2 = present(x2)
177  if (ispresentx0 .and. ispresentx1 .and. ispresentx2) then
178  brentminimum%Bracket = [x0, x1, x2]
179  else
180  if (ispresentx0) then
181  ax = x0
182  else
183  ax = 0._rk ! assume an initial starting point of zero. This is the worst cae scenario.
184  end if
185  if (ispresentx1) then
186  bx = x1
187  else
188  bx = ax + 1._rk
189  end if
190  call getbracket ( ax = ax &
191  , bx = bx &
192  , cx = cx &
193  , fa = fa &
194  , fb = fb &
195  , fc = fc &
196  , getfunc = getfunc &
197  )
198  brentminimum%Bracket = [ax, bx, cx]
199  end if
200 
201  a = min( brentminimum%Bracket(1), brentminimum%Bracket(3) )
202  b = max( brentminimum%Bracket(1), brentminimum%Bracket(3) )
203  v = brentminimum%Bracket(2)
204  w = v
205  x = v
206  e = 0._rk
207  fx = getfunc(x)
208  fv = fx
209  fw = fx
210  do iter = 1, itmax
211  brentminimum%niter = iter
212  xm = 0.5_rk * (a+b)
213  tol1 = xtol*abs(x) + zeps
214  tol2 = 2.0_rk*tol1
215  if (abs(x-xm) <= (tol2-0.5_rk*(b-a))) then
216  brentminimum%xmin = x
217  brentminimum%fmin = fx
218  return
219  end if
220  if (abs(e) > tol1) then
221  r = (x-w)*(fx-fv)
222  q = (x-v)*(fx-fw)
223  p = (x-v)*q-(x-w)*r
224  q = 2.0_rk*(q-r)
225  if (q > 0._rk) p = -p
226  q = abs(q)
227  etemp = e
228  e = d
229  if (abs(p) >= abs(0.5_rk*q*etemp) .or. p <= q*(a-x) .or. p >= q*(b-x)) then
230  e = merge(a-x,b-x, x >= xm )
231  d = cgold*e
232  else
233  d = p/q
234  u = x+d
235  if (u-a < tol2 .or. b-u < tol2) d = sign(tol1,xm-x)
236  end if
237  else
238  e = merge(a-x,b-x, x >= xm )
239  d = cgold*e
240  end if
241  u = merge(x+d,x+sign(tol1,d), abs(d) >= tol1 )
242  fu = getfunc(u)
243  if (fu <= fx) then
244  if (u >= x) then
245  a = x
246  else
247  b = x
248  end if
249  call shft(v,w,x,u)
250  call shft(fv,fw,fx,fu)
251  else
252  if (u < x) then
253  a = u
254  else
255  b = u
256  end if
257  if (fu <= fw .or. w == x) then
258  v = w
259  fv = fw
260  w = u
261  fw = fu
262  else if (fu <= fv .or. v == x .or. v == w) then
263  v = u
264  fv = fu
265  end if
266  end if
267  end do
268 
269  ! LCOV_EXCL_START
270  brentminimum%Err%occurred = .true.
271  brentminimum%Err%msg = procedure_name//": maximum number of iterations exceeded."
272  ! LCOV_EXCL_STOP
273  return

291  subroutine getbracket(ax,bx,cx,fa,fb,fc,getFunc)
292 #if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
293  !DEC$ ATTRIBUTES DLLEXPORT :: getBracket
294 #endif
295  use constants_mod, only: ik, rk, tiny_rk
296  use misc_mod, only : swap
297  implicit none
298 
299  real(RK), intent(inout) :: ax, bx
300  real(RK), intent(out) :: cx, fa, fb, fc
301  procedure(getfunc1d_proc) :: getFunc
302 
303  real(RK), parameter :: GOLD = 1.618034_rk
304  real(RK), parameter :: GLIMIT = 100.0_rk
305  real(RK), parameter :: TINY = tiny_rk ! 1.0e-20_RK
306  real(RK) :: fu, q, r, u, ulim
307 
308  fa = getfunc(ax)
309  fb = getfunc(bx)
310  if (fb > fa) then
311  call swap(ax,bx)
312  call swap(fa,fb)
313  end if
314 
315  cx = bx + gold*(bx-ax)
316  fc = getfunc(cx)
317  do
318  if (fb < fc) return
319  r = (bx-ax)*(fb-fc)
320  q = (bx-cx)*(fb-fa)
321  u = bx-((bx-cx)*q-(bx-ax)*r)/(2._rk*sign(max(abs(q-r),tiny),q-r))
322  ulim = bx+glimit*(cx-bx)
323  if ((bx-u)*(u-cx) > 0._rk) then
324  fu = getfunc(u)
325  if (fu < fc) then
326  ax = bx
327  fa = fb
328  bx = u
329  fb = fu
330  return
331  else if (fu > fb) then
332  cx = u
333  fc = fu
334  return
335  end if
336  u = cx+gold*(cx-bx)
337  fu = getfunc(u)
338  else if ((cx-u)*(u-ulim) > 0._rk) then
339  fu = getfunc(u)
340  if (fu < fc) then
341  bx = cx
342  cx = u
343  u = cx+gold*(cx-bx)
344  call shft(fb,fc,fu,getfunc(u))
345  end if
346  else if ((u-ulim)*(ulim-cx) >= 0._rk) then
347  u = ulim
348  fu = getfunc(u)
349  else
350  u = cx+gold*(cx-bx)
351  fu = getfunc(u)
352  end if
353  call shft(ax,bx,cx,u)
354  call shft(fa,fb,fc,fu)
355  end do

398 
399  if (present(dirmat)) then

401  else
402  allocate(powellminimum%DirMat(ndim,ndim), source = 0._rk)
403  do i = 1, ndim
404  powellminimum%DirMat(i,i) = 1._rk
405  end do
406  end if
407  if (present(ftol)) powellminimum%ftol = ftol
408 
409  powellminimum%fmin = getfuncmd(ndim,powellminimum%xmin)
410  pt = powellminimum%xmin
411  powellminimum%niter = 0
412  do
413 
414  powellminimum%niter = powellminimum%niter + 1
415  fp = powellminimum%fmin
416  ibig = 0
417  del = 0._rk
418  do i = 1, ndim
419  xit = powellminimum%DirMat(1:ndim,i)

425  return
426  end if
427  ! LCOV_EXCL_STOP

429  del = fptt - powellminimum%fmin
430  ibig = i
431  end if
432  end do
433 
434  if ( 2._rk*(fp-powellminimum%fmin) <= powellminimum%ftol*(abs(fp)+abs(powellminimum%fmin)) + tiny_rk ) return
435 
436  if (powellminimum%niter == itmax) then
437  ! LCOV_EXCL_START
438  powellminimum%Err%occurred = .true.
439  powellminimum%Err%msg = procedure_name//": maximum number of iterations exceeded."
440  return
441  end if
442  ! LCOV_EXCL_STOP
443 
444  ptt = 2._rk * powellminimum%xmin - pt
445  xit = powellminimum%xmin - pt
446  pt = powellminimum%xmin
447  fptt = getfuncmd(ndim,ptt)
448  if (fptt >= fp) cycle
449  t = 2._rk * (fp-2._rk*powellminimum%fmin+fptt) * (fp-powellminimum%fmin-del)**2 - del*(fp-fptt)**2
450  if (t >= 0.0) cycle
451  call linmin(getfuncmd, ndim, powellminimum%xmin, xit, powellminimum%fmin, powellminimum%Err)
452  if (powellminimum%Err%occurred) then
453  ! LCOV_EXCL_START
454  powellminimum%Err%msg = procedure_name//powellminimum%Err%msg
455  return
456  end if
457  ! LCOV_EXCL_STOP
458  powellminimum%DirMat(1:ndim,ibig) = powellminimum%DirMat(1:ndim,ndim)
459  powellminimum%DirMat(1:ndim,ndim) = xit
460 
461  end do
462 
463  end function minimizepowell
