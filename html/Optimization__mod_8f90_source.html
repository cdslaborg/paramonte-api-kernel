150 function minimizeBrent(getFunc, x0, x1, x2, xtol) result(BrentMinimum)
151#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
152 !DEC$ ATTRIBUTES DLLEXPORT :: minimizeBrent
153#endif
154 use Constants_mod, only: IK, RK
155 procedure(getFunc1D_proc) :: getFunc
156 real(RK) , intent(in), optional :: x0, x1, x2
157 real(RK) , intent(in), optional :: xtol
158 type(BrentMinimum_type) :: BrentMinimum
159
160 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@minimizeBrent"
161 integer(IK) , parameter :: ITMAX = 1000_IK ! the maximum number of iterations
162 real(RK) , parameter :: CGOLD = 0.3819660_RK ! Golden Section switch criterion
163 real(RK) , parameter :: ZEPS = sqrt(epsilon(1._RK))**3 ! tiny nonzero value
164
165 integer(IK) :: iter
166 logical :: isPresentX0, isPresentX1, isPresentX2
167 real(RK) :: a, b, d, e, etemp, fu, fv, fw, fx, p, q, r, tol1, tol2, u, v, w, x, xm
168 real(RK) :: ax, bx, cx, fa, fb, fc
169
170 BrentMinimum%Err%occurred = .false.
171
172 if (present(xtol)) BrentMinimum%xtol = xtol
173
174 isPresentX0 = present(x0)
175 isPresentX1 = present(x1)
176 isPresentX2 = present(x2)
177 if (isPresentX0 .and. isPresentX1 .and. isPresentX2) then
178 BrentMinimum%Bracket = [x0, x1, x2]
179 else
180 if (isPresentX0) then
181 ax = x0
182 else
183 ax = 0._RK ! assume an initial starting point of zero. This is the worst cae scenario.
184 end if
185 if (isPresentX1) then
186 bx = x1
187 else
188 bx = ax + 1._RK
189 end if
190 call getBracket( ax = ax &
191 , bx = bx &
192 , cx = cx &
193 , fa = fa &
194 , fb = fb &
195 , fc = fc &
196 , getFunc = getFunc &
197 )
198 BrentMinimum%Bracket = [ax, bx, cx]
199 end if
200
201 a = min( BrentMinimum%Bracket(1), BrentMinimum%Bracket(3) )
202 b = max( BrentMinimum%Bracket(1), BrentMinimum%Bracket(3) )
203 v = BrentMinimum%Bracket(2)
204 w = v
205 x = v
206 e = 0._RK
207 fx = getFunc(x)
208 fv = fx
209 fw = fx
210 do iter = 1, ITMAX
211 BrentMinimum%niter = iter
212 xm = 0.5_RK * (a+b)
213 tol1 = xtol*abs(x) + ZEPS
214 tol2 = 2.0_RK*tol1
215 if (abs(x-xm) <= (tol2-0.5_RK*(b-a))) then
216 BrentMinimum%xmin = x
217 BrentMinimum%fmin = fx
218 return
219 end if
220 if (abs(e) > tol1) then
221 r = (x-w)*(fx-fv)
222 q = (x-v)*(fx-fw)
223 p = (x-v)*q-(x-w)*r
224 q = 2.0_RK*(q-r)
225 if (q > 0._RK) p = -p
226 q = abs(q)
227 etemp = e
228 e = d
229 if (abs(p) >= abs(0.5_RK*q*etemp) .or. p <= q*(a-x) .or. p >= q*(b-x)) then
230 e = merge(a-x,b-x, x >= xm )
231 d = cgold*e
232 else
233 d = p/q
234 u = x+d
235 if (u-a < tol2 .or. b-u < tol2) d = sign(tol1,xm-x)
236 end if
237 else
238 e = merge(a-x,b-x, x >= xm )
239 d = cgold*e
240 end if
241 u = merge(x+d,x+sign(tol1,d), abs(d) >= tol1 )
242 fu = getFunc(u)
243 if (fu <= fx) then
244 if (u >= x) then
245 a = x
246 else
247 b = x
248 end if
249 call shft(v,w,x,u)
250 call shft(fv,fw,fx,fu)
251 else
252 if (u < x) then
253 a = u
254 else
255 b = u
256 end if
257 if (fu <= fw .or. w == x) then
258 v = w
259 fv = fw
260 w = u
261 fw = fu
262 else if (fu <= fv .or. v == x .or. v == w) then
263 v = u
264 fv = fu
265 end if
266 end if
267 end do
268
269 ! LCOV_EXCL_START
270 BrentMinimum%Err%occurred = .true.
271 BrentMinimum%Err%msg = PROCEDURE_NAME//": maximum number of iterations exceeded."
272 ! LCOV_EXCL_STOP
273 return

291 subroutine getBracket(ax,bx,cx,fa,fb,fc,getFunc)
292#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
293 !DEC$ ATTRIBUTES DLLEXPORT :: getBracket
294#endif
295 use Constants_mod, only: IK, RK, TINY_RK
296 use Misc_mod, only : swap
297 implicit none
298
299 real(RK), intent(inout) :: ax, bx
300 real(RK), intent(out) :: cx, fa, fb, fc
301 procedure(getFunc1D_proc) :: getFunc
302
303 real(RK), parameter :: GOLD = 1.618034_RK
304 real(RK), parameter :: GLIMIT = 100.0_RK
305 real(RK), parameter :: TINY = TINY_RK ! 1.0e-20_RK
306 real(RK) :: fu, q, r, u, ulim
307
308 fa = getFunc(ax)
309 fb = getFunc(bx)
310 if (fb > fa) then
311 call swap(ax,bx)
312 call swap(fa,fb)
313 end if
314
315 cx = bx + GOLD*(bx-ax)
316 fc = getFunc(cx)
317 do
318 if (fb < fc) return
319 r = (bx-ax)*(fb-fc)
320 q = (bx-cx)*(fb-fa)
321 u = bx-((bx-cx)*q-(bx-ax)*r)/(2._RK*sign(max(abs(q-r),TINY),q-r))
322 ulim = bx+GLIMIT*(cx-bx)
323 if ((bx-u)*(u-cx) > 0._RK) then
324 fu = getFunc(u)
325 if (fu < fc) then
326 ax = bx
327 fa = fb
328 bx = u
329 fb = fu
330 return
331 else if (fu > fb) then
332 cx = u
333 fc = fu
334 return
335 end if
336 u = cx+GOLD*(cx-bx)
337 fu = getFunc(u)
338 else if ((cx-u)*(u-ulim) > 0._RK) then
339 fu = getFunc(u)
340 if (fu < fc) then
341 bx = cx
342 cx = u
343 u = cx+GOLD*(cx-bx)
344 call shft(fb,fc,fu,getFunc(u))
345 end if
346 else if ((u-ulim)*(ulim-cx) >= 0._RK) then
347 u = ulim
348 fu = getFunc(u)
349 else
350 u = cx+GOLD*(cx-bx)
351 fu = getFunc(u)
352 end if
353 call shft(ax,bx,cx,u)
354 call shft(fa,fb,fc,fu)
355 end do
356
357 contains

373 function minimizePowell(ndim, getFuncMD, StartVec, DirMat, ftol) result(PowellMinimum)
374#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
375 !DEC$ ATTRIBUTES DLLEXPORT :: minimizePowell
376#endif
377 use Constants_mod, only: IK, RK, TINY_RK ! tiny = 1.0e-25_RK
378 implicit none
379
380 procedure(getFuncMD_proc) :: getFuncMD
381 integer(IK) , intent(in) :: ndim
382 real(RK) , intent(in) :: StartVec(ndim)
383 real(RK) , intent(in) , optional :: DirMat(ndim,ndim)
384 real(RK) , intent(in) , optional :: ftol
385 type(PowellMinimum_type) :: PowellMinimum
386
387 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@minimizeBrent"
388 integer(IK), parameter :: ITMAX = 1000
389
390 integer(IK) :: i, ibig
391 real(RK) :: del,fp,fptt,t
392 real(RK) :: pt(ndim), ptt(ndim), xit(ndim)
393
394 PowellMinimum%ndim = ndim
395 PowellMinimum%Err%occurred = .false.
396
397 allocate(PowellMinimum%xmin, source = StartVec)
398
399 if (present(DirMat)) then
400 PowellMinimum%DirMat = DirMat
401 else
402 allocate(PowellMinimum%DirMat(ndim,ndim), source = 0._RK)
403 do i = 1, ndim
404 PowellMinimum%DirMat(i,i) = 1._RK
405 end do
406 end if
407 if (present(ftol)) PowellMinimum%ftol = ftol
408
409 PowellMinimum%fmin = getFuncMD(ndim,PowellMinimum%xmin)
410 pt = PowellMinimum%xmin
411 PowellMinimum%niter = 0
412 do
413
414 PowellMinimum%niter = PowellMinimum%niter + 1
415 fp = PowellMinimum%fmin
416 ibig = 0
417 del = 0._RK
418 do i = 1, ndim
419 xit = PowellMinimum%DirMat(1:ndim,i)
420 fptt = PowellMinimum%fmin
421 call linmin(getFuncMD, ndim, PowellMinimum%xmin, xit, PowellMinimum%fmin, PowellMinimum%Err)
422 if (PowellMinimum%Err%occurred) then
423 ! LCOV_EXCL_START
424 PowellMinimum%Err%msg = PROCEDURE_NAME//PowellMinimum%Err%msg
425 return
426 end if
427 ! LCOV_EXCL_STOP
428 if (fptt - PowellMinimum%fmin > del) then
429 del = fptt - PowellMinimum%fmin
430 ibig = i
431 end if
432 end do
433
434 if ( 2._RK*(fp-PowellMinimum%fmin) <= PowellMinimum%ftol*(abs(fp)+abs(PowellMinimum%fmin)) + TINY_RK ) return
435
436 if (PowellMinimum%niter == ITMAX) then
437 ! LCOV_EXCL_START
438 PowellMinimum%Err%occurred = .true.
439 PowellMinimum%Err%msg = PROCEDURE_NAME//": maximum number of iterations exceeded."
440 return
441 end if
442 ! LCOV_EXCL_STOP
443
444 ptt = 2._RK * PowellMinimum%xmin - pt
445 xit = PowellMinimum%xmin - pt
446 pt = PowellMinimum%xmin
447 fptt = getFuncMD(ndim,ptt)
448 if (fptt >= fp) cycle
449 t = 2._RK * (fp-2._RK*PowellMinimum%fmin+fptt) * (fp-PowellMinimum%fmin-del)**2 - del*(fp-fptt)**2
450 if (t >= 0.0) cycle
451 call linmin(getFuncMD, ndim, PowellMinimum%xmin, xit, PowellMinimum%fmin, PowellMinimum%Err)
452 if (PowellMinimum%Err%occurred) then
453 ! LCOV_EXCL_START
454 PowellMinimum%Err%msg = PROCEDURE_NAME//PowellMinimum%Err%msg
455 return
456 end if
457 ! LCOV_EXCL_STOP
458 PowellMinimum%DirMat(1:ndim,ibig) = PowellMinimum%DirMat(1:ndim,ndim)
459 PowellMinimum%DirMat(1:ndim,ndim) = xit
460
461 end do
462
463 end function minimizePowell

467 subroutine linmin(getFuncMD, ndim, StartVec, DirVec, fmin, Err)
468#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
469 !DEC$ ATTRIBUTES DLLEXPORT :: linmin
470#endif
471 use Constants_mod, only: IK, RK
472 use Err_mod, only: Err_type
473 implicit none
474 procedure(getFuncMD_proc) :: getFuncMD
475 integer(IK) , intent(in) :: ndim
476 real(RK) , intent(inout) :: StartVec(ndim), DirVec(ndim)
477 real(RK) , intent(out) :: fmin
478 type(Err_type) , intent(out) :: Err
479 real(RK) , parameter :: XTOL = 1.0e-8_RK
480 real(RK) :: ax, bx, fa, fb, fx, xx
481 type(BrentMinimum_type) :: BrentMinimum
482#if WSL_ENABLED
483 getFuncMD_WSL => getFuncMD
484 DirVec_WSL = DirVec
485 StartVec_WSL = StartVec
486 ndim_WSL = ndim
487#endif
488 ax = 0.0
489 xx = 1.0
490 call getBracket(ax,xx,bx,fa,fx,fb,getFunc1D)
491 BrentMinimum = minimizeBrent(getFunc1D, ax, xx, bx, XTOL)
492 if (BrentMinimum%Err%occurred) then
493 ! LCOV_EXCL_START
494 Err = BrentMinimum%Err
495 return
496 ! LCOV_EXCL_STOP
497 else
498 Err%occurred = .false.
499 end if
500 fmin = BrentMinimum%fmin
501 DirVec = BrentMinimum%xmin * DirVec
502 StartVec = StartVec + DirVec
503#if WSL_ENABLED
504 nullify(getFuncMD_WSL)
505#else
506 contains
507 function getFunc1D(x) result(funcVal)
508 implicit none
509 real(RK), intent(in) :: x
510 real(RK) :: funcVal
511 real(RK), allocatable :: xt(:)
512 xt = StartVec + x * DirVec
513 funcVal = getFuncMD(ndim,xt)
514 end function getFunc1D
515#endif
516 end subroutine linmin
