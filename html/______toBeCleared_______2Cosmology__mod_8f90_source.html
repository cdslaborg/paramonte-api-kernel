16 ! returns the luminosity distance in units of Mpc for the given redshift z.
17 ! The input parameters is the redshift (z), while the required parameters are taken from module Constants:
18 ! LS2HC: which is the speed of light in units of km/s divided by the Hubble Constant in units of km/s/MPc.
19 ! For input redshifts z>zlim=0.1, the luminosity distance will be calculated according to approximation algorithm of Wickramasinghe & Okwatta (2010).
20 ! Note that for redshifts less than 0.1, the error in the calculated luminosity distance grows to more than 0.001.
21 ! This algorithm should therefore not be used for z<0.1, and only when the desired accuracy in Lum. dis. is 0.001.
22 ! Amir Shahmoradi, Wednesday July 14, 2016, 9:59 PM, ICES, The University of Texas at Austin.

32 ! Parameters required for the calculation of Luminosity distance, that are private to this function
33 real(rk), parameter :: alpha0 = 1._rk + 2._rk * omega_de / omega_dm
34 real(rk), parameter :: x0 = log( alpha0 + sqrt( alpha0**2 - 1._rk ) )
35 real(rk), parameter :: psi0=x0**0.33333333_rk*(1.5874010519682_rk-6.2992105236833e-3_rk*x0**2+7.5375168659459e-5_rk*x0**4)
36 real(rk), parameter :: lum_dis_factor = ls2hc / ( omega_de**0.16666666_rk * omega_dm**0.33333333_rk )
37
38 alpha = 1._rk + 2._rk * omega_de / ( omega_dm * (1._rk+z)**3 )
39 x = log( alpha + sqrt( alpha**2 - 1._rk ) )
40 psi = x**0.33333333_rk * ( 1.5874010519682_rk - 6.2992105236833e-3_rk * x**2 + 7.5375168659459e-5_rk * x**4 )
41 getlumdiswickram = lum_dis_factor * ( 1._rk + z ) * ( psi0 - psi )
42
43 end function getlumdiswickram
