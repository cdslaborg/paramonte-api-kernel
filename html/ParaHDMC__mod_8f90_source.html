470 function addstars(text)
471
472 use misc, only: write
473 use, intrinsic :: iso_fortran_env, only: output_unit
474 implicit none
475
476 character(*) , intent(in), optional :: text
477 character(mp_logFileRecordLength) :: addstars
478
479 integer(IK) , parameter :: stars_margin_len = len(mp_space)
480 character(*), parameter :: star = '*'
481 integer(IK) :: i,istart,iend,lentext
482
483 if (present(text)) then ! sandwich text between stars
484
485 lentext = len(trim(adjustl(text)))
486
487 ! initialize empty container
488 do i=1,mp_logfilerecordlength
489 addstars(i:i) = " "
490 end do
491
492 ! add margin
493 do i=1,stars_margin_len
494 addstars(i:i) = star
495 addstars(mp_logfilerecordlength-i+1:mp_logfilerecordlength-i+1) = star
496 end do
497
498 ! add text in between
499 istart = (mp_logfilerecordlength-lentext) / 2
500 iend = istart + lentext - 1
501 if(istart<=2*stars_margin_len .or. iend>=mp_logfilerecordlength-2*stars_margin_len) then
502 call write( output_unit,2,2, mp_brand // 'Internal error occurred while trying to use addStars()' )
503 end if
504 addstars(istart:iend) = trim(adjustl(text))
505
506 else
507
508 do i=1,mp_logfilerecordlength
509 addstars(i:i) = star
510 end do
511
512 end if
513
514 end function addstars

519 subroutine doforallneeded(doThis,action)
520 use misc, only: num2str
521 implicit none
522 external :: doThis
523 character(*), intent(in), optional :: action
524 character(:), allocatable :: when
525 !integer(IK) :: syncStat

527

530 else
531 when = "SYNC error occurred between coarray images: "
532 end if

535 if (mv_imageid>1) then
