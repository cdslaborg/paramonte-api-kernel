108
109 subroutine getClusteredPoint( self & ! LCOV_EXCL_LINE
110 , nd, ndmin, ndmax & ! LCOV_EXCL_LINE
111 , nc, ncmin, ncmax & ! LCOV_EXCL_LINE
112 , Size, sizeMin, sizeMax & ! LCOV_EXCL_LINE
113 , Center, centerMin, centerMax & ! LCOV_EXCL_LINE
114 , Std, stdmin, stdmax & ! LCOV_EXCL_LINE
115 , Eta, etaMin, etaMax & ! LCOV_EXCL_LINE
116 , dist & ! LCOV_EXCL_LINE
117 , hubEnabled & ! LCOV_EXCL_LINE
118 , relTol & ! LCOV_EXCL_LINE
119 )
120#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
121 !DEC$ ATTRIBUTES DLLEXPORT :: getClusteredPoint
122#endif
123 use MatrixDiag_mod, only: genEye
124 use MatrixCholesky_mod, only: getChoLow
125 use Matrix_mod, only: genInvMatFromChoLow
126 use DomainEllipsoid_mod, only: getRandUnif
127 use GeometryEllipsoid_mod, only: isInsideEllipsoid
128 use Statistics_mod, only: getLogProbMVU
129 use Statistics_mod, only: genRandCorMat
130 use DistMultiNorm_mod, only: getLogProbDenMVN
131 use Statistics_mod, only: genRandInt
132 use Constants_mod, only: RK, HUGE_RK
133 use String_mod, only: val2str !, genLowerCase
134 use Geometry_mod, only: getPairDistSq
135 use GeometryEllipsoid_mod, only: genLogVolUnitBall
136 use ProcessPoisson_mod, only: getMinSumPairDistSqPointIndex
137 use MathLogSumExp_mod, only: genLogSumExp
138 use Math_mod, only: getLogSubExp
139 use MathCumSum_mod, only: genCumSum
140 implicit none
141 class(ClusteredPoint_type), intent(inout) :: self
142 integer(IK) , intent(in), optional :: nd, ndmin, ndmax
143 integer(IK) , intent(in), optional :: nc, ncmin, ncmax
144 integer(IK) , intent(in), optional :: sizeMin, sizeMax
145 real(RK) , intent(in), optional :: stdMin, stdMax
146 real(RK) , intent(in), optional :: etaMin, etaMax
147 real(RK) , intent(in), optional :: centerMin, centerMax
148 real(RK) , intent(in), optional :: Center(:,:), Std(:,:), Eta(:)
149 logical , intent(in), optional :: hubEnabled
150 integer(IK) , intent(in), optional :: Size(:)
151 character(*), intent(in), optional :: dist
152 real(RK) , intent(in), optional :: relTol
153 integer(IK) , allocatable :: CumSumSize(:)
154 real(RK) , allocatable :: NormedPoint(:)
155 real(RK) , allocatable :: CumSumVolNormed(:)
156 real(RK) , allocatable :: VolNormed(:)
157 logical :: isUniformMixture
158 logical :: isUniform, isMember
159 logical :: isNormal
160 real(RK) :: dummy, maxLogVolNormed, relTolDef
161 integer(IK) :: i, j, k, ic, ip, sumOverlap
162
163 self%Err%occurred = .false.
164 if (present(hubEnabled)) self%hubEnabled = hubEnabled
165 if (present(relTol)) then
166 relTolDef = relTol
167 else
168 relTolDef = 0.01_RK
169 end if
170
171 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
172 ! ndmin, ndmax, nd
173 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
174
175 if (present(ndmin)) then
176 self%ndmin = ndmin
177 else
178 self%ndmin = 2
179 endif
180
181 if (present(ndmax)) then
182 self%ndmax = ndmax
183 else
184 self%ndmax = self%ndmin * 10_IK
185 endif
186
187 if (present(nd)) then
188 self%nd = nd
189 else
190 self%nd = genRandInt(self%ndmin, self%ndmax)
191 endif
192
193 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
194 ! ncmin, ncmax, nc
195 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
196
197 if (present(ncmin)) then
198 self%ncmin = ncmin
199 else
200 self%ncmin = 1_IK
201 endif
202
203 if (present(ncmax)) then
204 self%ncmax = ncmax
205 else
206 self%ncmax = self%ncmin * 10_IK
207 endif
208
209 if (present(nc)) then
210 self%nc = nc
211 else
212 self%nc = genRandInt(self%ncmin, self%ncmax)
213 endif
214
215 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
216 ! sizeMin, sizeMax, Size
217 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
218
219 if (present(sizeMin)) then
220 self%sizeMin = sizeMin
221 else
222 self%sizeMin = self%nd + 1_IK
223 endif
224
225 if (present(sizeMax)) then
226 self%sizeMax = sizeMax
227 else
228 self%sizeMax = self%sizeMin * 1000
229 endif
230
231 if (present(Size)) then
232 self%Size = Size
233 else
234 if (allocated(self%Size)) deallocate(self%Size) ! LCOV_EXCL_LINE ! GFortran crashes without this
235 if (allocated(CumSumSize)) deallocate(CumSumSize) ! LCOV_EXCL_LINE ! GFortran crashes without this
236 allocate(self%Size(self%nc), CumSumSize(0:self%nc))
237 CumSumSize(0) = 0._RK
238 do ic = 1, self%nc
239 self%Size(ic) = genRandInt(self%sizeMin, self%sizeMax)
240 CumSumSize(ic) = CumSumSize(ic-1) + self%Size(ic)
241 end do
242 endif
243
244 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
245 ! centerMin, centerMax, Size
246 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
247
248 if (present(centerMin)) then
249 self%centerMin = centerMin
250 else
251 self%centerMin = 0._RK
252 endif
253
254 if (present(centerMax)) then
255 self%centerMax = centerMax
256 else
257 self%centerMax = 1._RK
258 endif
259
260 if (present(Center)) then
261 self%Center = Center
262 else
263 if (allocated(self%Center)) deallocate(self%Center) ! LCOV_EXCL_LINE ! GFortran crashes without this
264 allocate(self%Center(self%nd,self%nc))
265 call random_number(self%Center)
266 do concurrent(ic = 1:self%nc)
267 self%Center(:,ic) = self%centerMin + self%Center(:,ic) * (self%centerMax - self%centerMin)
268 end do
269 endif
270
271 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
272 ! stdMin, stdMax, Size
273 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
274
275 if (present(stdMin)) then
276 self%stdMin = stdMin
277 else
278 self%stdMin = 0._RK
279 endif
280
281 if (present(stdMax)) then
282 self%stdMax = stdMax
283 else
284 self%stdMax = 1._RK
285 endif
286
287 if (present(Std)) then
288 self%Std = Std
289 else
290 if (allocated(self%Std)) deallocate(self%Std) ! LCOV_EXCL_LINE ! GFortran crashes without this
291 allocate(self%Std(self%nd,self%nc))
292 call random_number(self%Std)
293 do concurrent(ic = 1:self%nc)
294 self%Std(:,ic) = self%stdMin + self%Std(:,ic) * (self%stdMax - self%stdMin)
295 end do
296 endif
297
298 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
299 ! etaMin, etaMax, eta
300 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
301
302 if (present(etaMin)) then
303 self%etaMin = etaMin
304 else
305 self%etaMin = 0.1_RK
306 endif
307
308 if (present(etaMax)) then
309 self%etaMax = etaMax
310 else
311 self%etaMax = self%etaMin * (1._RK/self%etaMin)**2
312 endif
313
314 if (present(Eta)) then
315 self%Eta = Eta
316 else
317 if (allocated(self%Eta)) deallocate(self%Eta) ! LCOV_EXCL_LINE ! GFortran crashes without this
318 allocate(self%Eta(self%nc))
319 call random_number(self%Eta)
320 do concurrent(ic = 1:self%nc)
321 self%Eta(ic) = self%etaMin + self%Eta(ic) * (self%etaMax - self%etaMin)
322 end do
323 endif
324
325 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
326 ! Generate the representative matrices of the clusters
327 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
328
329 if (allocated(self%ChoDia)) deallocate(self%ChoDia) ! LCOV_EXCL_LINE ! GFortran crashes without this
330 if (allocated(self%InvCovMat)) deallocate(self%InvCovMat) ! LCOV_EXCL_LINE ! GFortran crashes without this
331 if (allocated(self%ChoLowCovUpp)) deallocate(self%ChoLowCovUpp) ! LCOV_EXCL_LINE ! GFortran crashes without this
332 if (allocated(self%LogVolNormed)) deallocate(self%LogVolNormed) ! LCOV_EXCL_LINE ! GFortran crashes without this
333 if (allocated(self%LogSqrtDetCovMat)) deallocate(self%LogSqrtDetCovMat) ! LCOV_EXCL_LINE ! GFortran crashes without this
334 if (allocated(CumSumVolNormed)) deallocate(CumSumVolNormed) ! LCOV_EXCL_LINE ! GFortran crashes without this
335
336 allocate(self%LogVolNormed(self%nc))
337 allocate(self%ChoDia(self%nd,self%nc))
338 allocate(self%InvCovMat(self%nd,self%nd,self%nc))
339 allocate(self%ChoLowCovUpp(self%nd,self%nd,self%nc))
340 allocate(self%LogSqrtDetCovMat(self%nc))
341 allocate(CumSumVolNormed(self%nc))
342
343 do ic = 1, self%nc
344
345 ! Generate random correlation matrix.
346
347 self%ChoDia(1,ic) = -1._RK
348 do k = 1, self%nd
349
350 if (k < self%nd) then
351 self%ChoLowCovUpp(:,:,ic) = genRandCorMat(self%nd, self%Eta(ic))
352 if (self%ChoLowCovUpp(1,1,ic) < 0._RK) cycle
353 else
354 self%ChoLowCovUpp(:,:,ic) = genEye(self%nd, self%nd)
355 end if
356
357 ! Compute the covariance matrix.
358
359 do j = 1, self%nd
360 do i = 1, self%nd
361 self%ChoLowCovUpp(i,j,ic) = self%ChoLowCovUpp(i,j,ic) * self%Std(i,ic) * self%Std(j,ic)
362 end do
363 end do
364
365 ! Compute the Cholesky factorization.
366
367 call getChoLow( nd = self%nd & ! LCOV_EXCL_LINE
368 , ChoLowMatUpp = self%ChoLowCovUpp(1:self%nd,1:self%nd,ic) & ! LCOV_EXCL_LINE
369 , ChoDia = self%ChoDia(1:self%nd,ic) & ! LCOV_EXCL_LINE
370 )
371 if (self%ChoDia(1,ic) < 0._RK) then
372 !self%Err%occurred = .true.
373 !self%Err%msg = "Singular Covariance matrix detected."
374 !write(*,"(A)") "ChoLowCovUpp:"
375 !write(*,"("//val2str(self%nd)//"(F15.8,:,' '))") self%ChoLowCovUpp(1:self%nd,1:self%nd,ic)
376 !error stop
377 cycle ! LCOV_EXCL_LINE
378 end if
379
380 exit
381
382 end do
383
384!do j = 1, self%nd
385! write(*,*) self%ChoLowCovUpp(1:self%nd,j,ic)
386!end do
387
388 ! Compute the inverse covariance matrix.
389
390 self%InvCovMat(1:self%nd,1:self%nd,ic) = genInvMatFromChoLow( nd = self%nd & ! LCOV_EXCL_LINE
391 , ChoLow = self%ChoLowCovUpp(1:self%nd,1:self%nd,ic) & ! LCOV_EXCL_LINE
392 , ChoDia = self%ChoDia(1:self%nd,ic) & ! LCOV_EXCL_LINE
393 )
394
395 self%LogSqrtDetCovMat(ic) = sum(log(self%ChoDia(1:self%nd,ic)))
396 self%LogVolNormed(ic) = sum(log(self%ChoDia(1:self%nd,ic)))
397
398 end do
399
400 maxLogVolNormed = maxval(self%LogVolNormed)
401 self%logSumVolNormed = genLogSumExp(self%LogVolNormed, maxLogVolNormed)
402 VolNormed = exp(self%LogVolNormed - maxLogVolNormed)
403 CumSumVolNormed(1:self%nc) = genCumSum(VolNormed)
404 CumSumVolNormed(1:self%nc) = CumSumVolNormed / CumSumVolNormed(self%nc)
405
406
407 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
408 ! Set the distribution
409 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
410
411 if (present(dist)) then
412 self%dist = dist
413 else
414 self%dist = "uniform-mixture"
415 end if
416
417 isUniform = self%dist == "uniform"
418 isNormal = self%dist == "normal-mixture"
419 isUniformMixture = self%dist == "uniform-mixture"
420
421 if (.not. (isUniformMixture .or. isUniform)) then
422 self%Err%occurred = .true.
423 self%Err%msg = "No point distribution other than uniform is currently supported."
424 return
425 end if
426
427 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
428 ! Generate the cluster members
429 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
430
431 self%np = sum(self%Size)
432
433 if (allocated(self%Point)) deallocate(self%Point) ! LCOV_EXCL_LINE ! GFortran crashes without this
434 if (allocated(self%LogProb)) deallocate(self%LogProb) ! LCOV_EXCL_LINE ! GFortran crashes without this
435 if (allocated(self%Overlap)) deallocate(self%Overlap) ! LCOV_EXCL_LINE ! GFortran crashes without this
436 if (allocated(self%Membership)) deallocate(self%Membership) ! LCOV_EXCL_LINE ! GFortran crashes without this
437 if (allocated(NormedPoint)) deallocate(NormedPoint) ! LCOV_EXCL_LINE ! GFortran crashes without this
438
439 allocate(self%LogProb(self%np))
440 allocate(self%Membership(self%np))
441 allocate(self%Point(self%nd,self%np))
442 allocate(self%Overlap(1:self%nc,self%np), source = 0_IK)
443 allocate(NormedPoint(self%nd))
444
445 if (isUniformMixture) then
446
447 do ic = 1, self%nc
448 do ip = CumSumSize(ic-1) + 1, CumSumSize(ic)
449
450 self%Membership(ip) = ic
451
452 call getRandUnif( Rand = self%Point(1:self%nd,ip) & ! LCOV_EXCL_LINE
453 , Mean = self%Center(1:self%nd,ic) & ! LCOV_EXCL_LINE
454 , ChoLow = self%ChoLowCovUpp(1:self%nd,1:self%nd,ic) & ! LCOV_EXCL_LINE
455 , ChoDia = self%ChoDia(1:self%nd,ic) & ! LCOV_EXCL_LINE
456 )
457
458 ! Compute the number of cluster memberships of each point.
459
460 do i = 1, self%nc
461 NormedPoint(1:self%nd) = self%Point(1:self%nd,ip) - self%Center(1:self%nd,i)
462 isMember = isInsideEllipsoid(nd = self%nd, NormedPoint = NormedPoint, InvRepMat = self%InvCovMat(1:self%nd,1:self%nd,i))
463 if (isMember) then
464 self%Overlap(ic,ip) = self%Overlap(ic,ip) + 1
465 end if
466 end do
467
468 end do
469 end do
470
471 elseif (isUniform) then
472
473 self%Size = 0 ! all cluster sizes will be determined after all point generations.
474
475 do ip = 1, self%np
476
477 ! Check for the distribution type.
478
479 loopAddClusterMember: do
480
481 call random_number(dummy)
482 ic = minloc(CumSumVolNormed, dim = 1, mask = CumSumVolNormed > dummy)
483 !ic = genRandInt(lowerBound = 1, upperBound = self%nc)
484
485 call getRandUnif( Rand = self%Point(1:self%nd,ip) & ! LCOV_EXCL_LINE
486 , Mean = self%Center(1:self%nd,ic) & ! LCOV_EXCL_LINE
487 , ChoLow = self%ChoLowCovUpp(1:self%nd,1:self%nd,ic) & ! LCOV_EXCL_LINE
488 , ChoDia = self%ChoDia(1:self%nd,ic) & ! LCOV_EXCL_LINE
489 )
490
491 ! Compute the number of cluster memberships of each point.
492
493 sumOverlap = 0_IK
494 do i = 1, self%nc
495 NormedPoint(1:self%nd) = self%Point(1:self%nd,ip) - self%Center(1:self%nd,i)
496 isMember = isInsideEllipsoid(nd = self%nd, NormedPoint = NormedPoint, InvRepMat = self%InvCovMat(1:self%nd,1:self%nd,i))
497 if (isMember) then
498 self%Overlap(ic,ip) = 1_IK
499 sumOverlap = sumOverlap + 1_IK
500 end if
501 end do
502
503 call random_number(dummy)
504 if (dummy < 1._RK / sumOverlap) then
505 self%Size(ic) = self%Size(ic) + 1_IK
506 self%Membership(ip) = ic
507 exit loopAddClusterMember
508 end if
509
510 cycle loopAddClusterMember
511
512 end do loopAddClusterMember
513
514 end do
515
516 end if
517
518 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
519 ! Compute the effective volume of a single Point
520 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
521
522 if (allocated(self%PointLogVolNormed)) deallocate(self%PointLogVolNormed) ! LCOV_EXCL_LINE ! GFortran crashes without this
523 allocate(self%PointLogVolNormed(self%nc))
524
525 do ic = 1, nc
526 if (self%Size(ic)>0_IK) then
527 self%PointLogVolNormed(ic) = self%LogVolNormed(ic) - log(real(self%Size(ic),RK))
528 else
529 self%PointLogVolNormed(ic) = HUGE_RK
530 end if
531 end do
532
533 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
534 ! Compute the Point probabilities and Estimate the total volume of all clusters while taking into account potential overlaps
535 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
536
537! if (sum(self%Size(1:nc)) > 2000) then
538!
539! ! This approach uses the simulated points to estimate the overlaps. However, it becomes increasingly biased in low counts.
540! self%logSumVolNormedEffective = self%logSumVolNormed
541! do ip = 1, self%np
542! if (isUniform) then
543! self%LogProb(ip) = 1._RK
544! do ic = 1, nc
545! if (ic /= self%Membership(ip) .and. self%Overlap(ic,ip) == 1) then
546! write(*,*) self%logSumVolNormedEffective, self%PointLogVolNormed(self%Membership(ip))
547! self%logSumVolNormedEffective = getLogSubExp(self%logSumVolNormedEffective,self%PointLogVolNormed(self%Membership(ip)))
548! end if
549! end do
550! elseif (isUniformMixture) then
551! self%LogProb(ip) = 0._RK
552! do ic = 1, self%nc
553! if (self%Overlap(ic,ip)==1) self%LogProb(ip) = self%LogProb(ip) - self%PointLogVolNormed(ic)
554! end do
555! end if
556! end do
557! !write(*,*) "self%logSumVolNormedEffective 1 = ", self%logSumVolNormedEffective
558!
559! else
560
561 block
562 use Geometry_mod, only: getLogSumVolEff
563 call getLogSumVolEff( nd = self%nd & ! LCOV_EXCL_LINE
564 , nc = self%nc & ! LCOV_EXCL_LINE
565 , relTol = relTolDef & ! LCOV_EXCL_LINE
566 , Center = self%Center & ! LCOV_EXCL_LINE
567 , ChoDia = self%ChoDia & ! LCOV_EXCL_LINE
568 , LogVol = self%LogVolNormed & ! LCOV_EXCL_LINE
569 , InvCovMat = self%InvCovMat & ! LCOV_EXCL_LINE
570 , logSumVol = self%logSumVolNormed & ! LCOV_EXCL_LINE
571 , ChoLowCovUpp = self%ChoLowCovUpp & ! LCOV_EXCL_LINE
572 , logSumVolEff = self%logSumVolNormedEffective & ! LCOV_EXCL_LINE
573 )
574 end block
575 !write(*,*) "self%logSumVolNormedEffective 2 = ", self%logSumVolNormedEffective
576
577! end if
578
579 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
580 ! hubify
581 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
582
583 if (self%hubEnabled) then
584 self%InnestPoint = self%Point(1:nd, getMinSumPairDistSqPointIndex(self%nd, self%np, self%Point))
585 self%Hub = Hub_type( self%np, PairDistSq = getPairDistSq(nd = self%nd, np = self%np, Point = self%Point) )
586 if (self%Hub%Err%occurred) then
587 ! LCOV_EXCL_START
588 self%Err%occurred = .true.
589 self%Err%msg = self%Hub%Err%msg
590 return
591 ! LCOV_EXCL_STOP
592 end if
593 end if
594
595 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
596 ! compute densities
597 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
598
599 self%logVolUnitBall = genLogVolUnitBall(self%nd)
600 if (allocated(self%LogDensity)) deallocate(self%LogDensity) ! LCOV_EXCL_LINE ! GFortran crashes without this
601 allocate(self%LogDensity(self%nc))
602 do concurrent(ic = 1:nc)
603 if (self%Size(ic) > 0_IK) self%LogDensity(ic) = log(real(self%Size(ic),RK)) - self%LogVolNormed(ic) - self%logVolUnitBall
604 end do
605
606 end subroutine getClusteredPoint
607
608!%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
609
610 subroutine writeClusteredPoint(self, fileUnit)
611 use Constants_mod, only: RK
612 implicit none
613 class(ClusteredPoint_type) , intent(in) :: self
614 integer(IK) , intent(in) :: fileUnit
615 character(*), parameter :: fileFormat = "(*(g0.8,:,','))"
