92 real(RK) , intent(in) :: logSqrtDetInvCovMat

95 real(RK) :: NormedPoint(nd)
96 real(RK) :: logProbSkewNorm
97 real(RK) , parameter :: LOG2 = log(2._RK)
98 real(RK) , parameter :: LOGHALF = log(0.5_RK)
99 NormedPoint = Point - Mean
100 logProbSkewNorm = LOG2 + nd*LOG_INVERSE_SQRT_TWO_PI + logSqrtDetInvCovMat & ! LCOV_EXCL_LINE
101 - 0.5_RK * dot_product(NormedPoint, matmul(InvCovMat,NormedPoint)) & ! LCOV_EXCL_LINE
102 + LOGHALF + log( 1._RK + erf(INVERSE_SQRT_TWO * dot_product(NormedPoint,Alpha)) )
103 end function getLogProbMSN_RK

121 function genRandMSN(nd,Mean,AugChoLow,AugChoDia) result(RandMSN)
122#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
123 !DEC$ ATTRIBUTES DLLEXPORT :: genRandMSN
124#endif
125 use DistMultiNorm_mod, only: genRandMVN
126 use DistUniNorm_mod, only: genRandGaus => genRand
127 use Constants_mod, only: IK, RK
128 implicit none
129 integer(IK), intent(in) :: nd
130 real(RK) , intent(in) :: Mean(nd)
131 real(RK) , intent(in) :: AugChoLow(0:nd,0:nd), AugChoDia(0:nd)
132 real(RK) :: RandMSN(nd), normrnd
133 logical :: switchDisabled
134 integer(IK) :: j, i
135 normrnd = genRandGaus()
136 switchDisabled = normrnd > 0._RK
137 RandMSN(1:nd) = AugChoLow(1:nd,0) * normrnd
138 do j = 1, nd
139 normrnd = genRandGaus()
140 RandMSN(j) = RandMSN(j) + AugChoDia(j) * normrnd
141 do i = j+1, nd
142 RandMSN(i) = RandMSN(i) + AugChoLow(i,j) * normrnd
143 end do
144 end do
145 if (switchDisabled) then
146 RandMSN = Mean(1:nd) + RandMSN
147 else
148 RandMSN = Mean(1:nd) - RandMSN
149 end if
150 end function genRandMSN

170 subroutine getAugChoFac(nd,Alpha,CovMat,AugChoLow,AugChoDia)
171 use Matrix_mod, only: getChoLow
172 use Constants_mod, only: IK, RK
173 implicit none
174 integer(IK) , intent(in) :: nd
175 real(RK) , intent(in) :: Alpha(nd)
176 real(RK) , intent(in) :: CovMat(nd,nd)
177 real(RK) , intent(out) :: AugChoLow(0:nd,0:nd), AugChoDia(0:nd)
178 real(RK) :: Delta(nd), coef
179 integer(IK) :: j
180 Delta(1:nd) = matmul(CovMat, Alpha)
181 coef = 1._RK / sqrt(1._RK + dot_product(Alpha,Delta))
182 Delta = Delta * coef
183 AugChoLow(0,0) = 1._RK
184 do j = 1, nd
185 AugChoLow(0,j) = Delta(j)
186 AugChoLow(1:j,j) = CovMat(1:j,j)
187 end do
188 call getChoLow(nd+1,AugChoLow,AugChoDia)
189 end subroutine getAugChoFac
