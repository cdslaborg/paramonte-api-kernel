100  !DEC$ ATTRIBUTES DLLEXPORT :: partition

102  use constants_mod, only: ik, rk
103  implicit none
104  real(rk) , intent(inout) :: array(:)
105  integer(IK), intent(out) :: marker
106  integer(IK) :: i, j
107  real(rk) :: temp
108  real(rk) :: x ! pivot point
109  x = array(1)
110  i= 0
111  j= size(array) + 1
112  do
113  j = j-1
114  do
115  if (array(j) <= x) exit
116  j = j-1
117  end do
118  i = i+1
119  do
120  if (array(i) >= x) exit
121  i = i+1
122  end do
123  if (i < j) then ! exchange array(i) and array(j)
124  temp = array(i)
125  array(i) = array(j)
126  array(j) = temp
127  elseif (i == j) then
128  marker = i+1
129  return
130  else
131  marker = i
132  return
133  endif
134  end do
135  end subroutine partition

164  real(rk) , intent(inout) :: point(np)
165  type(err_type) , intent(out) :: err
166 
167  character(*) , parameter :: procedure_name = module_name//"@sortAscending_RK()"
168  real(rk) :: dummy
169  integer(IK) :: k,i,j,jstack,m,r,istack(nstack)
170 
171  err%occurred = .false.
172 
173  jstack=0
174  m = 1
175  r = np
176  do
177  if (r-m < nn) then
178  do j = m+1,r
179  dummy = point(j)
180  do i = j-1,m,-1
181  if (point(i) <= dummy) exit
182  point(i+1) = point(i)
183  end do
184  point(i+1) = dummy
185  end do
186  if (jstack == 0) return
187  r = istack(jstack)
188  m = istack(jstack-1)
189  jstack = jstack-2
190  else
191  k = (m+r)/2
192  call swap(point(k),point(m+1))
193  if (point(m)>point(r)) call swap(point(m),point(r))
194  if (point(m+1)>point(r)) call swap(point(m+1),point(r))
195  if (point(m)>point(m+1)) call swap(point(m),point(m+1))
196  i = m+1
197  j = r
198  dummy = point(m+1)
199  do
200  do
201  i = i+1
202  if (point(i) >= dummy) exit
203  end do
204  do
205  j = j-1
206  if (point(j) <= dummy) exit
207  end do
208  if (j < i) exit
209  call swap(point(i),point(j))
210  end do
211  point(m+1) = point(j)
212  point(j) = dummy
213  jstack = jstack+2
214  if (jstack > nstack) then
215  ! LCOV_EXCL_START
216  err%occurred = .true.
217  err%msg = procedure_name//": NSTACK is too small."
218  return
219  ! LCOV_EXCL_STOP
220  end if
221  if (r-i+1 >= j-m) then
222  istack(jstack) = r
223  istack(jstack-1) = i
224  r = j-1
225  else
226  istack(jstack) = j-1
227  istack(jstack-1) = m
228  m = i
229  end if
230  end if
231  end do
232  end subroutine sortascending_rk
