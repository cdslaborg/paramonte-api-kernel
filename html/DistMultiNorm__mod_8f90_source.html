130 function construct(ndim, Mean, CovUpp, Std, Err) result(self)
131
132 use Constants_mod, only: LOG_INVERSE_SQRT_TWO_PI
133 use Matrix_mod, only: genInvMatFromChoLow
134 use Matrix_mod, only: getChoLow
135 use Matrix_mod, only: genEye
136 implicit none
137
138 integer(IK) , intent(in) :: ndim
139 real(RK) , intent(in) , optional :: Mean(ndim)
140 real(RK) , intent(in) , optional :: CovUpp(ndim,ndim)
141 real(RK) , intent(in) , optional :: Std(ndim)
142 type(Err_type) , intent(out) , optional :: Err
143 type(DistMultiNorm_type) :: self
144
145 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@construct()"
146
147 real(RK) :: inverseElongation
148 type(Err_type) :: ErrDef
149 integer(IK) :: i, j
150
151 self%ndim = ndim
152
153 blockErr: block
154
155 ErrDef%occurred = present(CovUpp) .and. present(Std)
156 if (ErrDef%occurred) then
157 ErrDef%msg = PROCEDURE_NAME//": The two input arguments `CovUpp` and `Std` cannot be present simultaneously."
158 exit blockErr
159 end if
160
161 ! Set mean vector.
162
163 if (present(Mean)) then
164 self%Mean = Mean
165 else
166 allocate(self%Mean(ndim), source = 0._RK)
167 end if
168
169 ! Set covariance matrix.
170
171 allocate(self%ChoLowCovUpp(ndim,ndim))
172 if (present(CovUpp)) then
173 allocate(self%Std(ndim))
174 do j = 1, ndim
175 self%Std(j) = sqrt(CovUpp(j,j))
176 do i = 1, j - 1
177 self%ChoLowCovUpp(i,j) = CovUpp(i,j)
178 end do
179 end do
180 else
181 if (present(Std)) then
182 self%Std = Std
183 do j = 1, ndim
184 do i = 1, j - 1
185 self%ChoLowCovUpp(i,j) = 0._RK
186 end do
187 self%ChoLowCovUpp(j,j) = Std(j)**2
188 end do
189 else
190 allocate(self%Std(ndim), source = 1._RK)
191 do j = 1, ndim
192 do i = 1, j - 1
193 self%ChoLowCovUpp(i,j) = 0._RK
194 end do
195 self%ChoLowCovUpp(j,j) = 1._RK
196 end do
197 end if
198 end if
199
200 ! Compute the Cholesky factorization of the covariance matrix.
201
202 allocate(self%ChoDia(ndim))
203 call getChoLow(ndim, self%ChoLowCovUpp, self%ChoDia)
204 ErrDef%occurred = self%ChoDia(1) < 0._RK
205 if (ErrDef%occurred) then
206 ErrDef%msg = PROCEDURE_NAME//": The Cholesky factorization of the covariance matrix of the MVN distribution failed."
207 exit blockErr
208 end if
209
210 ! Compute inverse covariance matrix and its determinant.
211
212 self%logDetInvCovMat = -2_IK * sum(log(self%ChoDia))
213 self%InvCovMat = genInvMatFromChoLow(ndim, self%ChoLowCovUpp, self%ChoDia)
214
215 ! Set up other MVN properties.
216
217 self%logNormFacPDF = getLogNormFacPDF(ndim, self%logDetInvCovMat)
218 self%entropy = getEntropy(ndim, self%logDetInvCovMat)
219
220 end block blockErr
221
222 if (present(Err)) then
223 Err = ErrDef
224 elseif (ErrDef%occurred) then
225 block
226 use iso_fortran_env, only: output_unit
227 write(output_unit,"(*(g0,:,' '))") ErrDef%msg
228 error stop
229 end block
230 end if
231
232 end function construct

500 ! This subroutine is legacy and slow. use [genRandMVN()](@ref getrandmvn) in this module.
501 ! Given the mean vector Mean and the covariance matrix CovMat, this subroutine generates a random vector x (of length nd>=2)
502 ! from an nd-dimensional multivariate normal distribution.
503 ! First a vector of nd standard normal random deviates is generated,
504 ! then this vector is multiplied by the Cholesky decomposition of the covariance matrix,
505 ! then the vector Mean is added to the resulting vector, and is stored in the output vector x.
506 ! ATTENTION: Only the upper half of the covariance matrix (plus the diagonal terms) need to be given in the input.
507 ! in the ouput, the upper half and diagonal part will still be the covariance matrix, while the lower half will be
508 ! the Cholesky decomposition elements (excluding its diagonal terms that are provided only in the vector ChoDia).
510 subroutine getMVNDev(nd,Mean,CovMat,X)
511#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
512 !DEC$ ATTRIBUTES DLLEXPORT :: getMVNDev
513#endif
514 use iso_fortran_env, only: output_unit
515 use Matrix_mod, only: getChoLow
516 use DistUniNorm_mod, only: genRand
517 implicit none
518 integer(IK), intent(in) :: nd
519 real(RK) , intent(in) :: Mean(nd), CovMat(nd,nd)
520 real(RK) , intent(out) :: X(nd)
521 real(RK) :: ChoLow(nd,nd), ChoDia(nd), DummyVec(nd)
522 integer(IK) :: i
523 ChoLow = CovMat
524 call getChoLow(nd,ChoLow,ChoDia)
525 if (ChoDia(1)<0._RK) then
526 ! LCOV_EXCL_START
527 write(output_unit,"(A)") "getChoLow() failed in getMVNDev()"
528 error stop
529 end if
530 ! LCOV_EXCL_STOP
531 do i=1,nd
532 DummyVec(i) = genRand()
533 x(i) = DummyVec(i) * ChoDia(i)
534 end do
535 do i=2,nd
536 x(i) = x(i) + dot_product(ChoLow(i,1:i-1),DummyVec(1:i-1))
537 end do
538 x = x + Mean
539 end subroutine getMVNDev
