120 SUBROUTINE ludcmp(a,n,np,indx,d)
121 INTEGER n,np,indx(n),NMAX
122 REAL*8 d,a(np,np),TINY
123 PARAMETER (NMAX=500,TINY=1.0e-20)
124 INTEGER i,imax,j,k
125 REAL*8 aamax,dum,sum,vv(NMAX)
126 d=1.
127 do 12 i=1,n
128 aamax=0.
129 do 11 j=1,n
130 if (abs(a(i,j)).gt.aamax) aamax=abs(a(i,j))
13111 continue
132 if (aamax.eq.0.) pause 'singular matrix in ludcmp'
133 vv(i)=1./aamax
13412 continue
135 do 19 j=1,n
136 do 14 i=1,j-1
137 sum=a(i,j)
138 do 13 k=1,i-1
139 sum=sum-a(i,k)*a(k,j)
14013 continue
141 a(i,j)=sum
14214 continue
143 aamax=0.
144 do 16 i=j,n
145 sum=a(i,j)
146 do 15 k=1,j-1
147 sum=sum-a(i,k)*a(k,j)
14815 continue
149 a(i,j)=sum
150 dum=vv(i)*abs(sum)
151 if (dum.ge.aamax) then
152 imax=i
153 aamax=dum
154 endif
15516 continue
156 if (j.ne.imax)then
157 do 17 k=1,n
158 dum=a(imax,k)
159 a(imax,k)=a(j,k)
160 a(j,k)=dum
16117 continue
162 d=-d
163 vv(imax)=vv(j)
164 endif
165 indx(j)=imax
166 if(a(j,j).eq.0.d0)a(j,j)=TINY
167 if(j.ne.n)then
168 dum=1./a(j,j)
169 do 18 i=j+1,n
170 a(i,j)=a(i,j)*dum
17118 continue
172 endif
17319 continue
174 return
175 end subroutine
