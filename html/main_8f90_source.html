256 SUBROUTINE ludcmp(a,n,np,indx,d)
257 INTEGER n,np,indx(n),NMAX
258 REAL*8 d,a(np,np),TINY
259 PARAMETER (NMAX=500,TINY=1.0e-20)
260 INTEGER i,imax,j,k
261 REAL*8 aamax,dum,sum,vv(NMAX)
262 d=1.
263 do 12 i=1,n
264 aamax=0.
265 do 11 j=1,n
266 if (abs(a(i,j)).gt.aamax) aamax=abs(a(i,j))
26711 continue
268 if (aamax.eq.0.) pause 'singular matrix in ludcmp'
269 vv(i)=1./aamax
27012 continue
271 do 19 j=1,n
272 do 14 i=1,j-1
273 sum=a(i,j)
274 do 13 k=1,i-1
275 sum=sum-a(i,k)*a(k,j)
27613 continue
277 a(i,j)=sum
27814 continue
279 aamax=0.
280 do 16 i=j,n
281 sum=a(i,j)
282 do 15 k=1,j-1
283 sum=sum-a(i,k)*a(k,j)
28415 continue
285 a(i,j)=sum
286 dum=vv(i)*abs(sum)
287 if (dum.ge.aamax) then
288 imax=i
289 aamax=dum
290 endif
29116 continue
292 if (j.ne.imax)then
293 do 17 k=1,n
294 dum=a(imax,k)
295 a(imax,k)=a(j,k)
296 a(j,k)=dum
29717 continue
298 d=-d
299 vv(imax)=vv(j)
300 endif
301 indx(j)=imax
302 if(a(j,j).eq.0.d0)a(j,j)=TINY
303 if(j.ne.n)then
304 dum=1./a(j,j)
305 do 18 i=j+1,n
306 a(i,j)=a(i,j)*dum
30718 continue
308 endif
30919 continue
310 return
311 END
