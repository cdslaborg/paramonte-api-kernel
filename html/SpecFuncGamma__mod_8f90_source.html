450 !
451 ! This algorithm borrows from the gser implementation of Numerical Recipes by Press et al. 1992.
452 pure function getGammaSeries(exponent,logGammaExponent,upperLim,tolerance) result(gammaSeries)
453#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
454 !DEC$ ATTRIBUTES DLLEXPORT :: getGammaSeries
455#endif
456 use Constants_mod, only: IK, RK, HUGE_RK
457 implicit none
458 real(RK), intent(in) :: exponent, logGammaExponent, upperLim
459 real(RK), intent(in), optional :: tolerance
460 real(RK) :: gammaSeries
461 integer(IK) , parameter :: ITMAX = 100_IK
462 real(RK) , parameter :: EPS_DEFAULT = epsilon(upperLim)
463 real(RK) :: ap,del,summ,eps
464 integer(IK) :: iter
465 if (present(tolerance)) then
466 eps = tolerance
467 else
468 eps = EPS_DEFAULT
469 end if
470 if (upperLim == 0._RK) then
471 gammaSeries = 0._RK
472 return
473 end if
474 ap = exponent
475 summ = 1.0_RK / exponent
476 del = summ
477 do iter = 1, ITMAX
478 ap = ap + 1.0_RK
479 del = del * upperLim / ap
480 summ = summ + del
481 if (abs(del) < abs(summ)*eps) exit
482 end do
483 if (iter > ITMAX) then
484 gammaSeries = -HUGE_RK
485 else
486 gammaSeries = summ * exp( exponent*log(upperLim) - upperLim - logGammaExponent )
487 end if
488 end function getGammaSeries

510 !
511 ! This algorithm borrows from the `gcf` implementation of Numerical Recipes by Press et al. 1992.
512 pure function getGammaContFrac(exponent,logGammaExponent,lowerLim,tolerance) result(gammaContFrac)
513#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
514 !DEC$ ATTRIBUTES DLLEXPORT :: getGammaContFrac
515#endif
516 use Constants_mod, only: IK, RK, HUGE_RK
517 implicit none
518 real(RK), intent(in) :: exponent, logGammaExponent, lowerLim
519 real(RK), optional, intent(in) :: tolerance
520 real(RK) :: gammaContFrac
521 integer(IK), parameter :: ITMAX = 100_IK
522 real(RK), parameter :: EPS_DEFAULT = epsilon(lowerLim)
523 real(RK), parameter :: FPMIN_DEFAULT = tiny(lowerLim) / EPS_DEFAULT
524 real(RK) :: eps, fpmin
525 real(RK) :: an,b,c,d,del,h
526 integer(IK) :: iter
527 if (lowerLim == 0._RK) then
528 gammaContFrac = 1._RK
529 return
530 end if
531 if (present(tolerance)) then
532 eps = tolerance
533 fpmin = tiny(lowerLim) / eps
534 else
535 eps = EPS_DEFAULT
536 fpmin = FPMIN_DEFAULT
537 end if
538 b = lowerLim + 1._RK - exponent
539 c = 1._RK / fpmin
540 d = 1._RK / b
541 h = d
542 do iter = 1, ITMAX
543 an = -iter * (iter-exponent)
544 b = b + 2.0_RK
545 d = an*d + b
546 if (abs(d) < fpmin) d = fpmin
547 c = b + an/c
548 if (abs(c) < fpmin) c = fpmin
549 d = 1.0_RK / d
550 del = d*c
551 h = h*del
552 if (abs(del-1._RK) <= eps) exit
553 end do
554 !if (iter > ITMAX) call nrerror('exponent too large, ITMAX too small in getGammaContFrac')
555 if (iter>ITMAX) then
556 gammaContFrac = -HUGE_RK
557 else
558 gammaContFrac = exp(exponent * log(lowerLim) - lowerLim - logGammaExponent) * h
559 end if
560 end function getGammaContFrac
