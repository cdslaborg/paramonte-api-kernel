309 !
310 ! This algorithm borrows from the gser implementation of Numerical Recipes by Press et al. 1992.
311 pure function getGammaSeries(exponent,logGammaExponent,upperLim,tolerance) result(gammaSeries)
312#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
313 !DEC$ ATTRIBUTES DLLEXPORT :: getGammaSeries
314#endif
315 use Constants_mod, only: IK, RK, HUGE_RK
316 implicit none
317 real(RK), intent(in) :: exponent, logGammaExponent, upperLim
318 real(RK), intent(in), optional :: tolerance
319 real(RK) :: gammaSeries
320 integer(IK) , parameter :: ITMAX = 100_IK
321 real(RK) , parameter :: EPS_DEFAULT = epsilon(upperLim)
322 real(RK) :: ap,del,summ,eps
323 integer(IK) :: iter
324 if (present(tolerance)) then
325 eps = tolerance
326 else
327 eps = EPS_DEFAULT
328 end if
329 if (upperLim == 0._RK) then
330 gammaSeries = 0._RK
331 return
332 end if
333 ap = exponent
334 summ = 1.0_RK / exponent
335 del = summ
336 do iter = 1, ITMAX
337 ap = ap + 1.0_RK
338 del = del * upperLim / ap
339 summ = summ + del
340 if (abs(del) < abs(summ)*eps) exit
341 end do
342 if (iter > ITMAX) then
343 gammaSeries = -HUGE_RK
344 else
345 gammaSeries = summ * exp( exponent*log(upperLim) - upperLim - logGammaExponent )
346 end if
347 end function getGammaSeries

369 !
370 ! This algorithm borrows from the `gcf` implementation of Numerical Recipes by Press et al. 1992.
371 pure function getGammaContFrac(exponent,logGammaExponent,lowerLim,tolerance) result(gammaContFrac)
372#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
373 !DEC$ ATTRIBUTES DLLEXPORT :: getGammaContFrac
374#endif
375 use Constants_mod, only: IK, RK, HUGE_RK
376 implicit none
377 real(RK), intent(in) :: exponent, logGammaExponent, lowerLim
378 real(RK), optional, intent(in) :: tolerance
379 real(RK) :: gammaContFrac
380 integer(IK), parameter :: ITMAX = 100_IK
381 real(RK), parameter :: EPS_DEFAULT = epsilon(lowerLim)
382 real(RK), parameter :: FPMIN_DEFAULT = tiny(lowerLim) / EPS_DEFAULT
383 real(RK) :: eps, fpmin
384 real(RK) :: an,b,c,d,del,h
385 integer(IK) :: iter
386 if (lowerLim == 0._RK) then
387 gammaContFrac = 1._RK
388 return
389 end if
390 if (present(tolerance)) then
391 eps = tolerance
392 fpmin = tiny(lowerLim) / eps
393 else
394 eps = EPS_DEFAULT
395 fpmin = FPMIN_DEFAULT
396 end if
397 b = lowerLim + 1._RK - exponent
398 c = 1._RK / fpmin
399 d = 1._RK / b
400 h = d
401 do iter = 1, ITMAX
402 an = -iter * (iter-exponent)
403 b = b + 2.0_RK
404 d = an*d + b
405 if (abs(d) < fpmin) d = fpmin
406 c = b + an/c
407 if (abs(c) < fpmin) c = fpmin
408 d = 1.0_RK / d
409 del = d*c
410 h = h*del
411 if (abs(del-1._RK) <= eps) exit
412 end do
413 !if (iter > ITMAX) call nrerror('exponent too large, ITMAX too small in getGammaContFrac')
414 if (iter>ITMAX) then
415 gammaContFrac = -HUGE_RK
416 else
417 gammaContFrac = exp(exponent * log(lowerLim) - lowerLim - logGammaExponent) * h
418 end if
419 end function getGammaContFrac
