131 subroutine queryImage(Image)
132#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)

140 ! setup general processor / coarray image variables
141
142#if CAF_ENABLED
143 Image%id = this_image()
144 Image%count = num_images()
145#elif MPI_ENABLED
146 block
147 use mpi
148 integer(IK) :: ierrMPI
149 logical :: isInitialized
150 call mpi_initialized( isInitialized, ierrMPI )

240 function constructForkJoin(processCount, lenProcessID, ProcessID, successProb, seqSecTime, parSecTime, comSecTime) result(ForkJoin) ! nonpure
241#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
242 !DEC$ ATTRIBUTES DLLEXPORT :: constructForkJoin
243#endif
244 use FitGeoCyclic_mod, only: fitGeoCyclicLogPDF ! LCOV_EXCL_LINE
245 use Constants_mod, only: IK, RK, SQRT_EPS_RK, NEGINF_RK
246 use Unique_mod, only: findUnique
247 use Sort_mod, only: indexArray
248 use String_mod, only: num2char
249
250 implicit none
251
252 integer(IK) , intent(in) :: processCount, lenProcessID, ProcessID(lenProcessID)
253 real(RK) , intent(in) :: successProb, seqSecTime, parSecTime, comSecTime
254 type(ForkJoin_type) :: ForkJoin
255
256 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME // "@constructForkJoin()"
257 real(RK) , parameter :: LOG_HALF = -0.693147180559945_RK
258
259 integer(IK) , allocatable :: Index(:)
260 integer(IK) :: i
261
262 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
263 ! check for invalid input
264 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
265
266 ForkJoin%Err%occurred = .false.
267
268 if (successProb<-SQRT_EPS_RK .or. successProb>1._RK+SQRT_EPS_RK) then
269 ForkJoin%Err%occurred = .true.
270 ForkJoin%Err%msg = PROCEDURE_NAME // ": successProb must be a number between zero and one. The input value is: " // num2char(successProb)
271 return
272 end if
273
274 if (processCount<1_IK) then
275 ForkJoin%Err%occurred = .true.
276 ForkJoin%Err%msg = PROCEDURE_NAME // ": processCount cannot be less than one. The input value is: " // num2char(processCount)
277 return
278 end if
279
280 if (processCount==1_IK .or. successProb <= SQRT_EPS_RK .or. successProb >= 1._RK+SQRT_EPS_RK) then
281 ForkJoin%Speedup%Maximum%value = 1._RK
282 ForkJoin%Speedup%Maximum%nproc = 1_IK
283 ForkJoin%Speedup%current = 1._RK
284 ForkJoin%Speedup%count = 1_IK
285 ForkJoin%Speedup%Scaling = [1._RK]
286 ForkJoin%Contribution%count = processCount
287 ForkJoin%Contribution%Identity = [( i, i = 1, processCount )]
288 allocate(ForkJoin%Contribution%Frequency(processCount), source = 0_IK)
289 ForkJoin%Contribution%Frequency(1) = lenProcessID
290 allocate(ForkJoin%Contribution%LogFrequency(processCount), source = NEGINF_RK)
291 ForkJoin%Contribution%LogFrequency(1) = log(real(ForkJoin%Contribution%Frequency(1),kind=RK))
292 ForkJoin%UniqueProcess%count = 1_IK
293 ForkJoin%UniqueProcess%Identity = [1_IK]
294 ForkJoin%UniqueProcess%Frequency = [lenProcessID]
295 ForkJoin%SuccessProb%current = successProb
296 ForkJoin%SuccessProb%effective = successProb
297 return
298 end if
299
300 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
301 ! compute the effective successProb (MCMC efficiency) from the Process contributions
302 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
303
304 call findUnique( lenVector = lenProcessID &
305 , Vector = ProcessID &
306 , lenUnique = ForkJoin%UniqueProcess%count &
307 , UniqueValue = ForkJoin%UniqueProcess%Identity &
308 , UniqueCount = ForkJoin%UniqueProcess%Frequency &
309 )
310 !maxContributorProcessID = ForkJoin%UniqueProcess%Identity(maxloc(ForkJoin%UniqueProcess%Frequency))
311
312 ! Sort ascending. This rearrangement may not be necessary anymore, since fitGeoCyclicLogPDF() is now order-agnostic.
313 ! From the IO report perspective, however, it is important to have it ordered.
314
315 allocate(Index(ForkJoin%UniqueProcess%count))
316 call indexArray(Array = ForkJoin%UniqueProcess%Identity(1:ForkJoin%UniqueProcess%count), Index = Index, failed = ForkJoin%Err%occurred)
317 if (ForkJoin%Err%occurred) then
318 ! LCOV_EXCL_START
319 ForkJoin%Err%msg = PROCEDURE_NAME//": Sorting of processor IDs failed."
320 return
321 ! LCOV_EXCL_STOP
322 end if
323
324 ! get all processes contributions
325
326 ForkJoin%UniqueProcess%Identity = ForkJoin%UniqueProcess%Identity(Index)
327 ForkJoin%UniqueProcess%Frequency = ForkJoin%UniqueProcess%Frequency(Index)
328
329 deallocate(Index)
330
331 ForkJoin%Contribution%count = processCount
332 if (allocated(ForkJoin%Contribution%LogFrequency)) deallocate(ForkJoin%Contribution%LogFrequency);
333 allocate(ForkJoin%Contribution%LogFrequency(ForkJoin%Contribution%count), source = 0._RK)
334 if (allocated(ForkJoin%Contribution%Frequency)) deallocate(ForkJoin%Contribution%Frequency);
335 allocate(ForkJoin%Contribution%Frequency(ForkJoin%Contribution%count), source = 0_IK)
336 if (allocated(ForkJoin%Contribution%Identity)) deallocate(ForkJoin%Contribution%Identity);
337 allocate(ForkJoin%Contribution%Identity(ForkJoin%Contribution%count))
338
339 ForkJoin%Contribution%Frequency(ForkJoin%UniqueProcess%Identity) = ForkJoin%UniqueProcess%Frequency
340 ForkJoin%Contribution%Identity = [(i, i = 1, ForkJoin%Contribution%count)]
341 if (ForkJoin%Contribution%Frequency(1)==0_IK) then
342 ! LCOV_EXCL_START
343 ForkJoin%Err%occurred = .true.
344 ForkJoin%Err%msg = PROCEDURE_NAME//": The contribution of the first process to the simulation is zero. This is highly unusual and requires further investigation."
345 return
346 ! LCOV_EXCL_STOP
347 end if
348 do i = 1, ForkJoin%Contribution%count
349 ! TODO: xxx a better solution instead of this ad-hoc approach would be to fit for the GeoCyclicCDF. Amir.
350 if (ForkJoin%Contribution%Frequency(i)/=0_IK) then
351 ForkJoin%Contribution%LogFrequency(i) = log(real(ForkJoin%Contribution%Frequency(i),kind=RK))
352 else
353 ForkJoin%Contribution%LogFrequency(i) = ForkJoin%Contribution%LogFrequency(i-1) + LOG_HALF
354 end if
355 end do
356
357 ForkJoin%SuccessProb%current = successProb
358
359 ! fit for the effective successProb
360
361 ForkJoin%SuccessProb%PowellMinimum = fitGeoCyclicLogPDF( maxNumTrial = processCount & ! LCOV_EXCL_LINE
362 , numTrial = ForkJoin%Contribution%count & ! LCOV_EXCL_LINE
363 , SuccessStep = ForkJoin%Contribution%Identity & ! LCOV_EXCL_LINE
364 , LogCount = ForkJoin%Contribution%LogFrequency & ! LCOV_EXCL_LINE
365 )
366 if (ForkJoin%SuccessProb%PowellMinimum%Err%occurred) then
367 ! LCOV_EXCL_START
368 ForkJoin%Err = ForkJoin%SuccessProb%PowellMinimum%Err
369 ForkJoin%Err%msg = PROCEDURE_NAME // ForkJoin%Err%msg
370 return
371 ! LCOV_EXCL_STOP
372 end if
373
374 ForkJoin%SuccessProb%effective = ForkJoin%SuccessProb%PowellMinimum%xmin(1)
375
376 ! fit for the effective successProb
377
378 call getForkJoinSpeedup( successProb = ForkJoin%SuccessProb%effective & ! max(mcmcSamplingEfficiency, ForkJoin%SuccessProb%PowellMinimum%xmin(1)) & ! avoid unstable estimates of the effective efficiency.
379 , seqSecTime = epsilon(seqSecTime) & ! time cost of the sequential section of the code, which is negligible here.
380 , parSecTime = parSecTime & ! the serial runtime for the parallel section of the code.
381 , comSecTimePerProc = comSecTime / (processCount - 1) & ! the communication overhead for each additional image beyond the leader.
382 , minMaxNumProc = 2 * processCount & ! speedup will be computed at least up to this process count, if not more.
383 , Speedup = ForkJoin%Speedup%Scaling & ! returned Speedup.
384 , lenSpeedup = ForkJoin%Speedup%count & ! length of the returned `Speedup%Scaling` vector.
385 , maxSpeedup = ForkJoin%Speedup%Maximum%value & ! returned maximum speedup.
386 , maxSpeedupNumProc = ForkJoin%Speedup%Maximum%nproc & ! returned number of processes for maximum speedup.
387 , Err = ForkJoin%Err &
388 )
389 if (ForkJoin%Err%occurred) then
390 ! LCOV_EXCL_START
391 ForkJoin%Err%msg = PROCEDURE_NAME // ForkJoin%Err%msg
392 return
393 ! LCOV_EXCL_STOP
394 end if
395 ForkJoin%Speedup%current = ForkJoin%Speedup%Scaling(processCount)
396
397 end function constructForkJoin

414 subroutine getForkJoinSpeedup(successProb, seqSecTime, parSecTime, comSecTimePerProc, minMaxNumProc, Speedup, lenSpeedup, maxSpeedupNumProc, maxSpeedup, Err)
415#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
416 !DEC$ ATTRIBUTES DLLEXPORT :: getForkJoinSpeedup
417#endif
418 use Statistics_mod, only: getLogProbGeoCyclic
419 use Constants_mod, only: IK, RK
420 use String_mod, only: num2char
421 use Misc_mod, only: resize
422
423 implicit none
424
425 real(RK) , intent(in) :: successProb, seqSecTime, parSecTime, comSecTimePerProc
426 integer(IK) , intent(in) :: minMaxNumProc ! must be at least 1
427 real(RK) , intent(out), allocatable :: Speedup(:)
428 integer(IK) , intent(out) :: lenSpeedup
429 real(RK) , intent(out) :: maxSpeedup
430 integer(IK) , intent(out) :: maxSpeedupNumProc
431 type(Err_type) , intent(out), optional :: Err
432
433 character(*) , parameter :: PROCEDURE_NAME = MODULE_NAME // "@constructForkJoin()"
434 integer(IK) , parameter :: ABS_MAX_NUM_PROC = 1000000_IK
435
436 real(RK) :: FirstImageContribution(1)
437 real(RK) :: serialRuntime, parSecTimePerProc, comSecTime
438 integer(IK) :: numProc, lenSpeedupNew !, maxNumProc
439 logical :: maxSpeedupFound, isPresentErr
440 integer(IK), parameter :: SuccessStep(*) = [1_IK]
441
442 isPresentErr = present(Err)
443 if (isPresentErr) Err%occurred = .false.
444
445 lenSpeedup = minMaxNumProc
446 allocate(Speedup(lenSpeedup))
447
448 numProc = 1
449 Speedup(numProc) = 1._RK
450 maxSpeedup = Speedup(numProc)
451 maxSpeedupNumProc = numProc
452 serialRuntime = seqSecTime + parSecTime ! serial runtime of the code per function call
453
454 maxSpeedupFound = .false.
455 loopOverProc: do
456
457 numProc = numProc + 1
458
459 ! resize allocation if needed
460
461 if (numProc>lenSpeedup) then
462 if (maxSpeedupFound) exit loopOverProc
463 lenSpeedupNew = 2 * lenSpeedup
464 call resize(Speedup, from = lenSpeedup, to = lenSpeedupNew)
465 lenSpeedup = lenSpeedupNew
466 end if
467
468 ! compute the fraction of work done by the first image
469
470 if (successProb==0._RK) then
471 FirstImageContribution(1) = 1._RK / numProc
472 else
473
474 FirstImageContribution = exp(getLogProbGeoCyclic( successProb = successProb &
475 , maxNumTrial = numProc &
476 , numTrial = 1_IK &
477 , SuccessStep = SuccessStep &
478 ) )
479 end if
480
481 ! effective runtime of the parallel-section of the code, when executed in parallel on numProc processes
482
483 parSecTimePerProc = parSecTime * FirstImageContribution(1)
484
485 ! communication time grows linearly with the number of processes.
486
487 comSecTime = (numProc-1_IK) * comSecTimePerProc
488
489 Speedup(numProc) = serialRuntime / (seqSecTime + parSecTimePerProc + comSecTime)
490 if (maxSpeedup < Speedup(numProc)) then
491 maxSpeedup = Speedup(numProc)
492 maxSpeedupNumProc = numProc
493 else
494 maxSpeedupFound = .true.
495 end if
496
497 if (numProc<ABS_MAX_NUM_PROC) cycle loopOverProc ! LCOV_EXCL_LINE
498
499 ! LCOV_EXCL_START
500 if (isPresentErr) then
501 Err%occurred = .true.
502 Err%msg = PROCEDURE_NAME // &
503 ": Failed to find the number of processes with which the maximum speedup occurs. The search continued up to " // &
504 num2char(ABS_MAX_NUM_PROC) // " processes."
505 end if
506 return
507 ! LCOV_EXCL_STOP
508
509 end do loopOverProc
510
511 end subroutine getForkJoinSpeedup
