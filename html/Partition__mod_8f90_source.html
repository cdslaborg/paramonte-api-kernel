497 function getBiasCorrectionScaleFactorSq(nd,npmin,npmax) result(BiasCorrectionScaleFactorSq)
498#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
499 !DEC$ ATTRIBUTES DLLEXPORT :: getBiasCorrectionScaleFactorSq
500#endif
501 use InterpLinear_mod, only: InterpLinear_type
502 use Constants_mod, only: IK, RK
503 implicit none
504
505 integer(IK) , intent(in) :: nd
506 integer(IK) , intent(in) :: npmin
507 integer(IK) , intent(in) :: npmax
508 real(RK) :: LogCount(npmin:npmax)
509 real(RK) :: BiasCorrectionScaleFactorSq(npmin:npmax)
510 integer(IK) :: ip, ndim, npminDef, icase
511 type(InterpLinear_type) :: Interp
512
513 type :: Case_type
514 real(RK), allocatable :: LogNumPointInSpheroid(:)
515 real(RK), allocatable :: LogMeanLogScaleFactor(:)
516 end type Case_type
517#include "Partition_mod.inc.scaleFactor.inc.f90"
518
519 ndim = nd
520 if (npmin < nd + 1_IK) error stop
521 if (ndim < lbound(Case, dim = 1)) error stop
522 if (ndim > ubound(Case, dim = 1)) ndim = ubound(Case, dim = 1)
523
524 LogCount(npmin:npmax) = [( log(real(ip,RK)), ip = npmin, npmax )]
525 Interp = InterpLinear_type(SortedX = Case(ndim)%LogNumPointInSpheroid, SortedY = Case(ndim)%LogMeanLogScaleFactor)
526 BiasCorrectionScaleFactorSq(npmin:npmax) = exp(2_IK*exp(Interp%predict(QueryX = LogCount)))
527
528#if DEBUG_ENABLED || TESTING_ENABLED || CODECOV_ENABLED
529 ! WARNING: scaleFactorSq must be larger than 1 in all circumstances. This is crucial later on.
530 if (any(BiasCorrectionScaleFactorSq < 1._RK)) error stop
531#else
532 where (BiasCorrectionScaleFactorSq < 1._RK); BiasCorrectionScaleFactorSq = 1._RK; end where
533#endif
534
535 end function getBiasCorrectionScaleFactorSq

539 ! ATTN Developer: Do NOT lower the case of `PURE` below. This is a macro.
540 PURE function genLogLikeFitness(count, logVolRatio) result(logLikeFitness)
541#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
542 !DEC$ ATTRIBUTES DLLEXPORT :: genLogLikeFitness
543#endif
544 use Constants_mod, only: IK, RK, NEGINF_RK
545 implicit none
546 integer(IK) , intent(in) :: count
547 real(RK) , intent(in) :: logVolRatio
548 real(RK) :: logLikeFitness
549 logLikeFitness = count * (1._RK + logVolRatio - exp(logVolRatio))
550#if DEBUG_ENABLED || TESTING_ENABLED || CODECOV_ENABLED
551 if (logLikeFitness < NEGINF_RK .or. logLikeFitness > 0._RK) then
552 write(*,*) .or."Internal error occurred: logLikeFitness < NEGINF_RK logLikeFitness > 0._RK"
553 write(*,*) "count, logVolRatio, logLikeFitness:", count, logVolRatio, logLikeFitness
554 error stop
555 end if
556#endif
557 end function genLogLikeFitness
