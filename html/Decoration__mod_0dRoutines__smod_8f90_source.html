524  padlength = iend
525  else
526  ispadzone = .false.
527  end if
528  if (string(istart:iend)==split) then
529  isendofsplitloc(iend) = 1_ik
530  else
531  isendofsplitloc(iend) = 0_ik
532  end if
533  istart = istart + 1_ik
534  iend = iend + 1_ik
535  end do blockfindsplit
536

538 
539  numsplitendloc = sum(isendofsplitloc)
540  if (numsplitendloc==0_ik) then
541  allocate( listoflines(1) )
542  listoflines(1)%record = string
543  return
544  else
545  ! xxx: here goes another GFortran 7.3 bug: EndOfSplitLoc is assumed already allocated, despite the first appearance here.
546  if (allocated(endofsplitloc)) deallocate(endofsplitloc)
