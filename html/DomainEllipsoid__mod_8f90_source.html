135 ! Developer note: The use of contiguous arrays as input arguments here is not encouraged
136 ! as the input argument `ndim` is necessary for when an identity domain must be constructed.
137 function construct(ndim, Center, RepUpp, Err) result(self)
138#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
139 !DEC$ ATTRIBUTES DLLEXPORT :: construct
140#endif
141 use Matrix_mod, only: getChoLow, genInvMatFromChoLow, genEye
142 use GeometryEllipsoid_mod, only: genLogVolUnitBall
143 use Err_mod, only: Err_type
144 integer(IK) , intent(in) :: ndim
145 real(RK) , intent(in) , optional :: Center(ndim)
146 real(RK) , intent(in) , optional :: RepUpp(ndim,ndim)
147 type(Err_type) , intent(out) , optional :: Err
148 type(DomainEllipsoid_type) :: self
149 integer(IK) :: i, j
150 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@construct()"
151 if (present(Err)) Err%occurred = .false.
152 self%ndim = ndim
153 if (present(Center)) then
154 self%Center = Center
155 else
156 allocate(self%Center(ndim), source = 0._RK)
157 end if
158 if (present(RepUpp)) then
159 self%ChoLowRepUpp = RepUpp
160 self%isSphere = all([((RepUpp(i,j), i = 1, j-1), j = 1, self%ndim)] < 1.e-8_RK) .and. all([( self%ChoLowRepUpp(i,i), i = 1, self%ndim )] == self%ChoLowRepUpp(1,1))
161 else
162 self%isSphere = .true.
163 self%ChoLowRepUpp = genEye(ndim,ndim,1._RK)
164 end if
165 if (self%isSphere) then
166 self%radiusSq = self%ChoLowRepUpp(1,1)
167 self%InvRepMat = genEye(self%ndim,self%ndim,1._RK/self%ChoLowRepUpp(1,1))
168 self%logVol = 0.5_RK * self%ndim * log(self%radiusSq) + genLogVolUnitBall(self%ndim)
169 allocate(self%ChoDia(self%ndim), source = sqrt(self%ChoLowRepUpp(1,1)))
170 else
171 self%radiusSq = -huge(1._RK)
172 allocate(self%ChoDia(self%ndim))
173 call getChoLow(self%ndim, self%ChoLowRepUpp, self%ChoDia)
174 if (self%ChoDia(1) < 0._RK) then
175 if (present(Err)) then
176 Err%occurred = .true.
177 Err%msg = PROCEDURE_NAME//": Cholesky factorization failed."
178 return
179 else
180 write(*,"(A)") PROCEDURE_NAME//": Cholesky factorization failed."
181 error stop
182 end if
183 end if
184 self%InvRepMat = genInvMatFromChoLow(self%ndim, self%ChoLowRepUpp, self%ChoDia)
185 self%logVol = sum(log(self%ChoDia)) + genLogVolUnitBall(self%ndim)
186 end if
187 end function construct

205 pure function isIdenticalDomain(Domain1, Domain2) result(equal)
206#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
207 !DEC$ ATTRIBUTES DLLEXPORT :: isIdenticalDomain
208#endif
209 implicit none
210 type(DomainEllipsoid_type) , intent(in) :: Domain1, Domain2
211 logical :: equal
212 equal = Domain1%ndim == Domain2%ndim .and. Domain1%logVol == Domain2%logVol
213 if (.not. equal) return
214 equal = equal .and. all(Domain1%Center == Domain2%Center)
215 if (.not. equal) return
216 equal = equal .and. all(Domain1%ChoDia == Domain2%ChoDia)
217 if (.not. equal) return
218 equal = equal .and. all(Domain1%ChoLowRepUpp == Domain2%ChoLowRepUpp)
219 if (.not. equal) return
220 equal = equal .and. all(Domain1%InvRepMat == Domain2%InvRepMat)
221 if (.not. equal) return
222 equal = equal .and. Domain1%radiusSq == Domain2%radiusSq
223 if (.not. equal) return
224 equal = equal .and. (Domain1%isSphere .eqv. Domain2%isSphere)
225 if (.not. equal) return
226 end function isIdenticalDomain

270 pure function contains(self, Point) result(isContained)
271#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
272 !DEC$ ATTRIBUTES DLLEXPORT :: contains
273#endif
274 class(DomainEllipsoid_type) , intent(in) :: self
275 real(RK) , intent(in) :: Point(self%ndim)
276 real(RK) :: PointNormed(self%ndim)
277 logical :: isContained
278 real(RK) :: sumTermsSq
279 integer(IK) :: i
280 if (self%isSphere) then
281 sumTermsSq = 0._RK
282 loopContainmentCheck: do i = 1, self%ndim
283 sumTermsSq = sumTermsSq + (Point(i) - self%Center(i))**2
284 isContained = sumTermsSq <= self%radiusSq
285 if (isContained) cycle loopContainmentCheck
286 return
287 end do loopContainmentCheck
288 else
289 PointNormed(1:self%ndim) = Point - self%Center
290 isContained = dot_product(PointNormed , matmul(self%InvRepMat, PointNormed)) <= 1._RK
291 end if
292 end function contains

341 function genRandUnitVolumeMVU(nd,logVolUnitBall) result(RandStanVolumeMVU)
342#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
343 !DEC$ ATTRIBUTES DLLEXPORT :: genRandUnitVolumeMVU
344#endif
345 use DistUniNorm_mod, only: genRand
346 implicit none
347 integer(IK), intent(in) :: nd
348 real(RK) , intent(in) :: logVolUnitBall
349 real(RK) :: RandStanVolumeMVU(nd), dummy, sumSqRandStanVolumeMVU
350 integer(IK) :: i,j
351 sumSqRandStanVolumeMVU = 0._RK
352 do j=1,nd
353 RandStanVolumeMVU(j) = genRand()
354 sumSqRandStanVolumeMVU = sumSqRandStanVolumeMVU + RandStanVolumeMVU(j)**2
355 end do
356 call random_number(dummy)
357 if (dummy > 0._RK) then
358 dummy = exp((log(dummy) - logVolUnitBall) / nd) / sqrt(sumSqRandStanVolumeMVU)
359 RandStanVolumeMVU = RandStanVolumeMVU * dummy ! a uniform random point from inside of nd-sphere
360 else
361 RandStanVolumeMVU = 0._RK
362 end if
363 end function genRandUnitVolumeMVU

475 function genRandMVU(nd,Mean,ChoLow,ChoDia) result(RandMVU)
476#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
477 !DEC$ ATTRIBUTES DLLEXPORT :: genRandMVU
478#endif
479 use DistUniNorm_mod, only: genRand
480 implicit none
481 integer(IK), intent(in) :: nd
482 real(RK) , intent(in) :: Mean(nd)
483 real(RK) , intent(in) :: ChoLow(nd,nd) ! Cholesky lower triangle, calculated from `RepMat`.
484 real(RK) , intent(in) :: ChoDia(nd) ! Cholesky diagonal terms, calculated from `RepMat`.
485 real(RK) :: RandMVU(nd), dummy, DummyVec(nd), sumSqDummyVec
486 integer(IK) :: i,j
487 sumSqDummyVec = 0._RK
488 do j=1,nd
489 DummyVec(j) = genRand()
490 sumSqDummyVec = sumSqDummyVec + DummyVec(j)**2
491 end do
492 call random_number(dummy)
493 dummy = dummy**(1._RK/nd) / sqrt(sumSqDummyVec)
494 DummyVec = DummyVec * dummy ! a uniform random point from inside of nd-sphere
495 RandMVU = Mean
496 do j = 1, nd
497 RandMVU(j) = RandMVU(j) + ChoDia(j) * DummyVec(j)
498 do i = j + 1, nd
499 RandMVU(i) = RandMVU(i) + ChoLow(i,j) * DummyVec(j)
500 end do
501 end do
502 end function genRandMVU

542!%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
543
544 ! This subroutine is legacy and slow. Use [genRandMVU()](@ref genRandMVU) in this same module.
545 ! Given the mean vector Mean and the representative matrix RepMat, this subroutine generates a random vector X (of length nd>=2)
546 ! from an nd-dimensional multivariate ellipsoidal uniform domain, such that getMVUDev() is randomly distributed inside the nd-dimensional ellipsoid.
547 ! ATTENTION: Only the upper half of the representative matrix (plus the diagonal terms) need to be given in the input.
548 ! in the ouput, the upper half and diagonal part will still be the representative matrix, while the lower half will be
549 ! the Cholesky decomposition elements (excluding its diagonal terms that are provided only in the vector ChoDia).
551 subroutine getMVUDev(nd,Mean,RepMat,X)
552#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
553 !DEC$ ATTRIBUTES DLLEXPORT :: getMVUDev
554#endif
555 use Matrix_mod, only: getChoLow
556 use DistUniNorm_mod, only: genRand
557 implicit none
558 integer(IK), intent(in) :: nd
559 real(RK) , intent(in) :: Mean(nd)
560 real(RK) , intent(in) :: RepMat(nd,nd)
561 real(RK) , intent(out) :: X(nd)
562 real(RK) :: ChoDia(nd), DummyVec(nd), ChoLow(nd,nd), dummy
563 integer(IK) :: i
564
565 ChoLow = RepMat
566 call getChoLow(nd,ChoLow,ChoDia)
567 if (ChoDia(1)<0._RK) then
568 ! LCOV_EXCL_START
569 error stop
570 !call abortProgram( output_unit , 1 , 1 , 'Statitistics@getMVUDev()@getChoLow() failed.' )
571 end if
572 ! LCOV_EXCL_STOP
573 do i=1,nd
574 DummyVec(i) = genRand()
575 end do
576 call random_number(dummy)
577 dummy = (dummy**(1._RK/real(nd,kind=RK)))/norm2(DummyVec) ! Now DummyVec is a uniformly drawn point from inside of nd-D sphere.
578 DummyVec = dummy*DummyVec
579
580 ! Now transform this point to a point inside the ellipsoid.
581 do i=1,nd
582 X(i) = DummyVec(i)*ChoDia(i)
583 end do
584
585 do i=2,nd
586 X(i) = X(i) + dot_product(ChoLow(i,1:i-1),DummyVec(1:i-1))
587 end do
588
589 X = X + Mean
590
591 end subroutine getMVUDev
