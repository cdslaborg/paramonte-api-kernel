124 ! Developer note: The use of contiguous arrays as input arguments here is not encouraged
125 ! as the input argument `ndim` is necessary for when an identity domain must be constructed.
126 function construct(ndim, Center, CovUpp, Err) result(self)
127#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
128 !DEC$ ATTRIBUTES DLLEXPORT :: construct
129#endif
130 use Matrix_mod, only: getChoLow, getInvMatFromChoLow, genEye
131 use GeometryEllipsoid_mod, only: genLogVolUnitBall
132 use Err_mod, only: Err_type
133 integer(IK) , intent(in) :: ndim
134 real(RK) , intent(in) , optional :: Center(ndim)
135 real(RK) , intent(in) , optional :: CovUpp(ndim,ndim)
136 type(Err_type) , intent(out) , optional :: Err
137 type(DomainEllipsoid_type) :: self
138 integer(IK) :: i, j
139 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@construct()"
140 if (present(Err)) Err%occurred = .false.
141 self%ndim = ndim
142 if (present(Center)) then
143 self%Center = Center
144 else
145 allocate(self%Center(ndim), source = 0._RK)
146 end if
147 if (present(CovUpp)) then
148 self%ChoLowCovUpp = CovUpp
149 self%isSphere = all([((CovUpp(i,j), i = 1, j-1), j = 1, self%ndim)] < 1.e-8_RK) .and. all([( self%ChoLowCovUpp(i,i), i = 1, self%ndim )] == self%ChoLowCovUpp(1,1))
150 else
151 self%isSphere = .true.
152 self%ChoLowCovUpp = genEye(ndim,ndim,1._RK)
153 end if
154 if (self%isSphere) then
155 self%radiusSq = self%ChoLowCovUpp(1,1)
156 self%InvCovMat = genEye(self%ndim,self%ndim,1._RK/self%ChoLowCovUpp(1,1))
157 self%logVol = 0.5_RK * self%ndim * log(self%radiusSq) + genLogVolUnitBall(self%ndim)
158 allocate(self%ChoDia(self%ndim), source = sqrt(self%ChoLowCovUpp(1,1)))
159 else
160 self%radiusSq = -huge(1._RK)
161 allocate(self%ChoDia(self%ndim))
162 call getChoLow(self%ndim, self%ChoLowCovUpp, self%ChoDia)
163 if (self%ChoDia(1) < 0._RK) then
164 if (present(Err)) then
165 Err%occurred = .true.
166 Err%msg = PROCEDURE_NAME//": Cholesky factorization failed."
167 return
168 else
169 write(*,"(A)") PROCEDURE_NAME//": Cholesky factorization failed."
170 error stop
171 end if
172 end if
173 self%InvCovMat = getInvMatFromChoLow(self%ndim, self%ChoLowCovUpp, self%ChoDia)
174 self%logVol = sum(log(self%ChoDia)) + genLogVolUnitBall(self%ndim)
175 end if
176 end function construct

180 pure function contains(self, Point) result(isContained)
181#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
182 !DEC$ ATTRIBUTES DLLEXPORT :: contains
183#endif
184 class(DomainEllipsoid_type) , intent(in) :: self
185 real(RK) , intent(in) :: Point(self%ndim)
186 real(RK) :: PointNormed(self%ndim)
187 logical :: isContained
188 real(RK) :: sumTermsSq
189 integer(IK) :: i
190 if (self%isSphere) then
191 sumTermsSq = 0._RK
192 loopContainmentCheck: do i = 1, self%ndim
193 sumTermsSq = sumTermsSq + (Point(i) - self%Center(i))**2
194 isContained = sumTermsSq <= self%radiusSq
195 if (isContained) cycle loopContainmentCheck
196 return
197 end do loopContainmentCheck
198 else
199 PointNormed(1:self%ndim) = Point - self%Center
200 isContained = dot_product(PointNormed , matmul(self%InvCovMat, PointNormed)) <= 1._RK
201 end if
202 end function contains

365 function getRandMVU(nd,Mean,ChoLow,ChoDia) result(RandMVU)
366#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
367 !DEC$ ATTRIBUTES DLLEXPORT :: getRandMVU
368#endif
369 use DistUniNorm_mod, only: getRand
370 implicit none
371 integer(IK), intent(in) :: nd
372 real(RK) , intent(in) :: Mean(nd)
373 real(RK) , intent(in) :: ChoLow(nd,nd) ! Cholesky lower triangle, calculated from the input CovMat.
374 real(RK) , intent(in) :: ChoDia(nd) ! Cholesky diagonal terms, calculated from the input CovMat.
375 real(RK) :: RandMVU(nd), dummy, DummyVec(nd), sumSqDummyVec
376 integer(IK) :: i,j
377 sumSqDummyVec = 0._RK
378 do j=1,nd
379 DummyVec(j) = getRand()
380 sumSqDummyVec = sumSqDummyVec + DummyVec(j)**2
381 end do
382 call random_number(dummy)
383 dummy = dummy**(1._RK/nd) / sqrt(sumSqDummyVec)
384 DummyVec = DummyVec * dummy ! a uniform random point from inside of nd-sphere
385 RandMVU = Mean
386 do j = 1, nd
387 RandMVU(j) = RandMVU(j) + ChoDia(j) * DummyVec(j)
388 do i = j + 1, nd
389 RandMVU(i) = RandMVU(i) + ChoLow(i,j) * DummyVec(j)
390 end do
391 end do
392 end function getRandMVU

434 ! This subroutine is legacy and slow. Use [getRandMVU()](@ref getRandMVU) in this same module.
435 ! Given the mean vector Mean and the covariance matrix CovMat, this subroutine generates a random vector X (of length nd>=2)
436 ! from an nd-dimensional multivariate ellipsoidal uniform domain, such that getMVUDev() is randomly distributed inside the nd-dimensional ellipsoid.
437 ! ATTENTION: Only the upper half of the covariance matrix (plus the diagonal terms) need to be given in the input.
438 ! in the ouput, the upper half and diagonal part will still be the covariance matrix, while the lower half will be
439 ! the Cholesky decomposition elements (excluding its diagonal terms that are provided only in the vector ChoDia).
441 subroutine getMVUDev(nd,Mean,CovMat,X)
442#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
443 !DEC$ ATTRIBUTES DLLEXPORT :: getMVUDev
444#endif
445 use Matrix_mod, only: getChoLow
446 use DistUniNorm_mod, only: getRand
447 implicit none
448 integer(IK), intent(in) :: nd
449 real(RK) , intent(in) :: Mean(nd)
450 real(RK) , intent(in) :: CovMat(nd,nd)
451 real(RK) , intent(out) :: X(nd)
452 real(RK) :: ChoDia(nd), DummyVec(nd), ChoLow(nd,nd), dummy
453 integer(IK) :: i
454
455 ChoLow = CovMat
456 call getChoLow(nd,ChoLow,ChoDia)
457 if (ChoDia(1)<0._RK) then
458 ! LCOV_EXCL_START
459 error stop
460 !call abortProgram( output_unit , 1 , 1 , 'Statitistics@getMVUDev()@getChoLow() failed.' )
461 end if
462 ! LCOV_EXCL_STOP
463 do i=1,nd
464 DummyVec(i) = getRand()
465 end do
466 call random_number(dummy)
467 dummy = (dummy**(1._RK/real(nd,kind=RK)))/norm2(DummyVec) ! Now DummyVec is a uniformly drawn point from inside of nd-D sphere.
468 DummyVec = dummy*DummyVec
469
470 ! Now transform this point to a point inside the ellipsoid.
471 do i=1,nd
472 X(i) = DummyVec(i)*ChoDia(i)
473 end do
474
475 do i=2,nd
476 X(i) = X(i) + dot_product(ChoLow(i,1:i-1),DummyVec(1:i-1))
477 end do
478
479 X = X + Mean
480
481 end subroutine getMVUDev
