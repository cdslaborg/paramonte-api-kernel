380 !
381 ! For more information, see: https://misc.flogisoft.com/bash/tip_colors_and_formatting
382 function style(string, attr, clfg, clbg) result(modifiedString)
383#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
384 !DEC$ ATTRIBUTES DLLEXPORT :: style
385#endif
386 use Constants_mod, only: ESC
387 implicit none
388 character(*), intent(in) :: string
389 character(*), intent(in), optional :: attr, clfg, clbg
390 character(:), allocatable :: modifiedString
391 integer(IK) :: i
392
393 ! cache the attribute / color list codes
394
395 if (.not.allocated(mc_AnsiAttributeList)) then
396 allocate(mc_AnsiAttributeList(ANSI_ATTRIBUTE_LIST_LEN))
397 mc_AnsiAttributeList(1)%value = "off" ; mc_AnsiAttributeList(1)%code = "0"
398 mc_AnsiAttributeList(2)%value = "bold" ; mc_AnsiAttributeList(2)%code = "1"
399 mc_AnsiAttributeList(3)%value = "bright" ; mc_AnsiAttributeList(3)%code = "1"
400 mc_AnsiAttributeList(4)%value = "dim" ; mc_AnsiAttributeList(4)%code = "2"
401 mc_AnsiAttributeList(5)%value = "underlined" ; mc_AnsiAttributeList(5)%code = "4"
402 mc_AnsiAttributeList(6)%value = "blinking" ; mc_AnsiAttributeList(6)%code = "5"
403 mc_AnsiAttributeList(7)%value = "reverse" ; mc_AnsiAttributeList(7)%code = "7"
404 mc_AnsiAttributeList(8)%value = "hidden" ; mc_AnsiAttributeList(8)%code = "8"
405 end if
406
407 if (.not.allocated(mc_AnsiForegroundColorList)) then
408 allocate(mc_AnsiForegroundColorList(ANSI_COLOR_LIST_LEN))
409 mc_AnsiForegroundColorList(1) %value = "black" ; mc_AnsiForegroundColorList(1) %code = "30"
410 mc_AnsiForegroundColorList(2) %value = "red" ; mc_AnsiForegroundColorList(2) %code = "31"
411 mc_AnsiForegroundColorList(3) %value = "green" ; mc_AnsiForegroundColorList(3) %code = "32"
412 mc_AnsiForegroundColorList(4) %value = "yellow" ; mc_AnsiForegroundColorList(4) %code = "33"
413 mc_AnsiForegroundColorList(5) %value = "blue" ; mc_AnsiForegroundColorList(5) %code = "34"
414 mc_AnsiForegroundColorList(6) %value = "magenta" ; mc_AnsiForegroundColorList(6) %code = "35"
415 mc_AnsiForegroundColorList(7) %value = "cyan" ; mc_AnsiForegroundColorList(7) %code = "36"
416 mc_AnsiForegroundColorList(8) %value = "light gray" ; mc_AnsiForegroundColorList(8) %code = "37"
417 mc_AnsiForegroundColorList(9) %value = "dark gray" ; mc_AnsiForegroundColorList(9) %code = "90"
418 mc_AnsiForegroundColorList(10)%value = "light red" ; mc_AnsiForegroundColorList(10)%code = "91"
419 mc_AnsiForegroundColorList(11)%value = "light green" ; mc_AnsiForegroundColorList(11)%code = "92"
420 mc_AnsiForegroundColorList(12)%value = "light yellow" ; mc_AnsiForegroundColorList(12)%code = "93"
421 mc_AnsiForegroundColorList(13)%value = "light blue" ; mc_AnsiForegroundColorList(13)%code = "94"
422 mc_AnsiForegroundColorList(14)%value = "light magenta" ; mc_AnsiForegroundColorList(14)%code = "95"
423 mc_AnsiForegroundColorList(15)%value = "light cyan" ; mc_AnsiForegroundColorList(15)%code = "96"
424 mc_AnsiForegroundColorList(16)%value = "white" ; mc_AnsiForegroundColorList(16)%code = "97"
425 end if
426
427 if (.not.allocated(mc_AnsiBackgroundColorList)) then
428 allocate(mc_AnsiBackgroundColorList(ANSI_COLOR_LIST_LEN))
429 mc_AnsiBackgroundColorList(1) %value = "black" ; mc_AnsiBackgroundColorList(1) %code = "40"
430 mc_AnsiBackgroundColorList(2) %value = "red" ; mc_AnsiBackgroundColorList(2) %code = "41"
431 mc_AnsiBackgroundColorList(3) %value = "green" ; mc_AnsiBackgroundColorList(3) %code = "42"
432 mc_AnsiBackgroundColorList(4) %value = "yellow" ; mc_AnsiBackgroundColorList(4) %code = "43"
433 mc_AnsiBackgroundColorList(5) %value = "blue" ; mc_AnsiBackgroundColorList(5) %code = "44"
434 mc_AnsiBackgroundColorList(6) %value = "magenta" ; mc_AnsiBackgroundColorList(6) %code = "45"
435 mc_AnsiBackgroundColorList(7) %value = "cyan" ; mc_AnsiBackgroundColorList(7) %code = "46"
436 mc_AnsiBackgroundColorList(8) %value = "light gray" ; mc_AnsiBackgroundColorList(8) %code = "47"
437 mc_AnsiBackgroundColorList(9) %value = "dark gray" ; mc_AnsiBackgroundColorList(9) %code = "100"
438 mc_AnsiBackgroundColorList(10)%value = "light red" ; mc_AnsiBackgroundColorList(10)%code = "101"
439 mc_AnsiBackgroundColorList(11)%value = "light green" ; mc_AnsiBackgroundColorList(11)%code = "102"
440 mc_AnsiBackgroundColorList(12)%value = "light yellow" ; mc_AnsiBackgroundColorList(12)%code = "103"
441 mc_AnsiBackgroundColorList(13)%value = "light blue" ; mc_AnsiBackgroundColorList(13)%code = "104"
442 mc_AnsiBackgroundColorList(14)%value = "light magenta" ; mc_AnsiBackgroundColorList(14)%code = "105"
443 mc_AnsiBackgroundColorList(15)%value = "light cyan" ; mc_AnsiBackgroundColorList(15)%code = "106"
444 mc_AnsiBackgroundColorList(16)%value = "white" ; mc_AnsiBackgroundColorList(16)%code = "107"
445 end if
446
447 ! construct the escape sequence
448
449 modifiedString = ESC//"[0"
450
451 if (present(attr)) then
452 do i = 1, ANSI_ATTRIBUTE_LIST_LEN
453 if (attr==mc_AnsiAttributeList(i)%value) then
454 modifiedString = modifiedString // ";" // mc_AnsiAttributeList(i)%code
455 exit
456 end if
457 end do
458 end if
459
460 if (present(clfg)) then
461 do i = 1, ANSI_COLOR_LIST_LEN
462 if (clfg==mc_AnsiForegroundColorList(i)%value) then
463 modifiedString = modifiedString // ";" // mc_AnsiForegroundColorList(i)%code
464 exit
465 end if
466 end do
467 end if
468
469 if (present(clbg)) then
470 do i = 1, ANSI_COLOR_LIST_LEN
471 if (clbg==mc_AnsiBackgroundColorList(i)%value) then
472 modifiedString = modifiedString // ";" // mc_AnsiBackgroundColorList(i)%code
473 exit
474 end if
475 end do
476 end if
477
478 modifiedString = modifiedString // "m" // string // ESC // "[0m"
479
480 end function style

494 pure function getGenericFormat(width,precision,delim,prefix) result(formatStr)
495#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
496 !DEC$ ATTRIBUTES DLLEXPORT :: getGenericFormat
497#endif
498 ! generates IO format strings, primarily for use in the output report files of ParaMonte
499 use Constants_mod, only: IK
500 use String_mod, only: num2char
501 implicit none
502 integer(IK) , intent(in), optional :: width
503 integer(IK) , intent(in), optional :: precision
504 character(*), intent(in), optional :: delim
505 character(*), intent(in), optional :: prefix
506 character(:), allocatable :: widthStr
507 character(:), allocatable :: formatStr
508 character(:), allocatable :: precisionStr
509 character(:), allocatable :: delimDefault
510
511 widthStr = "0"; if (present(width)) widthStr = num2char(width)
512 precisionStr = ""; if (present(precision)) precisionStr = "."//num2char(precision)
513 delimDefault = ""; if (present(delim)) delimDefault = ",:,'"//delim//"'"
514 formatStr = "*(g"//widthStr//precisionStr//delimDefault//"))"
515 if (present(prefix)) then
516 formatStr = "('" // prefix // "'," // formatStr
517 else
518 formatStr = "(" // formatStr
519 end if
520
521 end function getGenericFormat
