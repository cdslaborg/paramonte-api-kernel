207 module subroutine getSpecFromInputFile(self,nd)
208#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
209 !DEC$ ATTRIBUTES DLLEXPORT :: getSpecFromInputFile
210#endif
211 use Constants_mod, only: IK, RK, NLC
212 use String_mod, only: val2char
213
214 implicit none
215 class(ParaXXXX_type), intent(inout) :: self
216 integer(IK), intent(in) :: nd
217 character(*), parameter :: PROCEDURE_NAME = SUBMODULE_NAME//"@getSpecFromInputFile()"
218
219 ! initialize/nullify all general input options
220
221 call self%SpecBase%nullifyNameListVar(nd)
222#if defined PARADRAM || defined PARADISE
223 call self%SpecMCMC%nullifyNameListVar(nd)
224 call self%SpecDRAM%nullifyNameListVar(nd)
225#elif defined PARANEST
226 call self%SpecNest%nullifyNameListVar(nd)
227#endif
228
229 ! read input options if input file is provided
230
231 blockReadInputFile: if (self%inputFileArgIsPresent) then
232
233 blockInputFileType: if (self%InputFile%isInternal) then
234
235 ! read input file as an internal file
236
237 read(self%InputFile%Path%original,nml=ParaXXXX,iostat=self%InputFile%Err%stat)
238 self%Err = self%InputFile%getReadErr(self%InputFile%Err%stat,self%InputFile%Path%modified)
239
240 if (self%Err%occurred) then
241
242 if ( is_iostat_end(self%Err%stat) .or. is_iostat_eor(self%Err%stat) ) then
243
244 ! search for the paramonte namelist group in the file.
245
246 call self%warnUserAboutMissingNamelist(namelist = self%name)
247 read(self%InputFile%Path%original,nml=paramonte,iostat=self%InputFile%Err%stat)
248 self%Err = self%InputFile%getReadErr(self%InputFile%Err%stat,self%InputFile%Path%modified)
249
250 end if
251
252 if (self%Err%occurred) then
253
254 if (is_iostat_end(self%Err%stat) .or. is_iostat_eor(self%Err%stat)) then
255 call self%warnUserAboutMissingNamelist(namelist = "paramonte")
256 else
257 read(self%InputFile%Path%original,nml=ParaXXXX) ! LCOV_EXCL_LINE ! let the compiler print diagnostic messages, should any error happen.
258 end if
259
260 end if
261
262 end if
263
264 else blockInputFileType ! the input file is external
265
266 ! close input file if it is open
267
268 if (self%InputFile%isOpen) then
269 ! LCOV_EXCL_START
270 close(unit=self%InputFile%unit,iostat=self%InputFile%Err%stat)
271 self%Err = self%InputFile%getCloseErr(self%InputFile%Err%stat)
272 if (self%Err%occurred) then
273 self%Err%msg = PROCEDURE_NAME // ": Error occurred while attempting to close the user-provided input file='" // &
274 self%InputFile%Path%modified // "', unit=" // val2char(self%InputFile%unit) // "." // NLC // &
275 self%Err%msg
276 return
277 end if
278 ! LCOV_EXCL_STOP
279 end if
280
281 ! open input file
282
283 open( newunit = self%InputFile%unit &
284 , file = self%InputFile%Path%modified &
285 , status = self%InputFile%status &
286 , iostat = self%InputFile%Err%stat &
287#if INTEL_COMPILER_ENABLED && WINDOWS_ENABLED
288 , SHARED &
289#endif
290 )
291 self%Err = self%InputFile%getOpenErr(self%InputFile%Err%stat)
292 if (self%Err%occurred) then
293 ! LCOV_EXCL_START
294 self%Err%msg = PROCEDURE_NAME // ": Error occurred while attempting to open the user-provided input file='" // &
295 self%InputFile%Path%modified // "', unit=" // val2char(self%InputFile%unit) // "." // NLC // &
296 self%Err%msg
297 return
298 ! LCOV_EXCL_STOP
299 end if
300
301 ! read input file
302
303 read(self%InputFile%unit,nml=ParaXXXX,iostat=self%InputFile%Err%stat)
304 self%Err = self%InputFile%getReadErr(self%InputFile%Err%stat,self%InputFile%Path%modified)
305
306 if (self%Err%occurred) then
307
308 if ( is_iostat_end(self%Err%stat) .or. is_iostat_eor(self%Err%stat) ) then
309
310 ! search for the paraxxxx namelist group in the file.
311
312 rewind(self%InputFile%unit)
313 call self%warnUserAboutMissingNamelist(namelist = self%name)
314 read(self%InputFile%unit, nml=paramonte, iostat=self%InputFile%Err%stat) ! WARNING: "paraxxxx" is NOT the same as fpp macro name "ParaXXXX"
315 self%Err = self%InputFile%getReadErr(self%InputFile%Err%stat,self%InputFile%Path%modified)
316
317 end if
318
319 if (self%Err%occurred) then
320
321 if (is_iostat_end(self%Err%stat) .or. is_iostat_eor(self%Err%stat)) then
322
323 call self%warnUserAboutMissingNamelist(namelist = "paramonte")
324
325 else ! attempt to read the file one more time, without error handling, so that the compiler prints out the error message.
326
327 ! LCOV_EXCL_START
328 rewind(self%InputFile%unit)
329 read(self%InputFile%unit, nml=ParaXXXX)
330 ! LCOV_EXCL_STOP
331
332 end if
333
334 end if
335
336 end if
337
338 ! close input file
339
340 close(unit=self%InputFile%unit,iostat=self%InputFile%Err%stat)
341 ! LCOV_EXCL_START
342 self%Err = self%InputFile%getCloseErr(self%InputFile%Err%stat)
343 if (self%Err%occurred) then
344 self%Err%msg = PROCEDURE_NAME // ": Error occurred while attempting to close the user-provided input file='" // &
345 self%InputFile%Path%modified // "', unit=" // val2char(self%InputFile%unit) // "." // NLC // &
346 self%Err%msg
347 return
348 end if
349 ! LCOV_EXCL_STOP
350
351 end if blockInputFileType
352
353 end if blockReadInputFile
354
355 ! setup SpecBase variables that have been read form the input file
356
357 call self%SpecBase%setFromInputFile(Err = self%Err)
358 if (self%Err%occurred) then
359 self%Err%msg = PROCEDURE_NAME // self%Err%msg ! LCOV_EXCL_LINE
360 return ! LCOV_EXCL_LINE
361 end if
362
363#if defined PARADRAM || defined PARADISE
364
365 ! setup SpecMCMC variables that have been read form the input file
366
367 call self%SpecMCMC%setFromInputFile(Err = self%Err)
368 if (self%Err%occurred) then
369 self%Err%msg = PROCEDURE_NAME // self%Err%msg ! LCOV_EXCL_LINE
370 return ! LCOV_EXCL_LINE
371 end if
372
373 ! setup SpecDRAM variables that have been read form the input file
374
375 call self%SpecDRAM%setFromInputFile( self%Err )
376 if (self%Err%occurred) then
377 self%Err%msg = PROCEDURE_NAME // self%Err%msg ! LCOV_EXCL_LINE
378 return ! LCOV_EXCL_LINE
379 end if
380
381#elif defined PARANEST
382
383 ! setup SpecNest variables that have been read form the input file
384
385 call self%SpecNest%setFromInputFile( self%Err )
386 if (self%Err%occurred) then
387 self%Err%msg = PROCEDURE_NAME // self%Err%msg ! LCOV_EXCL_LINE
388 return ! LCOV_EXCL_LINE
389 end if
390
391#endif
392
393 end subroutine getSpecFromInputFile
