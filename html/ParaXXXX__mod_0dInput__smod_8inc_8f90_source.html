196 module subroutine getspecfrominputfile(self,nd)
197#if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
198 !DEC$ ATTRIBUTES DLLEXPORT :: getSpecFromInputFile
199#endif
200 use constants_mod, only: ik, rk
201 use string_mod, only: num2str
202
203 implicit none
204 class(ParaXXXX_type), intent(inout) :: self
205 integer(IK), intent(in) :: nd
206 character(*), parameter :: PROCEDURE_NAME = submodule_name//"@getSpecFromInputFile()"
207
208 ! initialize/nullify all general input options
209
210 call self%SpecBase%nullifyNameListVar(nd)
211 call self%SpecMCMC%nullifyNameListVar(nd)
212 call self%SpecDRAM%nullifyNameListVar(nd)
213
214 ! read input options if input file is provided
215
216 blockreadinputfile: if (self%inputFileArgIsPresent) then
217
218 blockinputfiletype: if (self%InputFile%isInternal) then
219
220 ! read input file as an internal file
221
222 read(self%InputFile%Path%original,nml=paraxxxx,iostat=self%InputFile%Err%stat)
223 self%Err = self%InputFile%getReadErr(self%InputFile%Err%stat,self%InputFile%Path%modified)
224
225 if (self%Err%occurred) then
226
227 if ( is_iostat_end(self%Err%stat) .or. is_iostat_eor(self%Err%stat) ) then
228
229 ! search for the paraxxxx namelist group in the file.
230
231 call self%warnUserAboutMissingNamelist(namelist = self%name)
232 read(self%InputFile%Path%original,nml=paramonte,iostat=self%InputFile%Err%stat)
233 self%Err = self%InputFile%getReadErr(self%InputFile%Err%stat,self%InputFile%Path%modified)
234
235 end if
236
237 if (self%Err%occurred) then
238
239 if (is_iostat_end(self%Err%stat) .or. is_iostat_eor(self%Err%stat)) then
240
241 call self%warnUserAboutMissingNamelist(namelist = "ParaXXXX")
242
243 else
244
245 ! LCOV_EXCL_START
246 read(self%InputFile%Path%original,nml=paraxxxx) ! let the compiler print diagnostic messages, should any error happen.
247 ! LCOV_EXCL_STOP
248
249 end if
250
251 end if
252
253 end if
254
255 else blockinputfiletype ! the input file is external
256
257 ! close input file if it is open
258
259 if (self%InputFile%isOpen) then
260 ! LCOV_EXCL_START
261 close(unit=self%InputFile%unit,iostat=self%InputFile%Err%stat)
262 self%Err = self%InputFile%getCloseErr(self%InputFile%Err%stat)
263 if (self%Err%occurred) then
264 self%Err%msg = procedure_name // ": Error occurred while attempting to close the user-provided input file='" // &
265 self%InputFile%Path%modified // "', unit=" // num2str(self%InputFile%unit) // ".\n" // &
266 self%Err%msg
267 return
268 end if
269 ! LCOV_EXCL_STOP
270 end if
271
272 ! open input file
273
274 open( newunit = self%InputFile%unit &
275 , file = self%InputFile%Path%modified &
276 , status = self%InputFile%status &
277 , iostat = self%InputFile%Err%stat &
278#if defined INTEL_COMPILER_ENABLED && defined OS_IS_WINDOWS
279 , shared &
280#endif
281 )
282 self%Err = self%InputFile%getOpenErr(self%InputFile%Err%stat)
283 if (self%Err%occurred) then
284 ! LCOV_EXCL_START
285 self%Err%msg = procedure_name // ": Error occurred while attempting to open the user-provided input file='" // &
286 self%InputFile%Path%modified // "', unit=" // num2str(self%InputFile%unit) // ".\n" // &
287 self%Err%msg
288 return
289 ! LCOV_EXCL_STOP
290 end if
291
292 ! read input file
293
294 read(self%InputFile%unit,nml=paraxxxx,iostat=self%InputFile%Err%stat)
295 self%Err = self%InputFile%getReadErr(self%InputFile%Err%stat,self%InputFile%Path%modified)
296
297 if (self%Err%occurred) then
298
299 if ( is_iostat_end(self%Err%stat) .or. is_iostat_eor(self%Err%stat) ) then
300
301 ! search for the paraxxxx namelist group in the file.
302
303 rewind(self%InputFile%unit)
304 call self%warnUserAboutMissingNamelist(namelist = self%name)
305 read(self%InputFile%unit, nml=paramonte, iostat=self%InputFile%Err%stat) ! WARNING: "paraxxxx" is NOT the same as fpp macro name "ParaXXXX"
306 self%Err = self%InputFile%getReadErr(self%InputFile%Err%stat,self%InputFile%Path%modified)
307
308 end if
309
310 if (self%Err%occurred) then
311
312 if (is_iostat_end(self%Err%stat) .or. is_iostat_eor(self%Err%stat)) then
313
314 call self%warnUserAboutMissingNamelist(namelist = "ParaXXXX")
315
316 else ! attempt to read the file one more time, without error handling, so that the compiler prints out the error message.
317
318 ! LCOV_EXCL_START
319 rewind(self%InputFile%unit)
320 read(self%InputFile%unit, nml=paraxxxx)
321 ! LCOV_EXCL_STOP
322
323 end if
324
325 end if
326
327 end if
328
329 ! close input file
330
331 close(unit=self%InputFile%unit,iostat=self%InputFile%Err%stat)
332 ! LCOV_EXCL_START
333 self%Err = self%InputFile%getCloseErr(self%InputFile%Err%stat)
334 if (self%Err%occurred) then
335 self%Err%msg = procedure_name // ": Error occurred while attempting to close the user-provided input file='" // &
336 self%InputFile%Path%modified // "', unit=" // num2str(self%InputFile%unit) // ".\n" // &
337 self%Err%msg
338 return
339 end if
340 ! LCOV_EXCL_STOP
341
342 end if blockinputfiletype
343
344 end if blockreadinputfile
345
346 ! setup SpecBase variables that have been read form the input file
347
348 call self%SpecBase%setFromInputFile(err = self%Err)
349 ! LCOV_EXCL_START
350 if (self%Err%occurred) then
351 self%Err%msg = procedure_name // self%Err%msg
352 return
353 end if
354 ! LCOV_EXCL_STOP
355
356 ! setup SpecMCMC variables that have been read form the input file
357
358 call self%SpecMCMC%setFromInputFile(err = self%Err)
359 ! LCOV_EXCL_START
360 if (self%Err%occurred) then
361 self%Err%msg = procedure_name // self%Err%msg
362 return
363 end if
364 ! LCOV_EXCL_STOP
365
366 ! setup SpecDRAM variables that have been read form the input file
367
368 call self%SpecDRAM%setFromInputFile( self%Err )
369 ! LCOV_EXCL_START
370 if (self%Err%occurred) then
371 self%Err%msg = procedure_name // self%Err%msg
372 return
373 end if
374 ! LCOV_EXCL_STOP
375
376 end subroutine getspecfrominputfile
