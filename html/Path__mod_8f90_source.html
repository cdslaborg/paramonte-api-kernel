157 subroutine query(Path,inputPath,OS)
158#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
159 !DEC$ ATTRIBUTES DLLEXPORT :: query
160#endif
161 use Err_mod, only: Err_type
162 use Constants_mod, only: IK
163 use System_mod, only: OS_type
164 use String_mod, only: replace
165 implicit none
166 class(Path_type), intent(inout) :: Path
167 character(*), intent(in), optional :: inputPath
168 type(OS_type), intent(in), optional :: OS
169 logical :: isUnixShell
170
171 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@query()"
172
173 Path%Err%occurred = .false.
174 Path%Err%msg = ""
175
176 if (present(inputPath)) then
177 Path%original = trim(adjustl(inputPath))
178 elseif (.not.allocated(Path%original)) then
179 Path%Err%occurred = .true.
180 Path%Err%msg = PROCEDURE_NAME//": Error occurred. Neither inputPath argument is given as input, nor Path%original is allocated to construct the Path object."
181 return
182 else
183 if ( len(trim(adjustl(Path%original)))==0 ) then
184 Path%Err%occurred = .true.
185 Path%Err%msg = PROCEDURE_NAME//": Error occurred. Neither inputPath argument is given as input, nor Path%original has a non-blank length > 0 to construct the Path object."
186 return
187 end if
188 end if
189
190 if (present(OS)) then
191 Path%shellSlash = OS%Shell%slash
192 isUnixShell = OS%Shell%isUnix
193 else
194 block
195 type(OS_type) :: OS
196 call OS%query()
197 if (OS%Err%occurred) then
198 ! LCOV_EXCL_START
199 Path%Err%stat = OS%Err%stat
200 Path%Err%occurred = OS%Err%occurred
201 Path%Err%msg = PROCEDURE_NAME // ": Error occurred while querying OS type.\n" // Path%Err%msg
202 end if
203 ! LCOV_EXCL_STOP
204 Path%shellSlash = OS%Shell%slash
205 isUnixShell = OS%Shell%isUnix
206 end block
207 if (Path%Err%occurred) return
208 end if
209
210 if (isUnixShell) then
211 ! if the path contains both / and \, then assume that it is already in linux style
212 if (index(Path%original,"/")==0) then ! path is given in Windows style
213 Path%modified = linify(Path%original)
214 else
215 Path%modified = Path%original
216 end if
217#if WINDOWS_ENABLED
218 else
219 Path%modified = winify(Path%original)
220#endif
221 end if
222
223 call Path%getDirNameExt( Path%modified, Path%shellSlash, Path%dir, Path%name, Path%ext )
224 Path%base = Path%dir // Path%name
225
226 end subroutine query

246 pure function winify(inputPath) result(outputPath) !,Err)!,ignoreWindowsReservedChars)
247#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
248 !DEC$ ATTRIBUTES DLLEXPORT :: winify
249#endif
250 !use Err_mod, only: Err_type
251 use Constants_mod, only: IK
252 use String_mod, only: replace
253 implicit none
254 character(len=*), intent(in) :: inputPath
255 character(:), allocatable :: outputPath
256 !type(Err_type), intent(out) :: Err
257 !logical, intent(in), optional :: ignoreWindowsReservedChars
258 !logical :: reservedCharInspectionNeeded
259 !character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@winify()"
260 character(:), allocatable :: outputPathDummy
261 integer(IK) :: i, j, outputPathLen
262
263 !Err%occurred = .false.
264 !Err%msg = ""
265
266 ! check if any character in the input path is Windows Reserved Character:
267
268 !reservedCharInspectionNeeded = .true.
269 !if (present(ignoreWindowsReservedChars)) reservedCharInspectionNeeded = .not. ignoreReservedChars
270 !if (reservedCharInspectionNeeded) then
271 ! do i = 1, len(WINDOWS_RESERVED_CHAR)
272 ! if ( index(inputPath,WINDOWS_RESERVED_CHAR(i:i)) /= 0 ) then
273 ! Err%occurred = .true.
274 ! Err%msg = PROCEDURE_NAME // ": Error occurred. Invalid Windows character '" // &
275 ! WINDOWS_RESERVED_CHAR(i:i) // "' detected in the input file path='" // inputPath // "'."
276 ! return
277 ! end if
278 ! end do
279 !end if
280
281 !if ( index(inputPath,"\\") /= 0 ) then
282 ! Err%occurred = .true.
283 ! Err%msg = PROCEDURE_NAME // ": Error occurred. Invalid Windows character '\' corresponding to '\\' detected &
284 ! & in the input file path='" // inputPath // "'."
285 ! return
286 !end if
287
288 ! note that multiple \ character in sequence is meaningless in Linux (basically \\ reduces to \‍),
289 ! and in Windows means the same as a single \. Therefore, reduce all sequential \ characters to a single \.
290
291 outputPath = trim(adjustl(inputPath))
292 loopRemoveMultipleSlash: do
293 outputPathDummy = replace(outputPath,"\\","\")
294 if (outputPathDummy==outputPath) exit loopRemoveMultipleSlash
295 outputPath = outputPathDummy
296 end do loopRemoveMultipleSlash
297 outputPathLen = len(outputPath)
298
299 ! Now check for the presence of any Linux Shell Escape Character in the input path without a preceding \.
300 ! If there is any, this would imply that the input path is a Windows path,
301 ! otherwise a escape character without preceding \ would be invalid in Linux.
302
303 if (outputPathLen==1_IK) then
304 if (outputPath=="/") outputPath = "\"
305 return
306 else
307 do i = 1, len(SHELL_ESCAPE_CHAR)
308 if (SHELL_ESCAPE_CHAR(i:i)/="\") then
309 do j = 2, outputPathLen
310 if (outputPath(j:j)==SHELL_ESCAPE_CHAR(i:i)) then
311 if (outputPath(j-1:j-1)/="\") return ! no escaping has occurred. Therefore, it is a windows path, there is no need for further winifying.
312 end if
313 end do
314 end if
315 end do
316 end if
317
318 ! By now, there is no way but to assume that the path is indeed a Linux path
319 ! Thus, correct for any Linux Shell Escape Character in the input path:
320
321 do i = 1, len(SHELL_ESCAPE_CHAR)
322 outputPath = replace(outputPath,"\"//SHELL_ESCAPE_CHAR(i:i),SHELL_ESCAPE_CHAR(i:i))
323 end do
324
325 ! Now remove any remaining backslash in the input path:
326 ! commented out: it is assumed that there are no dangling \ in the Linux path
327 !outputPath = replace(outputPath,"\","")
328
329 ! check if the file name contains white space. if so, put the entire name in quotations
330
331 if ( index(outputPath," ") /= 0 ) then
332 outputPath = '"' // outputPath // '"'
333 end if
334 outputPath = replace(outputPath,"/","\")
335
336 end function winify

348 pure function linify(inputPath) result(outputPath)
349#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
350 !DEC$ ATTRIBUTES DLLEXPORT :: linify
351#endif
352 use Constants_mod, only: IK
353 use String_mod, only: replace
354 implicit none
355 character(*), intent(in) :: inputPath
356 character(:), allocatable :: outputPath
357 character(:), allocatable :: outputPathDummy
358 integer(IK) :: i
359
360 ! check if the path is sandwiched between quotation marks. If so, remove them:
361 outputPath = trim(adjustl(inputPath))
362 i = len(outputPath)
363 if (i==0) return
364 if ( i>1 ) then
365 if ( (outputPath(1:1)=='"' .and. outputPath(i:i)=='"') .or. (outputPath(1:1)=="'" .and. outputPath(i:i)=="'") )then
366 outputPathDummy = outputPath(2:i-1)
367 else
368 outputPathDummy = outputPath
369 end if
370 end if
371
372 ! First change all backslashes to forward slash:
373 outputPath = replace(outputPathDummy,"\","/")
374
375 ! Now correct for any Linux Shell Escape Character in the input path:
376 do i = 1, len(SHELL_ESCAPE_CHAR)
377 if (SHELL_ESCAPE_CHAR(i:i)/="\") then
378 outputPathDummy = replace(outputPath,SHELL_ESCAPE_CHAR(i:i),"\"//SHELL_ESCAPE_CHAR(i:i))
379 outputPath = outputPathDummy
380 end if
381 end do
382
383 !! Now correct for any white spaces in outputPath:
384 !outputPath = replace(outputPath," ","\ ")
385
386 end function linify

473 subroutine getDirFullName(path,slash,dir,fullName)
474#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
475 !DEC$ ATTRIBUTES DLLEXPORT :: getDirFullName
476#endif
477 use Constants_mod, only: IK
478 implicit none
479 character(*) , intent(in) :: path
480 character(1) , intent(in) :: slash
481 character(:), allocatable, intent(out) :: dir, fullName
482
483 integer(IK) :: pathLen, slashPos
484
485 pathLen = len(path)
486
487 if ( pathLen==0 ) then
488 dir=""; fullName=""
489 return
490 end if
491
492 slashPos = index(path,slash,back=.true.)
493
494 if (slashPos==0) then ! it is all filename
495 dir = ""
496 fullName = path
497 elseif (slashPos==pathLen) then ! it is all directory
498 dir = path
499 fullName = ""
500 return
501 else
502 dir = path(1:slashPos)
503 fullName = path(slashPos+1:pathLen)
504 end if
505
506 end subroutine getDirFullName

559 function mkdir(dirPath,isUnixShell,wait) result(Err)
560#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
561 !DEC$ ATTRIBUTES DLLEXPORT :: mkdir
562#endif
563 use Constants_mod, only: IK, NLC
564 use System_mod, only: SysCmd_type, OS_type
565 use String_mod, only: val2str
566 use Err_mod, only: Err_type
567 implicit none
568 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@mkdir()"
569 character(*), intent(in) :: dirPath
570 logical, intent(in), optional :: isUnixShell, wait
571 type(Err_type) :: Err
572 type(SysCmd_type) :: SysCmd
573 type(OS_type) :: OS
574 logical :: isUnixShellDefault
575 character(:), allocatable :: command
576 integer(IK) :: itry
577
578
579
580 Err%occurred = .false.
581
582 if (present(isUnixShell)) then
583 isUnixShellDefault = isUnixShell
584 else
585 OS%Err%occurred = .false.
586 call OS%query()
587 if (OS%Err%occurred) then
588 ! LCOV_EXCL_START
589 command = 'mkdir "'//dirPath//'"'
590 ! LCOV_EXCL_STOP
591 else
592 isUnixShellDefault = OS%Shell%isUnix
593 end if
594 end if
595
596 if (.not. allocated(command)) then
597 if (isUnixShellDefault) then
598 command = 'mkdir -p "'//dirPath//'" > /dev/null 2>&1' ! -p enables nested mkdir
599#if WINDOWS_ENABLED
600 else
601 command = 'mkdir "'//dirPath//'" >nul 2>&1' ! WARNING: path has to be enclosed with "" to allow nested mkdir
602#endif
603 end if
604 end if
605
606 ! Try to create the folder for 10 times, and fail if all attempts fail.
607
608 loopTry: do itry = 1, 10
609 SysCmd = SysCmd_type(command, wait)
610 if (SysCmd%Err%occurred .and. .not. isdir(dirPath)) cycle loopTry
611 deallocate(command)
612 return
613 end do loopTry
614
615 ! LCOV_EXCL_START
616 Err%occurred = .true.
617 Err%stat = SysCmd%Err%stat
618 Err%msg = PROCEDURE_NAME // SysCmd%Err%msg //NLC//"execute_command_line() exitstat: " // val2str(SysCmd%exitstat)
619 ! LCOV_EXCL_STOP
620
621 end function mkdir
