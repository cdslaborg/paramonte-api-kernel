142 subroutine readDataGRB(inFilePath,outFilePath,isLgrb)
143#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
144 !DEC$ ATTRIBUTES DLLEXPORT :: readDataGRB
145#endif
146
147 use Parallelism_mod, only: Image_type
148 use Constants_mod, only: IK, RK
149 implicit none
150 character(*), intent(in) :: inFilePath, outFilePath
151 integer(IK) :: inFileUnit, outFileUnit, igrb
152 logical , intent(in) :: isLgrb
153 type(Image_type) :: Image
154
155 call Image%query()
156
157 if (isLgrb) then
158 GRB%count = NLGRB
159 else
160 GRB%count = NSGRB
161 end if
162
163 if (allocated(GRB%Event)) deallocate(GRB%Event); allocate(GRB%Event(GRB%count))
164 if (allocated(Trigger)) deallocate(Trigger); allocate(Trigger(GRB%count))
165
166 open( newunit = inFileUnit & ! LCOV_EXCL_LINE
167 , file = inFilePath & ! LCOV_EXCL_LINE
168 , status = "old" & ! LCOV_EXCL_LINE
169#if INTEL_COMPILER_ENABLED && WINDOWS_ENABLED
170 , SHARED & ! LCOV_EXCL_LINE
171#endif
172 )
173
174 if (Image%isFirst) then
175 ! LCOV_EXCL_START
176 open(newunit=outFileUnit,file=outFilePath,status="replace")
177 write(outFileUnit,"(9a30)" ) "trigger" &
178 , "logPbol_1eV_20MeV" &
179 , "logSbol_1eV_20MeV" &
180 , "logEpk" &
181 , "logEPR1024" &
182 , "logEFR" &
183 , "logFPR1024" &
184 , "logT90" &
185 , "logEffPF53"
186 end if
187 ! LCOV_EXCL_STOP
188
189 ! skip the header row in the input file
190 read(inFileUnit,*)
191
192 ! read BATSE GRB data
193 do igrb = 1, GRB%count
194
195 read(inFileUnit,* ) Trigger(igrb) & ! LCOV_EXCL_LINE
196 , GRB%Event(igrb)%logPF53 &
197 , GRB%Event(igrb)%logEpk &
198 , GRB%Event(igrb)%logSbol &
199 , GRB%Event(igrb)%logT90
200
201 ! convert all values to logarithm in base Neper
202
203 GRB%Event(igrb)%logPF53 = LOG_TEN * GRB%Event(igrb)%logPF53
204 GRB%Event(igrb)%logEpk = LOG_TEN * GRB%Event(igrb)%logEpk
205 GRB%Event(igrb)%logSbol = LOG_TEN * GRB%Event(igrb)%logSbol
206 GRB%Event(igrb)%logT90 = LOG_TEN * GRB%Event(igrb)%logT90
207
208 ! convert photon count data to energy in units of ergs
209
210 GRB%Event(igrb)%logPbol = getLogPbol( GRB%Event(igrb)%logEpk, GRB%Event(igrb)%logPF53 )
211 if (isLgrb) then
212 GRB%Event(igrb)%logSbol = getLogPbol( GRB%Event(igrb)%logEpk, GRB%Event(igrb)%logSbol )
213 else
214 GRB%Event(igrb)%logPF53 = GRB%Event(igrb)%logPF53 - THRESH_ERFC_AMP * erfc( (GRB%Event(igrb)%logT90-THRESH_ERFC_AVG) * THRESH_ERFC_STD_INV )
215 end if
216
217 ! write the converted data to output file
218
219 if (Image%isFirst) then
220 ! LCOV_EXCL_START
221 write(outFileUnit,"(I30,8E30.6)") Trigger(igrb) &
222 , GRB%Event(igrb)%logPbol &
223 , GRB%Event(igrb)%logSbol &
224 , GRB%Event(igrb)%logEpk &
225 , GRB%Event(igrb)%logEpk-GRB%Event(igrb)%logPbol &
226 , GRB%Event(igrb)%logEpk-GRB%Event(igrb)%logSbol &
227 , GRB%Event(igrb)%logSbol-GRB%Event(igrb)%logPbol &
228 , GRB%Event(igrb)%logT90 &
229 , GRB%Event(igrb)%logPF53
230
231 end if
232 ! LCOV_EXCL_STOP
233
234 end do
235
236 if (Image%isFirst) close(outFileUnit)
237
238 close(inFileUnit)
239
240!#if CAF_ENABLED
241! sync images(*)
242!
243! else
244!
245! sync images(1)
246! do igrb = 1, GRB%count
247! GRB%Event(igrb)%logPbol = GRB[1]%Event(igrb)%logPbol
248! GRB%Event(igrb)%logSbol = GRB[1]%Event(igrb)%logSbol
249! GRB%Event(igrb)%logPF53 = GRB[1]%Event(igrb)%logPF53
250! GRB%Event(igrb)%logEpk = GRB[1]%Event(igrb)%logEpk
251! GRB%Event(igrb)%logT90 = GRB[1]%Event(igrb)%logT90
252! end do
253!
254! end if
255!#endif
256
257 end subroutine readDataGRB

261 pure function getLog10PF53(log10epk,log10pbol) result(log10PF53)
262#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
263 !DEC$ ATTRIBUTES DLLEXPORT :: getLog10PF53
264#endif
265 ! Given Log10(Epk [KeV]) of an LGRB and its bolometric (0.0001-20000 KeV) peak flux [in units of Ergs/s], Log(Pbol),
266 ! this function calculates the corresponding Log10(peak photon flux) in the BATSE detection energy range [50,300] KeV.
267 ! Amir Shahmoradi, Wednesday June 27, 2012, 7:15 PM, Institute for Fusion Studies, The University of Texas at Austin.
268 ! Amir Shahmoradi, Wednesday June 27, 2012, 9:29 PM, Institute for Fusion Studies, The University of Texas at Austin.
269 use Constants_mod, only: RK
270 implicit none
271 real(RK), intent(in) :: log10epk,log10pbol
272 real(RK) :: log10PF53
273 if (log10epk<-2.915056638230699_RK) then
274 log10PF53 = log10pbol + 4.9200000000000000_RK
275 elseif (log10epk<1.5_RK) then
276 log10PF53 = log10pbol + 5.7361200000000000_RK + log10epk * &
277 ( 0.3093600000000000_RK + log10epk * &
278 ( 0.0045600000000000_RK + log10epk * &
279 ( 0.0015900000000000_RK + log10epk * &
280 ( 0.0001533360000000_RK - log10epk &
281 * 0.0003574800000000_RK ))))
282 elseif (log10epk<2.5_RK) then
283 log10PF53 = log10pbol + 1.9112800000000000_RK + log10epk * &
284 ( 39.710390000000000_RK - log10epk * &
285 ( 96.606280000000000_RK - log10epk * &
286 ( 109.24696000000000_RK - log10epk * &
287 ( 67.271800000000000_RK - log10epk * &
288 ( 23.402390000000000_RK - log10epk * &
289 ( 4.3454400000000000_RK - log10epk &
290 * 0.3360600000000000_RK ))))))
291 elseif (log10epk<4._RK) then
292 log10PF53 = log10pbol + 2.8020600000000000_RK + log10epk * &
293 ( 4.5690700000000000_RK - log10epk * &
294 ( 1.9277200000000000_RK - log10epk * &
295 ( 0.2938100000000000_RK - log10epk &
296 * 0.0148900000000000_RK )))
297 elseif (log10epk<5.4093868613659435_RK) then
298 log10PF53 = log10pbol - 10.465330000000000_RK + log10epk * &
299 ( 26.706370000000000_RK - log10epk * &
300 ( 14.476310000000000_RK - log10epk * &
301 ( 3.5404100000000000_RK - log10epk * &
302 ( 0.4095700000000000_RK - log10epk &
303 * 0.0183100000000000_RK ))))
304 else ! if (log10epk>=5.4093868613659435_RK) then
305 log10PF53 = log10pbol + 4.9200000000000000_RK
306 end if
307 end function getLog10PF53

316 ! this function calculates the corresponding Log(peak photon flux) in the BATSE detection energy range [50,300] KeV.

318 ! Amir Shahmoradi, Wednesday June 27, 2012, 9:29 PM, Institute for Fusion Studies, The University of Texas at Austin.
319 use Constants_mod, only: RK
320 implicit none
321 real(RK), intent(in) :: logEpk,logPbol
322 real(RK) :: logPF53
323 if (logEpk<-6.712165960423344_RK) then
324 logPF53 = logPbol + 11.32871865753070600_RK
325 elseif (logEpk<3.453877639491069_RK) then
326 logPF53 = logPbol + 13.20790440362500600_RK + logEpk * &
327 ( 0.309360000000000000_RK + logEpk * &

331 * 1.27171265917873e-05_RK ))))

333 logPF53 = logPbol + 4.400884836537660000_RK + logEpk * &
334 ( 39.71039000000000000_RK - logEpk * &
335 ( 41.95557432120050000_RK - logEpk * &
336 ( 20.60525451895990000_RK - logEpk * &
337 ( 5.510436247342930000_RK - logEpk * &
338 ( 0.832525333390336000_RK - logEpk * &
339 ( 0.067135977132248900_RK - logEpk &
340 * 0.002254876138523550_RK ))))))
341 elseif (logEpk<9.210340371976184_RK) then
342 logPF53 = logPbol + 6.451981585674900000_RK + logEpk * &
343 ( 4.569070000000000000_RK - logEpk * &
344 ( 0.837198158654537000_RK - logEpk * &
345 ( 0.055416002698982300_RK - logEpk &
346 * 0.001219684856402480_RK )))
347 elseif (logEpk<12.455573549219071_RK) then
348 logPF53 = logPbol - 24.09731285126340000_RK + logEpk * &

352 ( 0.033549115287895400_RK - logEpk &
