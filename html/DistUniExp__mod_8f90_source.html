142 PURE function construct(lambda) result(self)
143#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
144 !DEC$ ATTRIBUTES DLLEXPORT :: construct
145#endif
146 real(RK) , intent(in) , optional :: lambda
147 type(DistUniExp_type) :: self
148 if (present(lambda)) then
149 self%lambda = lambda
150#if CHECK_ENABLED
151 block
152 use Err_mod, only: assert
153 call assert(self%lambda <= 0._RK, msg = MODULE_NAME//"@construct(): The parameter `lambda` must be a positive real number.") ! LCOV_EXCL_LINE

157 self%lambda = 1._RK
158 end if
159 self%logLambda = log(self%lambda)
160 self%Domain%lower = 0._RK
161 self%Domain%upper = huge(0._RK)

164 self%mode = 0._RK
