123
124 use constants_mod, only: ik, rk, huge_rk
125
126 implicit none
127
128 integer(IK) , intent(in) :: nc, nd, np
129 real(rk) , intent(in) :: point(nd,np)
130 integer(IK) , intent(in) :: ntry
131 real(rk) , intent(in), optional :: initcenter(nd,nc)
132 integer(IK) , intent(in), optional :: nitermax, nzscimax
133 real(rk) , intent(in), optional :: reltolsq
134 type(kmeans_type) :: kmeans
135 type(kmeans_type), allocatable :: kmeanstry(:)
136 real(rk) :: potential
137 integer(IK) :: itry, itryleastpotential
138
139 if (allocated(kmeanstry)) deallocate(kmeanstry) ! LCOV_EXCL_LINE
140 allocate(kmeanstry(ntry))
141 potential = huge_rk
142
143 do itry = 1, ntry
144 kmeanstry(itry) = kmeans_type(nc, nd, np, point, initcenter, nitermax, nzscimax, reltolsq)
145 if (kmeanstry(itry)%potential < potential) then
146 potential = kmeanstry(itry)%potential
147 itryleastpotential = itry
148 end if
149 end do
150
151 kmeans = kmeanstry(itryleastpotential)

153 end function optimizekmeans

188 function runkmeans(nc, nd, np, Point, InitCenter, niterMax, nzsciMax, reltolSq) result(Kmeans)
189
190 use constants_mod, only: ik, rk, huge_rk
191
192 implicit none
193
194 integer(IK) , intent(in) :: nc, nd, np
195 real(rk) , intent(in) :: point(nd,np)
196 real(rk) , intent(in), optional :: initcenter(nd,nc)
197 integer(IK) , intent(in), optional :: nitermax, nzscimax
198 real(rk) , intent(in), optional :: reltolsq
199 type(kmeans_type) :: kmeans
200
201 character(*), parameter :: procedure_name = module_name//"@runKmeans()"
202 integer(IK) , parameter :: max_niter = 1e3_ik ! The maximum possible number of iterations for kmeans to converge.
203 integer(IK) , parameter :: max_nzsci = 1e1_ik ! The maximum possible number of iterations for which clusters of zero-size are identified.
204 real(rk) , parameter :: reltol_sq = 1.e-4_rk ! if the relative change in all Centers is less than this value, the algorithm has converged.
205
206 real(rk) :: potential ! The sum of all minimum distances squared.
207 real(rk) :: distancesq ! distance of each point ip from a given cluster center.
208 real(rk) :: sumpoint(nd,nc) ! The sum of the points in each cluster.
209 real(rk) :: mindistancesq(np) ! minimum distance of each point ip from all centers of the clusters.
210 integer(IK) :: membership(np) ! current (old) Memberships of the clusters.
211
212 logical :: convergenceoccurred
213 real(rk) :: reltolsqdefault
214 integer(IK) :: nitermaxdefault
215 integer(IK) :: nzscimaxdefault
216 integer :: i, j, ip, ic
217
218 kmeans%Err%occurred = .false.
219
220 allocate(kmeans%Size(nc), source = 0_ik)
221 allocate(kmeans%Center(nd,nc), source = 0._rk)
222 allocate(kmeans%Membership(np), source = 0_ik)
223 allocate(kmeans%MinDistanceSq(np), source = 0._rk)
224
225 reltolsqdefault = reltol_sq; if (present(reltolsq)) reltolsqdefault = reltolsq
226 nitermaxdefault = max_niter; if (present(nitermax)) nitermaxdefault = nitermax
227 nzscimaxdefault = max_nzsci; if (present(nzscimax)) nzscimaxdefault = nzscimax
228
229 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
230 ! Find the distances of all points to all cluster centers. If not given, initialize the centers via the Kmeans++ algorithm.
231 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
232
233 if(present(initcenter)) then
234 !%%%%%%%%%%%%%%%%%%%
235 ic = 1
236 do ip = 1, np
237 mindistancesq(ip) = sum( (initcenter(1:nd,ic) - point(1:nd,ip))**2 )
238 membership(ip) = ic
239 end do
240 !%%%%%%%%%%%%%%%%%%%
241 do ic = 2, nc - 1
242 do ip = 1, np
243 distancesq = sum( (initcenter(1:nd,ic) - point(1:nd,ip))**2 )
244 if (distancesq < mindistancesq(ip)) then
245 mindistancesq(ip) = distancesq
246 membership(ip) = ic
247 end if
248 end do
249 end do
250 !%%%%%%%%%%%%%%%%%%%
251 ic = nc
252 sumpoint = 0._rk
253 potential = 0._rk
254 do ip = 1, np
255 distancesq = sum( (initcenter(1:nd,ic) - point(1:nd,ip))**2 )
256 if (distancesq < mindistancesq(ip)) then
257 mindistancesq(ip) = distancesq
258 membership(ip) = ic
259 end if
260 sumpoint(1:nd,membership(ip)) = sumpoint(1:nd,membership(ip)) + point(1:nd,ip)
261 kmeans%Size(membership(ip)) = kmeans%Size(membership(ip)) + 1_ik
262 potential = potential + mindistancesq(ip)
263 end do
264 !%%%%%%%%%%%%%%%%%%%
265 else
266 call runkpp(nc, nd, np, point, sumpoint, membership, kmeans%Size, potential)
267 end if
268
269 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
270 ! Refine all centers until convergence.
271 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
272
273 kmeans%nzsci = 0_ik
274 kmeans%niter = 1_ik
275
276 loopcenterrefinement: do
277
278 ! compute the new cluster centers and create the new memberships.
279
280 !%%%%%%%%%%%%%%%%%%%
281 ic = 1
282 kmeans%Center(1:nd,ic) = sumpoint(1:nd,ic) / real(kmeans%Size(ic), kind = rk)
283 do ip = 1, np
284 kmeans%MinDistanceSq(ip) = sum( (kmeans%Center(1:nd,ic) - point(1:nd,ip))**2 )
285 kmeans%Membership(ip) = ic
286 end do
287 !%%%%%%%%%%%%%%%%%%%
288 do ic = 2, nc - 1
289 kmeans%Center(1:nd,ic) = sumpoint(1:nd,ic) / real(kmeans%Size(ic), kind = rk)
290 do ip = 1, np
291 distancesq = sum( (kmeans%Center(1:nd,ic) - point(1:nd,ip))**2 )
292 if (distancesq < kmeans%MinDistanceSq(ip)) then
293 kmeans%MinDistanceSq(ip) = distancesq
294 kmeans%Membership(ip) = ic
295 end if
296 end do
297 end do
298 !%%%%%%%%%%%%%%%%%%%
299 ic = nc
300 convergenceoccurred = .true.
301 kmeans%potential = 0._rk
302 kmeans%Center(1:nd,ic) = sumpoint(1:nd,ic) / real(kmeans%Size(ic), kind = rk)
303 do ip = 1, np
304 distancesq = sum( (kmeans%Center(1:nd,ic) - point(1:nd,ip))**2 )
305 if (distancesq < kmeans%MinDistanceSq(ip)) then
306 kmeans%MinDistanceSq(ip) = distancesq
307 kmeans%Membership(ip) = ic
308 end if
309 if (kmeans%Membership(ip)/=membership(ip)) then
310 convergenceoccurred = .false.
311 kmeans%Size(membership(ip)) = kmeans%Size(membership(ip)) - 1_ik
312 kmeans%Size(kmeans%Membership(ip)) = kmeans%Size(kmeans%Membership(ip)) + 1_ik
313 end if
314 kmeans%potential = kmeans%potential + kmeans%MinDistanceSq(ip)
315 end do
316 !%%%%%%%%%%%%%%%%%%%
317
318 if (any(kmeans%Size==0_ik)) then
319
320 kmeans%niter = 1_ik
321 kmeans%nzsci = kmeans%nzsci + 1_ik
322
323 if (kmeans%nzsci > nzscimaxdefault) then
324 !Kmeans%Err%msg = "Kmeans%nzsci > nzsciMaxDefault"
325 kmeans%Err%occurred = .true.
326 kmeans%Err%stat = 2_ik
327 return
328 end if
329
330 ! find new random Centers and start over.
331
332 call runkpp(nc, nd, np, point, sumpoint, membership, kmeans%Size, potential)
333
334 cycle loopcenterrefinement
335
336#if defined DEBUG_ENABLED || defined TESTING_ENABLED
337 elseif (any(kmeans%Size<0_ik)) then
338 block
339 use err_mod, only: abort
340 kmeans%Err%msg = procedure_name//": Internal error occurred - Kmeans%Size < 0. Please report this issue along with circumstances to the developers."
341 call abort(kmeans%Err)
342 error stop
343 end block
344#endif
345
346 end if
347
348 ! if `niterMax` has reached, quit.
349
350 if (kmeans%niter < nitermaxdefault) then
351
352 ! If convergence has occurred, return.
353
354 if (convergenceoccurred .or. abs(kmeans%potential-potential)/kmeans%potential < reltolsqdefault) return
355
356 kmeans%niter = kmeans%niter + 1_ik
357
358 do ip = 1, np
359 sumpoint(1:nd,membership(ip)) = sumpoint(1:nd,membership(ip)) - point(1:nd,ip)
360 sumpoint(1:nd,kmeans%Membership(ip)) = sumpoint(1:nd,kmeans%Membership(ip)) + point(1:nd,ip)
361 membership(ip) = kmeans%Membership(ip)
362 end do
363 potential = kmeans%potential
364
365 else
366
367 !Kmeans%Err%msg = "Kmeans%niter > niterMaxDefault"
368 kmeans%Err%occurred = .true.
369 kmeans%Err%stat = 1_ik
370 return
371
372 end if
373
374 end do loopcenterrefinement
375
376 end function runkmeans

404 subroutine runkpp(nc, nd, np, Point, SumPoint, Membership, Size, potential)
405
406 use statistics_mod, only: getrandint
407
408 implicit none
409
410 integer(IK) , intent(in) :: np, nd, nc
411 real(RK) , intent(in) :: Point(nd,np)
412 real(RK) , intent(out) :: SumPoint(nd,nc)
413 integer(IK) , intent(out) :: Membership(np)
414 integer(IK) , intent(out) :: Size(nc)
415 real(RK) , intent(out) :: potential
416
417 real(RK) :: dummy
418 real(RK) :: distanceSq
419 real(RK) :: Center(nd,nc)
420 real(RK) :: MinDistanceSq(np) ! An array representing the minimum distances of individual points from all clusters.
421 real(RK) :: CumSumDistSq(0:np) ! sum of distance squared of points up to the given index of the vector.
422 integer(IK) :: randomInteger
423 integer(IK) :: ic, ip, i
424
425 cumsumdistsq(0) = 0._rk ! This element must always remain zero.
426
427 ic = 1_ik
428 center(1:nd,ic) = point( 1:nd , getrandint(1,np) ) ! perhaps this random assignment is unnecessary, after all, Point is supposed to be of random order.
429 do ip = 1, np
430 mindistancesq(ip) = sum( (center(1:nd,ic) - point(1:nd,ip))**2 ) ! find the distance-squared of each point to the # ic cluster Center.
431 cumsumdistsq(ip) = cumsumdistsq(ip-1) + mindistancesq(ip)
432 membership(ip) = ic
433 end do
434
435 ic = 2_ik
436 call random_number(dummy)
437 dummy = dummy * cumsumdistsq(np)
438 center(1:nd,ic) = point( 1:nd, minloc(cumsumdistsq, dim = 1, mask = cumsumdistsq > dummy) - 1 ) ! -1 is essential to compensate for the 0 starting index of CumSumDistSq
439
440 ! find the rest of cluster centers
441
442 do ic = 2, nc - 1
443
444 do ip = 1, np
445 distancesq = sum( (center(1:nd,ic) - point(1:nd,ip))**2 )
446 if (distancesq < mindistancesq(ip)) then
447 mindistancesq(ip) = distancesq
448 membership(ip) = ic
449 end if
450 cumsumdistsq(ip) = cumsumdistsq(ip-1) + mindistancesq(ip)
451 end do
452
453 call random_number(dummy)
454 dummy = dummy * cumsumdistsq(np)
455 center(1:nd,ic+1) = point( 1:nd , minloc(cumsumdistsq, dim = 1, mask = cumsumdistsq > dummy) - 1 )
456
457 end do
458
459 Size = 0_ik
460 sumpoint = 0._rk
461 potential = 0._rk
462 do ip = 1, np
463 distancesq = sum( (center(1:nd,nc) - point(1:nd,ip))**2 )
464 if (distancesq < mindistancesq(ip)) then
465 mindistancesq(ip) = distancesq
466 membership(ip) = nc
467 end if
468 sumpoint(1:nd,membership(ip)) = sumpoint(1:nd,membership(ip)) + point(1:nd,ip)
469 Size(membership(ip)) = Size(membership(ip)) + 1_ik
470 potential = potential + mindistancesq(ip)
471 end do
472
473 end subroutine runkpp
