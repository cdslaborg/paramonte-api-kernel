96 function construct(ndim, LimitLower, LimitUpper, Err) result(self)
97#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
98 !DEC$ ATTRIBUTES DLLEXPORT :: construct

100 use Matrix_mod, only: genInvPosDefMat

102 use Err_mod, only: Err_type
103 integer(IK) , intent(in) :: ndim
104 real(RK) , intent(in) , optional :: LimitLower(ndim)
105 real(RK) , intent(in) , optional :: LimitUpper(ndim)
106 type(Err_type) , intent(out) , optional :: Err
107 type(DomainCube_type) :: self
108 character(:) , allocatable :: msg
109 self%ndim = ndim
110 if (present(LimitLower)) then
111 self%Limit%Lower = LimitLower
112 else
113 allocate(self%Limit%Lower(ndim), source = 0._RK)
114 end if
115 if (present(LimitUpper)) then
116 self%Limit%Upper = LimitUpper
117 else
118 allocate(self%Limit%Upper(ndim), source = 1._RK)
119 end if
120 if (any(self%Limit%Upper <= self%Limit%Lower)) then
121 msg = MODULE_NAME//"@construct(): The upper limits of the domain cannot be smaller than the lower limits."
122 if (present(Err)) then
123 Err%occurred = .true.
124 Err%msg = msg
125 else
126 write(*,"(*(g0,:,' '))") msg
127 error stop
128 end if
129 elseif (present(Err)) then
130 Err%occurred = .false.
131 end if
132 self%Size = self%Limit%Upper - self%Limit%Lower
133 self%logVol = sum(log(self%Size))
134 end function construct

138 pure function contains(self, Point) result(isContained)
139#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
140 !DEC$ ATTRIBUTES DLLEXPORT :: contains
141#endif
142 class(DomainCube_type) , intent(in) :: self
143 real(RK) , intent(in) :: Point(self%ndim)
144 logical :: isContained
145 integer(IK) :: i
146 loopContainmentCheck: do i = 1, self%ndim
147 isContained = self%Limit%Lower(i) <= Point(i) .and. Point(i) <= self%Limit%Upper(i)
148 if (isContained) cycle loopContainmentCheck
149 return
150 end do loopContainmentCheck
151 end function contains
