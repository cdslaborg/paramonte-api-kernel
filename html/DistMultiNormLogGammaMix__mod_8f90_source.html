138 PURE function logpdf(self,Point) result(logval)
139 use Constants_mod, only: LOG_HALF
140 use DistUniLogGamma_mod, only: getLogLogGammaPDF => getLogPDF
141 use DistUniNorm_mod, only: genNormLogPDF
142 use MathLogAddExp_mod, only: genLogAddExp
143 use MathLogSumExp_mod, only: genLogSumExp
144 use Misc_mod, only: getMinMax
145 implicit none
146 class(DistMultiNormLogGammaMix_type), intent(in) :: self
147 real(RK) , intent(in) :: Point(self%ndim)
148 real(RK) :: logval
149 real(RK) :: LogFunc(self%ndim)
150 real(RK) :: logLogGammaComp1, logNormComp1
151 real(RK) :: logLogGammaComp2, logNormComp2
152 real(RK) :: maxLogFunc, minLogFunc
153 real(RK) , parameter :: logNormFacPDF_logGamma = 0._RK ! -log_gamma(1._RK) - log(abs(1._RK))
154 integer(IK) :: idim, ndimPlusTwoOverTwo
155
156 logLogGammaComp1 = getLogLogGammaPDF(x = Point(1) + 10._RK, shape = 1._RK, logNormFacPDF = logNormFacPDF_logGamma)
157 logLogGammaComp2 = getLogLogGammaPDF(x = Point(1) - 10._RK, shape = 1._RK, logNormFacPDF = logNormFacPDF_logGamma)
158 call getMinMax(logLogGammaComp1, logLogGammaComp2, min = minLogFunc, max = maxLogFunc)
159 LogFunc(1) = LOG_HALF + genLogAddExp(smaller = minLogFunc, larger = maxLogFunc)
160
161 logNormComp1 = genNormLogPDF(point = Point(2) + 10._RK)
162 logNormComp2 = genNormLogPDF(point = Point(2) - 10._RK)
163 call getMinMax(logNormComp1, logNormComp2, min = minLogFunc, max = maxLogFunc)
164 LogFunc(2) = LOG_HALF + genLogAddExp(smaller = minLogFunc, larger = maxLogFunc)
165
166 ndimPlusTwoOverTwo = (self%ndim + 2) / 2
167 maxLogFunc = maxval(LogFunc(1:2))
168
169 do idim = 3, ndimPlusTwoOverTwo
170 LogFunc(idim) = getLogLogGammaPDF(logNormFacPDF = logNormFacPDF_logGamma, shape = 1._RK, x = Point(idim) - 10._RK)
171 if (maxLogFunc < LogFunc(idim)) maxLogFunc = LogFunc(idim)
172 end do
173 do idim = ndimPlusTwoOverTwo + 1, self%ndim
174 LogFunc(idim) = genNormLogPDF(point = Point(idim) - 10._RK)
175 if (maxLogFunc < LogFunc(idim)) maxLogFunc = LogFunc(idim)
176 end do
177 logval = genLogSumExp(LogFunc, maxLogFunc)
178 !write(*,*) maxLogFunc, maxval(LogFunc)
179 !read(*,*)
180
181 end function logpdf
