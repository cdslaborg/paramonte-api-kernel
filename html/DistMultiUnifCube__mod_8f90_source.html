115 function construct(ndim, SupportLower, SupportUpper, Err) result(self)
116
117 use Matrix_mod, only: getInvMatFromChoLow
118 use Matrix_mod, only: genDiagMat
119 use Matrix_mod, only: getChoLow
120 implicit none
121
122 integer(IK) , intent(in) :: ndim
123 real(RK) , intent(in) , optional :: SupportLower(ndim)
124 real(RK) , intent(in) , optional :: SupportUpper(ndim)
125 type(Err_type) , intent(out) , optional :: Err
126 type(DistMultiUnifCube_type) :: self
127
128 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@construct()"
129
130 real(RK) , parameter :: INVERSE_SQRT_TWELVE = sqrt(1._RK / 12._RK)
131 character(:), allocatable :: msg
132 integer(IK) :: i, j
133
134 self%Support = DomainCube_type(ndim, LimitLower = SupportLower, LimitUpper = SupportUpper, Err = Err)
135 if (present(Err)) then; if (Err%occurred) return; end if
136
137 self%ndim = ndim
138
139 ! Set mean vector.
140
141 self%Mean = self%Support%Limit%Lower + 0.5_RK * self%Support%Size
142
143 ! Set the standard deviation and covariance matrix.
144
145 self%Std = INVERSE_SQRT_TWELVE * (self%Support%Limit%Upper - self%Support%Limit%Lower)
146 self%ChoLowCovUpp = genDiagMat(ndim, ndim, Diag = self%Std**2)
147
148 ! Compute the Cholesky factorization of the covariance matrix.
149
150 allocate(self%ChoDia(ndim))
151 call getChoLow(ndim, self%ChoLowCovUpp, self%ChoDia)
152 if (self%ChoDia(1) < 0._RK) then
153 msg = PROCEDURE_NAME//": The Cholesky factorization of the covariance matrix of the distribution failed."
154 if (present(Err)) then
155 Err%occurred = .true.
156 Err%msg = msg
157 else
158 write(*,"(*(g0,:,' '))") msg
159 error stop
160 end if
161 end if
162
163 ! Compute inverse covariance matrix and its determinant.
164
165 self%logDetInvCovMat = sum(log(self%ChoDia))
166 ! \todo
167 ! This call is not really needed since the covariance matrix of uniform cubical distribution is diagonal.
168 self%InvCovMat = getInvMatFromChoLow(ndim, self%ChoLowCovUpp, self%ChoDia)
169
170 ! Set up other properties of the distribution.
171
172 self%logpdfval = -self%Support%logVol
173
174 end function construct
