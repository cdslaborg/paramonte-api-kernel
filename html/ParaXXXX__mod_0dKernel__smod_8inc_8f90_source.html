109#if CAF_ENABLED
110 real(RK) , allocatable :: co_AccRate(:)[:]
111 real(RK) , allocatable :: co_LogFuncState(:,:)[:] ! (0:nd,-1:proposalDelayedRejectionCount), -1 corresponds to the current accepted state
112 integer(IK) , save :: co_proposalFound_samplerUpdateOccurred(2)[*] ! merging these scalars would reduce the MPI communication overhead cost: co_proposalFound, co_samplerUpdateOccurred, co_counterDRS, 0 means false, 1 means true
113#else
114 real(RK) , allocatable :: co_LogFuncState(:,:)
115 real(RK) , allocatable :: co_AccRate(:)

117#endif

119 integer(IK) :: numFunCallAcceptedLastAdaptation ! number of function calls accepted at Last proposal adaptation occurrence
120 integer(IK) :: counterAUP ! counter for proposalAdaptationPeriod
121 integer(IK) :: counterAUC ! counter for proposalAdaptationCount
122 integer(IK) :: counterDRS ! counter for Delayed Rejection Stages
123 integer(IK) :: lastStateWeight ! This is used for passing the most recent verbose chain segment to the adaptive updater of the sampler
124 integer(IK) :: currentStateWeight ! counter for SampleWeight, used only in in restart mode
125 integer(IK) :: numFunCallAcceptedPlusOne ! counter for SampleWeight, used only in in restart mode
126 real(RK) :: uniformRnd ! used for random number generation.
127 real(RK) :: meanAccRateSinceStart ! used for restart file read. xxx \todo: this could be merged with self%Chain%MeanAccRate.
128 real(RK) :: logFuncDiff ! The difference between the log of the old and the new states. Used to avoid underflow.
129 !real(RK) :: adaptationMeasureDummy
130 real(RK) :: maxLogFuncRejectedProposal
131 logical :: samplerUpdateIsGreedy
132 logical :: samplerUpdateSucceeded
133 logical :: delayedRejectionRequested
134 logical :: noDelayedRejectionRequested
135 real(RK) , allocatable :: AdaptationMeasure(:), adaptationMeasurePlaceHolder(:)
136 integer(IK) :: adaptationMeasureLen
137 integer(IK) :: acceptedRejectedDelayedUnusedRestartMode
138 integer(IK) :: imageID, dumint
139 integer(IK) :: nd
140#if CAF_ENABLED || MPI_ENABLED
141 integer(IK) :: imageStartID, imageEndID
142 integer(IK) :: proposalFoundSinglChainMode ! used in singlChain delayed rejection. zero if the proposal is not accepted. 1 if the proposal is accepted.
143#if MPI_ENABLED
144 integer(IK) :: proposalFoundSinglChainModeReduced ! the reduced value by summing proposalFoundSinglChainModeReduced over all images.
145 real(RK) , allocatable :: AccRateMatrix(:,:) ! matrix of size (-1:self%SpecDRAM%ProposalDelayedRejectionCount%val,1:self%Image%count)
146 integer(IK) :: ndPlusOne
147 integer(IK) :: ierrMPI
148 integer(IK) :: proposalDelayedRejectionCountPlusTwo
149#endif
150 self%Stat%avgCommTimePerFunCall = 0._RK ! Until the reporting time, this is in reality, sumCommTimePerFunCall. This is meaningful only in singlChain parallelism
151 self%Stat%NumFunCall%acceptedRejectedDelayedUnused = self%Image%count ! used only in singlChain parallelism, and relevant only on the first image
152#endif
153 acceptedRejectedDelayedUnusedRestartMode = 0_IK ! used to compute more accurate timings in the restart mode
154 self%Stat%avgTimePerFunCalInSec = 0._RK
155 nd = self%nd%val
156
157 if (allocated(co_AccRate)) deallocate(co_AccRate)
158 if (allocated(co_LogFuncState)) deallocate(co_LogFuncState)
159#if CAF_ENABLED
160 allocate(co_LogFuncState(0:nd,-1:self%SpecDRAM%ProposalDelayedRejectionCount%val)[*])
161 allocate(co_AccRate(-1:self%SpecDRAM%ProposalDelayedRejectionCount%val)[*]) ! the negative element will contain counterDRS
162#else
163 allocate(co_LogFuncState(0:nd,-1:self%SpecDRAM%ProposalDelayedRejectionCount%val))
164 allocate(co_AccRate(-1:self%SpecDRAM%ProposalDelayedRejectionCount%val)) ! the negative element will contain counterDRS
165#endif
166 co_AccRate(-1) = 0._RK ! the real-value counterDRS, indicating the initial delayed rejection stage at which the first point is sampled
167 co_AccRate(0) = 1._RK ! initial acceptance rate for the first zeroth DR stage.
168 co_AccRate(1:self%SpecDRAM%ProposalDelayedRejectionCount%val) = 0._RK ! indicates the very first proposal acceptance on image 1
169
170#if MPI_ENABLED
171 if (allocated(AccRateMatrix)) deallocate(AccRateMatrix)
172 allocate(AccRateMatrix(-1:self%SpecDRAM%ProposalDelayedRejectionCount%val,1:self%Image%count)) ! the negative element will contain counterDRS
173 AccRateMatrix = 0._RK ! -huge(1._RK) ! debug
174 AccRateMatrix(0,1:self%Image%count) = 1._RK ! initial acceptance rate for the first zeroth DR stage.
175 ndPlusOne = nd + 1_IK
176 proposalDelayedRejectionCountPlusTwo = self%SpecDRAM%ProposalDelayedRejectionCount%val + 2_IK
177#endif
178
179 delayedRejectionRequested = self%SpecDRAM%ProposalDelayedRejectionCount%val > 0_IK
180 noDelayedRejectionRequested = .not. delayedRejectionRequested
181 if (delayedRejectionRequested) then
182 self%Stat%NumFunCall%acceptedRejectedDelayed = 0_IK ! Markov Chain counter
183 SumAccRateSinceStart%acceptedRejectedDelayed = 0._RK ! sum of acceptance rate
184 end if
185
186 !adaptationMeasure = 0._RK ! needed for the first output
187 samplerUpdateSucceeded = .true. ! needed to set up lastStateWeight and numFunCallAcceptedLastAdaptation for the first accepted proposal
188 SumAccRateSinceStart%acceptedRejected = 0._RK ! sum of acceptance rate
189 counterAUC = 0_IK ! counter for pproposalAdaptationCount.
190 counterAUP = 0_IK ! counter for proposalAdaptationPeriod.
191 self%Stat%NumFunCall%accepted = 0_IK ! Markov Chain acceptance counter.
192 self%Stat%NumFunCall%acceptedRejected = 0_IK ! Markov Chain counter
193 numFunCallAcceptedLastAdaptation = 0_IK
194 lastStateWeight = -huge(lastStateWeight)
195 meanAccRateSinceStart = 1._RK ! needed for the first restart output in fresh run.
196
197 call self%Timer%tic()
198
199 blockDryRunSetup: if (self%isFreshRun) then
200
201 adaptationMeasureLen = 100_IK
202
203 allocate(self%Chain%ProcessID( self%SpecMCMC%OutputChainSize%val))
204 allocate(self%Chain%DelRejStage( self%SpecMCMC%OutputChainSize%val))
205 allocate(self%Chain%MeanAccRate( self%SpecMCMC%OutputChainSize%val))
206 allocate(self%Chain%Adaptation( self%SpecMCMC%OutputChainSize%val))
207 allocate(self%Chain%BurninLoc( self%SpecMCMC%OutputChainSize%val))
208 allocate(self%Chain%Weight( self%SpecMCMC%OutputChainSize%val))
209 allocate(self%Chain%State(nd,self%SpecMCMC%OutputChainSize%val))
210 allocate(self%Chain%LogFunc( self%SpecMCMC%OutputChainSize%val))
211
212 else blockDryRunSetup
213
214 ! load the existing Chain file into self%Chain components
215
216 call self%Chain%read( chainFilePath = self%ChainFile%Path%original & ! LCOV_EXCL_LINE
217 , chainFileForm = self%SpecBase%OutputChainFileFormat%val & ! LCOV_EXCL_LINE
218 , targetChainSize = self%SpecMCMC%OutputChainSize%val & ! LCOV_EXCL_LINE
219 , delimiter = self%SpecBase%OutputDelimiter%val & ! LCOV_EXCL_LINE
220 , lenHeader = self%Chain%lenHeader & ! LCOV_EXCL_LINE
221 , Err = self%Err & ! LCOV_EXCL_LINE
222 , ndim = nd & ! LCOV_EXCL_LINE
223 )
224 if (self%Err%occurred) then
225 ! LCOV_EXCL_START
226 self%Err%msg = PROCEDURE_NAME//self%Err%msg
227 call self%abort( Err = self%Err, prefix = self%brand, newline = NLC, outputUnit = self%LogFile%unit )
228 return
229 ! LCOV_EXCL_STOP
230 end if
231
232 if (self%Chain%Size%compact<=CHAIN_RESTART_OFFSET) then
233 self%isFreshRun = .true.
234 self%isDryRun = .not. self%isFreshRun
235 end if
236
237 call self%Image%sync()
238
239 blockLeaderSetup: if (self%Image%isLeader) then
240
241 ! set up the chain file
242
243 block
244
245 use System_mod, only: RandomFileName_type, copyFile, removeFile
246 type(RandomFileName_type) :: RFN
247
248 ! create a copy of the chain file, just for the sake of not losing the simulation results
249
250 RFN = RandomFileName_type(dir = "", key = self%ChainFile%Path%original//".rst", ext="") ! temporary_restart_copy
251 call copyFile(pathOld=self%ChainFile%Path%original,pathNew=RFN%path,isUnixShell=self%OS%Shell%isUnix,Err=self%err)
252 if (self%Err%occurred) then
253 ! LCOV_EXCL_START
254 self%Err%msg = PROCEDURE_NAME//self%Err%msg
255 call self%abort( Err = self%Err, prefix = self%brand, newline = NLC, outputUnit = self%LogFile%unit )
256 exit blockLeaderSetup
257 return
258 ! LCOV_EXCL_STOP
259 end if
260
261 ! reopen the chain file to resume the simulation
262
263 open( newunit = self%ChainFile%unit & ! LCOV_EXCL_LINE
264 , file = self%ChainFile%Path%original & ! LCOV_EXCL_LINE
265 , form = self%ChainFile%Form%value & ! LCOV_EXCL_LINE
266 , status = self%ChainFile%status & ! LCOV_EXCL_LINE
267 , iostat = self%ChainFile%Err%stat & ! LCOV_EXCL_LINE
268#if INTEL_COMPILER_ENABLED && WINDOWS_ENABLED
269 , SHARED & ! LCOV_EXCL_LINE
270#endif
271 , position = self%ChainFile%Position%value )
272 self%Err = self%ChainFile%getOpenErr(self%ChainFile%Err%stat)
273 if (self%Err%occurred) then
274 ! LCOV_EXCL_START
275 self%Err%msg = PROCEDURE_NAME//": Error occurred while opening "//self%name//" "//self%ChainFile%suffix//" file='"//self%ChainFile%Path%original//"'."
276 call self%abort( Err = self%Err, prefix = self%brand, newline = NLC, outputUnit = self%LogFile%unit )
277 exit blockLeaderSetup
278 return
279 ! LCOV_EXCL_STOP
280 end if
281
282 ! rewrite the chain file
283
284 call self%Chain%write ( ndim = nd & ! LCOV_EXCL_LINE
285 , compactStartIndex = 1_IK & ! LCOV_EXCL_LINE
286 , compactEndIndex = self%Chain%Size%compact - CHAIN_RESTART_OFFSET & ! LCOV_EXCL_LINE
287 , chainFileUnit = self%ChainFile%unit & ! LCOV_EXCL_LINE
288 , chainFileForm = self%SpecBase%OutputChainFileFormat%val & ! LCOV_EXCL_LINE
289 , chainFileFormat = self%ChainFile%format & ! LCOV_EXCL_LINE
290 , proposalAdaptationPeriod = self%SpecDRAM%ProposalAdaptationPeriod%val & ! LCOV_EXCL_LINE
291 )
292
293 ! remove the temporary copy of the chain file
294
295 call removeFile(RFN%path, self%Err) ! Passing the optional Err argument, handles exceptions should any occur.
296
297 end block
298
299 adaptationMeasureLen = maxval(self%Chain%Weight(1:self%Chain%Size%compact-CHAIN_RESTART_OFFSET))
300
301 end if blockLeaderSetup
302
303#if (MPI_ENABLED || CAF_ENABLED) && (CODECOV_ENABLED || SAMPLER_TEST_ENABLED)
304 block; use Err_mod, only: bcastErr; call bcastErr(self%Err); end block
305#endif
306 if (self%Err%occurred) return
307
308 end if blockDryRunSetup
309
310 if (self%Image%isLeader) then
311 if (allocated(AdaptationMeasure)) deallocate(AdaptationMeasure); allocate(AdaptationMeasure(adaptationMeasureLen))
312 end if
313
314 if (self%isFreshRun) then ! this must be done separately from the above blockDryRunSetup
315
316 self%Chain%BurninLoc(1) = 1_IK
317
318 co_LogFuncState(1:nd,0) = self%SpecMCMC%ProposalStartPoint%Val ! proposal state
319 call self%Timer%toc()
320 co_LogFuncState(0,0) = getLogFunc( nd, co_LogFuncState(1:nd,0) ) ! proposal logFunc
321 call self%Timer%toc(); self%Stat%avgTimePerFunCalInSec = self%Stat%avgTimePerFunCalInSec + self%Timer%Time%delta
322
323 else
324
325 co_LogFuncState(1:nd,0) = self%Chain%State(1:nd,1) ! proposal state
326 co_LogFuncState(0,0) = self%Chain%LogFunc(1) ! proposal logFunc
327
328 end if
329
330 co_LogFuncState(0:nd,-1) = co_LogFuncState(0:nd,0) ! set current logFunc and State equal to the first proposal
331 self%Stat%LogFuncMode%val = -huge(self%Stat%LogFuncMode%val)
332 self%Stat%LogFuncMode%loc = 0_IK
333
334#if CAF_ENABLED || MPI_ENABLED
335 if (self%SpecBase%Parallelism%isSinglChain) then
336 imageStartID = 1_IK
337 imageEndID = self%Image%count
338 else ! isMultiChain
339 imageStartID = self%Image%id
340 imageEndID = self%Image%id
341 end if
342#else
343 imageID = 1_IK ! needed even in the case of serial run to assign a proper value to self%Chain%ProcessID
344#endif
345
346 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% start of loopMarkovChain %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
347 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
348 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
349
350 loopMarkovChain: do
351
352 co_proposalFound_samplerUpdateOccurred(2) = 0_IK ! at each iteration assume no samplerUpdateOccurred, unless it occurs
353
354#if CAF_ENABLED || MPI_ENABLED
355 blockLeaderImage: if (self%Image%isLeader) then
356#endif
357
358 co_proposalFound_samplerUpdateOccurred(1) = 0_IK ! co_proposalFound = .false.
359 samplerUpdateIsGreedy = counterAUC < self%SpecDRAM%ProposalAdaptationCountGreedy%val
360
361#if CAF_ENABLED || MPI_ENABLED
362 loopOverImages: do imageID = imageStartID, imageEndID
363#if CAF_ENABLED
364 call self%Timer%toc()
365 if (imageID/=self%Image%id) co_AccRate(-1:self%SpecDRAM%ProposalDelayedRejectionCount%val) = co_AccRate(-1:self%SpecDRAM%ProposalDelayedRejectionCount%val)[imageID] ! happens only in isSinglChain=TRUE
366 call self%Timer%toc(); self%Stat%avgCommTimePerFunCall = self%Stat%avgCommTimePerFunCall + self%Timer%Time%delta
367#elif MPI_ENABLED
368 if (imageID/=self%Image%id) co_AccRate(-1:self%SpecDRAM%ProposalDelayedRejectionCount%val) = AccRateMatrix(-1:self%SpecDRAM%ProposalDelayedRejectionCount%val,imageID) ! happens only in isSinglChain=TRUE
369#endif
370#endif
371
372 counterDRS = nint(co_AccRate(-1),kind=IK)
373 if (counterDRS > -1_IK) co_proposalFound_samplerUpdateOccurred(1) = 1_IK ! co_proposalFound = .true.
374
375 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% blockProposalAccepted %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
376 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
377
378 ! On the very first iteration, this block is (and must be) executed for imageID==1,
379 ! since it is for the first (starting) point, which is assumed to have been accepted
380 ! as the first point by the first coarray imageID.
381
382 blockProposalAccepted: if ( co_proposalFound_samplerUpdateOccurred(1) == 1_IK ) then ! co_proposalAccepted = .true.
383
384 currentStateWeight = 0_IK
385
386 ! communicate the accepted logFunc and State from the winning image to the leader/all images: co_LogFuncState
387#if MPI_ENABLED
388 if (self%SpecBase%Parallelism%isSinglChain) then
389 call self%Timer%toc()
390 ! broadcast winning image to all processes
391 call mpi_bcast ( imageID & ! buffer
392 , 1 & ! count
393 , mpi_integer & ! datatype
394 , 0 & ! root: broadcasting rank
395 , mpi_comm_world & ! comm
396 , ierrMPI & ! ierr
397 )
398 ! broadcast co_LogFuncState from the winning image to all others
399 call mpi_bcast ( co_LogFuncState(0:nd,-1) & ! buffer
400 , ndPlusOne & ! count
401 , mpi_double_precision & ! datatype
402 , imageID - 1_IK & ! root: broadcasting rank
403 , mpi_comm_world & ! comm
404 , ierrMPI & ! ierr
405 )
406 call self%Timer%toc(); self%Stat%avgCommTimePerFunCall = self%Stat%avgCommTimePerFunCall + self%Timer%Time%delta
407 end if
408#elif CAF_ENABLED
409 if (imageID/=self%Image%id) then ! Avoid remote connection for something that is available locally.
410 call self%Timer%toc()
411 co_LogFuncState(0:nd,-1) = co_LogFuncState(0:nd,-1)[imageID]
412 call self%Timer%toc(); self%Stat%avgCommTimePerFunCall = self%Stat%avgCommTimePerFunCall + self%Timer%Time%delta
413 end if
414#endif
415 ! Note: after every adaptive update of the sampler, counterAUP is reset to 0.
416 if (counterAUP==0_IK .and. samplerUpdateSucceeded) then
417 numFunCallAcceptedLastAdaptation = numFunCallAcceptedLastAdaptation + 1_IK
418 lastStateWeight = 0_IK
419 end if
420
421 blockFreshDryRun: if (self%isFreshRun) then
422
423 call writeOutput()
424
425 self%Stat%NumFunCall%accepted = self%Stat%NumFunCall%accepted + 1_IK
426
427 self%Chain%ProcessID(self%Stat%NumFunCall%accepted) = imageID
428 self%Chain%DelRejStage(self%Stat%NumFunCall%accepted) = counterDRS
429 self%Chain%Adaptation(self%Stat%NumFunCall%accepted) = 0._RK ! adaptationMeasure
430 self%Chain%Weight(self%Stat%NumFunCall%accepted) = 0_IK
431 self%Chain%LogFunc(self%Stat%NumFunCall%accepted) = co_LogFuncState(0,-1)
432 self%Chain%State(1:nd,self%Stat%NumFunCall%accepted) = co_LogFuncState(1:nd,-1)
433
434 ! find the burnin point
435
436 self%Chain%BurninLoc(self%Stat%NumFunCall%accepted) = getBurninLoc( lenLogFunc = self%Stat%NumFunCall%accepted &
437 , refLogFunc = self%Stat%LogFuncMode%val &
438 , LogFunc = self%Chain%LogFunc(1:self%Stat%NumFunCall%accepted) &
439 )
440
441 else blockFreshDryRun ! in restart mode: determine the correct value of co_proposalFound_samplerUpdateOccurred(1)
442
443 numFunCallAcceptedPlusOne = self%Stat%NumFunCall%accepted + 1_IK
444 if (numFunCallAcceptedPlusOne==self%Chain%Size%compact) then
445 self%isFreshRun = .true.
446 call writeOutput()
447 self%isDryRun = .not. self%isFreshRun
448 self%Chain%Weight(numFunCallAcceptedPlusOne) = 0_IK
449 SumAccRateSinceStart%acceptedRejected = self%Chain%MeanAccRate(self%Stat%NumFunCall%accepted) * real(self%Stat%NumFunCall%acceptedRejected,kind=RK)
450 if (delayedRejectionRequested) SumAccRateSinceStart%acceptedRejectedDelayed = self%Chain%MeanAccRate(self%Stat%NumFunCall%accepted) * real(self%Stat%NumFunCall%acceptedRejectedDelayed,kind=RK)
451 end if
452 self%Stat%NumFunCall%accepted = numFunCallAcceptedPlusOne
453 numFunCallAcceptedPlusOne = self%Stat%NumFunCall%accepted + 1_IK
454
455 end if blockFreshDryRun
456
457 if (self%Stat%LogFuncMode%val < self%Chain%LogFunc(self%Stat%NumFunCall%accepted)) then
458 self%Stat%LogFuncMode%val = self%Chain%LogFunc(self%Stat%NumFunCall%accepted)
459 self%Stat%LogFuncMode%loc = self%Stat%NumFunCall%accepted
460 end if
461
462 SumAccRateSinceStart%acceptedRejected = SumAccRateSinceStart%acceptedRejected + co_AccRate(counterDRS)
463
464 else blockProposalAccepted
465
466 counterDRS = self%SpecDRAM%ProposalDelayedRejectionCount%val
467 SumAccRateSinceStart%acceptedRejected = SumAccRateSinceStart%acceptedRejected + co_AccRate(counterDRS)
468
469 end if blockProposalAccepted
470
471 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
472 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% blockProposalAccepted %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
473
474 counterAUP = counterAUP + 1_IK
475 self%TimeFile%counterPRP = self%TimeFile%counterPRP + 1_IK
476
477 ! It is critical for this if block to occur before updating `self%Stat%NumFunCall%acceptedRejected` otherwise,
478 ! `numFunCallAcceptedRejectedLastReport` will be updated to `self%Stat%NumFunCall%acceptedRejected`
479 ! which can lead to "Floating-point exception - erroneous arithmetic operation" in the computation
480 ! of `inverseOutputProgressReportPeriod` when `blockLastSample` happens to be activated.
481 if (self%TimeFile%counterPRP == self%SpecBase%OutputProgressReportPeriod%val) then
482 call self%Timer%toc()
483 call self%reportProgress(timeElapsed = self%Timer%Time%total, timeLeft = getRemainingSimulationFraction() * self%Timer%Time%total)
484 end if
485
486 currentStateWeight = currentStateWeight + 1_IK
487 self%Stat%NumFunCall%acceptedRejected = self%Stat%NumFunCall%acceptedRejected + 1_IK
488
489 if (delayedRejectionRequested) then
490 SumAccRateSinceStart%acceptedRejectedDelayed = SumAccRateSinceStart%acceptedRejectedDelayed + sum(co_AccRate(0:counterDRS))
491 self%Stat%NumFunCall%acceptedRejectedDelayed = self%Stat%NumFunCall%acceptedRejectedDelayed + counterDRS + 1_IK
492 end if
493
494 if (self%isFreshRun) then ! these are used for adaptive proposal updating, so they have to be set on every accepted or rejected iteration (excluding delayed rejections)
495 self%Chain%MeanAccRate(self%Stat%NumFunCall%accepted) = SumAccRateSinceStart%acceptedRejected / real(self%Stat%NumFunCall%acceptedRejected,kind=RK)
496 self%Chain%Weight(self%Stat%NumFunCall%accepted) = self%Chain%Weight(self%Stat%NumFunCall%accepted) + 1_IK
497 if (adaptationMeasureLen<self%Chain%Weight(self%Stat%NumFunCall%accepted)) then
498 allocate(adaptationMeasurePlaceHolder(2*adaptationMeasureLen))
499 adaptationMeasurePlaceHolder(1:adaptationMeasureLen) = AdaptationMeasure
500 call move_alloc(from=adaptationMeasurePlaceHolder,to=AdaptationMeasure)
501 adaptationMeasureLen = 2_IK * adaptationMeasureLen
502 end if
503 else
504#if CAF_ENABLED || MPI_ENABLED
505 acceptedRejectedDelayedUnusedRestartMode = self%Stat%NumFunCall%acceptedRejectedDelayedUnused
506#else
507 acceptedRejectedDelayedUnusedRestartMode = self%Stat%NumFunCall%acceptedRejectedDelayed
508#endif
509 end if
510
511 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% last output write %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
