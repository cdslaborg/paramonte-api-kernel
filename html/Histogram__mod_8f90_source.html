212

218 thisXbin = getBin(X(ip),xmin,nxbin,xbinsize)
219 thisYbin = getBin(Y(ip),ymin,nybin,ybinsize)
220 Density(thisYbin,thisXbin) = Density(thisYbin,thisXbin) + 1._RK
221 end if
222 end do
223
224 Xbin = Xbin + 0.5_RK * xbinsize
225 Ybin = Ybin + 0.5_RK * ybinsize
226 method = getLowerCase(trim(adjustl(histType)))
227 if(method=="pdf") then
228 Density = Density / real(npEffective,kind=RK)
229 elseif(method=="pdf(y|x)") then
230 do i = 1,nxbin
231 Density(1:nybin,i) = Density(1:nybin,i) / sum(Density(1:nybin,i))
232 end do
233 elseif(method=="pdf(x|y)") then
234 do i = 1,nybin
235 Density(i,1:nxbin) = Density(i,1:nxbin) / sum(Density(i,1:nxbin))
236 end do
