133 function constructChainFileContents(ndim,allocSize,domainAxesNameList,chainFilePath,chainFileForm,lenHeader,delimiter,chainSize) result(CFC)
134#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
135 !DEC$ ATTRIBUTES DLLEXPORT :: constructChainFileContents
136#endif
137 implicit none
138 integer(IK) , intent(in) :: ndim
139 integer(IK) , intent(in), optional :: allocSize
140 character(*), intent(in), optional :: chainFileForm
141 character(*), intent(in), optional :: domainAxesNameList(ndim)
142 character(*), intent(in), optional :: chainFilePath
143 character(*), intent(in), optional :: delimiter
144 integer(IK) , intent(in), optional :: lenHeader
145 integer(IK) , intent(in), optional :: chainSize
146 type(ChainFileContents_type) :: CFC
147 integer(IK) :: icol
148
149 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@constructChainFileContents()"
150
151 CFC%Err%occurred = .false.
152 CFC%ndim = ndim
153
154 ! Set up the chain file column header.
155
156 allocate(CFC%Header(NUM_DEF_COL + ndim))
157 do icol = 1, NUM_DEF_COL
158 CFC%Header(icol)%value = trim(adjustl(COL_HEADER_DEFAULT(icol)))
159 end do
160
161 if (present(domainAxesNameList)) then
162 do icol = NUM_DEF_COL + 1, NUM_DEF_COL + ndim
163 CFC%Header(icol)%value = trim(adjustl(domainAxesNameList(icol-NUM_DEF_COL)))
164 end do
165 end if
166
167 ! Set up other components if given.
168
169 if (present(lenHeader)) CFC%lenHeader = lenHeader
170 if (present(delimiter)) CFC%delimiter = delimiter
171 if (present(allocSize)) CFC%size = allocSize
172
173 ! Read the chain file if the path is given.
174
175 if (present(chainFilePath)) then
176 if (present(chainFileForm)) then
177 call CFC%read(chainFilePath,chainFileForm,CFC%Err,lenHeader,ndim,delimiter,allocSize,chainSize)
178 else
179 CFC%Err%occurred = .true. ! LCOV_EXCL_LINE
180 CFC%Err%msg = PROCEDURE_NAME//"Internal Error occurred: When `chainFilePath` is present, chainFileForm must be also present." ! LCOV_EXCL_LINE
181 end if
182 else
183 if (present(allocSize)) call CFC%allocate(ndim,allocSize)
184 end if
185 if (CFC%Err%occurred) return ! LCOV_EXCL_LINE
186
187 end function constructChainFileContents

254 subroutine readChainFileContents(CFC,chainFilePath,chainFileForm,Err,lenHeader,ndim,delimiter,chainSize,allocSize)
255#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
256 !DEC$ ATTRIBUTES DLLEXPORT :: readChainFileContents
257#endif
258 use FileContents_mod, only: getNumRecordInFile
259 use Constants_mod, only: IK, RK, NLC, NEGINF_IK, NEGINF_RK
260 use String_mod, only: StrSplit_type, getLowerCase, val2char
261
262 implicit none
263
264 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME // "@readChainFileContents()"
265
266 class(ChainFileContents_type), intent(inout) :: CFC
267 character(*) , intent(in) :: chainFilePath
268 character(*) , intent(in) :: chainFileForm
269 type(Err_type) , intent(out) :: Err
270 character(*) , intent(in), optional :: delimiter
271 integer(IK) , intent(in), optional :: lenHeader, ndim, allocSize, chainSize
272 character(:) , allocatable :: chainFilePathTrimmed, thisForm
273 character(:) , allocatable :: chainFileFormLowerCase
274 type(StrSplit_type) :: Record
275 integer(IK) :: chainFileUnit, i, istate, isize, delimiterLen, chainSizeDefault
276 logical :: fileExists, fileIsOpen, delimHasBegun, delimHasEnded
277 logical :: isBinary
278 logical :: isAscii
279
280 Err%occurred = .false.
281 chainFilePathTrimmed = trim(adjustl(chainFilePath))
282 inquire(file=chainFilePathTrimmed,exist=fileExists,opened=fileIsOpen,number=chainFileUnit)
283
284 blockFileExistence: if (fileExists) then
285
286 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
287 ! set up chain file format
288 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
289
290 isAscii = .false.
291 isBinary = .false.
292 chainFileFormLowerCase = getLowerCase(chainFileForm)
293 if (chainFileFormLowerCase == "binary") then
294 isBinary = .true.
295 elseif (chainFileFormLowerCase == "ascii" .or. chainFileFormLowerCase == "compact" .or. chainFileFormLowerCase == "verbose") then
296 isAscii = .true.
297 else
298 Err%occurred = .true. ! LCOV_EXCL_LINE
299 Err%msg = PROCEDURE_NAME//": Unrecognized chain file form: "//chainFileForm ! LCOV_EXCL_LINE
300 return ! LCOV_EXCL_LINE
301 end if
302
303 if (isBinary) then
304 thisForm = "unformatted"
305 if (.not. present(ndim) .or. .not. present(lenHeader) .or. .not. present(delimiter)) then
306 Err%occurred = .true. ! LCOV_EXCL_LINE
307 Err%msg = PROCEDURE_NAME//": If the chain file is in binary form, chainSize, lenHeader, delimiter, and ndim must be provided by the user." ! LCOV_EXCL_LINE
308 return ! LCOV_EXCL_LINE
309 end if
310 else
311 thisForm = "formatted"
312 end if
313
314 ! If file is open, close the file.
315
316 if (fileIsOpen) then
317 if (chainFileUnit == -1_IK) then
318 Err%occurred = .true. ! LCOV_EXCL_LINE
319 Err%msg = PROCEDURE_NAME//": The file located at: "//chainFilePathTrimmed//NLC//"is open, but no unit is connected to the file."//NLC ! LCOV_EXCL_LINE
320 return ! LCOV_EXCL_LINE
321 else
322 close(chainFileUnit)
323 end if
324 end if
325
326 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
327 ! Get the number of records in file, minus header line.
328 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
329
330 blockSetOrGetChainSize: if (present(chainSize)) then
331
332 chainSizeDefault = chainSize
333
334 else blockSetOrGetChainSize
335
336 blockGetNumRecordBinaryAscii: if (isBinary) then
337
338 open( newunit = chainFileUnit & ! LCOV_EXCL_LINE
339 , file = chainFilePathTrimmed & ! LCOV_EXCL_LINE
340 , status = "old" & ! LCOV_EXCL_LINE
341 , form = thisForm & ! LCOV_EXCL_LINE
342 , iostat = Err%stat & ! LCOV_EXCL_LINE
343#if INTEL_COMPILER_ENABLED && WINDOWS_ENABLED
344 , SHARED & ! LCOV_EXCL_LINE
345#endif
346 )
347 if (Err%stat /= 0_IK) then
348 Err%occurred = .true. ! LCOV_EXCL_LINE
349 Err%msg = PROCEDURE_NAME//": Unable to open the file located at: "//chainFilePathTrimmed//NLC ! LCOV_EXCL_LINE
350 return ! LCOV_EXCL_LINE
351 end if
352
353 if (allocated(Record%value)) deallocate(Record%value)
354 allocate( character(lenHeader) :: Record%value )
355 read(chainFileUnit) Record%value
356 block
357 integer(IK) :: ProcessID ! LCOV_EXCL_LINE
358 real(RK) :: MeanAccRate ! LCOV_EXCL_LINE
359 real(RK) :: LogDomainSize! LCOV_EXCL_LINE
360 real(RK) :: LogIntegral ! LCOV_EXCL_LINE
361 real(RK) :: LogMaxRelErr ! LCOV_EXCL_LINE
362 real(RK) :: LogWeight ! LCOV_EXCL_LINE
363 real(RK) :: LogFunc ! LCOV_EXCL_LINE
364 real(RK), allocatable :: State(:) ! LCOV_EXCL_LINE
365
366 if (allocated(State)) deallocate(State); allocate(State(ndim)) ! takes care of potential GFortran bugs.
367 chainSizeDefault = 0_IK
368 loopFindchainSize: do
369 read(chainFileUnit,iostat=Err%stat) ProcessID, MeanAccRate, LogDomainSize, LogIntegral, LogMaxRelErr, LogWeight, LogFunc, State
370 if (Err%stat==0_IK) then
371 chainSizeDefault = chainSizeDefault + 1_IK
372 elseif (is_iostat_end(Err%stat)) then
373 exit loopFindchainSize
374 elseif (is_iostat_eor(Err%stat)) then
375 Err%occurred = .true. ! LCOV_EXCL_LINE
376 Err%msg = PROCEDURE_NAME//": Incomplete record detected while reading the input binary chain file at: "//chainFilePathTrimmed//NLC ! LCOV_EXCL_LINE
377 return ! LCOV_EXCL_LINE
378 else
379 Err%occurred = .true. ! LCOV_EXCL_LINE
380 Err%msg = PROCEDURE_NAME//": IO error occurred while reading the input binary chain file at: "//chainFilePathTrimmed//NLC ! LCOV_EXCL_LINE
381 return ! LCOV_EXCL_LINE
382 end if
383 end do loopFindchainSize
384 end block
385 close(chainFileUnit)
386
387 else blockGetNumRecordBinaryAscii ! is not binary
388
389 call getNumRecordInFile(chainFilePathTrimmed, chainSizeDefault, Err, exclude = "")
390 if (Err%occurred) then
391 Err%msg = PROCEDURE_NAME//Err%msg ! LCOV_EXCL_LINE
392 return ! LCOV_EXCL_LINE
393 end if
394 chainSizeDefault = chainSizeDefault - 1_IK ! subtract header
395
396 end if blockGetNumRecordBinaryAscii
397
398 end if blockSetOrGetChainSize
399
400 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
401 ! Set the number of elements in the Chain components.
402 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
403
404 if (present(allocSize)) then ! in restart mode, this must always be the case.
405 CFC%size = allocSize
406 else
407 CFC%size = chainSizeDefault
408 end if
409
410 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
411 ! Find the delimiter.
412 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
413
414 blockFindDelim: if (present(delimiter)) then
415
416 CFC%delimiter = delimiter
417
418 else blockFindDelim
419
420 if (allocated(CFC%delimiter)) deallocate(CFC%delimiter)
421 allocate( character(1023) :: CFC%delimiter )
422 if (allocated(Record%value)) deallocate(Record%value)
423 allocate( character(99999) :: Record%value )
424
425 open( newunit = chainFileUnit &
426 , file = chainFilePathTrimmed &
427 , status = "old" &
428 , form = thisForm &
429 , iostat = Err%stat &
430#if INTEL_COMPILER_ENABLED && WINDOWS_ENABLED
431 , SHARED &
432#endif
433 )
434 if (Err%stat/=0) then
435 Err%occurred = .true. ! LCOV_EXCL_LINE
436 Err%msg = PROCEDURE_NAME//": Unable to open the file located at: "//chainFilePathTrimmed//"."//NLC ! LCOV_EXCL_LINE
437 return ! LCOV_EXCL_LINE
438 end if
439
440 read(chainFileUnit,*) ! skip the header
441 read(chainFileUnit,"(A)") Record%value ! read the first numeric row in string format
442 close(chainFileUnit)
443
444 Record%value = trim(adjustl(Record%value))
445 delimHasEnded = .false.
446 delimHasBegun = .false.
447 delimiterLen = 0
448 loopSearchDelimiter: do i = 1, len(Record%value) - 1
449 if ( Record%isDigit(Record%value(i:i)) ) then
450 if (delimHasBegun) delimHasEnded = .true.
451 elseif (Record%value(i:i)=="." .or. Record%value(i:i)=="+" .or. Record%value(i:i)=="-") then
452 if (delimHasBegun) then
453 delimHasEnded = .true.
454 else
455 Err%occurred = .true. ! LCOV_EXCL_LINE
456 Err%msg = PROCEDURE_NAME//": The file located at: " // chainFilePathTrimmed //NLC//& ! LCOV_EXCL_LINE
457 "has unrecognizable format. Found "//Record%value(i:i)//" in the first column, while expecting positive integer."//NLC ! LCOV_EXCL_LINE
458 return ! LCOV_EXCL_LINE
459 end if
460 else
461 if (i==1_IK) then ! here it is assumed that the first column in chain file always contains integers
462 Err%occurred = .true. ! LCOV_EXCL_LINE
463 Err%msg = PROCEDURE_NAME//": The file located at: "//chainFilePathTrimmed//NLC//"has unrecognizable format."//NLC ! LCOV_EXCL_LINE
464 return ! LCOV_EXCL_LINE
465 else
466 delimHasBegun = .true.
467 delimiterLen = delimiterLen + 1_IK
468 CFC%delimiter(delimiterLen:delimiterLen) = Record%value(i:i)
469 end if
470 end if
471 if (delimHasEnded) exit loopSearchDelimiter
472 end do loopSearchDelimiter
473
474 if (.not.(delimHasBegun.and.delimHasEnded)) then
475 Err%occurred = .true. ! LCOV_EXCL_LINE
476 Err%msg = PROCEDURE_NAME//": The file located at: "//chainFilePathTrimmed//NLC//"has unrecognizable format. Could not identify the column delimiter."//NLC ! LCOV_EXCL_LINE
477 return ! LCOV_EXCL_LINE
478 else
479 CFC%delimiter = trim(adjustl(CFC%delimiter(1:delimiterLen)))
480 delimiterLen = len(CFC%delimiter)
481 if (delimiterLen==0_IK) then
482 CFC%delimiter = " "
483 delimiterLen = 1_IK
484 end if
485 end if
486
487 end if blockFindDelim
488
489 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
490 ! Find the number of dimensions of the state (the number of function variables).
491 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
492
493 if (present(ndim)) then
494 CFC%ndim = ndim
495 else
496 Record%Parts = Record%split(Record%value,CFC%delimiter,Record%nPart)
497 CFC%numDefCol = 0_IK
498 loopFindNumDefCol: do i = 1, Record%nPart
499 if ( index(string = Record%Parts(i)%value, substring = "SampleLogFunc" ) > 0_IK ) then
500 CFC%numDefCol = i
501 exit loopFindNumDefCol
502 end if
503 end do loopFindNumDefCol
504 if (CFC%numDefCol/=NUM_DEF_COL .or. CFC%numDefCol==0_IK) then
505 Err%occurred = .true. ! LCOV_EXCL_LINE
506 Err%msg = PROCEDURE_NAME//": Internal error occurred. CFC%numDefCol/=NUM_DEF_COL: " // val2char(CFC%numDefCol) // val2char(NUM_DEF_COL) ! LCOV_EXCL_LINE
507 return ! LCOV_EXCL_LINE
508 end if
509 CFC%ndim = Record%nPart - NUM_DEF_COL
510 end if
511
512 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
513 ! Allocate Chain components.
514 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
515
516 call CFC%allocate(CFC%ndim, CFC%size)
517
518 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
519 ! Reopen the file to read the contents.
520 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
521
522 open( newunit = chainFileUnit & ! LCOV_EXCL_LINE
523 , file = chainFilePathTrimmed & ! LCOV_EXCL_LINE
524 , status = "old" & ! LCOV_EXCL_LINE
525 , form = thisForm & ! LCOV_EXCL_LINE
526 , iostat = Err%stat & ! LCOV_EXCL_LINE
527#if INTEL_COMPILER_ENABLED && WINDOWS_ENABLED
528 , SHARED & ! LCOV_EXCL_LINE
529#endif
530 )
531 if (Err%stat /= 0_IK) then
532 Err%occurred = .true. ! LCOV_EXCL_LINE
533 Err%msg = PROCEDURE_NAME//": Unable to open the file located at: "//chainFilePathTrimmed //"."//NLC ! LCOV_EXCL_LINE
534 return ! LCOV_EXCL_LINE
535 end if
536
537 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
538 ! First read the column headers.
539 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
540
541 if (allocated(Record%value)) deallocate(Record%value) ! set up the record string that keeps the contents of each line
542 if (isBinary) then
543 allocate( character(lenHeader) :: Record%value )
544 read(chainFileUnit) Record%value
545 else
546 allocate( character(99999) :: Record%value ) ! such huge allocation is rather redundant and is good for a ~4000 dimensional objective function.
547 read(chainFileUnit, "(A)" ) Record%value
548 end if
549 CFC%Header = Record%split(trim(adjustl(Record%value)), CFC%delimiter, Record%npart)
550 !write(*,*) (CFC%Header(i)%value, i = 1, NUM_DEF_COL + CFC%ndim)
551
552 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
553 ! Read the chain.
554 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
555
556 isize = 0_IK
557 istate = 0_IK
558 CFC%maxLogWeight = NEGINF_RK
559 blockReadChainBinaryAscii: if (isBinary) then
560
561 loopReadBinary: do
562 isize = isize + 1_IK
563 istate = istate + 1_IK
564 if (istate > chainSizeDefault) exit loopReadBinary
565 if (isize > CFC%size) isize = 1_IK
566 read(chainFileUnit, iostat=Err%stat ) CFC%ProcessID(isize) & ! LCOV_EXCL_LINE
567 , CFC%MeanAccRate(isize) & ! LCOV_EXCL_LINE
568 , CFC%LogDomainSize(isize) & ! LCOV_EXCL_LINE
569 , CFC%LogIntegral(isize) & ! LCOV_EXCL_LINE
570 , CFC%LogMaxRelErr(isize) & ! LCOV_EXCL_LINE
571 , CFC%LogWeight(isize) & ! LCOV_EXCL_LINE
572 , CFC%LogFunc(isize) & ! LCOV_EXCL_LINE
573 , CFC%State(1:CFC%ndim,isize)
574 if (CFC%maxLogWeight < CFC%LogWeight(isize)) CFC%maxLogWeight = CFC%LogWeight(isize)
575 if (is_iostat_eor(Err%stat) .or. is_iostat_end(Err%stat)) then
576 call warnUserAboutCorruptChainFile(istate) ! LCOV_EXCL_LINE
577 exit loopReadBinary ! LCOV_EXCL_LINE
578 end if
579 end do loopReadBinary
580
581 elseif (isAscii) then blockReadChainBinaryAscii
582
583 associate(numColTot => CFC%numDefCol + CFC%ndim)
584 loopReadAscii: do
585 isize = isize + 1_IK
586 istate = istate + 1_IK
587 if (istate > chainSizeDefault) exit loopReadAscii
588 if (isize > CFC%size) isize = 1_IK
589 read(chainFileUnit, "(A)", iostat = Err%stat) Record%value
590 if (is_iostat_eor(Err%stat) .or. is_iostat_end(Err%stat)) then
591 call warnUserAboutCorruptChainFile(istate) ! LCOV_EXCL_LINE
592 !write(*,*) (CFC%Header(i)%value, i = 1, NUM_DEF_COL + CFC%ndim)
593 !write(*,*) "chainSizeDefault = ", chainSizeDefault
594 !write(*,*) "chainSize = ", chainSize
595 exit loopReadAscii ! LCOV_EXCL_LINE
596 end if
597 Record%Parts = Record%split(trim(adjustl(Record%value)),CFC%delimiter,Record%nPart)
598 if (Record%nPart<numColTot) then
599 call warnUserAboutCorruptChainFile(istate) ! LCOV_EXCL_LINE
600 exit loopReadAscii ! LCOV_EXCL_LINE
601 else
602 read(Record%Parts(1)%value,*) CFC%ProcessID(isize)
603 read(Record%Parts(2)%value,*) CFC%MeanAccRate(isize)
604 read(Record%Parts(3)%value,*) CFC%LogDomainSize(isize)
605 read(Record%Parts(4)%value,*) CFC%LogIntegral(isize)
606 read(Record%Parts(5)%value,*) CFC%LogMaxRelErr(isize)
607 read(Record%Parts(6)%value,*) CFC%LogWeight(isize)
608 read(Record%Parts(7)%value,*) CFC%LogFunc(isize)
609 do i = 1, CFC%ndim
610 read(Record%Parts(NUM_DEF_COL+i)%value,*) CFC%State(i,isize)
611 end do
612 if (CFC%maxLogWeight < CFC%LogWeight(isize)) CFC%maxLogWeight = CFC%LogWeight(isize)
613 end if
614 end do loopReadAscii
615 end associate
616
617 end if blockReadChainBinaryAscii
618
619 close(chainFileUnit)
620
621 ! Set the rest of elements to null values.
622
623 if (isize <= CFC%size) call CFC%nullify(startIndex = isize, endIndex = CFC%size)
624
625 else blockFileExistence
626
627 Err%occurred = .true. ! LCOV_EXCL_LINE
628 Err%msg = PROCEDURE_NAME//": The chain file does not exist in the given file path: "//chainFilePathTrimmed ! LCOV_EXCL_LINE
