318  pure function arth_rk(first,increment,n) result(arth)
319 #if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
320  !DEC$ ATTRIBUTES DLLEXPORT :: arth_RK
321 #endif
322  use constants_mod, only: ik, rk
323  implicit none
324  real(rk) , intent(in) :: first,increment
325  integer(IK) , intent(in) :: n
326  real(rk) :: arth(n)
327  integer(IK) :: k,k2
328  real(rk) :: temp
329  if (n > 0) arth(1)=first
330  if (n <= npar_arth) then
331  do k = 2,n
332  arth(k) = arth(k-1) + increment
333  end do
334  else
335  do k = 2, npar2_arth
336  arth(k) = arth(k-1) + increment
337  end do
338  temp = increment * npar2_arth
339  k = npar2_arth
340  do
341  if (k >= n) exit
342  k2 = k+k
343  arth(k+1:min(k2,n)) = temp + arth(1:min(k,n-k))
344  temp = temp + temp
345  k = k2
346  end do
347  end if
348  end function arth_rk

388  pure function zroots_unity(n,nn)
389 #if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
390  !DEC$ ATTRIBUTES DLLEXPORT :: zroots_unity
391 #endif
392  use constants_mod, only: ik, rk, ck, twopi
393  implicit none
394  integer(IK), intent(in) :: n, nn
395  complex(CK) :: zroots_unity(nn)
396  integer(IK) :: k
397  real(rk) :: theta
398  zroots_unity(1) = 1._rk
399  theta = twopi / n
400  k = 1
401  do
402  if (k >= nn) exit
403  zroots_unity(k+1) = cmplx(cos(k*theta),sin(k*theta),kind=rk)
404  zroots_unity(k+2:min(2*k,nn)) = zroots_unity(k+1) * zroots_unity(2:min(k,nn-k))
405  k = 2 * k
406  end do
407  end function zroots_unity

452 #if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
453  !DEC$ ATTRIBUTES DLLEXPORT :: findUnique_IK
454 #endif

456  implicit none
457  integer(IK) , intent(in) :: lenvector
458  integer(IK) , intent(in) :: vector(lenvector)
459  integer(IK) , intent(out), allocatable :: uniquevalue(:)
460  integer(IK) , intent(out), allocatable :: uniquecount(:)
461  integer(IK) , intent(out), optional :: lenunique
462  integer(IK) :: lenuniq, i, j
463  logical :: isunique
464  allocate(uniquevalue(lenvector))
465  allocate(uniquecount(lenvector), source = 0_ik)
466  lenuniq = 0
467  do i = 1, lenvector
468  isunique = .true.
469  loopsearchunique: do j = 1, lenuniq
470  if (uniquevalue(j)==vector(i)) then
471  uniquecount(j) = uniquecount(j) + 1
472  isunique = .false.
473  exit loopsearchunique
474  end if
475  end do loopsearchunique
476  if (isunique) then
477  lenuniq = lenuniq + 1
478  uniquevalue(lenuniq) = vector(i)
479  uniquecount(lenuniq) = uniquecount(lenuniq) + 1
480  end if
481  end do
482  uniquevalue = uniquevalue(1:lenuniq)
483  uniquecount = uniquecount(1:lenuniq)
484  if (present(lenunique)) lenunique = lenuniq
485  end subroutine findunique_ik
