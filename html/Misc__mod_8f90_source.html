145#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
146 !DEC$ ATTRIBUTES DLLEXPORT :: mesh_RK

148 use Constants_mod, only: IK, RK
149 real(RK), intent(in) , contiguous :: X(:), Y(:)
150 real(RK) , allocatable :: MeshCrd(:,:)
151 integer(IK) :: ix, iy, counter
152 allocate(MeshCrd(2, size(X) * size(Y)))
153 counter = 0_IK
154 do iy = 1, size(Y)
155 do ix = 1, size(X)
156 counter = counter + 1_IK

158 MeshCrd(2,counter) = Y(iy)
159 end do
160 end do
161 end function mesh_RK

396 pure function arth_RK(first,increment,n) result(arth)
397#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
398 !DEC$ ATTRIBUTES DLLEXPORT :: arth_RK
399#endif
400 use Constants_mod, only: IK, RK
401 implicit none
402 real(RK) , intent(in) :: first,increment
403 integer(IK) , intent(in) :: n
404 real(RK) :: arth(n)
405 integer(IK) :: k,k2
406 real(RK) :: temp
407 if (n > 0) arth(1)=first
408 if (n <= NPAR_ARTH) then
409 do k = 2,n
410 arth(k) = arth(k-1) + increment
411 end do
412 else
413 do k = 2, NPAR2_ARTH
414 arth(k) = arth(k-1) + increment
415 end do
416 temp = increment * NPAR2_ARTH
417 k = NPAR2_ARTH
418 do
419 if (k >= n) exit
420 k2 = k+k
421 arth(k+1:min(k2,n)) = temp + arth(1:min(k,n-k))
422 temp = temp + temp
423 k = k2
424 end do
425 end if
426 end function arth_RK

466 pure function zroots_unity(n,nn)
467#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
468 !DEC$ ATTRIBUTES DLLEXPORT :: zroots_unity
469#endif
470 use Constants_mod, only: IK, RK, CK, TWO_PI
471 implicit none
472 integer(IK), intent(in) :: n, nn
473 complex(CK) :: zroots_unity(nn)
474 integer(IK) :: k
475 real(RK) :: theta
476 zroots_unity(1) = 1._RK
477 theta = TWO_PI / n
478 k = 1
479 do
480 if (k >= nn) exit
481 zroots_unity(k+1) = cmplx(cos(k*theta),sin(k*theta),kind=RK)
482 zroots_unity(k+2:min(2*k,nn)) = zroots_unity(k+1) * zroots_unity(2:min(k,nn-k))
483 k = 2 * k
484 end do
485 end function zroots_unity
