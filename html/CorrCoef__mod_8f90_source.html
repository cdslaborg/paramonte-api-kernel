121  ! LCOV_EXCL_STOP
122  call crank(ndata,workspace2,sg)
123 
124  workspace1 = workspace1 - workspace2
125  dstarstar = dot_product(workspace1,workspace1)
126  en = ndata
127  en3n = en**3-en
128  aved = en3n/6.0_rk-(sf+sg)/12.0_rk
129  fac = (1.0_rk-sf/en3n)*(1.0_rk-sg/en3n)
130  vard = ((en-1.0_rk)*en**2*(en+1.0_rk)**2/36.0_rk)*fac
131  dstarstarsignif = (dstarstar-aved)/sqrt(vard)
132  dstarstarprob = erfc( real( abs(dstarstarsignif)/sqrt2 , kind=spr ) )
133  rho = (1.0_rk-(6.0_rk/en3n)*(dstarstar+(sf+sg)/12.0_rk))/sqrt(fac)
134  fac = (1.0_rk+rho)*(1.0_rk-rho)
135  if (fac > 0.0) then
136  t = rho*sqrt((en-2.0_rk)/fac)
137  df = en-2.0_rk
138  rhoprob = getbetacdf(0.5_rk*df,0.5_rk,df/(df+t**2))
139  else
