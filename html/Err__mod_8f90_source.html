114 subroutine abort(Err, prefix, newline, outputUnit, renabled)
115#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
116 !DEC$ ATTRIBUTES DLLEXPORT :: abort
117#endif
118 use, intrinsic :: iso_fortran_env, only: output_unit
119 use Decoration_mod, only: write
120 use Constants_mod, only: BELL
121 use Constants_mod, only: NLC
122 implicit none
123 type(Err_type) , intent(in), optional :: Err
124 character(*) , intent(in), optional :: prefix, newline
125 integer(IK) , intent(in), optional :: outputUnit
126 logical , intent(in), optional :: renabled
127
128 logical :: renabledDefault
129 character(:), allocatable :: pfx, msg, nlstr
130 character(63) :: dummyChar1, imageChar !, dummyChar2
131
132 if (present(renabled)) then
133 renabledDefault = renabled
134 else
135 renabledDefault = SOFT_EXIT_ENABLED
136 end if
137
138 if (present(Err)) then
139
140#if CAF_ENABLED
141 write(imageChar ,"(g0)") this_image()
142#elif MPI_ENABLED
143 block
144 use mpi
145 integer :: imageID, ierrMPI
146 call mpi_comm_rank(mpi_comm_world, imageID, ierrMPI)
147 write(imageChar ,"(g0)") imageID + 1
148 end block
149#else
150 imageChar = "1"
151#endif
152
153 if (present(newline)) then
154 nlstr = newline
155 else
156 nlstr = NLC
157 end if
158
159 if (Err%stat==Err%statNull) then ! it is a null error code, ignore it and do not report the error code
160 msg = Err%msg
161 else
162 write(dummyChar1,"(g0)") Err%stat
163 !write(dummyChar2,"(g0)") Err%statNull
164 !msg = Err%msg // nlstr // "Error Code: " // trim(adjustl(dummyChar1)) // ". Null Error Code: " // trim(adjustl(dummyChar2)) // "."
165 msg = Err%msg // nlstr // "Error Code: " // trim(adjustl(dummyChar1)) // "."
166 end if
167
168 if (present(prefix)) then
169 call informUser(msg,prefix//" - FATAL: ",nlstr,outputUnit)
170 pfx = prefix
171 else
172 call informUser(msg," - FATAL: ",nlstr,outputUnit)
173 pfx = ""
174 end if
175
176 if (present(outputUnit)) then
177 if (outputUnit/=output_unit) then
178 call write(outputUnit,1,0,1, pfx // " - Please correct the error(s) and rerun the program." )
179 call write(outputUnit,0,0,1, pfx // " - If the cause of the error cannot be diagnosed, please report it at:" )
180 call write(outputUnit,0,0,1, pfx // " -" )
181 call write(outputUnit,0,0,1, pfx // " - https://github.com/cdslaborg/paramonte/issues" )
182 call write(outputUnit,0,0,1, pfx // " -" )
183 call write(outputUnit,0,2,1, pfx // " - Gracefully exiting on image " // trim(adjustl(imageChar)) // "." )
184 end if
185 end if
186
187 ! notify the user on screen too
188
189 if (.not. mv_isTestingMode) then
190 ! LCOV_EXCL_START
191 call write(output_unit,1,0,1, pfx // " - FATAL: Runtime error occurred." )
192 call write(output_unit,0,0,1, pfx // " - FATAL: For more information, see the output '*_report.txt' file (if generated)." )
193 call write(output_unit,0,2,1, pfx // " - FATAL: Gracefully exiting on image " // trim(adjustl(imageChar)) // "." )
194 end if
195 ! LCOV_EXCL_STOP
196
197 flush(output_unit) ! call execute_command_line(" ")
198 flush(outputUnit)
199
200 ! wait for one second:
201 block
202 use Constants_mod, only: RK, IK64
203 integer(IK64) :: countOld, countNew, countMax
204 real(RK) :: countRate
205 call system_clock( count=countOld, count_rate=countRate, count_max=countMax )
206 if (countOld/=-huge(0_IK64) .and. countRate/=0._RK .and. countMax==0_IK64) then
207 ! LCOV_EXCL_START
208 loopWait: do
209 call system_clock( count = countNew )
210 if (countNew == countMax) then
211 if (renabledDefault) return
212 write(output_unit,"(A)") repeat(BELL,3)
213 error stop
214 elseif ( real(countNew - countOld, kind = RK) / countRate >= 2._RK ) then
215 exit loopWait
216 end if
217 cycle
218 end do loopWait
219 ! LCOV_EXCL_STOP
220 end if
221 end block
222
223 end if
224
225 if (renabledDefault .or. mv_isTestingMode) return
226
227 write(output_unit,"(A)") repeat(BELL,3)
228
229#if MPI_ENABLED
230 block
231 use mpi
232 integer :: ierrMPI, errcode
233 errcode = 1; call mpi_abort(mpi_comm_world, errcode, ierrMPI) ! LCOV_EXCL_LINE
234 end block
235#else
236 error stop ! LCOV_EXCL_LINE
237#endif
238
239 end subroutine abort

328 subroutine informUser(msg,prefix,newline,outputUnit,wrapSplit,wrapWidth,marginTop,marginBot)
329#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
330 !DEC$ ATTRIBUTES DLLEXPORT :: informUser
331#endif
332 use, intrinsic :: iso_fortran_env, only: output_unit
333 use Decoration_mod, only: write, getListOfLines, wrapText
334 use String_mod, only: Str_type
335 use Constants_mod, only: IK
336 implicit none
337 character(*), intent(in) :: msg
338 character(*), intent(in), optional :: prefix, newline, wrapSplit
339 integer(IK) , intent(in), optional :: outputUnit, wrapWidth, marginTop, marginBot
340
341 integer(IK) :: stdout, sizeList, sizeListJustified
342 integer(IK) :: irecord, ijustified, width
343 integer(IK) :: padTop, padBot, padTopCurrent, padBotCurrent
344 character(:) , allocatable :: pfx, split
345 type(Str_type) , allocatable :: List(:), ListJustified(:)
346
347 if (present(outputUnit)) then
348 stdout = outputUnit
349 ! LCOV_EXCL_START
350 else
351 stdout = output_unit
352 end if
353 ! LCOV_EXCL_STOP
354 if (present(prefix)) then
355 pfx = prefix
356 else
357 pfx = ""
358 end if
359 if (present(wrapSplit)) then
360 split = wrapSplit
361 else
362 split = " "
363 end if
364 if (present(wrapWidth)) then
365 width = wrapWidth
366 else
367 width = 100_IK
368 end if
369 if (present(marginTop)) then
370 padTop = marginTop
371 else
372 padTop = 1_IK
373 end if
374 if (present(marginBot)) then
375 padBot = marginBot
376 else
377 padBot = 0_IK
378 end if
379
380 List = getListOfLines(string=msg,delimiter=newline)
381 sizeList = size(List)
382 do irecord = 1, sizeList
383 ListJustified = wrapText( string = List(irecord)%value &
384 , width = width &
385 , split = split &
386 , pad = " " &
387 )
388 sizeListJustified = size(ListJustified)
389 do ijustified = 1, sizeListJustified
390 padTopCurrent = 0_IK
391 padBotCurrent = 0_IK
392 if (irecord==1 .and. ijustified==1_IK) padTopCurrent = padTop ! the very first line
393 if (irecord==sizeList .and. ijustified==sizeListJustified) padBotCurrent = padBot ! the very last line
394 call write(stdout, padTopCurrent, padBotCurrent, 1_IK, pfx // ListJustified(ijustified)%value)
395 end do
396 end do
397 if (.not.present(marginBot)) call write(stdout)
398
399 end subroutine informUser
