702 subroutine setupParaMonte(self,nd,name,inputFile)
703#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
704 !DEC$ ATTRIBUTES DLLEXPORT :: setupParaMonte
705#endif
706 use, intrinsic :: iso_fortran_env, only: output_unit
707 use Decoration_mod, only: INDENT
708 use Constants_mod, only: IK, NLC
709 use String_mod, only: getLowerCase, val2char
710 use System_mod, only: OS_type
711 implicit none
712 class(ParaMonte_type), intent(inout) :: self
713 integer(IK), intent(in) :: nd
714 character(*), intent(in) :: name !, date, version
715 character(*), intent(in), optional :: inputFile
716 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME // "@setupParaMonte()"
717
718 self%LogFile%unit = output_unit ! temporarily set the report file to stdout.
719 self%Err%occurred = .false.
720 self%Err%msg = ""
721
722 self%Timer = Timer_type(self%Err%occurred)
723 if (self%Err%occurred) then
724 ! LCOV_EXCL_START
725 self%Err%msg = PROCEDURE_NAME // ": Error occurred while setting up the " // self%name // "timer. There is no processor clock."
726 call self%abort( Err = self%Err, prefix = self%brand, newline = NLC, outputUnit = self%LogFile%unit )
727 return
728 end if
729 ! LCOV_EXCL_STOP
730
731 self%nd%val = nd
732 self%Decor = Decoration_type() ! initialize the TAB character and decoration symbol to the default values.
733
734 self%name = name
735 self%brand = INDENT // self%name
736#if INTEL_COMPILER_ENABLED || GNU_COMPILER_ENABLED
737 self%date = "Build: " // __TIMESTAMP__
738#else
739 self%date = "Unknown Release Date"
740#endif
741#if defined PARAMONTE_VERSION
742 self%version = "Version " // PARAMONTE_VERSION
743#else
744 ! WARNING: The following ParaMonte library version tag as it will be replaced by the version
745 ! WARNING: that is generated by the preprocessing build script of the ParaMonte library.
746 ! WARNING: This is superior to the above method of using the compiler Preprocessor
747 ! WARNING: since CMAKE triggers a complete lengthy rebuild of the library when the
748 ! WARNING: only the library version has changed.
749 self%version = "Unknown ParaMonte Version"
750#include "ParaMonte_mod@version@kernel.inc.f90"
751#endif
752
753 ! setup general processor / coarray image variables
754
755 call self%Image%query()
756
757 ! setup formatting variables
758
759 self%nd%char = val2char(self%nd%val)
760
761 ! determine OS. Should be only needed by the Leader processes. But apparently not.
762
763 call self%OS%query()
764 if (self%OS%Err%occurred) then
765 ! LCOV_EXCL_START
766 self%Err = self%OS%Err
767 self%Err%msg = PROCEDURE_NAME // ": Error occurred while querying OS type."//NLC//self%Err%msg
768 call self%abort( Err = self%Err, prefix = self%brand, newline = NLC, outputUnit = self%LogFile%unit )
769 return
770 end if
771 ! LCOV_EXCL_STOP
772
773 ! This is where SystemInfo used to live, but not anymore.
774
775#if ! (CODECOV_ENABLED || SAMPLER_TEST_ENABLED)
776 blockSplashByFirstImage: if (self%Image%isFirst) then
777 call self%addSplashScreen()
778 call self%noteUserAboutEnvSetup()
779 end if blockSplashByFirstImage
780#endif
781
782 ! check if input file exists by all images
783
784 self%InputFile%isInternal = .false.
785 self%inputFileArgIsPresent = present(inputFile)
786 blockInputFileExistence: if (self%inputFileArgIsPresent) then
787 self%InputFile = File_type( path=inputFile, status="old", OS=self%OS )
788 if (self%InputFile%Err%occurred) then
789 ! LCOV_EXCL_START
790 self%Err = self%InputFile%Err
791 self%Err%msg = PROCEDURE_NAME//": Error occurred while attempting to setup the user's input file='"//inputFile//"'."//NLC//self%Err%msg
792 call self%abort( Err = self%Err, prefix = self%brand, newline = NLC, outputUnit = self%LogFile%unit )
793 return
794 end if
795 ! LCOV_EXCL_STOP
796 ! determine if the file is internal
797 self%InputFile%isInternal = self%inputFileArgIsPresent .and. .not.self%InputFile%exists .and. index(getLowerCase(inputFile),"&"//getLowerCase(self%name)) > 0
798 if (.not.(self%InputFile%isInternal .or. self%InputFile%exists)) then
799 ! file is given, but is neither a path to an external file, nor an internal file containing a namelist
800 ! Therefore, there must be an error/mistake by the user.
801 self%Err%msg = "The user's input file='"//inputFile//"' is neither the path to an existing " // &
802 "external input file nor a string containing the input "//self%name//" specifications namelist. &
803 &This may be due to the user's mistake, providing a wrong path to the input external file or &
804 &a wrong list of input specifications for the " //self%name// " simulation. " //self%name//" will &
805 &assume no input specifications file is given by the user..."
806 if (self%Image%isFirst) call self%warn(msg=self%Err%msg, prefix=self%brand, newline=NLC, outputUnit=self%LogFile%unit)
807 self%inputFileArgIsPresent = .false.
808 end if
809 end if blockInputFileExistence
810
811 if (self%Image%isFirst) call self%warnUserAboutInputFilePresence()
812
813 ! Set the default and null values for ParaMonte SpecBase
814
815 self%SpecBase = SpecBase_type(nd=self%nd%val,methodName=self%name,imageID=self%Image%id,imageCount=self%Image%count)
816
817 end subroutine setupParaMonte
