333  ! WARNING: that is generated by the preprocessing build script of the ParaMonte library.

335  ! WARNING: since CMAKE triggers a complete lengthy rebuild of the library when the

341  ! setup general processor / coarray image variables
342 
343  call self%Image%query()
344 
345  ! setup formatting variables
346 
347  self%nd%str = num2str(self%nd%val)
348 
349  ! determine OS. Should be only needed by the Leader processes. But apparently not.
350 
351  call self%OS%query()
352  if (self%OS%Err%occurred) then
353  ! LCOV_EXCL_START
354  self%Err = self%OS%Err
355  self%Err%msg = procedure_name // ": Error occurred while querying OS type."//nlc//self%Err%msg
356  call self%abort( err = self%Err, prefix = self%brand, newline = nlc, outputunit = self%LogFile%unit )
357  return
358  end if
359  ! LCOV_EXCL_STOP
360 
361  ! This is where SystemInfo used to live, but not anymore.
362 
363 #if ! (defined CODECOV_ENABLED || defined SAMPLER_TEST_ENABLED)
364  blocksplashbyfirstimage: if (self%Image%isFirst) then
365  call self%addSplashScreen()
366  call self%noteUserAboutEnvSetup()
367  end if blocksplashbyfirstimage
368 #endif
369 
370  ! check if input file exists by all images
371 
372  self%InputFile%isInternal = .false.
373  self%inputFileArgIsPresent = present(inputfile)
374  blockinputfileexistence: if (self%inputFileArgIsPresent) then
375  self%InputFile = file_type( path=inputfile, status="old", os=self%OS )
376  if (self%InputFile%Err%occurred) then
377  ! LCOV_EXCL_START
378  self%Err = self%InputFile%Err
379  self%Err%msg = procedure_name//": Error occurred while attempting to setup the user's input file='"//inputfile//"'."//nlc//self%Err%msg
380  call self%abort( err = self%Err, prefix = self%brand, newline = nlc, outputunit = self%LogFile%unit )
381  return
382  end if
383  ! LCOV_EXCL_STOP
384  ! determine if the file is internal
385  self%InputFile%isInternal = self%inputFileArgIsPresent .and. .not.self%InputFile%exists .and. index(getlowercase(inputfile),"&"//getlowercase(self%name)) > 0
386  if (.not.(self%InputFile%isInternal .or. self%InputFile%exists)) then
387  ! file is given, but is neither a path to an external file, nor an internal file containing a namelist
388  ! Therefore, there must be an error/mistake by the user.
389  self%Err%msg = "The user's input file='"//inputfile//"' is neither the path to an existing " // &
390  "external input file nor a string containing the input "//self%name//" specifications namelist. &
391  &This may be due to the user's mistake, providing a wrong path to the input external file or &
392  &a wrong list of input specifications for the " //self%name// " simulation. " //self%name//" will &
393  &assume no input specifications file is given by the user..."
394  if (self%Image%isFirst) call self%warn(msg=self%Err%msg, prefix=self%brand, newline=nlc, outputunit=self%LogFile%unit)
395  self%inputFileArgIsPresent = .false.
396  end if
397  end if blockinputfileexistence
398 
399  if (self%Image%isFirst) call self%warnUserAboutInputFilePresence()
400 
401  ! Set the default and null values for ParaMonte SpecBase
402 
403  self%SpecBase = specbase_type(nd=self%nd%val,methodname=self%name,imageid=self%Image%id,imagecount=self%Image%count)
404 
405  end subroutine setupparamonte
