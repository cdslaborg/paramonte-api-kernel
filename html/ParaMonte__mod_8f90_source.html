496 subroutine setupParaMonte(self,nd,name,inputFile)
497#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
498 !DEC$ ATTRIBUTES DLLEXPORT :: setupParaMonte
499#endif
500 use, intrinsic :: iso_fortran_env, only: output_unit
501 use Decoration_mod, only: INDENT
502 use Constants_mod, only: IK, NLC
503 use String_mod, only: getLowerCase, num2char
504 use System_mod, only: OS_type
505 implicit none
506 class(ParaMonte_type), intent(inout) :: self
507 integer(IK), intent(in) :: nd
508 character(*), intent(in) :: name !, date, version
509 character(*), intent(in), optional :: inputFile
510 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME // "@setupParaMonte()"
511
512 self%LogFile%unit = output_unit ! temporarily set the report file to stdout.
513 self%Err%occurred = .false.
514 self%Err%msg = ""
515
516 self%Timer = Timer_type(self%Err%occurred)
517 if (self%Err%occurred) then
518 ! LCOV_EXCL_START
519 self%Err%msg = PROCEDURE_NAME // ": Error occurred while setting up the " // self%name // "timer. There is no processor clock."
520 call self%abort( Err = self%Err, prefix = self%brand, newline = NLC, outputUnit = self%LogFile%unit )
521 return
522 end if
523 ! LCOV_EXCL_STOP
524
525 self%nd%val = nd
526 self%Decor = Decoration_type() ! initialize the TAB character and decoration symbol to the default values.
527
528 self%name = name
529 self%brand = INDENT // self%name
530#if INTEL_COMPILER_ENABLED || GNU_COMPILER_ENABLED
531 self%date = "Build: " // __TIMESTAMP__
532#else
533 self%date = "Unknown Release Date"
534#endif
535#if defined PARAMONTE_VERSION
536 self%version = "Version " // PARAMONTE_VERSION
537#else
538 ! WARNING: The following ParaMonte library version tag as it will be replaced by the version
539 ! WARNING: that is generated by the preprocessing build script of the ParaMonte library.
540 ! WARNING: This is superior to the above method of using the compiler Preprocessor
541 ! WARNING: since CMAKE triggers a complete lengthy rebuild of the library when the
542 ! WARNING: only the library version has changed.
543 self%version = "Unknown ParaMonte Version"
544#include "ParaMonte_mod@version@kernel.inc.f90"
545#endif
546
547 ! setup general processor / coarray image variables
548
549 call self%Image%query()
550
551 ! setup formatting variables
552
553 self%nd%char = num2char(self%nd%val)
554
555 ! determine OS. Should be only needed by the Leader processes. But apparently not.
556
557 call self%OS%query()
558 if (self%OS%Err%occurred) then
559 ! LCOV_EXCL_START
560 self%Err = self%OS%Err
561 self%Err%msg = PROCEDURE_NAME // ": Error occurred while querying OS type."//NLC//self%Err%msg
562 call self%abort( Err = self%Err, prefix = self%brand, newline = NLC, outputUnit = self%LogFile%unit )
563 return
564 end if
565 ! LCOV_EXCL_STOP
566
567 ! This is where SystemInfo used to live, but not anymore.
568
569#if ! (CODECOV_ENABLED || SAMPLER_TEST_ENABLED)
570 blockSplashByFirstImage: if (self%Image%isFirst) then
571 call self%addSplashScreen()
572 call self%noteUserAboutEnvSetup()
573 end if blockSplashByFirstImage
574#endif
575
576 ! check if input file exists by all images
577
578 self%InputFile%isInternal = .false.
579 self%inputFileArgIsPresent = present(inputFile)
580 blockInputFileExistence: if (self%inputFileArgIsPresent) then
581 self%InputFile = File_type( path=inputFile, status="old", OS=self%OS )
582 if (self%InputFile%Err%occurred) then
583 ! LCOV_EXCL_START
584 self%Err = self%InputFile%Err
585 self%Err%msg = PROCEDURE_NAME//": Error occurred while attempting to setup the user's input file='"//inputFile//"'."//NLC//self%Err%msg
586 call self%abort( Err = self%Err, prefix = self%brand, newline = NLC, outputUnit = self%LogFile%unit )
587 return
588 end if
589 ! LCOV_EXCL_STOP
590 ! determine if the file is internal
591 self%InputFile%isInternal = self%inputFileArgIsPresent .and. .not.self%InputFile%exists .and. index(getLowerCase(inputFile),"&"//getLowerCase(self%name)) > 0
592 if (.not.(self%InputFile%isInternal .or. self%InputFile%exists)) then
593 ! file is given, but is neither a path to an external file, nor an internal file containing a namelist
594 ! Therefore, there must be an error/mistake by the user.
595 self%Err%msg = "The user's input file='"//inputFile//"' is neither the path to an existing " // &
596 "external input file nor a string containing the input "//self%name//" specifications namelist. &
597 &This may be due to the user's mistake, providing a wrong path to the input external file or &
598 &a wrong list of input specifications for the " //self%name// " simulation. " //self%name//" will &
599 &assume no input specifications file is given by the user..."
600 if (self%Image%isFirst) call self%warn(msg=self%Err%msg, prefix=self%brand, newline=NLC, outputUnit=self%LogFile%unit)
601 self%inputFileArgIsPresent = .false.
602 end if
603 end if blockInputFileExistence
604
605 if (self%Image%isFirst) call self%warnUserAboutInputFilePresence()
606
607 ! Set the default and null values for ParaMonte SpecBase
608
609 self%SpecBase = SpecBase_type(nd=self%nd%val,methodName=self%name,imageID=self%Image%id,imageCount=self%Image%count)
610
611 end subroutine setupParaMonte
