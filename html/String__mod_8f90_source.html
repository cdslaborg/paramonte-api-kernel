1150 function split(value,delim,npart) result(Parts)
1151#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
1152 !DEC$ ATTRIBUTES DLLEXPORT :: split
1153#endif
1154 implicit none
1155 character(len=*) , intent(in) :: value, delim
1156 integer(IK) , intent(out), optional :: npart
1157 type(Str_type) , allocatable :: Parts(:)
1158 integer(IK) , allocatable :: PartEnd(:)
1159 integer(IK) , allocatable :: PartBegin(:)
1160 integer(IK) :: dlmlenMinusOne
1161 integer(IK) :: strlen, dlmlen, npartMax, ipart, ibeg, iend, i
1162 logical :: npartIsPresent
1163
1164 dlmlen = len(delim)
1165 strlen = len(value)
1166 npartIsPresent = present(npart)
1167
1168 ! if dlm is empty, return the whole value split character by character
1169
1170 if (dlmlen==0_IK) then
1171 allocate(Parts(strlen))
1172 do ipart = 1, strlen
1173 Parts(ipart)%value = value(ipart:ipart)
1174 end do
1175 if (npartIsPresent) npart = strlen
1176 return
1177 end if
1178
1179 npartMax = 1_IK + strlen / dlmlen ! There can be at most strlen + 1 splits
1180 allocate(PartBegin(npartMax), PartEnd(npartMax)) ! This will contain the beginning and the ends of the splits.
1181 dlmlenMinusOne = dlmlen - 1_IK
1182
1183 ibeg = 0_IK
1184 ipart = 1_IK
1185 PartBegin(ipart) = 1_IK
1186 loopParseString: do
1187
1188 ibeg = ibeg + 1_IK
1189 iend = ibeg + dlmlenMinusOne
1190
1191 if (strlen<iend) then ! the remaining part of the value is shorter than the delim
1192 PartEnd(ipart) = strlen
1193 exit loopParseString
1194 elseif ( value(ibeg:iend) == delim ) then
1195 PartEnd(ipart) = ibeg - 1_IK
1196 ipart = ipart + 1_IK
1197 PartBegin(ipart) = iend + 1_IK
1198 ibeg = iend
1199 end if
1200
1201 end do loopParseString
1202
1203 allocate(Parts(ipart))
1204 do i = 1, ipart
1205 Parts(i)%value = value(PartBegin(i):PartEnd(i))
1206 end do
1207 if (present(npart)) npart = ipart
1208
1209 deallocate(PartBegin, PartEnd)
1210
1211 end function split

1239 function splitStr(value,delim,nPart) result(Parts)
1240#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
1241 !DEC$ ATTRIBUTES DLLEXPORT :: splitStr
1242#endif
1243
1244 implicit none
1245 character(len=*) , intent(in) :: value, delim
1246 integer(IK) , intent(out), optional :: nPart
1247 character(len=:) , allocatable :: dummyStr
1248 type(Str_type) , allocatable :: Parts(:)
1249 integer(IK) :: maxNumSplit
1250 integer(IK) :: valueLen, delimLen, splitCounter, currentPos
1251
1252 dummyStr = value
1253 delimLen = len(delim)
1254 valueLen = len(dummyStr)
1255
1256 if (delimLen==0) then
1257 allocate(Parts(1))
1258 Parts(1)%value = value
1259 if (present(nPart)) nPart = 1_IK
1260 return
1261 end if
1262
1263 maxNumSplit = 1 + valueLen / delimLen
1264 allocate(Parts(maxNumSplit))
1265 splitCounter = 1
1266 loopParseString: do
1267 if (valueLen<delimLen) then
1268 Parts(splitCounter)%value = dummyStr
1269 exit loopParseString
1270 elseif (valueLen==delimLen) then
1271 ! Note that in Fortran: 'amir '=='amir' = .true.
1272 ! https://software.intel.com/en-us/forums/intel-visual-fortran-compiler-for-windows/topic/275823)
1273 if (dummyStr==delim) then
1274 Parts(splitCounter)%value = ""
1275 else
1276 Parts(splitCounter)%value = dummyStr
1277 end if
1278 exit loopParseString
1279 elseif (dummyStr(1:delimLen)==delim) then
1280 dummyStr = dummyStr(delimLen+1:valueLen)
1281 valueLen = len(dummyStr)
1282 cycle loopParseString
1283 else
1284 currentPos = 2
1285 loopSearchString: do
1286 if (dummyStr(currentPos:currentPos+delimLen-1)==delim) then
1287 Parts(splitCounter)%value = dummyStr(1:currentPos-1)
1288 if (currentPos+delimLen>valueLen) then
1289 exit loopParseString
1290 else
1291 splitCounter = splitCounter + 1
1292 dummyStr = dummyStr(currentPos+delimLen:valueLen)
1293 valueLen = len(dummyStr)
1294 cycle loopParseString
1295 end if
1296 else
1297 currentPos = currentPos + 1
1298 if (valueLen<currentPos+delimLen-1) then
1299 Parts(splitCounter)%value = dummyStr
1300 exit loopParseString
1301 end if
1302 cycle loopSearchString
1303 end if
1304 end do loopSearchString
1305 end if
1306 end do loopParseString
1307 Parts = Parts(1:splitCounter)
1308 if (present(nPart)) nPart = splitCounter
1309
1310 end function splitStr
