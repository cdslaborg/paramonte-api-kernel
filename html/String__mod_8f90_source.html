1152#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
1153 !DEC$ ATTRIBUTES DLLEXPORT :: split
1154#endif
1155 use Constants_mod, only: IK, LK
1156 character(*,SK) , intent(in) :: value, delim
1157 integer(IK) , intent(out), optional :: npart
1158 type(String_type) , allocatable :: Parts(:)
1159 integer(IK) , allocatable :: PartEnd(:)
1160 integer(IK) , allocatable :: PartBegin(:)
1161 integer(IK) :: dlmlenMinusOne
1162 integer(IK) :: strlen, dlmlen, npartMax, ipart, ibeg, iend, i
1163 logical(LK) :: npartIsPresent
1164
1165 dlmlen = len(delim)
1166 strlen = len(value)
1167 npartIsPresent = present(npart)
1168
1169 ! if dlm is empty, return the whole value split character by character
1170
1171 if (dlmlen == 0_IK) then
1172 allocate(Parts(strlen))
1173 do ipart = 1, strlen
1174 Parts(ipart)%value = value(ipart:ipart)
1175 end do
1176 if (npartIsPresent) npart = strlen
1177 return
1178 end if
1179
1180 npartMax = 1_IK + strlen / dlmlen ! There can be at most strlen + 1 splits
1181 allocate(PartBegin(npartMax), PartEnd(npartMax)) ! This will contain the beginning and the ends of the splits.
1182 dlmlenMinusOne = dlmlen - 1_IK
1183
1184 ibeg = 0_IK
1185 ipart = 1_IK
1186 PartBegin(ipart) = 1_IK
1187 loopParseString: do
1188
1189 ibeg = ibeg + 1_IK
1190 iend = ibeg + dlmlenMinusOne
1191
1192 if (strlen<iend) then ! the remaining part of the value is shorter than the delim
1193 PartEnd(ipart) = strlen
1194 exit loopParseString
1195 elseif ( value(ibeg:iend) == delim ) then
1196 PartEnd(ipart) = ibeg - 1_IK
1197 ipart = ipart + 1_IK
1198 PartBegin(ipart) = iend + 1_IK
1199 ibeg = iend
1200 end if
1201
1202 end do loopParseString
1203
1204 allocate(Parts(ipart))
1205 do i = 1, ipart
1206 Parts(i)%value = value(PartBegin(i):PartEnd(i))
1207 end do
1208 if (present(npart)) npart = ipart
1209
1210 deallocate(PartBegin, PartEnd)
1211
1212 end function split

1240 function splitStr(value,delim,nPart) result(Parts)
1241#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
1242 !DEC$ ATTRIBUTES DLLEXPORT :: splitStr
1243#endif
1244
1245 implicit none
1246 character(*,SK) , intent(in) :: value, delim
1247 integer(IK) , intent(out), optional :: nPart
1248 character(:,SK) , allocatable :: dummyStr
1249 type(String_type) , allocatable :: Parts(:)
1250 integer(IK) :: maxNumSplit
1251 integer(IK) :: valueLen, delimLen, splitCounter, currentPos
1252
1253 dummyStr = value
1254 delimLen = len(delim)
1255 valueLen = len(dummyStr)
1256
1257 if (delimLen==0) then
1258 allocate(Parts(1))
1259 Parts(1)%value = value
1260 if (present(nPart)) nPart = 1_IK
1261 return
1262 end if
1263
1264 maxNumSplit = 1 + valueLen / delimLen
1265 allocate(Parts(maxNumSplit))
1266 splitCounter = 1
1267 loopParseString: do
1268 if (valueLen<delimLen) then
1269 Parts(splitCounter)%value = dummyStr
1270 exit loopParseString
1271 elseif (valueLen==delimLen) then
1272 ! Note that in Fortran: 'amir '=='amir' = .true.
1273 ! https://software.intel.com/en-us/forums/intel-visual-fortran-compiler-for-windows/topic/275823)
1274 if (dummyStr==delim) then
1275 Parts(splitCounter)%value = ""
1276 else
1277 Parts(splitCounter)%value = dummyStr
1278 end if
1279 exit loopParseString
1280 elseif (dummyStr(1:delimLen)==delim) then
1281 dummyStr = dummyStr(delimLen+1:valueLen)
1282 valueLen = len(dummyStr)
1283 cycle loopParseString
1284 else
1285 currentPos = 2
1286 loopSearchString: do
1287 if (dummyStr(currentPos:currentPos+delimLen-1)==delim) then
1288 Parts(splitCounter)%value = dummyStr(1:currentPos-1)
1289 if (currentPos+delimLen>valueLen) then
1290 exit loopParseString
1291 else
1292 splitCounter = splitCounter + 1
1293 dummyStr = dummyStr(currentPos+delimLen:valueLen)
1294 valueLen = len(dummyStr)
1295 cycle loopParseString
1296 end if
1297 else
1298 currentPos = currentPos + 1
1299 if (valueLen<currentPos+delimLen-1) then
1300 Parts(splitCounter)%value = dummyStr
1301 exit loopParseString
1302 end if
1303 cycle loopSearchString
1304 end if
1305 end do loopSearchString
1306 end if
1307 end do loopParseString
1308 Parts = Parts(1:splitCounter)
1309 if (present(nPart)) nPart = splitCounter
1310
1311 end function splitStr
