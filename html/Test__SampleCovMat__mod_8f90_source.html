175 function test_genCovUpp_1() result(assertion)
176
177 use Constants_mod, only: IK, RK
178 use SampleShift_mod, only: genShifted
179 implicit none
180
181 logical :: assertion
182 real(RK) , parameter :: tolerance = 1.e-12_RK
183 integer(IK) , parameter :: nd = 3_IK
184 integer(IK) , parameter :: np = 5_IK
185 integer(IK) :: i, j
186 real(RK) , parameter :: Sample(nd,np) = reshape( [ 0.706046088019609_RK, 0.031832846377421_RK, 0.276922984960890_RK &
187 , 0.046171390631154_RK, 0.097131781235848_RK, 0.823457828327293_RK &
188 , 0.694828622975817_RK, 0.317099480060861_RK, 0.950222048838355_RK &
189 , 0.034446080502909_RK, 0.438744359656398_RK, 0.381558457093008_RK &
190 , 0.765516788149002_RK, 0.795199901137063_RK, 0.186872604554379_RK &
191 ], shape = shape(Sample) )
192
193 real(RK) , parameter :: Mean(nd) = [0.449401794055698_RK, 0.336001673693518_RK, 0.523806784754785_RK]
194 real(RK) , parameter :: ChoLowCovUpp_ref(nd,nd) = transpose(reshape( [ +0.140203078481164_RK, +0.029035771029083_RK, -0.031754793421423_RK &
195 , +0.077545140669886_RK, +0.092838466390969_RK, -0.043469498536710_RK &
196 , -0.084806768876261_RK, -0.125205309782421_RK, +0.116582891117029_RK &
197 ] , shape = shape(ChoLowCovUpp_ref) ))
198 real(RK) :: ChoLowCovUpp_diff(nd,nd)
199 real(RK) :: ChoLowCovUpp(nd,nd)
200
201 ChoLowCovUpp = genCovUpp(genShifted(Sample,-Mean, dim = 2_IK), biased = .false.)
202 ChoLowCovUpp_diff = abs(ChoLowCovUpp - ChoLowCovUpp_ref) / abs(ChoLowCovUpp_ref)
203 assertion = all([((ChoLowCovUpp_diff(i,j), i = 1, j), j = 1, nd)] <= tolerance)
204
205 if (Test%isVerboseMode .and. .not. assertion) then
206 ! LCOV_EXCL_START
207 write(Test%outputUnit,"(*(g0,:,', '))")
208 write(Test%outputUnit,"(*(g0,:,', '))") "ChoLowCovUpp_ref ", ChoLowCovUpp_ref
209 write(Test%outputUnit,"(*(g0,:,', '))") "ChoLowCovUpp ", ChoLowCovUpp
210 write(Test%outputUnit,"(*(g0,:,', '))") "ChoLowCovUpp_diff ", ChoLowCovUpp_diff
211 write(Test%outputUnit,"(*(g0,:,', '))")
212 ! LCOV_EXCL_STOP
213 end if
214
215 end function test_genCovUpp_1
