72  function constructscalefactor(nd,methodName) result(ScaleFactorObj)
73 #if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
74  !DEC$ ATTRIBUTES DLLEXPORT :: constructScaleFactor
75 #endif
76  use constants_mod, only: rk, ik, null_sk
77  use string_mod, only: num2str
78  use decoration_mod, only: tab
79  implicit none
80  integer(IK), intent(in) :: nd
81  character(*), intent(in) :: methodname
82  type(scalefactor_type) :: scalefactorobj
83 
84  scalefactorobj%defStr = "gelman"
85  scalefactorobj%defVal = 2.38_rk/sqrt(real(nd,kind=rk)) ! Gelman, Roberts, Gilks (1996): Efficient Metropolis Jumping Rules
86  scalefactorobj%null = repeat(null_sk, max_len_string_scale_factor)
87  scalefactorobj%desc = &
88  "scaleFactor is a real-valued positive number (which must be given as string), by the square of which the &
89  &covariance matrix of the proposal distribution of the MCMC sampler is scaled. In other words, &
90  &the proposal distribution will be scaled in every direction by the value of scaleFactor. &
91  &It can also be given in units of the string keyword 'gelman' (which is case-INsensitive) after the paper:\n\n" &
92  // tab // "Gelman, Roberts, and Gilks (1996): 'Efficient Metropolis Jumping Rules'.\n\n&
93  &The paper finds that the optimal scaling factor for a Multivariate Gaussian proposal distribution for the &
94  &Metropolis-Hastings Markov Chain Monte Carlo sampling of a target Multivariate Normal Distribution &
95  &of dimension ndim is given by:\n\n&
96  & scaleFactor = 2.38/sqrt(ndim) , in the limit of ndim -> Infinity.\n\n&
97  &Multiples of the gelman scale factors are also acceptable as input and can be specified like the following examples:\n\n&
98  & scaleFactor = '1'\n\n&
99  & multiplies the ndim-dimensional proposal covariance matrix by 1, essentially no change occurs to &
100  &the covariance matrix.\n\n" // &
101  ' scaleFactor = "1"\n\n' // &
102  " same as the previous example. The double-quotation marks act the same way as single-quotation marks.\n\n&
103  & scaleFactor = '2.5'\n\n&
104  & multiplies the ndim-dimensional proposal covariance matrix by 2.5.\n\n&
105  & scaleFactor = '2.5*Gelman'\n\n&
106  & multiplies the ndim-dimensional proposal covariance matrix by 2.5 * 2.38/sqrt(ndim).\n\n" // &
107  ' scaleFactor = "2.5 * gelman"\n\n' // &
108  " same as the previous example, but with double-quotation marks. space characters are ignored.\n\n" // &
109  ' scaleFactor = "2.5 * gelman*gelman*2"\n\n' // &
110  " equivalent to gelmanFactor-squared multiplied by 5.\n\n&
111  &Note, however, that the result of Gelman et al. paper applies only to multivariate normal proposal distributions, in &
112  &the limit of infinite dimensions. Therefore, care must be taken when using Gelman's scaling factor with non-Gaussian &
113  &proposals and target objective functions. Note that only the product symbol (*) can be parsed &
114  &in the string value of scaleFactor. The presence of other mathematical symbols or multiple appearances of the product &
115  &symbol will lead to a simulation crash. Also, note that the prescription of an acceptance range specified by the input &
116  &variable 'targetAcceptanceRate' will lead to dynamic modification of the initial input value of scaleFactor throughout sampling &
117  &for adaptiveUpdateCount times. &
118  &The default scaleFactor string-value is 'gelman' (for all proposals), which is subsequently converted to 2.38/sqrt(ndim)."
119  end function constructscalefactor

150  ! ATTN: This subroutine also assigns the value of ScaleFactor. It MUST be executed by all images.
151  subroutine checkforsanity(ScaleFactorObj,Err,methodName)
152 #if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
153  !DEC$ ATTRIBUTES DLLEXPORT :: checkForSanity
154 #endif
155  use constants_mod, only: ik, rk
156  use string_mod, only: num2str, string_type
157  use err_mod, only: err_type
158  implicit none
159  class(scalefactor_type), intent(inout) :: scalefactorobj
160  character(*), intent(in) :: methodname
161  type(err_type), intent(inout) :: err
162  character(*), parameter :: procedure_name = "@checkForSanity()"
163  type(string_type) :: string
164  integer(IK) :: i
165  real(rk) :: temp
166 
167  ! First convert the scaleFactor string to real value:
168 
169  string%value = string%replaceStr(scalefactorobj%str," ","") ! remove the white spaces
170  if (len_trim(adjustl(string%value))==0) then
171  err%occurred = .true.
172  err%msg = err%msg // &
173  module_name // procedure_name // ": Error occurred. &
174  &The input string value (" // scalefactorobj%str // ") for the variable scaleFactor &
175  &is empty. Make sure the input string follows the syntax rules of " &
176  // methodname // " for this variable. Otherwise drop it from the input list. " &
177  // methodname // " will automatically assign an appropriate value to it.\n\n"
178  return
179  end if
180 
181  ! Now split the string by "*" to real coefficient and character (gelman) parts for further evaluations
182 
183  string%Parts = string%split(string = string%value, delim = "*", npart = string%nPart)
184  scalefactorobj%val = 1._rk
185  do i = 1, string%nPart
186  if ( string%getLowerCase( string%Parts(i)%record ) == "gelman" ) then
187  scalefactorobj%val = scalefactorobj%val * scalefactorobj%defVal
188  else
189  temp = string%str2real64( str=string%Parts(i)%record, iostat=err%stat )
190  if ( err%stat/=0 ) then
191  err%occurred = .true.
192  err%msg = err%msg // & ! LCOV_EXCL_LINE
193  module_name // procedure_name // ": Error occurred while reading real number.\n"//& ! LCOV_EXCL_LINE
194  "The input string value for the variable scaleFactor (" // scalefactorobj%str // ") does not appear to follow "//& ! LCOV_EXCL_LINE
195  "the standard syntax rules of "// methodname // " for this variable. '" // string%Parts(i)%record // & ! LCOV_EXCL_LINE
196  "' cannot be parsed into any meaningful token. Please correct the input value, or drop it from the input list, "//& ! LCOV_EXCL_LINE
197  "in which case, " // methodname // " will automatically assign an appropriate value to it.\n\n"
198  return
199  else
200  scalefactorobj%val = scalefactorobj%val * temp
201  end if
202  end if
203  end do
204 
205  ! Now check if the real value is positive
206 
207  if (scalefactorobj%val<=0) then
208  err%occurred = .true.
209  err%msg = err%msg // &
210  module_name // procedure_name // ": Error occurred. &
211  &The input string value (" // scalefactorobj%str // ") translates to a negative real value: " // &
212  num2str(scalefactorobj%val) // ". &
213  &Make sure the input string follows the syntax rules of " // methodname // " for this variable. &
214  &Otherwise drop it from the input list. " // methodname // &
215  " will automatically assign an appropriate value to it.\n\n"
216  return
217  end if
218 
219  end subroutine checkforsanity
