108  , CFC &
109  , Err &

112  , sampleRefinementCount &
113  , sampleRefinementMethod &

121  use crosscorr_mod, only: getbatchmeansiac, getcumsumiac, getmaxcumsumiac

127  character(*), parameter :: procedure_name = module_name//"@getRefinedChain()"
128 
129  class(refinedchain_type) , intent(inout) :: RefinedChain
130  type(chainfilecontents_type), intent(inout) :: CFC
131  type(err_type) , intent(out) :: Err
132  character(*) , intent(in), optional :: sampleRefinementMethod
133  integer(IK) , intent(in), optional :: burninLoc, refinedChainSize, sampleRefinementCount
134  integer(IK) :: burninLocDefault, i, countCompact, ndimPlusOne
135  integer(IK) :: maxRefinementCurrent, maxRefinementCount
136  real(RK) :: integratedAutoCorrTime, ndimPlusOneInverse
137  logical :: refinedChainSizeIsPresent
138  logical :: maxRefinementCountIsReached
139  character(:) , allocatable :: sampleRefinementMethodDefault, sampleRefinementMethodDefaultLowerCase, srmethod
140  type(count_type), allocatable :: DumCount(:)
141  real(RK) , allocatable :: DumIAC(:,:), DumVec(:)
142  integer(IK) , allocatable :: SampleWeight(:)
143  type(method_type) :: Method
144  type(chainfilecontents_type) :: DumCFC

146  err%occurred = .false.
147 
148  ! if the size of the refined sample is given as input, then generate the requested sample straight
149 
150  refinedchainsizeispresent = present(refinedchainsize)
151  if (refinedchainsizeispresent) then ! ignore sampleRefinementCount, even if it is given by the user
152  maxrefinementcount = 1_ik
153  else
154  maxrefinementcount = 20_ik ! this is a temporary maximum value, to be increased later if needed
155  if (present(samplerefinementcount)) maxrefinementcount = samplerefinementcount
156  end if
157 
158  ! this is to avoid memory overflow due to extremely large maxRefinementCount requested by the user
159 
160  maxrefinementcurrent = min(2_ik, maxrefinementcount)
161 
162  ! compute ndim and the initial chain size
163 
164  refinedchain%numRefinement = 0_ik
165 
166  if (cfc%ndim == 0_ik) then
167  refinedchain%ndim = size(cfc%State(:,1))
168  cfc%ndim = refinedchain%ndim
169  else
170  refinedchain%ndim = cfc%ndim
171  end if
172 
173  ! allocate components
174 
175  if (allocated(refinedchain%IAC)) deallocate(refinedchain%IAC)
176  if (allocated(refinedchain%Count)) deallocate(refinedchain%Count)
177  allocate(refinedchain%IAC(0:refinedchain%ndim,0:maxrefinementcurrent))
178  allocate(refinedchain%Count(0:maxrefinementcurrent))
179  if (cfc%Count%compact == 0_ik) then
180  refinedchain%Count(refinedchain%numRefinement)%compact = size(cfc%State(1,:))
181  cfc%Count%compact = refinedchain%Count(refinedchain%numRefinement)%compact
182  else
183  refinedchain%Count(refinedchain%numRefinement)%compact = cfc%Count%compact
184  end if
185  if (cfc%Count%verbose == 0_ik) cfc%Count%verbose = sum(cfc%Weight(1:cfc%Count%compact))
186  refinedchain%Count(refinedchain%numRefinement)%verbose = cfc%Count%verbose
187 
188  ! define the AIC computation method
189 
190  samplerefinementmethoddefault = batch_means_method_name; if (present(samplerefinementmethod)) samplerefinementmethoddefault = samplerefinementmethod
191  samplerefinementmethoddefaultlowercase = getlowercase(samplerefinementmethoddefault)
192 
193  if (index(samplerefinementmethoddefaultlowercase,getlowercase(batch_means_method_name))>0) then
194  method%isBatchMeans = .true.
195  elseif (index(samplerefinementmethoddefaultlowercase,getlowercase(cutoff_autocorr_method_name))>0 .or. index(samplerefinementmethoddefaultlowercase,"cutoff")>0) then
196  method%isCutoffAutoCorr = .true.
197  elseif (index(samplerefinementmethoddefaultlowercase,getlowercase(max_cumsum_autocorr_method_name))>0 .or. index(samplerefinementmethoddefaultlowercase,"cumsum")>0) then
198  method%isMaxCumSumAutoCorr = .true.
199  else
200  ! LCOV_EXCL_START
201  err%occurred = .true.
202  err%msg = procedure_name // ": Unknown unsupported IAC computation method name: " // samplerefinementmethoddefault
203  return
204  ! LCOV_EXCL_STOP
205  end if
206 
207  ! define the chain types to use for the AIC computation
208 
209  method%isViaCompactChain = index(samplerefinementmethoddefaultlowercase,"compact") > 0
210  method%isViaVerboseChain = index(samplerefinementmethoddefaultlowercase,"verbose") > 0
211  if (.not.(method%isViaCompactChain .or. method%isViaVerboseChain)) then
212  method%isViaCompactChain = .true.
213  method%isViaVerboseChain = .true.
214  end if
215 
216  ! define the statistic to use for the AIC computation
217 
218  srmethod = replacestr(string=samplerefinementmethoddefaultlowercase,search=" ",substitute="-")
219  method%isAvg = index(srmethod,"-avg") > 0 .or. index(srmethod,"-average") > 0
220  method%isMed = index(srmethod,"-med") > 0 .or. index(srmethod,"-median") > 0
221  method%isMin = index(srmethod,"-min") > 0 .or. index(srmethod,"-minimum") > 0
222  method%isMax = index(srmethod,"-max") > 0 .or. index(srmethod,"-maximum") > 0
223  if ( method%isAvg .and. (method%isMed .or. method%isMax .or. method%isMin) ) err%occurred = .true.
224  if ( method%isMed .and. (method%isMax .or. method%isMin) ) err%occurred = .true.
225  if ( method%isMax .and. method%isMin ) err%occurred = .true.
226  if (.not.(method%isAvg .or. method%isMed .or. method%isMin .or. method%isMax)) method%isAvg = .true. ! default method of AIC summarization.
227 
228  if (method%isAvg) ndimplusoneinverse = 1._rk / (refinedchain%ndim + 1_ik)
229  if (method%isMed) ndimplusone = refinedchain%ndim + 1_ik
230 
231  ! assign the column headers
232 
233  if (allocated(cfc%ColHeader)) then
234  if (allocated(refinedchain%ColHeader)) deallocate(refinedchain%ColHeader)
235  allocate(refinedchain%ColHeader(0:refinedchain%ndim))
236  do i = 0, refinedchain%ndim
237  refinedchain%ColHeader(i)%record = cfc%ColHeader(i+num_def_col)%record
238  end do
239  else
240  ! LCOV_EXCL_START
241  err%occurred = .true.
242  err%msg = procedure_name // ": Internal error occurred. CFC%ColHeader is NULL."
243  return
244  ! LCOV_EXCL_STOP
245  end if
246 
247  if (present(burninloc)) then
248  burninlocdefault = burninloc
249  else
250  burninlocdefault = cfc%BurninLoc(cfc%Count%compact)
251  end if
252 
253  refinedchain%Count(refinedchain%numRefinement)%compact = cfc%Count%compact - burninlocdefault + 1
254  if (allocated(refinedchain%LogFuncState)) deallocate(refinedchain%LogFuncState)
255  allocate(refinedchain%LogFuncState(0:refinedchain%ndim,refinedchain%Count(refinedchain%numRefinement)%compact))
256  refinedchain%LogFuncState(0 ,1:refinedchain%Count(refinedchain%numRefinement)%compact) = cfc%LogFunc(burninlocdefault:cfc%Count%compact)
257  refinedchain%LogFuncState(1:refinedchain%ndim,1:refinedchain%Count(refinedchain%numRefinement)%compact) = cfc%State(1:refinedchain%ndim, burninlocdefault:cfc%Count%compact)
258 
259  ! check if there are more than 1 sample points in the burnin-subtracted CFC
260 
261  if (refinedchain%Count(refinedchain%numRefinement)%compact==0_ik) then
262  ! LCOV_EXCL_START
263  err%occurred = .true.
264  err%msg = procedure_name // ": The size of the refined sample is zero."
265  return
266  ! LCOV_EXCL_STOP
267  elseif (refinedchain%Count(refinedchain%numRefinement)%compact==1_ik) then
268  if (allocated(refinedchain%Weight)) deallocate(refinedchain%Weight)
269  allocate(refinedchain%Weight(refinedchain%Count(refinedchain%numRefinement)%compact))
270  if (refinedchainsizeispresent) then
271  refinedchain%Weight = refinedchainsize
272  else
273  refinedchain%Weight = 1
274  end if
275  if (allocated(refinedchain%IAC)) deallocate(refinedchain%IAC)
276  if (allocated(refinedchain%Count)) deallocate(refinedchain%Count)
277  allocate(refinedchain%IAC(0:refinedchain%ndim,0:0))
278  allocate(refinedchain%Count(0:0))
279  refinedchain%IAC = 0._rk
280  refinedchain%Count(refinedchain%numRefinement)%verbose = sum(refinedchain%Weight)
281  return
282  end if
283 
284  refinedchain%Weight = cfc%Weight(burninlocdefault:cfc%Count%compact) ! Weight is intentionally separately assigned from State here
285 
286  !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
287 
288  looprefinement: do
289 
290  countcompact = refinedchain%Count(refinedchain%numRefinement)%compact
291 
292  ! set the sample weight
293 
294  if (method%isViaCompactChain) then
295  if (allocated(sampleweight)) deallocate(sampleweight)
296  elseif (method%isViaVerboseChain) then
297  sampleweight = refinedchain%Weight
298  endif
299 
300  ! obtain the IAC for each individual variable
301 
302  do i = 0, refinedchain%ndim
303  dumvec = refinedchain%LogFuncState(i,1:refinedchain%Count(refinedchain%numRefinement)%compact)
304  if (method%isBatchMeans) then
305  refinedchain%IAC(i,refinedchain%numRefinement) = getbatchmeansiac( np = countcompact &
306  , point = dumvec & ! LCOV_EXCL_LINE
307  , weight = sampleweight & ! LCOV_EXCL_LINE
308  )
309  elseif (method%isCutoffAutoCorr) then
310  refinedchain%IAC(i,refinedchain%numRefinement) = getcumsumiac( np = countcompact &
311  , point = dumvec & ! LCOV_EXCL_LINE
312  , weight = sampleweight & ! LCOV_EXCL_LINE
313  )
314  elseif (method%isMaxCumSumAutoCorr) then
315  refinedchain%IAC(i,refinedchain%numRefinement) = getmaxcumsumiac( np = countcompact &
316  , point = dumvec & ! LCOV_EXCL_LINE
317  , weight = sampleweight & ! LCOV_EXCL_LINE
318  )
319  end if
320  end do
321 
322  if (refinedchainsizeispresent) then
323  integratedautocorrtime = real(sum(refinedchain%Weight),kind=rk) / real(refinedchainsize,kind=rk)
324  else
325  if (method%isAvg) then
326  integratedautocorrtime = sum( refinedchain%IAC(0:refinedchain%ndim,refinedchain%numRefinement) ) * ndimplusoneinverse
327  elseif (method%isMed) then
328  call getmedian(lenarray=ndimplusone,array=refinedchain%IAC(0:refinedchain%ndim,refinedchain%numRefinement),median=integratedautocorrtime,err=err)
329  if (err%occurred) then
330  ! LCOV_EXCL_START
331  err%msg = procedure_name//err%msg
332  return
333  ! LCOV_EXCL_STOP
334  end if
335  elseif (method%isMax) then
336  integratedautocorrtime = maxval( refinedchain%IAC(0:refinedchain%ndim,refinedchain%numRefinement) )
337  elseif (method%isMin) then
338  integratedautocorrtime = minval( refinedchain%IAC(0:refinedchain%ndim,refinedchain%numRefinement) )
339  end if
340  end if
341 
342  ! so far, we have computed the max IAC of the sample, but no refinement. Refine the sample only if needed.
343 
344  maxrefinementcountisreached = refinedchain%numRefinement==maxrefinementcount
345  if (integratedautocorrtime<2._rk .or. maxrefinementcountisreached) then
346  if (method%isViaCompactChain .and. method%isViaVerboseChain) then
347  !if (maxRefinementCountIsReached) maxRefinementCount = maxRefinementCount * 2_IK
348  maxrefinementcount = maxrefinementcount * 2_ik
349  method%isViaCompactChain = .false.
350  cycle looprefinement
351  end if
352  exit looprefinement
353  end if
354 
355  ! generate the refined sample, dump it in CFC, then put it back into RefinedChain to start over again
356 
357  refinedchain%numRefinement = refinedchain%numRefinement + 1_ik
358 
359  ! reallocate to bigger array if nedded
360 
361  if (refinedchain%numRefinement>maxrefinementcurrent) then
362  call move_alloc( from = refinedchain%IAC, to = dumiac )
363  call move_alloc( from = refinedchain%Count, to = dumcount )
364  maxrefinementcurrent = min( maxrefinementcurrent*2 , maxrefinementcount )
365  allocate( refinedchain%IAC( 0:refinedchain%ndim , 0:maxrefinementcurrent ) )
366  allocate( refinedchain%Count( 0:maxrefinementcurrent ) )
367  refinedchain%IAC(0:refinedchain%ndim,0:refinedchain%numRefinement-1) = dumiac
368  refinedchain%Count(0:refinedchain%numRefinement-1) = dumcount
369  end if
370 
371  if (integratedautocorrtime<2._rk) cycle looprefinement ! no need for refinement. should happen only when transitioning from compact to verbose
372 
373  call refineweightedsample ( nd = refinedchain%ndim & ! LCOV_EXCL_LINE
374  , np = countcompact & ! LCOV_EXCL_LINE
375  , skip = integratedautocorrtime & ! LCOV_EXCL_LINE
376  , sample = refinedchain%LogFuncState & ! LCOV_EXCL_LINE
377  , weight = refinedchain%Weight & ! LCOV_EXCL_LINE
378  , refinedchain = dumcfc%State & ! LCOV_EXCL_LINE
379  , refinedweight = dumcfc%Weight & ! LCOV_EXCL_LINE
380  , pointcount = refinedchain%Count(refinedchain%numRefinement) &
381  , refinedchainsize = refinedchainsize & ! LCOV_EXCL_LINE
382  )
383  refinedchain%Weight = dumcfc%Weight
384  refinedchain%LogFuncState = dumcfc%State
385 
386  end do looprefinement
387 
388  !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
389 
390  end subroutine getrefinedchain

416  pure function getrefinedweight(np,Weight,skip,refinedChainSize) result(RefinedWeight)
417 #if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
418  !DEC$ ATTRIBUTES DLLEXPORT :: getRefinedWeight
419 #endif
420  use math_mod, only: getcumsum
421  use constants_mod, only: ik, rk
422  implicit none
423  real(rk), intent(in) :: skip
424  integer(IK), intent(in) :: np, weight(np)
425  integer(IK) , intent(in) , optional :: refinedchainsize
426  integer(IK) :: refinedweight(np)
427  integer(IK) :: ip, refinedchainsizecounter, offset
428  real(rk) :: sumskips, cumsumweight(np)
429  logical :: refinedchainsizeispresent
430  refinedchainsizeispresent = present(refinedchainsize)
431  if (refinedchainsizeispresent) refinedchainsizecounter = 0_ik
432  cumsumweight = real(getcumsum(np,weight),kind=rk)
433  sumskips = skip
434  offset = 1_ik
435  ip = offset
436  refinedweight = 0_ik
437  loopoverallsample: do
438  loopovercurrentsample: do
439  if (sumskips>cumsumweight(ip)) then
440  exit loopovercurrentsample
441  elseif (refinedchainsizeispresent) then
442  if (refinedchainsizecounter==refinedchainsize) exit loopoverallsample
443  refinedchainsizecounter = refinedchainsizecounter + 1_ik
444  end if
445  refinedweight(ip) = refinedweight(ip) + 1_ik
446  sumskips = sumskips + skip
447  cycle loopovercurrentsample
448  end do loopovercurrentsample
449  if (ip==np) then
450  if (refinedchainsizeispresent) then
451  if (refinedchainsizecounter<refinedchainsize) then
452  offset = offset + 1_ik
453  if (offset==np) offset = 1_ik
454  ip = offset
455  sumskips = skip
456  if (offset/=1_ik) sumskips = sumskips + cumsumweight(ip-1)
457  cycle loopoverallsample ! LCOV_EXCL_LINE
458  end if
459  end if
460  exit loopoverallsample
461  end if
462  ip = ip + 1_ik
463  end do loopoverallsample
464  end function getrefinedweight
