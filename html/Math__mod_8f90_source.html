153 PURE function getLogSubExp_RK(logValueLarger,logValueSmaller) result(logSubExp)
154#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
155 !DEC$ ATTRIBUTES DLLEXPORT :: getLogSubExp_RK
156#endif
157 use Constants_mod, only: RK, LOGTINY_RK
158 real(RK) , intent(in) :: logValueLarger, logValueSmaller
159 real(RK) :: logSubExp, diff
160#if CHECK_ENABLED
161 block
162 use iso_fortran_env, only: output_unit
163 if (logValueSmaller > logValueLarger) then
164 write(output_unit,"(*(g0,:,' '))") "logValueSmaller > logValueLarger:", logValueSmaller, logValueLarger
165 error stop
166 end if
167 end block
168#endif
169 logSubExp = logValueLarger
170 diff = logValueSmaller - logValueLarger
171 if (diff>LOGTINY_RK) logSubExp = logSubExp + log( 1._RK - exp(diff) )
172 end function getLogSubExp_RK

187 pure function getSumExp_RK(LogValue, maxLogValue) result(sumExp)
188#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
189 !DEC$ ATTRIBUTES DLLEXPORT :: getSumExp_RK
190#endif
191 use Constants_mod, only: IK, RK, LOGTINY_RK
192 real(RK), intent(in), contiguous :: LogValue(:)
193 real(RK), intent(in) :: maxLogValue
194 real(RK) :: sumExp
195 integer(IK) :: i
196#if DEBUG_ENABLED
197 real(RK) :: dummy
198#endif
199 sumExp = 0._RK
200 do i = 1, size(LogValue)
201#if DEBUG_ENABLED
202 dummy = LogValue(i) - maxLogValue
203 if (dummy > LOGTINY_RK) then
204 dummy = exp(dummy)
205 else
206 dummy = 0._RK
207 end if
208 sumExp = sumExp + dummy
209#else
210 sumExp = sumExp + exp(LogValue(i) - maxLogValue)
211#endif
212 end do
213 end function getSumExp_RK
