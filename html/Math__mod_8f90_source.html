143#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)

146 use Constants_mod, only: RK, LOGTINY_RK
147 real(RK) , intent(in) :: logValueLarger, logValueSmaller

149#if CHECK_ENABLED
150 block
151 use iso_fortran_env, only: output_unit
152 if (logValueSmaller > logValueLarger) then

154 error stop
155 end if
156 end block
157#endif
158 logSubExp = logValueLarger
159 diff = logValueSmaller - logValueLarger

211 pure function getLogSumExp_RK(LogValue, maxLogValue) result(logSumExp)
212#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
213 !DEC$ ATTRIBUTES DLLEXPORT :: getLogSumExp_RK

277 pure function getSumExp_RK(LogValue, maxLogValue) result(sumExp)
278#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
279 !DEC$ ATTRIBUTES DLLEXPORT :: getSumExp_RK
280#endif
281 use Constants_mod, only: IK, RK, LOGTINY_RK
282 real(RK), intent(in), contiguous :: LogValue(:)
283 real(RK), intent(in) :: maxLogValue
284 real(RK) :: sumExp
285 integer(IK) :: i
286#if DEBUG_ENABLED
287 real(RK) :: dummy
288#endif
289 sumExp = 0._RK
290 do i = 1, size(LogValue)
291#if DEBUG_ENABLED

293 if (dummy > LOGTINY_RK) then
294 dummy = exp(dummy)
295 else
296 dummy = 0._RK
297 end if

301#endif
302 end do
