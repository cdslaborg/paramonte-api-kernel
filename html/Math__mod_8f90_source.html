579  pure function getellvolcoef(nd) result(ellVolCoef)
580 #if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
581  !DEC$ ATTRIBUTES DLLEXPORT :: getEllVolCoef
582 #endif
583  use constants_mod, only: ik, rk, pi
584  implicit none
585  integer(IK), intent(in) :: nd
586  integer(IK) :: i,k
587  real(rk), parameter :: four_pi = pi * 4._rk
588  real(rk) :: ellvolcoef
589  if (mod(nd,2_ik)==0_ik) then ! nd is even
590 
591  ellvolcoef = pi
592  do i = 2_ik, nd / 2_ik
593  ellvolcoef = ellvolcoef * pi / i ! nd = 2k ; ellVolCoef = PI^(k) / Factorial(k)
594  end do
595 
596  else ! nd is an odd integer
597 
598  ! nd = 2k-1 ; gamma(nd/2 + 1) = gamma(k + 1/2) ; gamma(k+1/2) = sqrt(PI) * (2k)! / (4^k * k!)
599  k = (nd + 1_ik) / 2_ik
600 
601  ! This is to avoid an extra unnecessary division of ellVolCoef by PI
602  ellvolcoef = 4._rk / (k + 1_ik)
603 
604  do i = k+2_ik, 2_ik*k
605  ! ellVolCoef = PI^(k-1/2) / gamma(k+1/2) = PI^(k+1) * 4^k * k! / (2k)!
606  ellvolcoef = ellvolcoef * four_pi / i
607  end do
608 
609  end if
610  end function getellvolcoef

895  real(rk), intent(in) :: exponent, loggammaexponent, lowerlim
896  real(rk), optional, intent(in) :: tolerance
897  real(rk) :: gammacontfrac
898  integer(IK), parameter :: itmax = 100_ik
899  real(rk), parameter :: eps_default = epsilon(lowerlim)
900  real(rk), parameter :: fpmin_default = tiny(lowerlim) / eps_default
901  real(rk) :: eps, fpmin
902  real(rk) :: an,b,c,d,del,h
903  integer(IK) :: iter
904  if (lowerlim == 0._rk) then
905  gammacontfrac = 1._rk
906  return
907  end if
908  if (present(tolerance)) then
909  eps = tolerance
910  fpmin = tiny(lowerlim) / eps
911  else
912  eps = eps_default
913  fpmin = fpmin_default
914  end if
915  b = lowerlim + 1._rk - exponent
916  c = 1._rk / fpmin
917  d = 1._rk / b
918  h = d
919  do iter = 1, itmax
920  an = -iter * (iter-exponent)
921  b = b + 2.0_rk
922  d = an*d + b
923  if (abs(d) < fpmin) d = fpmin
924  c = b + an/c
925  if (abs(c) < fpmin) c = fpmin
926  d = 1.0_rk / d
927  del = d*c
928  h = h*del
929  if (abs(del-1._rk) <= eps) exit
930  end do

932  if (iter>itmax) then
933  gammacontfrac = -huge_rk
934  else

936  end if
