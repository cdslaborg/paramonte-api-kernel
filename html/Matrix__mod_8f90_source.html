190  real(rk) , intent(out) :: sqrtdetinvposdefmat !
191  real(rk) :: choleskylower(nd,nd) ! Cholesky factor
192  real(rk) :: diagonal(nd) ! diagonal terms of the inverse matrix
193  real(rk) :: summ

209  end if

212  choleskylower(i,i) = 1._rk / diagonal(i)
213  do j = i+1,nd
214  summ = 0._rk
215  do k = i,j-1
216  summ = summ - choleskylower(j,k) * choleskylower(k,i)
217  end do
218  choleskylower(j,i) = summ / diagonal(j)
219  end do
220  end do
221  do i = 1,nd
222  do j = i,nd
223  matinvmat(j,i) = dot_product(choleskylower(j:nd,j),choleskylower(j:nd,i))

226  end do

512  subroutine getlu(nd,MatrixLU,Permutation,parity) ! ,errorOccurred)
513 #if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
514  !DEC$ ATTRIBUTES DLLEXPORT :: getLU
515 #endif
516  use constants_mod, only: rk, ik
517  use, intrinsic :: iso_fortran_env, only: output_unit
518  integer(IK) , intent(in) :: nd
519  integer(IK) , intent(out) :: Permutation(nd)
520  real(RK) , intent(inout) :: MatrixLU(nd,nd)
521  real(RK) , intent(out) :: parity
522  !logical , intent(out) :: errorOccurred
523  real(RK) , parameter :: TINY = 1.e-20_rk
524  real(RK) :: aamax,dum,summ,vv(nd)
525  integer(IK) :: i,imax,j,k
526  !errorOccurred = .false.
527  parity = 1._rk
528  do i = 1,nd
529  aamax = 0._rk
530  do j=1,nd
531  if ( abs(matrixlu(i,j)) > aamax ) aamax = abs( matrixlu(i,j) )
532  end do
533  if (aamax == 0._rk) then
534  ! LCOV_EXCL_START
535  write(output_unit,"(A)") "Statistics@getLU() failed. Singular matrix detected."
536  error stop
537  !errorOccurred = .true.
538  !return
539  ! LCOV_EXCL_STOP
540  end if
541  vv(i) = 1._rk/aamax
542  end do
543  do j=1,nd
544  do i=1,j-1
545  summ = matrixlu(i,j)
546  do k=1,i-1
547  summ = summ - matrixlu(i,k) * matrixlu(k,j)
548  end do
549  matrixlu(i,j) = summ
550  end do
551  aamax = 0._rk
552  do i = j, nd
553  summ = matrixlu(i,j)
554  do k = 1, j-1
555  summ = summ - matrixlu(i,k) * matrixlu(k,j)
556  end do
557  matrixlu(i,j) = summ
558  dum = vv(i) * abs(summ)
559  if (dum >= aamax) then
560  imax = i
561  aamax = dum
562  end if
563  end do
564  if (j /= imax)then
565  do k=1,nd
566  dum = matrixlu(imax,k)
567  matrixlu(imax,k) = matrixlu(j,k)
568  matrixlu(j,k) = dum
569  end do
570  parity = - parity
571  vv(imax) = vv(j)
572  end if
573  permutation(j) = imax
574  if (matrixlu(j,j) == 0._rk) matrixlu(j,j) = tiny
575  if (j /= nd) then
576  dum = 1._rk / matrixlu(j,j)
577  do i = j+1,nd
578  matrixlu(i,j) = matrixlu(i,j) * dum
579  end do
580  endif
581  end do
582  end subroutine getlu
