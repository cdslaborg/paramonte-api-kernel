279 pure subroutine getChoLow(nd,ChoLowCovUpp,ChoDia)
280#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
281 !DEC$ ATTRIBUTES DLLEXPORT :: getChoLow
282#endif
283 use Constants_mod, only: RK, IK
284 implicit none
285 integer(IK), intent(in) :: nd
286 real(RK) , intent(inout) :: ChoLowCovUpp(nd,nd) ! Upper triangle + diagonal is input matrix, lower is output.
287 real(RK) , intent(out) :: ChoDia(nd)
288 real(RK) :: summ
289 integer(IK) :: i
290 do i=1,nd
291 summ = ChoLowCovUpp(i,i) - dot_product(ChoLowCovUpp(i,1:i-1),ChoLowCovUpp(i,1:i-1))
292 if (summ <= 0._RK) then
293 ChoDia(1) = -1._RK
294 return
295 end if
296 ChoDia(i) = sqrt(summ)
297 ChoLowCovUpp(i+1:nd,i) = ( ChoLowCovUpp(i,i+1:nd) - matmul(ChoLowCovUpp(i+1:nd,1:i-1),ChoLowCovUpp(i,1:i-1)) ) / ChoDia(i)
298 end do
299 end subroutine getChoLow

432 pure subroutine getInvPosDefMatSqrtDet(nd,MatInvMat,sqrtDetInvPosDefMat)
433#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
434 !DEC$ ATTRIBUTES DLLEXPORT :: getInvPosDefMatSqrtDet
435#endif
436 use Constants_mod, only: RK, IK ! LCOV_EXCL_LINE
437 implicit none
438 integer(IK), intent(in) :: nd
439 real(RK) , intent(inout) :: MatInvMat(nd,nd) ! input: upper half is covariance matrix, output: inverse matrix
440 real(RK) , intent(out) :: sqrtDetInvPosDefMat !
441 real(RK) :: ChoLow(nd,nd) ! The lower triangle of the Cholesky factorization.
442 real(RK) :: ChoDia(nd) ! The diagonal elements of the inverse matrix
443 real(RK) :: summ
444 integer(IK) :: i,j,k
445 if (nd==1_IK) then
446 MatInvMat(1,1) = 1._RK / MatInvMat(1,1)
447 sqrtDetInvPosDefMat = MatInvMat(1,1)
448 return
449 end if
450 do j=1,nd
451 do i=1,j
452 ChoLow(i,j) = MatInvMat(i,j)
453 end do
454 end do
455 call getChoLow(nd,ChoLow,ChoDia)
456 if (ChoDia(1)<0._RK) then
457 sqrtDetInvPosDefMat = -1._RK
458 return
459 end if
460 sqrtDetInvPosDefMat = 1._RK / product(ChoDia)
461 do i = 1,nd
462 ChoLow(i,i) = 1._RK / ChoDia(i)
463 do j = i+1,nd
464 summ = 0._RK
465 do k = i,j-1
466 summ = summ - ChoLow(j,k) * ChoLow(k,i)
467 end do
468 ChoLow(j,i) = summ / ChoDia(j)
469 end do
470 end do
471 do i = 1,nd
472 do j = i,nd
473 MatInvMat(j,i) = dot_product(ChoLow(j:nd,j),ChoLow(j:nd,i))
474 end do
475 MatInvMat(i,i:nd) = MatInvMat(i:nd,i)
476 end do
477 end subroutine getInvPosDefMatSqrtDet

499 pure function getInvMatFromChoLow(nd,ChoLow,ChoDia) result(InvMatFromChoLow)
500#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
501 !DEC$ ATTRIBUTES DLLEXPORT :: getInvMatFromChoLow
502#endif
503 use Constants_mod, only: RK, IK
504 implicit none
505 integer(IK), intent(in) :: nd
506 real(RK) , intent(in) :: ChoDia(nd)
507 real(RK) , intent(in) :: ChoLow(nd,nd)
508 real(RK) :: InvMatFromChoLow(nd,nd)
509 real(RK) :: summ
510 integer(IK) :: i,j,k

512 InvMatFromChoLow(1,1) = 1._RK / ChoDia(1)**2

515 InvMatFromChoLow = 0._RK

869 subroutine solveLinearSystem(nd, LU, Permutation, InputSolution)
870#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
871 !DEC$ ATTRIBUTES DLLEXPORT :: solveLinearSystem
872#endif
873 use Constants_mod, only: RK, IK
874 integer(IK), intent(in) :: nd
875 integer(IK), intent(in) :: Permutation(nd)
876 real(RK) , intent(in) :: LU(nd,nd)
877 real(RK) , intent(inout) :: InputSolution(nd)
878 integer(IK) :: i,ii
879 real(RK) :: summ
880 ii = 0
881 do i=1,nd
882 summ = InputSolution(Permutation(i))
883 InputSolution(Permutation(i)) = InputSolution(i)
884 if (ii /= 0) then
885 summ = summ - dot_product( LU(i,ii:i-1) , InputSolution(ii:i-1) )
886 else if (summ /= 0._RK) then
887 ii = i
888 end if
889 InputSolution(i) = summ
890 end do
891 do i = nd,1,-1
892 InputSolution(i) = ( InputSolution(i) - dot_product( LU(i,i+1:nd) , InputSolution(i+1:nd)) ) / LU(i,i)
893 end do
894 end subroutine solveLinearSystem

939 end if

947 end do

949 end do

952 summ = MatrixLU(i,j)
953 do k = 1, j-1
954 summ = summ - MatrixLU(i,k) * MatrixLU(k,j)
955 end do
956 MatrixLU(i,j) = summ
957 dum = vv(i) * abs(summ)
958 if (dum >= aamax) then
959 imax = i
960 aamax = dum
961 end if
962 end do
963 if (j /= imax)then
964 do k=1,nd
965 dum = MatrixLU(imax,k)
966 MatrixLU(imax,k) = MatrixLU(j,k)
967 MatrixLU(j,k) = dum
968 end do
969 parity = - parity
970 vv(imax) = vv(j)
971 end if
972 Permutation(j) = imax
973 if (MatrixLU(j,j) == 0._RK) MatrixLU(j,j) = TINY
974 if (j /= nd) then
975 dum = 1._RK / MatrixLU(j,j)
976 do i = j+1,nd
977 MatrixLU(i,j) = MatrixLU(i,j) * dum
978 end do
979 endif
