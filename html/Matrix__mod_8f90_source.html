127 function genDet(Matrix,failed) result(determinant)
128#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
129 !DEC$ ATTRIBUTES DLLEXPORT :: genDet
130#endif
131 use Constants_mod, only: RK, IK
132 implicit none
133 real(RK) , intent(in) , contiguous :: Matrix(:,:)
134 logical , intent(out) , optional :: failed
135 real(RK) :: DummyMat(size(Matrix,1),size(Matrix,1))
136 real(RK) :: determinant
137 logical :: failed_def
138 integer(IK) :: Permutation(size(Matrix,1))
139 integer(IK) :: j, ndim
140 ndim = size(Matrix,1)
141 DummyMat = Matrix
142 call getLUP(DummyMat,Permutation,determinant,failed_def) ! This returns determinant as +-1.
143 if (present(failed)) then
144 failed = failed_def
145 if (failed) return
146 elseif (failed_def) then
147 error stop ! LCOV_EXCL_LINE
148 end if
149 do j = 1, ndim
150 determinant = determinant * DummyMat(j,j)
151 end do
152 end function genDet

171 function genSqrtDetPosDefMat(nd,PosDefMat) result(sqrtDetPosDefMat)
172#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
173 !DEC$ ATTRIBUTES DLLEXPORT :: genSqrtDetPosDefMat
174#endif
175 use Constants_mod, only: RK, IK
176 implicit none
177 integer(IK), intent(in) :: nd
178 real(RK) , intent(in) :: PosDefMat(nd,nd)
179 real(RK) :: ChoLow(nd,nd)
180 real(RK) :: ChoDia(nd)
181 real(RK) :: sqrtDetPosDefMat
182 integer(IK) :: i,j
183 do j=1,nd
184 do i=1,j
185 ChoLow(i,j) = PosDefMat(i,j)
186 end do
187 end do
188 call getChoLow(nd, ChoLow, ChoDia)
189 if (ChoDia(1)<0._RK) then
190 sqrtDetPosDefMat = -1._RK
191 return
192 end if
193 sqrtDetPosDefMat = product(ChoDia)
194 end function genSqrtDetPosDefMat

211 pure subroutine getLogSqrtDetPosDefMat(nd,PosDefMat,logSqrtDetPosDefMat,failed)
212#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
213 !DEC$ ATTRIBUTES DLLEXPORT :: getLogSqrtDetPosDefMat
214#endif
215 use Constants_mod, only: RK, IK
216 implicit none
217 integer(IK), intent(in) :: nd
218 real(RK) , intent(inout) :: PosDefMat(nd,nd) ! Upper triangle + diagonal is input matrix, lower is output.
219 real(RK) , intent(out) :: logSqrtDetPosDefMat
220 logical , intent(out) :: failed
221 real(RK) :: ChoDia(nd)
222 real(RK) :: summ
223 integer(IK) :: i
224 failed = .false.
225 do i=1,nd
226 summ = PosDefMat(i,i) - dot_product(PosDefMat(i,1:i-1),PosDefMat(i,1:i-1))
227 if (summ <= 0._RK) then
228 failed = .true.
229 return
230 end if
231 ChoDia(i) = sqrt(summ)
232 PosDefMat(i+1:nd,i) = ( PosDefMat(i,i+1:nd) - matmul(PosDefMat(i+1:nd,1:i-1),PosDefMat(i,1:i-1)) ) / ChoDia(i)
233 end do
234 logSqrtDetPosDefMat = sum(log(ChoDia))
235 end subroutine getLogSqrtDetPosDefMat

256 pure logical function isPosDef(nd,Matrix)
257#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
258 !DEC$ ATTRIBUTES DLLEXPORT :: isPosDef
259#endif
260 use Constants_mod, only: RK, IK
261 implicit none
262 integer(IK), intent(in) :: nd
263 real(RK) , intent(in) :: Matrix(nd,nd)
264 real(RK) :: Array(nd,nd),p(nd)
265 real(RK) :: dummySum
266 integer(IK) :: i,j,k
267 isPosDef = .true.
268 do i = 1, nd
269 do j = i, nd
270 dummySum = Matrix(i,j)
271 do k = i-1,1,-1
272 dummySum = dummySum - Array(i,k) * Array(j,k)
273 end do
274 if(i==j)then
275 if(dummySum<=0._RK) then
276 isPosDef = .false.
277 return
278 end if
279 p(i) = sqrt(dummySum)
280 else
281 Array(j,i)=dummySum/p(i)
282 endif
283 end do
284 end do
285 end function isPosDef

325 pure function genSortedPosDefMat(rank,PosDefMatUpper,lenColIndx,ColIndx,ColIndxMap) result(SortedPosDefMatUpper)
326#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
327 !DEC$ ATTRIBUTES DLLEXPORT :: genSortedPosDefMat
328#endif
329 use Constants_mod, only: RK, IK
330 integer(IK), intent(in) :: rank, lenColIndx, ColIndx(lenColIndx), ColIndxMap(lenColIndx)
331 real(RK), intent(in) :: PosDefMatUpper(rank,rank)
332 real(RK) :: SortedPosDefMatUpper(rank,rank)
333 integer(IK) :: iRow, iCol, iRowNew, iColNew, i
334 loopIndx: do i = 1, lenColIndx
335 do iCol = 1, rank
336 iColNew = iCol
337 if (iColNew==ColIndx(i)) then
338 iColNew = ColIndxMap(i)
339 elseif (iColNew==ColIndxMap(i)) then
340 iColNew = ColIndx(i)
341 end if
342 do iRow = 1, iCol
343 iRowNew = iRow
344 if (iRowNew==ColIndx(i)) then
345 iRowNew = ColIndxMap(i)
346 elseif (iRowNew==ColIndxMap(i)) then
347 iRowNew = ColIndx(i)
348 end if
349 if (iRowNew>iColNew) then
350 SortedPosDefMatUpper(iRow,iCol) = PosDefMatUpper(iColNew,iRowNew)
351 else
352 SortedPosDefMatUpper(iRow,iCol) = PosDefMatUpper(iRowNew,iColNew)
353 end if
354 end do
355 end do
356 end do loopIndx
357 end function genSortedPosDefMat

510 PURE subroutine getRegresCoef(rankPDM,rankS11,rankS22,PosDefMat,RegresCoefMat,failed,SchurComplement)
511#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
512 !DEC$ ATTRIBUTES DLLEXPORT :: getRegresCoef
513#endif
514 use Constants_mod, only: RK, IK ! LCOV_EXCL_LINE
515 implicit none
516 integer(IK) , intent(in) :: rankPDM, rankS11, rankS22
517 real(RK) , intent(in) :: PosDefMat(rankPDM,rankPDM)
518 real(RK) , intent(out) :: RegresCoefMat(rankS11,rankS22)
519 logical , intent(out) :: failed
520 real(RK) , intent(out), optional :: SchurComplement(rankS11,rankS11)
521 real(RK) :: InvS22(rankS22,rankS22), S22(rankS22,rankS22)
522 integer(IK) :: startS22
523 startS22 = rankS11 + 1
524 S22 = PosDefMat(startS22:rankPDM,startS22:rankPDM)
525 if (rankS22==1) then
526 InvS22(1,1) = 1._RK/S22(1,1)
527 else
528 InvS22 = genInvPosDefMat(rankS22,S22)
529 end if
530 failed = InvS22(1,1) < 0._RK
531 if (failed) return
532 RegresCoefMat = matmul( PosDefMat(1:rankS11,startS22:rankPDM) , InvS22 )
533 if (present(SchurComplement)) then
534 SchurComplement = PosDefMat(1:rankS11,1:rankS11) - matmul( RegresCoefMat , PosDefMat(startS22:rankPDM,1:rankS11) )
535 end if
536 end subroutine getRegresCoef
