85 function construct(methodName) result(self)
86#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
87 !DEC$ ATTRIBUTES DLLEXPORT :: construct
88#endif
89 use, intrinsic :: iso_fortran_env, only: output_unit
90 use Constants_mod, only: NULL_SK, IK, PMSM
91 use Decoration_mod, only: TAB
92 use String_mod, only: val2str
93 implicit none
94 character(*), intent(in) :: methodName
95 type(Parallelism_type) :: self
96
97 self%isSinglChain = .false.
98 self%isMultiChain = .false.
99 self%singlChain = "singleChain"
100 self%multiChain = "multiChain"
101 self%def = self%singlChain
102 self%null = repeat(NULL_SK, MAX_LEN_PARALLELISM_MODEL)
103
104 self%desc = &
105 "parallelism is a string variable that represents the parallelization method to be used in "// methodName //". &
106 &The string value must be enclosed by either single or double quotation marks when provided as input. &
107 &Two options are currently supported:\n\n&
108 & parallelism = '" // self%multiChain // "'\n\n&
109 & This method uses the Prefect Parallelism scheme in which multiple MCMC chains are generated &
110 &independently of each other. In this case, multiple output MCMC chain files will also be generated.\n\n&
111 & parallelism = '" // self%singlChain // "'\n\n&
112 & This method uses the fork-style parallelization scheme. &
113 &A single MCMC chain file will be generated in this case. At each MCMC step multiple proposal steps &
114 &will be checked in parallel until one proposal is accepted.\n\n&
115 &Note that in serial mode, there is no parallelism. Therefore, this option does not affect non-parallel simulations &
116 &and its value is ignored. The serial mode is equivalent to either of the parallelism methods with only one simulation &
117 &image (processor, core, or thread). &
118 &The default value is parallelism = '" // self%def // "'. &
119 &Note that the input values are case-INsensitive and white-space characters are ignored."
120 end function construct

135 subroutine set(self,parallelism)
136#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
137 !DEC$ ATTRIBUTES DLLEXPORT :: set
138#endif
139 use String_mod, only: genLowerCase, replace
140 implicit none
141 class(Parallelism_type), intent(inout) :: self
142 character(*), intent(in) :: parallelism
143 character(:), allocatable :: parallelismLowerCase
144 self%val = trim(adjustl(replace(parallelism," ", "")))
145 if (self%val==trim(adjustl(self%null))) self%val = trim(adjustl(self%def))
146 parallelismLowerCase = genLowerCase(self%val)
147 self%isSinglChain = parallelismLowerCase == genLowerCase(self%singlChain)
148 self%isMultiChain = parallelismLowerCase == genLowerCase(self%multiChain)
149 end subroutine set

153 subroutine checkForSanity(Parallelism,Err,methodName)
154#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
155 !DEC$ ATTRIBUTES DLLEXPORT :: checkForSanity
156#endif
157 use Err_mod, only: Err_type
158 use String_mod, only: val2str
159 use Constants_mod, only: NLC
160 implicit none
161 class(Parallelism_type), intent(in) :: Parallelism
162 character(*), intent(in) :: methodName
163 type(Err_type), intent(inout) :: Err
164 character(*), parameter :: PROCEDURE_NAME = "@checkForSanity()"
165 if ( .not.(Parallelism%isSinglChain .or. Parallelism%isMultiChain) ) then
166 ! LCOV_EXCL_START
167 Err%occurred = .true.
168 Err%msg = Err%msg // &
169 MODULE_NAME // PROCEDURE_NAME // ": Error occurred. &
170 &The input requested parallelization method (" // Parallelism%val // &
171 ") represented by variable parallelism cannot be anything other than &
172 &'singleChain' or 'multiChain'. If you don't know an appropriate value &
173 &for Parallelism, drop it from the input list. " // methodName // &
174 " will automatically assign an appropriate value to it."//NLC//NLC
175 end if
176 ! LCOV_EXCL_STOP
177 end subroutine checkForSanity
