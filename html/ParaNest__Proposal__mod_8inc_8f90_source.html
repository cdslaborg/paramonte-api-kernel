453 end if
454 end do loopCheckInclusion
455 end do

457 write(*,*)
458 write(*,*)

640 !write(*,*) "isMultiParition", Proposal%count

649 end if
650 end do

653 call random_number(unifrnd)

660 end if blockMembershipCount
