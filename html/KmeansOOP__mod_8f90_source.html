221 subroutine runKmeans(Kmeans, Point, nd, np, nc, nt, niterMax, nzsciMax, relTol, InitCenter)
222#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
223 !DEC$ ATTRIBUTES DLLEXPORT :: runKmeans
224#endif
225 use Constants_mod, only: IK, RK, NEGINF_IK
226
227 implicit none
228
229 class(KmeansOOP_type), intent(inout) :: Kmeans
230 integer(IK) , intent(in) :: nd, np, nc, nt
231 real(RK) , intent(in) :: Point(nd,np) ! This must appear after nd, np declarations.
232 integer(IK) , intent(in), optional :: niterMax, nzsciMax
233 real(RK) , intent(in), optional :: relTol, InitCenter(nd,nc,nt)
234
235 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@runKmeans()"
236
237 real(RK) :: potential ! The sum of all minimum distances squared.
238 real(RK) :: distanceSq ! distance of each point ip from a given cluster center.
239 real(RK) :: minPotential ! The minimum of sum of all minimum distances squared across all Kmeans tries.
240 real(RK) :: SumPoint(nd,nc) ! The sum of the points in each cluster.
241 real(RK) :: MinDistanceSq(np) ! minimum distance of each point ip from all centers of the clusters.
242 integer(IK) :: Membership(np) ! current (old) Memberships of the clusters.
243
244 logical :: isPresentInitCenter
245 logical :: convergenceOccurred
246 integer :: ip, ic, it
247
248 Kmeans%itopt = 0_IK
249 Kmeans%failed = .true.
250
251 if (present(relTol)) Kmeans%relTol = relTol
252 if (present(niterMax)) Kmeans%niterMax = niterMax
253 if (present(nzsciMax)) Kmeans%nzsciMax = nzsciMax
254
255 if (nd > Kmeans%ndMax .or. np > Kmeans%npMax .or. nc > Kmeans%ncMax .or. nt > Kmeans%ntMax) then
256 call Kmeans%allocate(nd, np, nc, nt)
257 else
258 Kmeans%nd = nd
259 Kmeans%np = np
260 Kmeans%nc = nc
261 Kmeans%nt = nt
262 end if
263
264 minPotential = huge(minPotential)
265 isPresentInitCenter = present(InitCenter)
266
267 loopMultipleKmeansTry: do it = 1, Kmeans%nt
268
269 Kmeans%Try(it)%Err%occurred = .false.
270 Kmeans%Try(it)%Err%stat = NEGINF_IK
271
272 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
273 ! Find the distances of all points to all cluster centers. If not given, initialize the centers via the Kmeans++ algorithm.
274 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
275
276 if(isPresentInitCenter) then
277 !%%%%%%%%%%%%%%%%%%%
278 ic = 1
279 do ip = 1, np
280 MinDistanceSq(ip) = sum( (InitCenter(1:nd,ic,it) - Point(1:nd,ip))**2 )
281 Membership(ip) = ic
282 end do
283 !%%%%%%%%%%%%%%%%%%%
284 do ic = 2, nc - 1
285 do ip = 1, np
286 distanceSq = sum( (InitCenter(1:nd,ic,it) - Point(1:nd,ip))**2 )
287 if (distanceSq < MinDistanceSq(ip)) then
288 MinDistanceSq(ip) = distanceSq
289 Membership(ip) = ic
290 end if
291 end do
292 end do
293 !%%%%%%%%%%%%%%%%%%%
294 ic = nc
295 SumPoint = 0._RK
296 potential = 0._RK
297 do ip = 1, np
298 distanceSq = sum( (InitCenter(1:nd,ic,it) - Point(1:nd,ip))**2 )
299 if (distanceSq < MinDistanceSq(ip)) then
300 MinDistanceSq(ip) = distanceSq
301 Membership(ip) = ic
302 end if
303 SumPoint(1:nd,Membership(ip)) = SumPoint(1:nd,Membership(ip)) + Point(1:nd,ip)
304 Kmeans%Try(it)%Size(Membership(ip)) = Kmeans%Try(it)%Size(Membership(ip)) + 1_IK
305 potential = potential + MinDistanceSq(ip)
306 end do
307 !%%%%%%%%%%%%%%%%%%%
308 else
309 call runKPP(nc, nd, np, Point, SumPoint, Membership, Kmeans%Try(it)%Size, potential)
310 end if
311
312 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
313 ! Refine all centers until convergence.
314 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
315
316 Kmeans%Try(it)%nzsci = 0_IK
317 Kmeans%Try(it)%niter = 1_IK
318
319 loopCenterRefinement: do
320
321 ! compute the new cluster centers and create the new memberships.
322
323 !%%%%%%%%%%%%%%%%%%%
324 ic = 1
325 Kmeans%Try(it)%Center(1:nd,ic) = SumPoint(1:nd,ic) / real(Kmeans%Try(it)%Size(ic), kind = RK)
326 do ip = 1, np
327 Kmeans%Try(it)%NormedPoint(1:nd,ip,ic) = Kmeans%Try(it)%Center(1:nd,ic) - Point(1:nd,ip)
328 Kmeans%Try(it)%MinDistanceSq(ip) = sum( Kmeans%Try(it)%NormedPoint(1:nd,ip,ic)**2 )
329 Kmeans%Try(it)%Membership(ip) = ic
330 end do
331 !%%%%%%%%%%%%%%%%%%%
332 do ic = 2, nc - 1
333 Kmeans%Try(it)%Center(1:nd,ic) = SumPoint(1:nd,ic) / real(Kmeans%Try(it)%Size(ic), kind = RK)
334 do ip = 1, np
335 Kmeans%Try(it)%NormedPoint(1:nd,ip,ic) = Kmeans%Try(it)%Center(1:nd,ic) - Point(1:nd,ip)
336 distanceSq = sum( Kmeans%Try(it)%NormedPoint(1:nd,ip,ic)**2 )
337 if (distanceSq < Kmeans%Try(it)%MinDistanceSq(ip)) then
338 Kmeans%Try(it)%MinDistanceSq(ip) = distanceSq
339 Kmeans%Try(it)%Membership(ip) = ic
340 end if
341 end do
342 end do
343 !%%%%%%%%%%%%%%%%%%%
344 ic = nc
345 convergenceOccurred = .true.
346 Kmeans%Try(it)%potential = 0._RK
347 Kmeans%Try(it)%Center(1:nd,ic) = SumPoint(1:nd,ic) / real(Kmeans%Try(it)%Size(ic), kind = RK)
348 do ip = 1, np
349 Kmeans%Try(it)%NormedPoint(1:nd,ip,ic) = Kmeans%Try(it)%Center(1:nd,ic) - Point(1:nd,ip)
350 distanceSq = sum( Kmeans%Try(it)%NormedPoint(1:nd,ip,ic)**2 )
351 if (distanceSq < Kmeans%Try(it)%MinDistanceSq(ip)) then
352 Kmeans%Try(it)%MinDistanceSq(ip) = distanceSq
353 Kmeans%Try(it)%Membership(ip) = ic
354 end if
355 if (Kmeans%Try(it)%Membership(ip)/=Membership(ip)) then
356 convergenceOccurred = .false.
357 Kmeans%Try(it)%Size(Membership(ip)) = Kmeans%Try(it)%Size(Membership(ip)) - 1_IK
358 Kmeans%Try(it)%Size(Kmeans%Try(it)%Membership(ip)) = Kmeans%Try(it)%Size(Kmeans%Try(it)%Membership(ip)) + 1_IK
359 end if
360 Kmeans%Try(it)%potential = Kmeans%Try(it)%potential + Kmeans%Try(it)%MinDistanceSq(ip)
361 end do
362 !%%%%%%%%%%%%%%%%%%%
363
364 if (any(Kmeans%Try(it)%Size==0_IK)) then
365 ! LCOV_EXCL_START
366 Kmeans%Try(it)%niter = 1_IK
367 Kmeans%Try(it)%nzsci = Kmeans%Try(it)%nzsci + 1_IK
368
369 if (Kmeans%Try(it)%nzsci > Kmeans%nzsciMax) then
370 !Kmeans%Try(it)%Err%msg = "Kmeans%Try(it)%nzsci > Kmeans%nzsciMax"
371 Kmeans%Try(it)%Err%occurred = .true.
372 Kmeans%Try(it)%Err%stat = 2_IK
373 exit loopCenterRefinement
374 end if
375
376 ! find new random Centers and start over.
377
378 call runKPP(nc, nd, np, Point, SumPoint, Membership, Kmeans%Try(it)%Size, potential)
379
380 cycle loopCenterRefinement
381
382#if DEBUG_ENABLED || TESTING_ENABLED
383 elseif (any(Kmeans%Try(it)%Size<0_IK)) then
384 block
385 use Err_mod, only: abort
386 Kmeans%Try(it)%Err%msg = PROCEDURE_NAME//": Internal error occurred - Kmeans%Try(it)%Size < 0. Please report this issue along with circumstances to the developers."
387 call abort(Kmeans%Try(it)%Err)
388 return
389 end block
390#endif
391 end if
392 ! LCOV_EXCL_STOP
393
394 ! if `niterMax` has reached, quit.
395
396 if (Kmeans%Try(it)%niter < Kmeans%niterMax) then
397
398 ! If convergence has occurred, exit loopCenterRefinement.
399
400 if (convergenceOccurred .or. abs(Kmeans%Try(it)%potential-potential)/Kmeans%Try(it)%potential < Kmeans%relTol) exit loopCenterRefinement
401
402 Kmeans%Try(it)%niter = Kmeans%Try(it)%niter + 1_IK
403
404 do ip = 1, np
405 SumPoint(1:nd,Membership(ip)) = SumPoint(1:nd,Membership(ip)) - Point(1:nd,ip)
406 SumPoint(1:nd,Kmeans%Try(it)%Membership(ip)) = SumPoint(1:nd,Kmeans%Try(it)%Membership(ip)) + Point(1:nd,ip)
407 Membership(ip) = Kmeans%Try(it)%Membership(ip)
408 end do
409 potential = Kmeans%Try(it)%potential
410
411 else
412
413 !Kmeans%Try(it)%Err%msg = "Kmeans%Try(it)%niter > Kmeans%niterMax"
414 Kmeans%Try(it)%Err%occurred = .true.
415 Kmeans%Try(it)%Err%stat = 1_IK
416 exit loopCenterRefinement
417
418 end if
419
420 end do loopCenterRefinement
421
422 if (.not. Kmeans%Try(it)%Err%occurred) then
423 if (Kmeans%Try(it)%potential < minPotential) Kmeans%itopt = it
424 Kmeans%failed = .false.
425 end if
426
427 end do loopMultipleKmeansTry
428
429 end subroutine runKmeans

458 subroutine runKPP(nc, nd, np, Point, SumPoint, Membership, Size, potential)
459#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
460 !DEC$ ATTRIBUTES DLLEXPORT :: runKPP
461#endif
462 use Statistics_mod, only: genRandInt
463
464 implicit none
465
466 integer(IK) , intent(in) :: np, nd, nc
467 real(RK) , intent(in) :: Point(nd,np)
468 real(RK) , intent(out) :: SumPoint(nd,nc)
469 integer(IK) , intent(out) :: Membership(np)
470 integer(IK) , intent(out) :: Size(nc)
471 real(RK) , intent(out) :: potential
472
473 real(RK) :: dummy
474 real(RK) :: distanceSq
475 real(RK) :: Center(nd,nc)
476 real(RK) :: MinDistanceSq(np) ! An array representing the minimum distances of individual points from all clusters.
477 real(RK) :: CumSumDistSq(0:np) ! sum of distance squared of points up to the given index of the vector.
478 integer(IK) :: ic, ip
479
480 CumSumDistSq(0) = 0._RK ! This element must always remain zero.
481
482 ic = 1_IK
483 Center(1:nd,ic) = Point( 1:nd , genRandInt(1,np) ) ! perhaps this random assignment is unnecessary, after all, Point is supposed to be of random order.
484 do ip = 1, np
485 MinDistanceSq(ip) = sum( (Center(1:nd,ic) - Point(1:nd,ip))**2 ) ! find the distance-squared of each point to the # ic cluster Center.
486 CumSumDistSq(ip) = CumSumDistSq(ip-1) + MinDistanceSq(ip)
487 Membership(ip) = ic
488 end do
489
490 ic = 2_IK
491 call random_number(dummy)
492 dummy = dummy * CumSumDistSq(np)
493 Center(1:nd,ic) = Point( 1:nd, minloc(CumSumDistSq, dim = 1, mask = CumSumDistSq > dummy) - 1 ) ! -1 is essential to compensate for the 0 starting index of CumSumDistSq
494
495 ! find the rest of cluster centers
496
497 do ic = 2, nc - 1
498
499 do ip = 1, np
500 distanceSq = sum( (Center(1:nd,ic) - Point(1:nd,ip))**2 )
501 if (distanceSq < MinDistanceSq(ip)) then
502 MinDistanceSq(ip) = distanceSq
503 Membership(ip) = ic
504 end if
505 CumSumDistSq(ip) = CumSumDistSq(ip-1) + MinDistanceSq(ip)
506 end do
507
508 call random_number(dummy)
509 dummy = dummy * CumSumDistSq(np)
510 Center(1:nd,ic+1) = Point( 1:nd , minloc(CumSumDistSq, dim = 1, mask = CumSumDistSq > dummy) - 1 )
511
512 end do
513
514 Size = 0_IK
515 SumPoint = 0._RK
516 potential = 0._RK
517 do ip = 1, np
518 distanceSq = sum( (Center(1:nd,nc) - Point(1:nd,ip))**2 )
519 if (distanceSq < MinDistanceSq(ip)) then
520 MinDistanceSq(ip) = distanceSq
521 Membership(ip) = nc
522 end if
523 SumPoint(1:nd,Membership(ip)) = SumPoint(1:nd,Membership(ip)) + Point(1:nd,ip)
524 Size(Membership(ip)) = Size(Membership(ip)) + 1_IK
525 potential = potential + MinDistanceSq(ip)
526 end do
527
528 end subroutine runKPP
