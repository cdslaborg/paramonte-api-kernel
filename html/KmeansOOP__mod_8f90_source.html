222 subroutine runkmeans(Kmeans, Point, nd, np, nc, nt, niterMax, nzsciMax, relTol, InitCenter)
223#if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
224 !DEC$ ATTRIBUTES DLLEXPORT :: runKmeans
225#endif
226 use constants_mod, only: ik, rk, neginf_ik
227
228 implicit none
229
230 real(RK) , intent(in) :: Point(nd,np)
231 integer(IK) , intent(in) :: nd, np, nc, nt
232 integer(IK) , intent(in), optional :: niterMax, nzsciMax
233 real(RK) , intent(in), optional :: relTol, InitCenter(nd,nc,nt)
234 class(kmeansoop_type) :: Kmeans
235
236 character(*), parameter :: PROCEDURE_NAME = module_name//"@runKmeans()"
237
238 real(RK) :: potential ! The sum of all minimum distances squared.
239 real(RK) :: distanceSq ! distance of each point ip from a given cluster center.
240 real(RK) :: minPotential ! The minimum of sum of all minimum distances squared across all Kmeans tries.
241 real(RK) :: SumPoint(nd,nc) ! The sum of the points in each cluster.
242 real(RK) :: MinDistanceSq(np) ! minimum distance of each point ip from all centers of the clusters.
243 integer(IK) :: Membership(np) ! current (old) Memberships of the clusters.
244
245 logical :: isPresentInitCenter
246 logical :: convergenceOccurred
247 integer :: i, j, ip, ic, it
248
249 kmeans%itopt = 0_ik
250 kmeans%failed = .true.
251
252 if (present(reltol)) kmeans%relTol = reltol
253 if (present(nitermax)) kmeans%niterMax = nitermax
254 if (present(nzscimax)) kmeans%nzsciMax = nzscimax
255
256 if (nd > kmeans%ndMax .or. np > kmeans%npMax .or. nc > kmeans%ncMax .or. nt > kmeans%ntMax) then
257 call kmeans%allocate(nd, np, nc, nt)
258 else
259 kmeans%nd = nd
260 kmeans%np = np
261 kmeans%nc = nc
262 kmeans%nt = nt
263 end if
264
265 minpotential = huge(minpotential)
266 ispresentinitcenter = present(initcenter)
267
268 loopmultiplekmeanstry: do it = 1, kmeans%nt
269
270 kmeans%Try(it)%Err%occurred = .false.
271 kmeans%Try(it)%Err%stat = neginf_ik
272
273 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
274 ! Find the distances of all points to all cluster centers. If not given, initialize the centers via the Kmeans++ algorithm.
275 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
276
277 if(ispresentinitcenter) then
278 !%%%%%%%%%%%%%%%%%%%
279 ic = 1
280 do ip = 1, np
281 mindistancesq(ip) = sum( (initcenter(1:nd,ic,it) - point(1:nd,ip))**2 )
282 membership(ip) = ic
283 end do
284 !%%%%%%%%%%%%%%%%%%%
285 do ic = 2, nc - 1
286 do ip = 1, np
287 distancesq = sum( (initcenter(1:nd,ic,it) - point(1:nd,ip))**2 )
288 if (distancesq < mindistancesq(ip)) then
289 mindistancesq(ip) = distancesq
290 membership(ip) = ic
291 end if
292 end do
293 end do
294 !%%%%%%%%%%%%%%%%%%%
295 ic = nc
296 sumpoint = 0._rk
297 potential = 0._rk
298 do ip = 1, np
299 distancesq = sum( (initcenter(1:nd,ic,it) - point(1:nd,ip))**2 )
300 if (distancesq < mindistancesq(ip)) then
301 mindistancesq(ip) = distancesq
302 membership(ip) = ic
303 end if
304 sumpoint(1:nd,membership(ip)) = sumpoint(1:nd,membership(ip)) + point(1:nd,ip)
305 kmeans%Try(it)%Size(membership(ip)) = kmeans%Try(it)%Size(membership(ip)) + 1_ik
306 potential = potential + mindistancesq(ip)
307 end do
308 !%%%%%%%%%%%%%%%%%%%
309 else
310 call runkpp(nc, nd, np, point, sumpoint, membership, kmeans%Try(it)%Size, potential)
311 end if
312
313 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
314 ! Refine all centers until convergence.
315 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
316
317 kmeans%Try(it)%nzsci = 0_ik
318 kmeans%Try(it)%niter = 1_ik
319
320 loopcenterrefinement: do
321
322 ! compute the new cluster centers and create the new memberships.
323
324 !%%%%%%%%%%%%%%%%%%%
325 ic = 1
326 kmeans%Try(it)%Center(1:nd,ic) = sumpoint(1:nd,ic) / real(kmeans%Try(it)%Size(ic), kind = rk)
327 do ip = 1, np
328 kmeans%Try(it)%NormedPoint(1:nd,ip,ic) = kmeans%Try(it)%Center(1:nd,ic) - point(1:nd,ip)
329 kmeans%Try(it)%MinDistanceSq(ip) = sum( kmeans%Try(it)%NormedPoint(1:nd,ip,ic)**2 )
330 kmeans%Try(it)%Membership(ip) = ic
331 end do
332 !%%%%%%%%%%%%%%%%%%%
333 do ic = 2, nc - 1
334 kmeans%Try(it)%Center(1:nd,ic) = sumpoint(1:nd,ic) / real(kmeans%Try(it)%Size(ic), kind = rk)
335 do ip = 1, np
336 kmeans%Try(it)%NormedPoint(1:nd,ip,ic) = kmeans%Try(it)%Center(1:nd,ic) - point(1:nd,ip)
337 distancesq = sum( kmeans%Try(it)%NormedPoint(1:nd,ip,ic)**2 )
338 if (distancesq < kmeans%Try(it)%MinDistanceSq(ip)) then
339 kmeans%Try(it)%MinDistanceSq(ip) = distancesq
340 kmeans%Try(it)%Membership(ip) = ic
341 end if
342 end do
343 end do
344 !%%%%%%%%%%%%%%%%%%%
345 ic = nc
346 convergenceoccurred = .true.
347 kmeans%Try(it)%potential = 0._rk
348 kmeans%Try(it)%Center(1:nd,ic) = sumpoint(1:nd,ic) / real(kmeans%Try(it)%Size(ic), kind = rk)
349 do ip = 1, np
350 kmeans%Try(it)%NormedPoint(1:nd,ip,ic) = kmeans%Try(it)%Center(1:nd,ic) - point(1:nd,ip)
351 distancesq = sum( kmeans%Try(it)%NormedPoint(1:nd,ip,ic)**2 )
352 if (distancesq < kmeans%Try(it)%MinDistanceSq(ip)) then
353 kmeans%Try(it)%MinDistanceSq(ip) = distancesq
354 kmeans%Try(it)%Membership(ip) = ic
355 end if
356 if (kmeans%Try(it)%Membership(ip)/=membership(ip)) then
357 convergenceoccurred = .false.
358 kmeans%Try(it)%Size(membership(ip)) = kmeans%Try(it)%Size(membership(ip)) - 1_ik
359 kmeans%Try(it)%Size(kmeans%Try(it)%Membership(ip)) = kmeans%Try(it)%Size(kmeans%Try(it)%Membership(ip)) + 1_ik
360 end if
361 kmeans%Try(it)%potential = kmeans%Try(it)%potential + kmeans%Try(it)%MinDistanceSq(ip)
362 end do
363 !%%%%%%%%%%%%%%%%%%%
364
365 if (any(kmeans%Try(it)%Size==0_ik)) then
366
367 kmeans%Try(it)%niter = 1_ik
368 kmeans%Try(it)%nzsci = kmeans%Try(it)%nzsci + 1_ik
369
370 if (kmeans%Try(it)%nzsci > kmeans%nzsciMax) then
371 !Kmeans%Try(it)%Err%msg = "Kmeans%Try(it)%nzsci > Kmeans%nzsciMax"
372 kmeans%Try(it)%Err%occurred = .true.
373 kmeans%Try(it)%Err%stat = 2_ik
374 exit loopcenterrefinement
375 end if
376
377 ! find new random Centers and start over.
378
379 call runkpp(nc, nd, np, point, sumpoint, membership, kmeans%Try(it)%Size, potential)
380
381 cycle loopcenterrefinement
382
383#if defined DEBUG_ENABLED || defined TESTING_ENABLED
384 elseif (any(kmeans%Try(it)%Size<0_ik)) then
385 block
386 use err_mod, only: abort
387 kmeans%Try(it)%Err%msg = procedure_name//": Internal error occurred - Kmeans%Try(it)%Size < 0. Please report this issue along with circumstances to the developers."
388 call abort(kmeans%Try(it)%Err)
389 return
390 end block
391#endif
392
393 end if
394
395 ! if `niterMax` has reached, quit.
396
397 if (kmeans%Try(it)%niter < kmeans%niterMax) then
398
399 ! If convergence has occurred, exit loopCenterRefinement.
400
401 if (convergenceoccurred .or. abs(kmeans%Try(it)%potential-potential)/kmeans%Try(it)%potential < kmeans%relTol) exit loopcenterrefinement
402
403 kmeans%Try(it)%niter = kmeans%Try(it)%niter + 1_ik
404
405 do ip = 1, np
406 sumpoint(1:nd,membership(ip)) = sumpoint(1:nd,membership(ip)) - point(1:nd,ip)
407 sumpoint(1:nd,kmeans%Try(it)%Membership(ip)) = sumpoint(1:nd,kmeans%Try(it)%Membership(ip)) + point(1:nd,ip)
408 membership(ip) = kmeans%Try(it)%Membership(ip)
409 end do
410 potential = kmeans%Try(it)%potential
411
412 else
413
414 !Kmeans%Try(it)%Err%msg = "Kmeans%Try(it)%niter > Kmeans%niterMax"
415 kmeans%Try(it)%Err%occurred = .true.
416 kmeans%Try(it)%Err%stat = 1_ik
417 exit loopcenterrefinement
418
419 end if
420
421 end do loopcenterrefinement
422
423 if (.not. kmeans%Try(it)%Err%occurred) then
424 if (kmeans%Try(it)%potential < minpotential) kmeans%itopt = it
425 kmeans%failed = .false.
426 end if
427
428 end do loopmultiplekmeanstry
429
430 end subroutine runkmeans

458 subroutine runkpp(nc, nd, np, Point, SumPoint, Membership, Size, potential)
459#if INTEL_COMPILER_ENABLED && defined DLL_ENABLED && (OS_IS_WINDOWS || defined OS_IS_DARWIN)
460 !DEC$ ATTRIBUTES DLLEXPORT :: runKPP
461#endif
462 use statistics_mod, only: getrandint
463
464 implicit none
465
466 integer(IK) , intent(in) :: np, nd, nc
467 real(RK) , intent(in) :: Point(nd,np)
468 real(RK) , intent(out) :: SumPoint(nd,nc)
469 integer(IK) , intent(out) :: Membership(np)
470 integer(IK) , intent(out) :: Size(nc)
471 real(RK) , intent(out) :: potential
472
473 real(RK) :: dummy
474 real(RK) :: distanceSq
475 real(RK) :: Center(nd,nc)
476 real(RK) :: MinDistanceSq(np) ! An array representing the minimum distances of individual points from all clusters.
477 real(RK) :: CumSumDistSq(0:np) ! sum of distance squared of points up to the given index of the vector.
478 integer(IK) :: randomInteger
479 integer(IK) :: ic, ip, i
480
481 cumsumdistsq(0) = 0._rk ! This element must always remain zero.
482
483 ic = 1_ik
484 center(1:nd,ic) = point( 1:nd , getrandint(1,np) ) ! perhaps this random assignment is unnecessary, after all, Point is supposed to be of random order.
485 do ip = 1, np
486 mindistancesq(ip) = sum( (center(1:nd,ic) - point(1:nd,ip))**2 ) ! find the distance-squared of each point to the # ic cluster Center.
487 cumsumdistsq(ip) = cumsumdistsq(ip-1) + mindistancesq(ip)
488 membership(ip) = ic
489 end do
490
491 ic = 2_ik
492 call random_number(dummy)
493 dummy = dummy * cumsumdistsq(np)
494 center(1:nd,ic) = point( 1:nd, minloc(cumsumdistsq, dim = 1, mask = cumsumdistsq > dummy) - 1 ) ! -1 is essential to compensate for the 0 starting index of CumSumDistSq
495
496 ! find the rest of cluster centers
497
498 do ic = 2, nc - 1
499
500 do ip = 1, np
501 distancesq = sum( (center(1:nd,ic) - point(1:nd,ip))**2 )
502 if (distancesq < mindistancesq(ip)) then
503 mindistancesq(ip) = distancesq
504 membership(ip) = ic
505 end if
506 cumsumdistsq(ip) = cumsumdistsq(ip-1) + mindistancesq(ip)
507 end do
508
509 call random_number(dummy)
510 dummy = dummy * cumsumdistsq(np)
511 center(1:nd,ic+1) = point( 1:nd , minloc(cumsumdistsq, dim = 1, mask = cumsumdistsq > dummy) - 1 )
512
513 end do
514
515 Size = 0_ik
516 sumpoint = 0._rk
517 potential = 0._rk
518 do ip = 1, np
519 distancesq = sum( (center(1:nd,nc) - point(1:nd,ip))**2 )
520 if (distancesq < mindistancesq(ip)) then
521 mindistancesq(ip) = distancesq
522 membership(ip) = nc
523 end if
524 sumpoint(1:nd,membership(ip)) = sumpoint(1:nd,membership(ip)) + point(1:nd,ip)
525 Size(membership(ip)) = Size(membership(ip)) + 1_ik
526 potential = potential + mindistancesq(ip)
527 end do
528
529 end subroutine runkpp
