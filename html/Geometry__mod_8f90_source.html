103 pure function genMinDistEdge(np, PairDistSq) result(MinDistEdge)
104#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
105 !DEC$ ATTRIBUTES DLLEXPORT :: genMinDistEdge
106#endif
107 use Constants_mod, only: IK, RK, HUGE_RK ! LCOV_EXCL_LINE
108 implicit none
109 integer(IK) , intent(in) :: np
110 real(RK) , intent(in) :: PairDistSq(np,np)
111 type(MinDistEdge_type) :: MinDistEdge
112
113 integer(IK) :: pindex1
114 integer(IK) :: pindex2
115 integer(IK) :: jp
116
117 allocate(MinDistEdge%Index%Value(np), MinDistEdge%LenSq%Value(np))
118 MinDistEdge%np = np
119
120 jp = 1_IK
121 pindex2 = minloc(PairDistSq(2:np,jp), dim = 1, kind = IK) + 1_IK
122 MinDistEdge%Index%Value(jp) = pindex2
123 MinDistEdge%LenSq%Value(jp) = PairDistSq(pindex2,jp)
124
125 jp = np
126 pindex1 = minloc(PairDistSq(1:jp-1,jp), dim = 1, kind = IK)
127 MinDistEdge%Index%Value(jp) = pindex1
128 MinDistEdge%LenSq%Value(jp) = PairDistSq(pindex1,jp)
129
130 do concurrent(jp = 2:np-1)
131 pindex1 = minloc(PairDistSq(1:jp-1,jp), dim = 1, kind = IK)
132 pindex2 = minloc(PairDistSq(jp+1:np,jp), dim = 1, kind = IK) + jp
133 if (PairDistSq(pindex1,jp) < PairDistSq(pindex2,jp)) then
134 MinDistEdge%Index%Value(jp) = pindex1
135 MinDistEdge%LenSq%Value(jp) = PairDistSq(pindex1,jp)
136 else
137 MinDistEdge%Index%Value(jp) = pindex2
138 MinDistEdge%LenSq%Value(jp) = PairDistSq(pindex2,jp)
139 end if
140 end do
141
142 end function genMinDistEdge

348 PURE subroutine getDistSortedExpDiff(nd, np, Point, RefPoint, DistSortedExpDiff, Index)
349#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
350 !DEC$ ATTRIBUTES DLLEXPORT :: getDistSortedExpDiff
351#endif
352 use Constants_mod, only: IK, RK ! LCOV_EXCL_LINE
353 use Sort_mod, only: sortIndex
354 implicit none
355 integer(IK) , intent(in) :: nd, np
356 real(RK) , intent(in) :: Point(nd,np)
357 real(RK) , intent(in) :: RefPoint(nd)
358 real(RK) , intent(out) :: DistSortedExpDiff(np)
359 integer(IK) , intent(out) :: Index(np)
360 real(RK) :: DistFromRefPoint(np)
361 real(RK) :: DistFromRefPointSortedExp(np)
362 integer(IK) :: exponent
363 exponent = nd / 2_IK
364 if (2_IK * exponent == nd) then ! nd is even
365 ! Avoid redundant sqrt() in computing distances when nd is even.
366 DistFromRefPoint(1:np) = getDistSqFromRefPoint(nd, np, Point, RefPoint)
367 else ! nd is odd
368 exponent = nd
369 DistFromRefPoint(1:np) = getDistFromRefPoint(nd, np, Point, RefPoint)
370 end if
371 call sortIndex(DistFromRefPoint, Index)
372 DistFromRefPointSortedExp(1:np) = DistFromRefPoint(Index)**exponent
373 DistSortedExpDiff(1:np) = DistFromRefPointSortedExp(1)
374 DistSortedExpDiff(2:np) = DistFromRefPointSortedExp(2:np) - DistFromRefPointSortedExp(1:np-1)
375 end subroutine getDistSortedExpDiff

487 !
488 ! scratch:
489 ! \param[out] Overlap : An array of size `(nc,nc)` representing the diagonal elements of the Cholesky lower triangle of the covariance matrices of the ellipsoids.
490 subroutine getLogSumVolEff( nd & ! LCOV_EXCL_LINE
491 , nc & ! LCOV_EXCL_LINE
492 , relTol & ! LCOV_EXCL_LINE
493 , LogVol & ! LCOV_EXCL_LINE
494 , Center & ! LCOV_EXCL_LINE
495 , ChoDia & ! LCOV_EXCL_LINE
496 , logSumVol & ! LCOV_EXCL_LINE
497 , InvCovMat & ! LCOV_EXCL_LINE
498 , ChoLowCovUpp & ! LCOV_EXCL_LINE
499 , logSumVolEff & ! LCOV_EXCL_LINE
500 , ScaleFactorSq & ! LCOV_EXCL_LINE
501 !, Overlap & ! LCOV_EXCL_LINE
502 )
503#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
504 !DEC$ ATTRIBUTES DLLEXPORT :: getLogSumVolEff
505#endif
506 use DomainEllipsoid_mod, only: getRandUnif
507 use Constants_mod, only: RK, NEGINF_RK
508 use Math_mod, only: genCumSum
509 implicit none
510
511 integer , intent(in) :: nd
512 integer , intent(in) :: nc
513 real(RK) , intent(in) :: relTol
514 real(RK) , intent(in) :: logSumVol
515 real(RK) , intent(in) :: LogVol(nc)
516 real(RK) , intent(in) :: Center(nd,nc)
517 real(RK) , intent(in) :: ChoDia(nd,nc)
518 real(RK) , intent(in) :: InvCovMat(nd,nd,nc)
519 real(RK) , intent(in) :: ChoLowCovUpp(nd,nd,nc)
520 real(RK) , intent(out) :: logSumVolEff
521 real(RK) , intent(in) , optional :: ScaleFactorSq(nc)
522 !integer , intent(out) :: Overlap(nc,nc)
523
524 integer :: ic,jc
525 integer :: nsim,isim
526 integer :: membershipCount
527 real(RK) :: CumSumVolNormed(nc)
528 real(RK) :: ScaleFacSq(nc)
529 real(RK) :: VolNormed(nc)
530 real(RK) :: SimPoint(nd)
531 real(RK) :: isimAccepted
532 real(RK) :: unifrnd
533 real(RK) :: mahalSq
534
535 if (present(ScaleFactorSq)) then
536 ScaleFacSq(1:nc) = ScaleFactorSq
537 else
538 ScaleFacSq(1:nc) = 1._RK
539 end if
540
541 ! Compute the expected number of simulated points in each bounded region.
542
543 nsim = nint(1._RK / relTol**2)
544 VolNormed = exp(LogVol - logSumVol)
545 CumSumVolNormed(1:nc) = genCumSum(VolNormed)
546 CumSumVolNormed(1:nc) = CumSumVolNormed(1:nc) / CumSumVolNormed(nc)
547
548#if DEBUG_ENABLED || TESTING_ENABLED || CODECOV_ENABLED
549 if (abs(CumSumVolNormed(nc) - 1._RK)>1.e-6) then
550 write(*,*) "abs(CumSumVolNormed(nc) - 1._RK) > 1.e-6", CumSumVolNormed(nc)
551 error stop
552 end if
553#endif
554
555 ! Simulate points within each cluster and compute the effective sum of volumes by excluding overlaps.
556
557 isim = 0._RK
558 isimAccepted = 0._RK
559 loopGeneratePoint: do
560
561 call random_number(unifrnd)
562 ic = minloc(CumSumVolNormed, mask = CumSumVolNormed >= unifrnd, dim = 1)
563
564 call getRandUnif( Rand = SimPoint & ! LCOV_EXCL_LINE
565 , Mean = Center(1:nd,ic) & ! LCOV_EXCL_LINE
566 , ChoDia = ChoDia(1:nd,ic) & ! LCOV_EXCL_LINE
567 , ChoLow = ChoLowCovUpp(1:nd,1:nd,ic) & ! LCOV_EXCL_LINE
568 )
569
570
571 membershipCount = 1
572 do jc = 1, nc
573 if (jc/=ic) then
574 SimPoint(1:nd) = SimPoint(1:nd) - Center(1:nd,jc)
575 mahalSq = dot_product( SimPoint(1:nd) , matmul(InvCovMat(1:nd,1:nd,jc), SimPoint(1:nd)) )
576 SimPoint(1:nd) = SimPoint(1:nd) + Center(1:nd,jc)
577 if (mahalSq < ScaleFacSq(jc)) membershipCount = membershipCount + 1
578 end if
579 end do
580
581 isim = isim + 1
582 isimAccepted = isimAccepted + 1._RK / membershipCount
583 if (isim < nsim) cycle loopGeneratePoint
584 exit loopGeneratePoint
585
586 end do loopGeneratePoint
587
588 if (isimAccepted < isim) then
589 logSumVolEff = log(isimAccepted / isim) + logSumVol
590 else
591 logSumVolEff = logSumVol
592 end if
593
594 end subroutine getLogSumVolEff
