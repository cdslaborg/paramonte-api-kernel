175 !
176 ! \warning
177 ! Developer note: The use of `contiguous` arrays as input arguments here is not encouraged
178 ! as the input argument `ndim` is necessary for when an identity domain must be constructed.
179 function construct(ndim, Mean, RepUpp, isCovUpp, Err) result(self)
180#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
181 !DEC$ ATTRIBUTES DLLEXPORT :: construct
182#endif
183 use MatrixDiag_mod, only: genDiag
184 use MatrixScale_mod, only: genScaledMat, operator(.udtimes.)
185 use MatrixCopy_mod, only: copyUppDia2UppDia
186 use Matrix_mod, only: genInvMatFromChoLow
187 use Err_mod, only: Err_type
188 integer(IK) , intent(in) :: ndim
189 real(RK) , intent(in) , optional :: Mean(ndim)
190 real(RK) , intent(in) , optional :: RepUpp(ndim,ndim)
191 logical , intent(in) , optional :: isCovUpp
192 type(Err_type) , intent(out) , optional :: Err
193 type(DistMultiUnifEllipsoid_type) :: self
194 logical :: isCovUpp_default
195 real(RK) :: ndimPlusTwoInverse
196 real(RK) :: ndimPlusTwoInverseSqrt
197 integer(IK) :: i, j
198 character(*,SK) , parameter :: PROCEDURE_NAME = MODULE_NAME//"@construct()"
199
200 if (present(isCovUpp) .and. present(RepUpp)) then
201 isCovUpp_default = isCovUpp
202 else
203 isCovUpp_default = .false.
204 end if
205 if (isCovUpp_default) then
206 self%Support = DomainEllipsoid_type(ndim, Center = Mean, RepUpp = RepUpp .udtimes. (ndim + 2_IK), Err = Err)
207 else
208 self%Support = DomainEllipsoid_type(ndim, Center = Mean, RepUpp = RepUpp, Err = Err)
209 end if
210 if (present(Err)) then; if (Err%occurred) return; end if
211
212 self%ndim = ndim
213
214 ! Compute the covariance matrix and its Cholesky factor.
215
216
217 ndimPlusTwoInverse = 1._RK / real(ndim+2_IK,RK)
218 ndimPlusTwoInverseSqrt = sqrt(ndimPlusTwoInverse)
219 if (isCovUpp_default) then
220 allocate(self%ChoLowCovUpp(self%ndim,self%ndim), self%ChoDia(self%ndim))
221 call copyUppDia2UppDia(from = RepUpp, to = self%ChoLowCovUpp)
222 do j = 1, self%ndim
223 do i = j + 1, self%ndim
224 self%ChoLowCovUpp(i,j) = self%Support%ChoLowRepUpp(i,j) * ndimPlusTwoInverseSqrt
225 end do
226 self%ChoDia(j) = self%Support%ChoDia(j) * ndimPlusTwoInverseSqrt
227 end do
228 else
229 self%ChoLowCovUpp = genScaledMat(self%Support%ChoLowRepUpp, facupp = ndimPlusTwoInverse, facdia = ndimPlusTwoInverse, faclow = ndimPlusTwoInverseSqrt)
230 self%ChoDia = self%Support%ChoDia * ndimPlusTwoInverseSqrt
231 end if
232
233 ! Compute inverse covariance matrix and its determinant.
234
235 self%logDetInvCovMat = -2_IK * sum(log(self%ChoDia))
236
237 ! Compute the standard deviation of the distribution along each dimension.
238
239 self%Std = sqrt(genDiag(self%ChoLowCovUpp))
240
241 ! Compute inverse covariance matrix and its determinant.
242
243 self%InvCovMat = genInvMatFromChoLow(ndim,self%ChoLowCovUpp,self%ChoDia)
244
245 ! Set up other properties of the distribution.
246
247 self%logpdfval = -self%Support%logVol
248
249 end function construct

377! !> \brief
378! !> Generate the covariance matrix of the Ellipsoidal MVU distribution from the representative matrix of the distribution.
379! !>
380! !> \details
381! !> The representative matrix is the matrix \f$A\f$ such that for any point \f$X\f$ within the support of the distribution,
382! !> the following relationship holds,
383! !>
384! !> \f{eqnarray*}{
385! !>
386! !> X^T ~ A^{-1} ~ X \leq 1 ~,
387! !>
388! !> \f}
389! !>
390! !> In such a case, the covariance of the ellipsoidal distribution is given by
391! !>
392! !> \f{eqnarray*}{
393! !>
394! !> \text{CovMat} = \frac{A}{p + 2} ~,
395! !>
396! !> \f}
397! !>
398! !> \param[inout] ChoLowRepUpp : The upper triangle and diagonals of the **square** representative matrix of the ellipsoidal support of the distribution.
399! !> The lower-triangle is assumed to contain the lower-triangle Cholesky factorization of the representative matrix.
400! !> The representative matrix must be a `contiguous` array of shape `(nd,nd)` where `nd` is the number of dimensions
401! !> of the domain of definition of the distribution.
402! !> \return
403! !> `ChoLowCovUpp` : The upper triangle and diagonals of the covariance matrix of the distribution.
404! !> The lower triangle contains the Cholesky factorization of the covariance matrix.
405! !>
406! !> \author
407! !> Amir Shahmoradi, April 23, 2017, 12:36 AM, ICES, University of Texas at Austin
408! pure function genChoLowCovUppFromChoLowRepUpp(ChoLowRepUpp) result(ChoLowCovUpp)
409!#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
410! !DEC$ ATTRIBUTES DLLEXPORT :: genChoLowCovUppFromChoLowRepUpp
411!#endif
412! real(RK), intent(in), contiguous :: ChoLowRepUpp(:,:)
413! real(RK) :: ChoLowCovUpp(size(ChoLowRepUpp,dim=1),size(ChoLowRepUpp,dim=2))
414! integer(IK) :: icol, ndim
415! real(RK) :: ndimPlusTwoInverse
416! real(RK) :: ndimPlusTwoInverseSqrt
417! ndim = size(ChoLowRepUpp,dim=1,kind=IK)
418! ndimPlusTwoInverse = 1._RK / (ndim + 2_IK)
419! ndimPlusTwoInverseSqrt = sqrt(ndimPlusTwoInverse)
420! ChoLowCovUpp(1,1) = ndimPlusTwoInverse * ChoLowRepUpp(1,1)
421! ChoLowCovUpp(2:ndim,1) = ndimPlusTwoInverseSqrt * ChoLowRepUpp(2:ndim,1)
422! do icol = 1, ndim - 1
423! ChoLowCovUpp(1:icol,icol) = ndimPlusTwoInverse * ChoLowRepUpp(1:icol,icol)
424! ChoLowCovUpp(icol+1:ndim,icol) = ndimPlusTwoInverseSqrt * ChoLowRepUpp(icol+1:ndim,icol)
425! end do
426! ChoLowCovUpp(1:ndim,ndim) = ndimPlusTwoInverse * ChoLowRepUpp(1:ndim,ndim)
427! end function genChoLowCovUppFromChoLowRepUpp
