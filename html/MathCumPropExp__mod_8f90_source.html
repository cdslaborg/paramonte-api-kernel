416 pure function genCumPropExp_RK(Array, maxArray, lenArray) result(CumPropExp)
417#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
418 !DEC$ ATTRIBUTES DLLEXPORT :: genCumPropExp_RK
419#endif
420 use Constants_mod, only: LOGTINY_RK, RK, IK
421 integer(IK) , intent(in) :: lenArray
422 real(RK) , intent(in) :: Array(lenArray)
423 real(RK) , intent(in) :: maxArray
424 real(RK) :: CumPropExp(lenArray)
425 real(RK) :: invCumPropExp
426 integer(IK) :: i
427 CumPropExp(1) = Array(1) - maxArray
428 if (CumPropExp(1) < LOGTINY_RK) then
429 CumPropExp(1) = 0._RK
430 else
431 CumPropExp(1) = exp(CumPropExp(1))
432 end if
433 do i = 2, lenArray
434 CumPropExp(i) = Array(i) - maxArray
435 if (CumPropExp(i) < LOGTINY_RK) then
436 CumPropExp(i) = CumPropExp(i-1)
437 else
438 CumPropExp(i) = CumPropExp(i-1) + exp(CumPropExp(i))
439 end if
440 end do
441 invCumPropExp = 1._RK / CumPropExp(lenArray)
442 CumPropExp = CumPropExp * invCumPropExp
443 end function genCumPropExp_RK
