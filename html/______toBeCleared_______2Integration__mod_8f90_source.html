175 recursive subroutine doquadtrap(getFunc,xmin,xmax,integral,n)
176 implicit none
177 integer , intent(in) :: n
178 real(rk), intent(in) :: xmin,xmax
179 real(rk), intent(out) :: integral
180 integer :: it,j
181 real(rk) :: del,sum,tnm,x
182 procedure(integrand) :: getfunc
183
184 if (n==1) then
185 integral=0.5_rk*(xmax-xmin)*(getfunc(xmin)+getfunc(xmax))
186 else
187 it=2**(n-2)
188 tnm=it
189 del=(xmax-xmin)/tnm
190 x=xmin+0.5*del
191 sum=0._rk
192 do j=1,it

195 end do
