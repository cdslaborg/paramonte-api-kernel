171 pure function logpdf(self,Point) result(logval)
172 use Math_mod, only: getLogSumExp
173 implicit none
174 class(DistMultiNormShellMix_type) , intent(in) :: self
175 real(RK) , intent(in) :: Point(self%ndim)
176 real(RK) :: logval
177 real(RK) :: NormedPoint(self%ndim)
178 real(RK) :: LogFunc(self%nshell)
179 real(RK) :: maxLogFunc
180 integer(IK) :: imode
181 maxLogFunc = -huge(maxLogFunc)
182 do imode = 1, self%nshell
183 NormedPoint(1:self%ndim) = Point - self%Mean(1:self%ndim,imode)
184 LogFunc(imode) = self%LogAdd(imode) - self%Coef(imode) * (norm2(NormedPoint) - self%Radius(imode))**2
185 if (maxLogFunc < LogFunc(imode)) maxLogFunc = LogFunc(imode)
186 end do
187 logval = getLogSumExp(LogFunc, maxLogFunc)
188 end function logpdf
