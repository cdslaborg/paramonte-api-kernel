180 subroutine finalize()
181
182 use Constants_mod, only: RK, IK
183 use Decoration_mod, only: style
184 use String_mod, only: val2char
185 use Err_mod, only: abort
186 implicit none
187 real(RK) :: percentageTestPassed
188 integer(IK) :: ntotal
189 character(:), allocatable :: color
190 character(:), allocatable :: msg
191 integer(IK) :: i
192
193 ntotal = mv_npass + mv_nfail
194
195 percentageTestPassed = 0._RK
196 if (ntotal/=0_IK) percentageTestPassed = 100_IK * mv_npass / real(ntotal, kind=RK)
197
198 if (mv_Image%isFirst) then
199
200 msg = style( val2char(ntotal)//" tests performed. ", "bright", "yellow")
201
202 color = "green"; if (nint(percentageTestPassed,kind=IK)==0_IK) color = "red"
203 msg = msg // style( val2char(percentageTestPassed,"(f0.2)")//"% of "//val2char(ntotal)//" tests passed. ", "bright", color)
204
205 color = "green"; if (mv_nfail>0_IK) color = "red"
206 msg = msg // style( val2char(mv_nfail,"(g0)")//" tests failed. ", "bright", color)
207
208 msg = msg // style( "The total elapsed wall-clock time: " // val2char(mv_timeTotal,"(f0.6)")//" seconds.", "bright", "yellow" )
209
210 write(output_unit, "(*(g0,:,' '))")
211 write(output_unit, "(*(g0,:,' '))") msg
212 write(output_unit, "(*(g0,:,' '))")
213
214 if (mv_nfail>0_IK) then
215
216 write(output_unit, "(*(g0,:,' '))")
217 write(output_unit, "(*(g0,:,' '))") style("The following tests FAILED:", "bright", "red")
218 write(output_unit, "(*(g0,:,' '))")
219 do i = 1, mv_nfail
220 write(output_unit, "(*(g0,:,' '))") style(" "//val2char(i,"(g0)")//" - "//mv_FailedTestFuncName(i)%value, "bright", "red")
221 end do
222 write(output_unit, "(*(g0,:,' '))")
223 write(output_unit, "(*(g0,:,' '))") style("Errors occurred while running the ParaMonte tests. Please report this issue at:", "bright", "red")
224 write(output_unit, "(*(g0,:,' '))")
225 write(output_unit, "(*(g0,:,' '))") style(" https://github.com/cdslaborg/paramonte/issues", "bright", "cyan")
226#if ! DEBUG_ENABLED
227 write(output_unit, "(*(g0,:,' '))")
228 write(output_unit, "(*(g0,:,' '))") style("To get more information about the failure, you may also compile and run the test in debug mode.", "bright", "red")
229#endif
230 write(output_unit, "(*(g0,:,' '))")
231
232 end if
233
234 end if
235
236 if (mv_nfail>0_IK) then
237 call abort()
238 else
239 call mv_Image%finalize()
240 end if
241
242 end subroutine finalize

246 function constructTest(moduleName, inDir, outDir) result(Test)
247
248 use Decoration_mod, only: style
249 use Constants_mod, only: NLC
250 use String_mod, only: val2char
251 use Path_mod, only: Path_type
252 use Err_mod, only: Err_type, abort
253
254 implicit none
255
256 character(*), intent(in) :: moduleName
257 character(*), intent(in), optional :: inDir, outDir
258 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@constructTest()"
259 type(Test_type) :: Test
260 type(Path_type) :: Path
261 type(Err_type) :: Err
262
263 ! Set up image counts and the user timer.
264
265 Test%Image = mv_Image
266 Test%outputUnit = output_unit
267 Test%Timer = Timer_type(Err%occurred)
268
269 ! Set up the module specs.
270
271 Test%Module%name = moduleName
272 Test%Module%Timer = Timer_type(Err%occurred)
273 if (Err%occurred) then
274 Err%msg = PROCEDURE_NAME // ": Error occurred while setting up the test timer. There is no processor clock."
275 call abort( Err = Err, prefix = "ParaMonteUnitTesting", newline = NLC, outputUnit = Test%outputUnit )
276 return
277 end if
278
279 ! Set up the tests input path.
280
281 if (present(inDir)) then
282 Test%inDir = inDir
283 else
284 Test%inDir = "./input"
285 end if
286 Path = Path_type(Test%inDir)
287 if (Path%Err%occurred) then
288 call abort( Err = Path%Err &
289 , newline = "\n" &
290 , outputUnit = Test%outputUnit &
291 )
292 else
293 Test%inDir = Path%modified
294 end if
295
296 ! Set up the tests output path.
297
298 if (present(outDir)) then
299 Test%outDir = outDir
300 else
301 Test%outDir = "./output"
302 end if
303 Path = Path_type(Test%outDir)
304 if (Path%Err%occurred) then
305 call abort( Err = Path%Err &
306 !, prefix = "FATAL: " &
307 , newline = "\n" &
308 , outputUnit = Test%outputUnit &
309 )
310 else
311 Test%outDir = Path%modified
312 if (Test%Image%isFirst) then
313 Path%Err = Path%mkdir(Test%outDir)
314 if (Path%Err%occurred) then
315 call abort( Err = Path%Err &
316 !, prefix = "FATAL: " &
317 , newline = "\n" &
318 , outputUnit = Test%outputUnit &
319 )
320 end if
321 end if
322 end if
323
324 call mv_Image%sync()
325
326 end function constructTest

332 subroutine runTest(Test, test_func, test_func_name)
333
334 use String_mod, only: isInteger
335 implicit none
336 class(Test_type), intent(inout) :: Test
337 procedure(test_func_proc) :: test_func
338 character(*), intent(in) :: test_func_name
339 logical :: assertion
340
341 Test%Subject%Name%value = trim(adjustl(test_func_name(6:))) ! remove "test_"
342 Test%Subject%Name%Parts = Test%Subject%Name%split(value = Test%Subject%Name%value, delim = "_", npart = Test%Subject%Name%nPart)
343 if ( Test%Subject%Name%nPart==1_IK .or. .not. isInteger(Test%Subject%Name%Parts(Test%Subject%Name%nPart)%value) ) then
344 Test%Func%id = "#1"
345 else
346 Test%Subject%Name%value = Test%Subject%Name%value(1:len(Test%Subject%Name%value)-len("_"//Test%Subject%Name%Parts(Test%Subject%Name%nPart)%value))
347 Test%Func%id = "#"//Test%Subject%Name%Parts(Test%Subject%Name%nPart)%value
348 !else
349 ! Test%Err%occurred = .true.
350 ! Test%Err%msg = "Invalid function name passed to runTest() method of the Test object: " // trim(adjustl(test_func_name))
351 ! call abort ( Err = Test%Err &
352 ! , newline = "\n" &
353 ! , outputUnit = Test%outputUnit &
354 ! )
355 end if
356
357
358 Test%Func%caseCounter = 0_IK
359 Test%Func%name = "Test_"//Test%Module%name(2:)//"@"//trim(adjustl(test_func_name))
360
361 call Test%Func%Timer%tic()
362 assertion = test_func()
363 call Test%assert(assertion)
364
365 end subroutine runTest

386 subroutine assert(Test, assertion, desc)
387
388 use String_mod, only: val2char, padChar
389 use Err_mod, only: abort
390 implicit none
391
392 class(Test_type), intent(inout) :: Test
393 logical , intent(in) :: assertion
394 character(*), intent(in), optional :: desc
395 character(:), allocatable :: statusMsg
396 character(:), allocatable :: desc_def
397 character(:), allocatable :: caseID
398 integer(IK) :: imageID
399
400 call Test%Func%Timer%toc()
401
402 Test%Func%caseCounter = Test%Func%caseCounter + 1_IK
403 caseID = "-"//val2char(Test%Func%caseCounter)
404
405 if (present(desc)) then
406 desc_def = desc
407 else
408 desc_def = "Not Available."
409 end if
410
411 if (assertion) then
412
413 mv_npass = mv_npass + 1
414 statusMsg = mc_passedString
415
416 else
417
418 statusMsg = mc_failedString
419 mv_nfail = mv_nfail + 1
420
421 ! set up the allocation for the names of failed tests
422
423 if (mv_nfail==mv_nfailMax) then
424 block
425 type(Str_type), allocatable :: mv_FailedTestFuncNameTemp(:)
426 mv_nfailMax = 2 * mv_nfailMax
427 allocate(mv_FailedTestFuncNameTemp(mv_nfailMax))
428 mv_FailedTestFuncNameTemp(1:mv_nfail) = mv_FailedTestFuncName
429 call move_alloc(from=mv_FailedTestFuncNameTemp, to=mv_FailedTestFuncName)
430 end block
431 end if
432
433 mv_FailedTestFuncName(mv_nfail)%value = Test%Func%name
434
435 end if
436
437 mv_testCounter = mv_testCounter + 1
438
439 do imageID = 1, Test%Image%count
440
441 if (imageID==Test%Image%id) then
442 write(Test%outputUnit,"(*(g0,:,' '))") "["//adjustr(val2char(mv_testCounter,len=4_IK))//"] testing" &
443 , padChar( Test%Module%name//"@"//Test%Subject%Name%value//" "//Test%Func%id//caseID//" ","." , 79_IK ) &
444 , statusMsg, "in", adjustr(val2char(Test%Func%Timer%Time%delta,"(f0.4)",8_IK)), "seconds on image "//val2char(Test%Image%id)
445
446 end if
447
448#if CAF_ENABLED || MPI_ENABLED
449 call execute_command_line(" ", cmdstat = Test%Err%stat)
450 flush(output_unit)
451 call Test%Image%sync()
452#endif
453
454 end do
455
456 block
457 use System_mod, only: sleep
458 use Err_mod, only: Err_type
459 type(Err_type) :: Err
460 call sleep(0.001_RK, Err)
461 end block
462
463 if (Test%isVerboseMode .and. .not. assertion) then
464 Test%Err%occurred = .true.
465 Test%Err%msg = "The test assertion is FALSE.\nThe test description: "//trim(adjustl(desc_def))
466 call abort( Err = Test%Err &
467 , prefix = "ParaMonteTest @ "//Test%Func%name//"()" &
468 , newline = "\n" &
469 , outputUnit = Test%outputUnit &
470 )
471 error stop
472 end if
473
474 end subroutine assert

479 use Decoration_mod, only: style
480 use String_mod, only: val2char, padChar
481 implicit none

492 mv_timeTotal = mv_timeTotal + Test%Module%Timer%Time%delta
493 call mv_Image%sync()
494 end subroutine finalizeTest

499 ! Return the negative natural logarithm of MVN distribution evaluated at the input vector `Point` of length `ndim`.
500 function getLogFuncMVN(ndim,Point) result(logFunc) BIND_C
501 implicit none
502 integer(IK) , intent(in) :: ndim
503 real(RK) , intent(in) :: Point(ndim)
504 real(RK) :: logFunc
505#if CFI_ENABLED
506 value :: ndim
507#endif
508 real(RK) , parameter :: LOG_INVERSE_SQRT_TWO_PI = log(1._RK / sqrt(2._RK * acos(-1._RK)))
509
510 !block
511 ! use System_mod, only: sleep
512 ! use Err_mod, only: Err_type
513 ! type(Err_type) :: Err
514 ! call sleep(seconds=5000.e-6_RK,Err=Err)
515 !end block
516
517 !block
518 ! real(RK), allocatable :: unifrnd(:,:)
519 ! allocate(unifrnd(200,20))
520 ! call random_number(unifrnd)
521 ! logFunc = sum(unifrnd) - 0.5_RK * sum(Point**2) - sum(unifrnd)
522 ! deallocate(unifrnd)
523 !end block
524
525 logFunc = ndim * LOG_INVERSE_SQRT_TWO_PI - 0.5_RK * sum(Point**2_IK)
526
527 !block
528 ! integer(IK), parameter :: nmode = 2_IK
529 ! real(RK) :: LogAmplitude(nmode), Mean(nmode), InvCovMat(nmode), LogSqrtDetInvCovMat(nmode)
530 ! LogAmplitude = [1._RK, 1._RK]
531 ! Mean = [0._RK, 7._RK]
532 ! InvCovMat = [1._RK,1._RK]
533 ! LogSqrtDetInvCovMat = [1._RK,1._RK]
534 ! logFunc = getLogProbGausMix ( nmode = 2_IK &
535 ! , nd = 1_IK &
536 ! , np = 1_IK &
537 ! , LogAmplitude = LogAmplitude &
538 ! , Mean = Mean &
539 ! , InvCovMat = InvCovMat &
540 ! , LogSqrtDetInvCovMat = LogSqrtDetInvCovMat &
541 ! , Point = Point(1) &
542 ! )
543 !end block
544
545 end function getLogFuncMVN

550 ! Return the negative natural logarithm of MVN distribution evaluated at the input vector `Point` of length `ndim`.
551 function getLogFuncBanana2D(ndim,Point) result(logFunc) BIND_C
552 use DistMultiNorm_mod, only: getLogProbDenMVN
553 use MathLogSumExp_mod, only: genLogSumExp
554 implicit none
555 integer(IK) , intent(in) :: ndim
556 real(RK) , intent(in) :: Point(ndim)
557 real(RK) :: logFunc
558#if CFI_ENABLED
559 value :: ndim
560#endif
561 integer(IK) , parameter :: NPAR = 2_IK ! sum(Banana,gaussian) normalization factor
562 real(RK) , parameter :: normfac = 0.3_RK ! 0.6_RK ! sum(Banana,gaussian) normalization factor
563 real(RK) , parameter :: lognormfac = log(normfac) ! sum(Banana,gaussian) normalization factor
564 real(RK) , parameter :: a = 0.7_RK, b = 1.5_RK ! parameters of Banana function
565 real(RK) , parameter :: MeanB(NPAR) = [ -5.0_RK , 0._RK ] ! mean vector of Banana function
566 real(RK) , parameter :: MeanG(NPAR) = [ 3.5_RK , 0._RK ] ! mean vector of Gaussian function
567 real(RK) , parameter :: CovMatB(NPAR,NPAR) = reshape([0.25_RK,0._RK,0._RK,0.81_RK],shape=shape(CovMatB)) ! Covariance matrix of Banana function
568 real(RK) , parameter :: CovMatG(NPAR,NPAR) = reshape([0.15_RK,0._RK,0._RK,0.15_RK],shape=shape(CovMatB)) ! Covariance matrix of Gaussian function
569 real(RK) , parameter :: InvCovMatB(NPAR,NPAR) = reshape([4._RK,0._RK,0._RK,1.23456790123457_RK],shape=shape(InvCovMatB)) ! Inverse Covariance matrix of Banana function
570 real(RK) , parameter :: InvCovMatG(NPAR,NPAR) = reshape([6.66666666666667_RK,0._RK,0._RK,6.66666666666667_RK],shape=shape(InvCovMatG)) ! Inverse Covariance matrix of Gaussian function
571 real(RK) , parameter :: logSqrtDetInvCovMatB = log(sqrt(4.93827160493827_RK)) ! Determinant of the Inverse Covariance matrix of Banana function
572 real(RK) , parameter :: logSqrtDetInvCovMatG = log(sqrt(44.4444444444445_RK)) ! Determinant of the Inverse Covariance matrix of Gaussian function
573 real(RK) :: PointSkewed(NPAR) ! transformed parameters that transform the Gaussian to the Banana function
574 real(RK) :: LogProb(2) ! logProbMVN, logProbBanana
575
576 PointSkewed(1) = -Point(1)
577 PointSkewed(2) = +Point(2)
578
579 ! Gaussian function
580
581 LogProb(1) = lognormfac + getLogProbDenMVN(nd = NPAR, Mean = MeanG, InvCovMat = InvCovMatG, logSqrtDetInvCovMat = logSqrtDetInvCovMatG, Point = PointSkewed) ! logProbMVN
582
583 ! Do variable transformations for the Skewed-Gaussian (banana) function.
584
585 PointSkewed(2) = a * PointSkewed(2)
586 PointSkewed(1) = PointSkewed(1)/a - b*(PointSkewed(2)**2 + a**2)
587
588 ! Banana function
589
590 LogProb(2) = getLogProbDenMVN(nd = NPAR, Mean = MeanB, InvCovMat = InvCovMatB, logSqrtDetInvCovMat = logSqrtDetInvCovMatB, Point = PointSkewed) ! logProbBanana
591
592 !MeanBnew(2) = a * MeanBnew(2)
593 !MeanBnew(1) = MeanBnew(1)/a - b*(MeanBnew(2)**2 + a**2)
594
595 !logFunc = LogProb(2)
596 !logFunc = lognormfac + LogProb(1)
597 logFunc = genLogSumExp(LogProb, maxval(LogProb))
598
599 end function getLogFuncBanana2D

622 function openFile(Test, path, label, status, position) result(File)
623
624 use String_mod, only: val2char, getLowerCase
625 implicit none
626
627 class(Test_type), intent(in) :: Test
628 character(*), intent(in), optional :: path, label, status, position
629 type(File_type) :: File
630

632

647 , status = statusDefault & ! LCOV_EXCL_LINE
648 , position = positionDefault & ! LCOV_EXCL_LINE
649#if INTEL_COMPILER_ENABLED && WINDOWS_ENABLED
650 , SHARED & ! LCOV_EXCL_LINE
651#endif
652 )
653
654 end function openFile
