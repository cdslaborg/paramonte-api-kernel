154 subroutine finalize()
155
156 use Constants_mod, only: RK, IK
157 use Decoration_mod, only: style
158 use String_mod, only: num2char
159 use Err_mod, only: abort
160 implicit none
161 real(RK) :: percentageTestPassed
162 integer(IK) :: ntotal
163 character(:), allocatable :: color
164 character(:), allocatable :: msg
165 integer(IK) :: i
166
167 ntotal = mv_npass + mv_nfail
168
169 percentageTestPassed = 0._RK
170 if (ntotal/=0_IK) percentageTestPassed = 100_IK * mv_npass / real(ntotal, kind=RK)
171
172 if (mv_Image%isFirst) then
173
174 msg = style( num2char(ntotal)//" tests performed. ", "bright", "yellow")
175
176 color = "green"; if (nint(percentageTestPassed,kind=IK)==0_IK) color = "red"
177 msg = msg // style( num2char(percentageTestPassed,"(f0.2)")//"% of "//num2char(ntotal)//" tests passed. ", "bright", color)
178
179 color = "green"; if (mv_nfail>0_IK) color = "red"
180 msg = msg // style( num2char(mv_nfail,"(g0)")//" tests failed. ", "bright", color)
181
182 msg = msg // style( "The total elapsed wall-clock time: " // num2char(mv_timeTotal,"(f0.6)")//" seconds.", "bright", "yellow" )
183
184 write(output_unit, "(*(g0,:,' '))")
185 write(output_unit, "(*(g0,:,' '))") msg
186 write(output_unit, "(*(g0,:,' '))")
187
188 if (mv_nfail>0_IK) then
189
190 write(output_unit, "(*(g0,:,' '))")
191 write(output_unit, "(*(g0,:,' '))") style("The following tests FAILED:", "bright", "red")
192 write(output_unit, "(*(g0,:,' '))")
193 do i = 1, mv_nfail
194 write(output_unit, "(*(g0,:,' '))") style(" "//num2char(i,"(g0)")//" - "//mv_FailedTestFuncName(i)%value, "bright", "red")
195 end do
196 write(output_unit, "(*(g0,:,' '))")
197 write(output_unit, "(*(g0,:,' '))") style("Errors occurred while running the ParaMonte tests. Please report this issue at:", "bright", "red")
198 write(output_unit, "(*(g0,:,' '))")
199 write(output_unit, "(*(g0,:,' '))") style(" https://github.com/cdslaborg/paramonte/issues", "bright", "cyan")
200#if ! DEBUG_ENABLED
201 write(output_unit, "(*(g0,:,' '))")
202 write(output_unit, "(*(g0,:,' '))") style("To get more information about the failure, you may also compile and run the test in debug mode.", "bright", "red")
203#endif
204 write(output_unit, "(*(g0,:,' '))")
205
206 end if
207
208 end if
209
210 if (mv_nfail>0_IK) then
211 call abort()
212 else
213 call mv_Image%finalize()
214 end if
215
216 end subroutine finalize

220 function constructTest(moduleName, inDir, outDir) result(Test)
221
222 use Decoration_mod, only: style
223 use Constants_mod, only: NLC
224 use String_mod, only: num2char
225 use Path_mod, only: Path_type
226 use Err_mod, only: abort
227
228 implicit none
229
230 character(*), intent(in) :: moduleName
231 character(*), intent(in), optional :: inDir, outDir
232 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@constructTest()"
233 type(Test_type) :: Test
234 type(Path_type) :: Path
235
236 ! set up image counts
237
238 Test%Image = mv_Image
239 Test%outputUnit = output_unit
240 Test%moduleName = moduleName
241
242 ! set up the timer
243
244 Test%Timer = Timer_type(Test%Err%occurred)
245 if (Test%Err%occurred) then
246 Test%Err%msg = PROCEDURE_NAME // ": Error occurred while setting up the test timer. There is no processor clock."
247 call abort( Err = Test%Err, prefix = "ParaMonteUnitTesting", newline = NLC, outputUnit = Test%outputUnit )
248 return
249 end if
250
251 ! set up the tests input path
252
253 if (present(inDir)) then
254 Test%inDir = inDir
255 else
256 Test%inDir = "./input"
257 end if
258 Path = Path_type(Test%inDir)
259 if (Path%Err%occurred) then
260 call abort( Err = Path%Err &
261 , newline = "\n" &
262 , outputUnit = Test%outputUnit &
263 )
264 else
265 Test%inDir = Path%modified
266 end if
267
268 ! set up the tests output path
269
270 if (present(outDir)) then
271 Test%outDir = outDir
272 else
273 Test%outDir = "./output"
274 end if
275 Path = Path_type(Test%outDir)
276 if (Path%Err%occurred) then
277 call abort( Err = Path%Err &
278 !, prefix = "FATAL: " &
279 , newline = "\n" &
280 , outputUnit = Test%outputUnit &
281 )
282 else
283 Test%outDir = Path%modified
284 if (Test%Image%isFirst) then
285 Path%Err = Path%mkdir(Test%outDir)
286 if (Path%Err%occurred) then
287 call abort( Err = Path%Err &
288 !, prefix = "FATAL: " &
289 , newline = "\n" &
290 , outputUnit = Test%outputUnit &
291 )
292 end if
293 end if
294 end if
295
296 call mv_Image%sync()
297
298 end function constructTest

304 subroutine runTest(Test, test_func, test_func_name)
305
306 use String_mod, only: StrSplit_type, num2char, isInteger, replace, padChar
307 use Err_mod, only: abort
308 implicit none
309 class(Test_type), intent(inout) :: Test
310 procedure(test_func_proc) :: test_func
311 character(*), intent(in) :: test_func_name
312 character(:), allocatable :: counterStr, funcName
313 type(StrSplit_type) :: Message
314 logical :: assertion
315 integer(IK) :: imageID
316
317 Test%funcName = "Test_"//Test%moduleName(2:)//"@"//trim(adjustl(test_func_name))
318
319 funcName = trim(adjustl(test_func_name(6:))) ! remove "test_"
320 Message%Parts = Message%split(value = funcName, delim = "_", npart = Message%nPart)
321 if ( Message%nPart==1_IK .or. .not. isInteger(Message%Parts(Message%nPart)%value) ) then
322 counterStr = "#1"
323 else
324 funcName = funcName(1:len(funcName)-len("_"//Message%Parts(Message%nPart)%value))
325 counterStr = "#"//Message%Parts(Message%nPart)%value
326 !else
327 ! Test%Err%occurred = .true.
328 ! Test%Err%msg = "Invalid function name passed to runTest() method of the Test object: " // trim(adjustl(test_func_name))
329 ! call abort ( Err = Test%Err &
330 ! , newline = "\n" &
331 ! , outputUnit = Test%outputUnit &
332 ! )
333 end if
334
335 call Test%Timer%toc()
336 assertion = test_func()
337 call Test%Timer%toc()
338
339 if (assertion) then
340
341 mv_npass = mv_npass + 1
342 Message%value = mc_passedString
343
344 else
345
346 Message%value = mc_failedString
347 mv_nfail = mv_nfail + 1
348
349 ! set up the allocation for the names of failed tests
350
351 if (mv_nfail==mv_nfailMax) then
352 block
353 type(Str_type), allocatable :: mv_FailedTestFuncNameTemp(:)
354 mv_nfailMax = 2 * mv_nfailMax
355 allocate(mv_FailedTestFuncNameTemp(mv_nfailMax))
356 mv_FailedTestFuncNameTemp(1:mv_nfail) = mv_FailedTestFuncName
357 call move_alloc(from=mv_FailedTestFuncNameTemp, to=mv_FailedTestFuncName)
358 end block
359 end if
360
361 mv_FailedTestFuncName(mv_nfail)%value = Test%funcName
362
363 end if
364
365 mv_testCounter = mv_testCounter + 1
366
367 do imageID = 1, Test%Image%count
368
369 if (imageID==Test%Image%id) then
370 write(Test%outputUnit,"(*(g0,:,' '))") "["//adjustr(num2char(mv_testCounter,len=4_IK))//"] testing" &
371 , padChar( Test%moduleName//"@"//funcName//" "//counterStr//" ","." , 79_IK ) &
372 , Message%value, "in", adjustr(num2char(Test%Timer%Time%delta,"(f0.4)",8_IK)), "seconds on image "//num2char(Test%Image%id)
373
374 end if
375
376#if CAF_ENABLED || MPI_ENABLED
377 call execute_command_line(" ", cmdstat = Test%Err%stat)
378 flush(output_unit)
379 call Test%Image%sync()
380#endif
381
382 end do
383
384 block
385 use System_mod, only: sleep
386 use Err_mod, only: Err_type
387 type(Err_type) :: Err
388 call sleep(0.02_RK, Err)
389 end block
390
391 if (Test%isVerboseMode .and. .not. assertion) then
392 Test%Err%occurred = .true.
393 Test%Err%msg = "The test assertion is FALSE."
394 call abort( Err = Test%Err &
395 , prefix = "ParaMonteTest @ "//Test%funcName//"()" &
396 , newline = "\n" &
397 , outputUnit = Test%outputUnit &
398 )
399 error stop
400 end if
401
402 end subroutine runTest

484#if CFI_ENABLED
485 function getLogFuncBanana2D(ndim,Point) result(logFunc) bind(C)

501 integer(IK) , parameter :: NPAR = 2_IK ! sum(Banana,gaussian) normalization factor

505 real(RK) , parameter :: MeanB(NPAR) = [ -5.0_RK , 0._RK ] ! mean vector of Banana function
506 real(RK) , parameter :: MeanG(NPAR) = [ 3.5_RK , 0._RK ] ! mean vector of Gaussian function

510 real(RK) , parameter :: InvCovMatG(NPAR,NPAR) = reshape([6.66666666666667_RK,0._RK,0._RK,6.66666666666667_RK],shape=shape(InvCovMatG)) ! Inverse Covariance matrix of Gaussian function
511 real(RK) , parameter :: logSqrtDetInvCovMatB = log(sqrt(4.93827160493827_RK)) ! Determinant of the Inverse Covariance matrix of Banana function
