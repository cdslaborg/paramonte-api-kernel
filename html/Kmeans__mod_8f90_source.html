187 function getKmeans(nd, np, nc, nt, Point, InitCenter, relTol, niterMax, nfailMax, minSize) result(Kmeans)
188#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
189 !DEC$ ATTRIBUTES DLLEXPORT :: getKmeans
190#endif
191 use Constants_mod, only: IK, RK, POSINF_RK
192
193 implicit none
194
195 integer , intent(in) :: nd, np, nc, nt
196 real(RK) , intent(in) :: Point(nd,np)
197 real(RK) , intent(in) , optional :: InitCenter(nd,nc)
198 real(RK) , intent(in) , optional :: relTol
199 integer , intent(in) , optional :: minSize
200 integer , intent(in) , optional :: nfailMax
201 integer , intent(in) , optional :: niterMax
202
203 type(Kmeans_type) :: Kmeans
204 type(Kmeans_type), allocatable :: KmeansTry(:)
205
206 integer :: itry
207 integer :: itryLeastPotential
208 integer :: nfailMaxDefault
209 integer :: niterMaxDefault
210 integer :: minSizeDefault
211 real(RK) :: relTolDefault
212 real(RK) :: potential
213
214 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@getKmeans()"
215
216 niterMaxDefault = 1e3 ; if (present(niterMax)) niterMaxDefault = niterMax
217 nfailMaxDefault = 1e1 ; if (present(nfailMax)) nfailMaxDefault = nfailMax
218 minSizeDefault = 1 ; if (present(minSize)) minSizeDefault = minSize
219 relTolDefault = 1.e-4_RK ; if (present(relTol)) relTolDefault = relTol
220
221 if (allocated(KmeansTry)) deallocate(KmeansTry) ! LCOV_EXCL_LINE
222 allocate(KmeansTry(nt))
223 potential = POSINF_RK
224
225 ! Compute the optimal Kmeans
226
227 if (nt > 1) then
228
229 itryLeastPotential = 0
230 do itry = 1, nt
231 KmeansTry(itry) = Kmeans_type(nd, np, nc)
232 call runKmeans( nd = nd & ! LCOV_EXCL_LINE
233 , np = np & ! LCOV_EXCL_LINE
234 , nc = nc & ! LCOV_EXCL_LINE
235 , niterMax = niterMaxDefault & ! LCOV_EXCL_LINE
236 , nfailMax = nfailMaxDefault & ! LCOV_EXCL_LINE
237 , minSize = minSizeDefault & ! LCOV_EXCL_LINE
238 , relTol = relTolDefault & ! LCOV_EXCL_LINE
239 , Point = Point & ! LCOV_EXCL_LINE
240 , Size = KmeansTry(itry)%Size & ! LCOV_EXCL_LINE
241 , Center = KmeansTry(itry)%Center & ! LCOV_EXCL_LINE
242 , Membership = KmeansTry(itry)%Membership & ! LCOV_EXCL_LINE
243 !, NormedPoint = KmeansTry(itry)%NormedPoint & ! LCOV_EXCL_LINE
244 , MinDistanceSq = KmeansTry(itry)%MinDistanceSq & ! LCOV_EXCL_LINE
245 , potential = KmeansTry(itry)%potential & ! LCOV_EXCL_LINE
246 , niter = KmeansTry(itry)%niter & ! LCOV_EXCL_LINE
247 , nfail = KmeansTry(itry)%nfail & ! LCOV_EXCL_LINE
248 , Err = KmeansTry(itry)%Err & ! LCOV_EXCL_LINE
249 , InitCenter = InitCenter & ! LCOV_EXCL_LINE
250 )
251 if ( KmeansTry(itry)%potential < potential .and. .not. any(KmeansTry(itry)%Size < minSizeDefault) ) then
252 potential = KmeansTry(itry)%potential
253 itryLeastPotential = itry
254 end if
255 end do
256
257 if (itryLeastPotential > 0) then
258 Kmeans = KmeansTry(itryLeastPotential)
259 else
260 ! LCOV_EXCL_START
261 Kmeans%Err%msg = "All Kmeans clustering tries failed."
262 Kmeans%Err%occurred = .true.
263 Kmeans%Err%stat = 1
264 return
265 ! LCOV_EXCL_STOP
266 end if
267
268 else
269
270 Kmeans = Kmeans_type(nd, np, nc)
271 call runKmeans( nd = nd & ! LCOV_EXCL_LINE
272 , np = np & ! LCOV_EXCL_LINE
273 , nc = nc & ! LCOV_EXCL_LINE
274 , niterMax = niterMaxDefault & ! LCOV_EXCL_LINE
275 , nfailMax = nfailMaxDefault & ! LCOV_EXCL_LINE
276 , minSize = minSizeDefault & ! LCOV_EXCL_LINE
277 , relTol = relTolDefault & ! LCOV_EXCL_LINE
278 , Point = Point & ! LCOV_EXCL_LINE
279 , Size = Kmeans%Size & ! LCOV_EXCL_LINE
280 , Center = Kmeans%Center & ! LCOV_EXCL_LINE
281 , Membership = Kmeans%Membership & ! LCOV_EXCL_LINE
282 , MinDistanceSq = Kmeans%MinDistanceSq & ! LCOV_EXCL_LINE
283 , potential = Kmeans%potential & ! LCOV_EXCL_LINE
284 , niter = Kmeans%niter & ! LCOV_EXCL_LINE
285 , nfail = Kmeans%nfail & ! LCOV_EXCL_LINE
286 , Err = Kmeans%Err & ! LCOV_EXCL_LINE
287 , InitCenter = InitCenter & ! LCOV_EXCL_LINE
288 )
289#if DEBUG_ENABLED || TESTING_ENABLED
290 if (any(Kmeans%Size < minSizeDefault)) then ! @todo xxx : This need further enhancement. Is it even necessary?
291 ! LCOV_EXCL_START
292 if (.not. Kmeans%Err%occurred .and. Kmeans%Err%stat /= 2) then
293 Kmeans%Err%msg = "The Kmeans clustering sizes are smaller than the requested input minimum cluster size."
294 write(*,*) Kmeans%Err%msg, minSizeDefault, Kmeans%Size
295 Kmeans%Err%occurred = .true.
296 Kmeans%Err%stat = 1
297 error stop
298 !return
299 end if
300 ! LCOV_EXCL_STOP
301 else
302 block
303 integer :: ic
304 do ic = 1, nc
305 if (Kmeans%Size(ic) == 0) then
306 if (any(Kmeans%Membership(1:np) == ic)) then
307 write(*,*) "Internal error occurred: Zero-sized cluster has associated members. ic, ip = ", ic
308 error stop
309 end if
310 end if
311 end do
312 end block
313 end if
314#endif
315 end if
316
317#if DEBUG_ENABLED || TESTING_ENABLED
318 block
319 integer :: sizeDum, ip, ic
320 real(RK) :: CenterDum(nd)
321 do ic = 1, nc
322 sizeDum = 0
323 CenterDum = 0._RK
324 do ip = 1, np
325 if (Kmeans%Membership(ip) == ic) then
326 CenterDum(1:nd) = CenterDum(1:nd) + Point(1:nd,ip)
327 sizeDum = sizeDum + 1
328 end if
329 end do
330 CenterDum(1:nd) = CenterDum(1:nd) / sizeDum
331 if (any(abs(CenterDum(1:nd) - Kmeans%Center(1:nd,ic)) > 1.e-6_RK)) then
332 if (sizeDum /= Kmeans%Size(ic)) then; write(*,*) "sizeDum, Kmeans%Size = ", sizeDum, Kmeans%Size; error stop; endif
333 write(*,*) "nd, minSize, ic, nc = ", nd, minSizeDefault, ic, nc
334 write(*,*) "Kmeans%Center(1:nd) = ", Kmeans%Center(1:nd,ic)
335 write(*,*) "CenterDum(1:nd) = ", CenterDum(1:nd)
336 write(*,*) "Kmeans%Size(1:nc) = ", Kmeans%Size
337 write(*,*) "Kmeans%Err%occurred = ", Kmeans%Err%occurred
338 write(*,*) "Kmeans%Err%stat = ", Kmeans%Err%stat
339 error stop
340 endif
341 end do
342 end block
343#endif
344
345 end function getKmeans

384 ! Amir Shahmoradi, April 03, 2017, 2:16 PM, ICES, University of Texas at Austin
385 subroutine runKmeans( nd, np, nc & ! LCOV_EXCL_LINE
386 , nfailMax & ! LCOV_EXCL_LINE
387 , niterMax & ! LCOV_EXCL_LINE
388 , minSize & ! LCOV_EXCL_LINE
389 , relTol & ! LCOV_EXCL_LINE
390 , Point & ! LCOV_EXCL_LINE
391 , Size & ! LCOV_EXCL_LINE
392 , niter & ! LCOV_EXCL_LINE
393 , nfail & ! LCOV_EXCL_LINE
394 , Center & ! LCOV_EXCL_LINE
395 , potential & ! LCOV_EXCL_LINE
396 , Membership & ! LCOV_EXCL_LINE
397 , MinDistanceSq & ! LCOV_EXCL_LINE
398 , Err & ! LCOV_EXCL_LINE
399 , InitCenter & ! LCOV_EXCL_LINE
400 )
401#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
402 !DEC$ ATTRIBUTES DLLEXPORT :: runKmeans
403#endif
404 use Constants_mod, only: IK, RK, POSINF_RK
405
406 implicit none
407
408 integer , intent(in) :: nc, nd, np, minSize, nfailMax, niterMax
409 real(RK) , intent(in) :: Point(nd,np)
410 real(RK) , intent(in) :: relTol
411 integer , intent(out) :: nfail
412 integer , intent(out) :: niter
413 real(RK) , intent(out) :: potential
414 real(RK) , intent(out) :: Center(nd,nc)
415 !real(RK) , intent(out) :: NormedPoint(nd,np,nc)
416 real(RK) , intent(out) :: MinDistanceSq(np)
417 integer , intent(out) :: Membership(np)
418 integer , intent(out) :: Size(nc)
419 type(Err_type) , intent(out) :: Err
420 real(RK) , intent(in), optional :: InitCenter(nd,nc)
421
422 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@runKmeans()"
423
424 integer :: MembershipOld(np) ! current (old) Memberships of the clusters.
425 logical :: convergenceOccurred
426 integer :: ip, ic
427 real(RK) :: distanceSq ! distance of each point ip from a given cluster center.
428 real(RK) :: potentialOld ! The sum of all minimum distances squared.
429 real(RK) :: SumPoint(nd,nc) ! The sum of the points in each cluster.
430
431 Err%occurred = .false.
432
433 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
434 ! Find the distances of all points to all cluster centers. If not given, initialize the centers via the Kmeans++ algorithm.
435 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
436
437 if(present(InitCenter)) then
438 !%%%%%%%%%%%%%%%%%%%
439 ic = 1
440 Size(ic) = 0
441 do ip = 1, np
442 MinDistanceSq(ip) = sum( (InitCenter(1:nd,ic) - Point(1:nd,ip))**2 )
443 MembershipOld(ip) = ic
444 end do
445 !%%%%%%%%%%%%%%%%%%%
446 do ic = 2, nc - 1
447 Size(ic) = 0
448 do ip = 1, np
449 distanceSq = sum( (InitCenter(1:nd,ic) - Point(1:nd,ip))**2 )
450 if (distanceSq < MinDistanceSq(ip)) then
451 MinDistanceSq(ip) = distanceSq
452 MembershipOld(ip) = ic
453 end if
454 end do
455 end do
456 !%%%%%%%%%%%%%%%%%%%
457 ic = nc
458 Size(ic) = 0
459 SumPoint = 0._RK
460 potentialOld = 0._RK
461 do ip = 1, np
462 distanceSq = sum( (InitCenter(1:nd,ic) - Point(1:nd,ip))**2 )
463 if (distanceSq < MinDistanceSq(ip)) then
464 MinDistanceSq(ip) = distanceSq
465 MembershipOld(ip) = ic
466 end if
467 SumPoint(1:nd,MembershipOld(ip)) = SumPoint(1:nd,MembershipOld(ip)) + Point(1:nd,ip)
468 Size(MembershipOld(ip)) = Size(MembershipOld(ip)) + 1
469 potentialOld = potentialOld + MinDistanceSq(ip)
470 end do
471 !%%%%%%%%%%%%%%%%%%%
472 else
473 call runKPP(nc, nd, np, Point, SumPoint, MembershipOld, Size, potentialOld)
474 end if
475
476 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
477 ! Refine all centers until convergence.
478 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
479
480 nfail = 0
481 niter = 0
482
483 loopCenterRefinement: do
484
485 ! compute the new cluster centers and create the new memberships.
486
487 !%%%%%%%%%%%%%%%%%%%
488
489 ic = 1
490 if (Size(ic) > 0) then
491 Center(1:nd,ic) = SumPoint(1:nd,ic) / real(Size(ic), kind = RK)
492 do concurrent(ip = 1:np)
493 !NormedPoint(1:nd,ip,ic) = Center(1:nd,ic) - Point(1:nd,ip)
494 !MinDistanceSq(ip) = sum( NormedPoint(1:nd,ip,ic)**2 )
495 MinDistanceSq(ip) = sum( (Center(1:nd,ic) - Point(1:nd,ip))**2 )
496 Membership(ip) = 1
497 end do
498 else
499 MinDistanceSq(1:np) = POSINF_RK
500 end if
501
502 !%%%%%%%%%%%%%%%%%%%
503
504 do ic = 2, nc - 1
505 if (Size(ic) > 0) then
506 Center(1:nd,ic) = SumPoint(1:nd,ic) / real(Size(ic), kind = RK)
507 do concurrent(ip = 1:np)
508 !NormedPoint(1:nd,ip,ic) = Center(1:nd,ic) - Point(1:nd,ip)
509 !distanceSq = sum( NormedPoint(1:nd,ip,ic)**2 )
510 distanceSq = sum( (Center(1:nd,ic) - Point(1:nd,ip))**2 )
511 if (distanceSq < MinDistanceSq(ip)) then
512 MinDistanceSq(ip) = distanceSq
513 Membership(ip) = ic
514 end if
515 end do
516 end if
517 end do
518
519 !%%%%%%%%%%%%%%%%%%%
520
521 ic = nc
522 potential = 0._RK
523 convergenceOccurred = .true.
524 if (Size(ic) > 0) then
525 Center(1:nd,ic) = SumPoint(1:nd,ic) / real(Size(ic), kind = RK)
526 do ip = 1, np
527 !NormedPoint(1:nd,ip,ic) = Center(1:nd,ic) - Point(1:nd,ip)
528 !distanceSq = sum( NormedPoint(1:nd,ip,ic)**2 )
529 distanceSq = sum( (Center(1:nd,ic) - Point(1:nd,ip))**2 )
530 if (distanceSq < MinDistanceSq(ip)) then
531 MinDistanceSq(ip) = distanceSq
532 Membership(ip) = ic
533 end if
534 if (Membership(ip)/=MembershipOld(ip)) then
535 SumPoint(1:nd,MembershipOld(ip)) = SumPoint(1:nd,MembershipOld(ip)) - Point(1:nd,ip)
536 SumPoint(1:nd,Membership(ip)) = SumPoint(1:nd,Membership(ip)) + Point(1:nd,ip)
537 Size(MembershipOld(ip)) = Size(MembershipOld(ip)) - 1
538 Size(Membership(ip)) = Size(Membership(ip)) + 1
539 convergenceOccurred = .false.
540 end if
541 potential = potential + MinDistanceSq(ip)
542 end do
543 else
544 do ip = 1, np
545 if (Membership(ip)/=MembershipOld(ip)) then
546 SumPoint(1:nd,MembershipOld(ip)) = SumPoint(1:nd,MembershipOld(ip)) - Point(1:nd,ip)
547 SumPoint(1:nd,Membership(ip)) = SumPoint(1:nd,Membership(ip)) + Point(1:nd,ip)
548 Size(MembershipOld(ip)) = Size(MembershipOld(ip)) - 1
549 Size(Membership(ip)) = Size(Membership(ip)) + 1
550 convergenceOccurred = .false.
551 end if
552 potential = potential + MinDistanceSq(ip)
553 end do
554 end if
555
556 !%%%%%%%%%%%%%%%%%%%
557
558#if DEBUG_ENABLED || TESTING_ENABLED || CODECOV_ENABLED
559 if (any(Size<0)) then
560 ! LCOV_EXCL_START
561 block
562 use Err_mod, only: abort
563 Err%msg = PROCEDURE_NAME//": Internal error occurred - Size < 0. Please report this issue along with circumstances to the developers at: https://github.com/cdslaborg/paramonte/issues"
564 call abort(Err)
565 error stop
566 return
567 end block
568 ! LCOV_EXCL_STOP
569 end if
570#endif
571
572 if (any(Size < minSize)) then
573 ! LCOV_EXCL_START
574 niter = 0
575 nfail = nfail + 1
576 if (nfail > nfailMax) then
577 !Err%msg = "nfail > nfailMax"
578 Err%occurred = .true.
579 Err%stat = 2
580 exit loopCenterRefinement
581 end if
582 call runKPP(nc, nd, np, Point, SumPoint, MembershipOld, Size, potentialOld) ! find new random Centers and start over.
583 cycle loopCenterRefinement
584 ! LCOV_EXCL_STOP
585 end if
586
587 ! if `niterMax` has reached, quit.
588
589 if (niter < niterMax) then
590
591 ! If convergence has occurred, return.
592
593#if DEBUG_ENABLED || TESTING_ENABLED || CODECOV_ENABLED
594 if (potential < log(tiny(potential))) then
595 write(*,*) "Internal error occurred: potential == 0."
596 write(*,*) "potential, Size", potential, Size
597 error stop
598 end if
599#endif
600 if (convergenceOccurred .or. abs(potential-potentialOld)/potential < relTol) exit loopCenterRefinement
601
602 niter = niter + 1
603
604 MembershipOld(1:np) = Membership(1:np)
605 potentialOld = potential
606
607 else
608
609 !Err%msg = "niter > niterMax"
610 Err%stat = 1
611 Err%occurred = .true.
612 exit loopCenterRefinement
613
614 end if
615
616 end do loopCenterRefinement
617
618 do concurrent(ic = 1:nc)
619 if (Size(ic) >= 0) Center(1:nd,ic) = SumPoint(1:nd,ic) / real(Size(ic), kind = RK)
620 end do
621
622 end subroutine runKmeans

676 subroutine getProp(Kmeans, nd, np, Point, Index, inclusionFraction, pointLogVolNormed)
677#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
678 !DEC$ ATTRIBUTES DLLEXPORT :: getProp
679#endif
680 use Constants_mod, only: IK, RK, NEGINF_RK
681 use MathLogSumExp_mod, only: genLogSumExp
682 use MatrixDiag_mod, only: genDiag
683 use Matrix_mod, only: genInvMatFromChoLow
684 use MatrixCholesky_mod, only: getChoLow
685
686 implicit none
687
688 class(Kmeans_type) , intent(inout) :: Kmeans
689 integer(IK) , intent(in) :: nd, np
690 real(RK) , intent(inout) :: Point(nd,np)
691 real(RK) , intent(in) , optional :: pointLogVolNormed
692 real(RK) , intent(in) , optional :: inclusionFraction
693 integer(IK) , intent(inout) , optional :: Index(np)
694
695 integer(IK) , allocatable :: KmeansMemberCounter(:)
696 integer(IK) :: i, j, ic, ip, nc
697 integer(IK) :: ipstart, ipend
698
699 real(RK) :: NormedPoint(nd,np)
700 real(RK) :: scaleFactorSqInverse
701 real(RK) :: pointLogVolNormedDefault
702 real(RK) :: maxLogVolNormed
703 real(RK) :: proposalScaleFactor
704
705 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@getProp()"
706
707 nc = size(Kmeans%Size)
708
709 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
710 ! Reorder Point based on the identified clusters.
711 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
712
713 call Kmeans%Prop%allocate(nd,np,nc)
714
715 if (allocated(KmeansMemberCounter)) deallocate(KmeansMemberCounter) ! LCOV_EXCL_LINE
716 allocate(KmeansMemberCounter(nc))
717
718 ! Compute the cumulative sum of the cluster Size array.
719
720 Kmeans%Prop%CumSumSize(0) = 0_IK
721 loopDetermineClusterBoundaries: do ic = 1, nc
722 KmeansMemberCounter(ic) = Kmeans%Prop%CumSumSize(ic-1)
723 Kmeans%Prop%CumSumSize(ic) = KmeansMemberCounter(ic) + Kmeans%Size(ic)
724 end do loopDetermineClusterBoundaries
725
726 if (present(Index)) then
727 do ip = 1, np
728 KmeansMemberCounter(Kmeans%Membership(ip)) = KmeansMemberCounter(Kmeans%Membership(ip)) + 1
729 NormedPoint(1:nd,KmeansMemberCounter(Kmeans%Membership(ip))) = Point(1:nd,ip)
730 Kmeans%Prop%Index(KmeansMemberCounter(Kmeans%Membership(ip))) = Index(ip)
731 end do
732 Index = Kmeans%Prop%Index

804 Kmeans%Prop%InvCovMat(1:nd,1:nd,ic) = Kmeans%Prop%InvCovMat(1:nd,1:nd,ic) * scaleFactorSqInverse

830!write(*,*)
831 else

833 end if

836 maxLogVolNormed = max( maxLogVolNormed, Kmeans%Prop%LogVolNormed(ic) )
837 !Kmeans%Prop%LogDenNormed(ic) = log(real(Kmeans%Prop%EffectiveSize(ic),RK)) - Kmeans%Prop%LogVolNormed(ic)

845 end if blockMinimumClusterSize

853 !%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

899 Kmeans%Prop%MahalSq(ip,ic) = sum( ( Point(1:nd,ip) - Kmeans%Center(1:nd,ic) )**2 )
900 end do
901
902 Kmeans%Prop%ScaleFactorSq(ic) = max(exp(2*pointLogVolNormedDefault/nd), maxval(Kmeans%Prop%MahalSq(ipstart:ipend,ic)))
903 scaleFactorSqInverse = 1._RK / Kmeans%Prop%ScaleFactorSq(ic)

905
906 !Kmeans%Prop%MahalSq(ip,ic) = Kmeans%Prop%MahalSq(ip,ic) * scaleFactorSqInverse
907
908 Kmeans%Prop%ChoDia(1:nd,ic) = proposalScaleFactor
909 Kmeans%Prop%LogVolNormed(ic) = nd * log(proposalScaleFactor)
910 !Kmeans%Prop%LogDenNormed(ic) = log(real(Kmeans%Size(ic),RK)) - Kmeans%Prop%LogVolNormed(ic)
911 Kmeans%Prop%ChoLowCovUpp(1:nd,1:nd,ic) = genDiag(nd, nd, Kmeans%Prop%ScaleFactorSq(ic))
912 Kmeans%Prop%InvCovMat(1:nd,1:nd,ic) = genDiag(nd, nd, scaleFactorSqInverse)

971 subroutine runKPP(nc, nd, np, Point, SumPoint, Membership, Size, potential)
972#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
973 !DEC$ ATTRIBUTES DLLEXPORT :: runKPP
974#endif
975 use Statistics_mod, only: genRandInt
976
977 implicit none
978
979 integer , intent(in) :: np, nd, nc
980 real(RK) , intent(in) :: Point(nd,np)
981 real(RK) , intent(out) :: SumPoint(nd,nc)
982 integer , intent(out) :: Membership(np)
983 integer , intent(out) :: Size(nc)
984 real(RK) , intent(out) :: potential
985
986 real(RK) :: dummy
987 real(RK) :: distanceSq
988 real(RK) :: Center(nd,nc)
989 real(RK) :: MinDistanceSq(np) ! An array representing the minimum distances of individual points from all clusters.
990 real(RK) :: CumSumDistSq(0:np) ! sum of distance squared of points up to the given index of the vector.
991 integer :: ic, ip
992
993 CumSumDistSq(0) = 0._RK ! This element must always remain zero.
994
995 ic = 1
996 Center(1:nd,ic) = Point( 1:nd , genRandInt(1,np) ) ! perhaps this random assignment is unnecessary, after all, Point is supposed to be of random order.
997 do ip = 1, np
998 MinDistanceSq(ip) = sum( (Center(1:nd,ic) - Point(1:nd,ip))**2 ) ! find the distance-squared of each point to the # ic cluster Center.
999 CumSumDistSq(ip) = CumSumDistSq(ip-1) + MinDistanceSq(ip)
1000 Membership(ip) = ic
1001 end do
1002
1003 ic = 2
1004 call random_number(dummy)
1005 dummy = dummy * CumSumDistSq(np)
1006 Center(1:nd,ic) = Point( 1:nd , minloc(CumSumDistSq, dim = 1, mask = CumSumDistSq > dummy) - 1 ) ! -1 is essential to compensate for the 0 starting index of CumSumDistSq
1007
1008 ! Find the rest of cluster centers.
1009
1010 do ic = 2, nc - 1
1011
1012 do ip = 1, np
1013 distanceSq = sum( (Center(1:nd,ic) - Point(1:nd,ip))**2 )
1014 if (distanceSq < MinDistanceSq(ip)) then
1015 MinDistanceSq(ip) = distanceSq
1016 Membership(ip) = ic
1017 end if
1018 CumSumDistSq(ip) = CumSumDistSq(ip-1) + MinDistanceSq(ip)
1019 end do
1020
1021 call random_number(dummy)
1022 dummy = dummy * CumSumDistSq(np)
1023 Center(1:nd,ic+1) = Point( 1:nd , minloc(CumSumDistSq, dim = 1, mask = CumSumDistSq > dummy) - 1 )
1024
1025 end do
1026
1027 Size = 0
1028 SumPoint = 0._RK
1029 potential = 0._RK
1030 do ip = 1, np
1031 distanceSq = sum( (Center(1:nd,nc) - Point(1:nd,ip))**2 )
1032 if (distanceSq < MinDistanceSq(ip)) then
1033 MinDistanceSq(ip) = distanceSq
1034 Membership(ip) = nc
1035 end if
1036 SumPoint(1:nd,Membership(ip)) = SumPoint(1:nd,Membership(ip)) + Point(1:nd,ip)
1037 Size(Membership(ip)) = Size(Membership(ip)) + 1
1038 potential = potential + MinDistanceSq(ip)
1039 end do
1040
1041 end subroutine runKPP
