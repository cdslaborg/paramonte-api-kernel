354 ! The Gaussian PDF with sigma = 1/sqrt(2)
355 function getTestFuncOpenInterval_1(x) result(funcVal)
356 use Constants_mod, only: RK, INVERSE_SQRT_PI, NEGLOGINF_RK
357 implicit none
358 real(RK), intent(in) :: x
359 real(RK) :: funcVal
360 real(RK), parameter :: LOG_INVERSE_SQRT_PI = log(INVERSE_SQRT_PI)
361 funcVal = LOG_INVERSE_SQRT_PI - x**2
362 if (funcVal<NEGLOGINF_RK) then ! This takes care of the GNU Fortran 9.1 test crash in debug mode.
363 funcVal = 0._RK
364 else
365 funcVal = exp(funcVal)
366 end if
367 end function getTestFuncOpenInterval_1
