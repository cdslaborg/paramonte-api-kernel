171 pure function constructEllipsoid(nd, np, Point, isBounding) result(Ellipsoid)
172#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
173 !DEC$ ATTRIBUTES DLLEXPORT :: constructEllipsoid
174#endif
175 use Constants_mod, only: IK, RK ! LCOV_EXCL_LINE
176 use Matrix_mod, only: getChoLow, genInvMatFromChoLow ! LCOV_EXCL_LINE
177 implicit none
178 integer(IK) , intent(in) :: nd, np
179 real(RK) , intent(in) :: Point(nd,np)
180 logical , intent(in) :: isBounding
181 type(Ellipsoid_type) :: Ellipsoid
182
183 real(RK) :: scaleFactorSqInverse
184 real(RK) :: NormedPoint(nd,np)
185 integer(IK) :: id, ip
186
187 ! Compute the mean.
188
189 Ellipsoid%Center = sum(Point, dim = 2) / np
190
191 do concurrent(ip = 1:np)
192 NormedPoint(1:nd,ip) = Point(1:nd,ip) - Ellipsoid%Center
193 end do
194
195 ! Compute the upper representative matrix of the cluster representative matrices.
196
197 allocate(Ellipsoid%ChoLowCovUpp(1:nd,1:nd), Ellipsoid%ChoDia(1:nd), source = 0._RK)
198 do ip = 1, np
199 do id = 1, nd
200 Ellipsoid%ChoLowCovUpp(1:id,id) = Ellipsoid%ChoLowCovUpp(1:id,id) + NormedPoint(1:id,ip) * NormedPoint(id,ip)
201 end do
202 end do
203 Ellipsoid%ChoLowCovUpp(1:nd,1:nd) = Ellipsoid%ChoLowCovUpp(1:nd,1:nd) / real(np - 1, RK)
204
205 ! Compute the Cholesky Factor of the cluster representative matrices.
206
207 call getChoLow(nd, Ellipsoid%ChoLowCovUpp(1:nd,1:nd), Ellipsoid%ChoDia(1:nd))
208 if (Ellipsoid%ChoDia(1) < 0._RK) return
209
210 ! Compute the inverse of the cluster representative matrices.
211
212 Ellipsoid%InvCovMat = genInvMatFromChoLow(nd, Ellipsoid%ChoLowCovUpp, Ellipsoid%ChoDia)
213
214 ! Compute the MahalSq of the points.
215
216 allocate(Ellipsoid%MahalSq(np))
217
218! if (isBounding) then
219
220 ! Compute the scaleFcator of the bounding region.
221
222 Ellipsoid%scaleFactorSq = NEGINF_RK
223 do ip = 1, np
224 Ellipsoid%MahalSq(ip) = dot_product( NormedPoint(1:nd,ip) , matmul(Ellipsoid%InvCovMat, NormedPoint(1:nd,ip)) )
225 if (Ellipsoid%scaleFactorSq < Ellipsoid%MahalSq(ip)) Ellipsoid%scaleFactorSq = Ellipsoid%MahalSq(ip)
226 end do
227
228 Ellipsoid%scaleFactor = sqrt(Ellipsoid%scaleFactorSq)
229 scaleFactorSqInverse = 1._RK / Ellipsoid%scaleFactorSq
230
231 Ellipsoid%ChoDia(1:nd) = Ellipsoid%ChoDia * Ellipsoid%scaleFactor
232 Ellipsoid%InvCovMat(1:nd,1:nd) = Ellipsoid%InvCovMat * scaleFactorSqInverse
233 do concurrent(id = 1:nd); Ellipsoid%ChoLowCovUpp(id+1:nd,id) = Ellipsoid%ChoLowCovUpp(id+1:nd,id) * Ellipsoid%scaleFactor; end do
234
235! else
236!
237! do concurrent(ip = 1:np)
238! Ellipsoid%MahalSq(ip) = dot_product( NormedPoint(1:nd,ip) , matmul(Ellipsoid%InvCovMat, NormedPoint(1:nd,ip)) )
239! end do
240!
241! Ellipsoid%scaleFactor = 1._RK
242! Ellipsoid%scaleFactorSq = 1._RK
243! Ellipsoid%logVolNormed = sum(log(Ellipsoid%ChoDia(1:nd)))
244!
245! end if
246
247 Ellipsoid%logVolNormed = sum(log(Ellipsoid%ChoDia(1:nd)))
248
249 end function constructEllipsoid

304 pure function getEllVolCoef(nd) result(ellVolCoef)
305#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
306 !DEC$ ATTRIBUTES DLLEXPORT :: getEllVolCoef
307#endif
308 use Constants_mod, only: IK, RK, PI
309 implicit none
310 integer(IK), intent(in) :: nd
311 integer(IK) :: i,k
312 real(RK), parameter :: FOUR_PI = PI * 4._RK
313 real(RK) :: ellVolCoef
314 if (mod(nd,2_IK)==0_IK) then ! nd is even
315
316 ellVolCoef = PI
317 do i = 2_IK, nd / 2_IK
318 ellVolCoef = ellVolCoef * PI / i ! nd = 2k ; ellVolCoef = PI^(k) / Factorial(k)
319 end do
320
321 else ! nd is an odd integer
322
323 ! nd = 2k-1 ; gamma(nd/2 + 1) = gamma(k + 1/2) ; gamma(k+1/2) = sqrt(PI) * (2k)! / (4^k * k!)
324 k = (nd + 1_IK) / 2_IK
325
326 ! This is to avoid an extra unnecessary division of ellVolCoef by PI
327 ellVolCoef = 4._RK / (k + 1_IK)
328
329 do i = k+2_IK, 2_IK*k
330 ! ellVolCoef = PI^(k-1/2) / gamma(k+1/2) = PI^(k+1) * 4^k * k! / (2k)!
331 ellVolCoef = ellVolCoef * FOUR_PI / i
332 end do
333
334 end if
335 end function getEllVolCoef

408 pure function genLogVolUnitBall(nd) result(logVolUnitBall)

439 pure function getLogSurfUnitBall(nd) result(logSurfUnitBall)
440#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
441 !DEC$ ATTRIBUTES DLLEXPORT :: getLogSurfUnitBall
442#endif
443 use Constants_mod, only: IK, RK, PI, LOG_TWO, NEGINF_RK
444 use SpecFuncGamma_mod, only: getLogFactorial
445 implicit none
446 integer(IK) , intent(in) :: nd
447 real(RK) , parameter :: LOG_PI = log(PI)
448 real(RK) :: logSurfUnitBall
449 real(RK) :: ndHalf_RK
450 integer(IK) :: ndHalf_IK
451 if (mod(nd,2_IK) == 0_IK) then ! nd is even
452 if (nd > 0_IK) then
453 ndHalf_IK = nd / 2_IK
454 logSurfUnitBall = LOG_TWO + ndHalf_IK * LOG_PI - getLogFactorial(ndHalf_IK-1_IK) ! nd = 2k ; logSurfUnitBall = PI^(k) / Factorial(k)
455 else
456 logSurfUnitBall = NEGINF_RK
457 end if
458 else ! nd is an odd integer
459 ndHalf_RK = 0.5_RK * real(nd,kind=RK)
460 logSurfUnitBall = LOG_TWO + ndHalf_RK * LOG_PI - log_gamma(ndHalf_RK)
461 end if
462 end function getLogSurfUnitBall

548 function getLogVolChoLowEllipsoidFromRepMatUpp(ChoLowRepMatUpp,ChoDia,failed,logVolUnitBall) result(logVolEllipsoid)
549#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
550 !DEC$ ATTRIBUTES DLLEXPORT :: getLogVolChoLowEllipsoidFromRepMatUpp
551#endif
552 use Matrix_mod, only: getChoLow, genUpperFromUpper
553 use Constants_mod, only: IK, RK
554 implicit none
555 real(RK) , intent(inout) , contiguous :: ChoLowRepMatUpp(:,:)
556 real(RK) , intent(out) , optional :: ChoDia(size(ChoLowRepMatUpp,dim=1))
557 logical , intent(out) , optional :: failed
558 real(RK) , intent(in) , optional :: logVolUnitBall
559 real(RK) :: logVolEllipsoid
560 real(RK) :: logVolUnitBallDef
561 real(RK) , allocatable :: ChoDiaDef(:), ChoLowRepMatUppDef(:,:)
562 integer(IK) :: nd
563 nd = size(ChoLowRepMatUpp,dim=1,kind=IK)
564 if (present(logVolUnitBall)) then
565 logVolUnitBallDef = logVolUnitBall
566 else
567 logVolUnitBallDef = genLogVolUnitBall(nd)
568 end if
569 if (present(ChoDia)) then
570 call getChoLow(nd,ChoLowRepMatUpp,ChoDia)
571 if (ChoDia(1) > 0._RK) then
572 logVolEllipsoid = genLogVolUnitBall(nd) + sum(log(ChoDia))
573 return
574 end if
575 else
576 allocate(ChoDiaDef(nd))
577 ChoLowRepMatUppDef = genUpperFromUpper(nd, ChoLowRepMatUpp, diagEnabled = .true.)
578 call getChoLow(nd,ChoLowRepMatUppDef,ChoDiaDef)
579 if (ChoDiaDef(1) > 0._RK) then
580 logVolEllipsoid = genLogVolUnitBall(nd) + sum(log(ChoDiaDef))
581 deallocate(ChoDiaDef,ChoLowRepMatUppDef)
582 return
583 end if
584 deallocate(ChoDiaDef)
585 end if
586 if (present(failed)) then
587 failed = .true.
588 return
589 else
590 write(*,"(A)") MODULE_NAME//"@getLogVolChoLowEllipsoidFromRepMatUpp(): Cholesky factorization failed."
591 error stop
592 end if
593 end function getLogVolChoLowEllipsoidFromRepMatUpp

659 function genRandPointOnEllipsoid(nd,Mean,ChoLow,ChoDia) result(RandPointOnEllipsoid)
660#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
661 !DEC$ ATTRIBUTES DLLEXPORT :: genRandPointOnEllipsoid
662#endif
663 use DistUniNorm_mod, only: genRand
664 implicit none
665 integer(IK), intent(in) :: nd
666 real(RK) , intent(in) :: Mean(nd)
667 real(RK) , intent(in) :: ChoLow(nd,nd) ! Cholesky lower triangle, calculated from the MVN CovMat.
668 real(RK) , intent(in) :: ChoDia(nd) ! Cholesky diagonal terms, calculated from the MVN CovMat.
669 real(RK) :: RandPointOnEllipsoid(nd), DummyVec(nd), sumSqDummyVec
670 integer(IK) :: i,j
671 sumSqDummyVec = 0._RK
672 do j=1,nd
673 DummyVec(j) = genRand()
674 sumSqDummyVec = sumSqDummyVec + DummyVec(j)**2
675 end do
676 DummyVec = DummyVec / sqrt(sumSqDummyVec) ! DummyVec is a random point on the surface of nd-sphere.
677 RandPointOnEllipsoid = 0._RK
678 do j = 1,nd
679 RandPointOnEllipsoid(j) = RandPointOnEllipsoid(j) + ChoDia(j) * DummyVec(j)
680 do i = j+1,nd
681 RandPointOnEllipsoid(i) = RandPointOnEllipsoid(i) + ChoLow(i,j) * DummyVec(j)
682 end do
683 end do
684 RandPointOnEllipsoid = RandPointOnEllipsoid + Mean
685 end function genRandPointOnEllipsoid
