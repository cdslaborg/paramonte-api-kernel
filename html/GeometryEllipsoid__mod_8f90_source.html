258 function constructEllipsoid(nd, np, Point, isBounding, failed) result(Ellipsoid)
259#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
260 !DEC$ ATTRIBUTES DLLEXPORT :: constructEllipsoid
261#endif
262 use Constants_mod, only: IK, RK ! LCOV_EXCL_LINE
263 use Matrix_mod, only: genInvMatFromChoLow
264 use MatrixCholesky_mod, only: getChoLow
265 implicit none
266 integer(IK) , intent(in) :: nd, np
267 real(RK) , intent(in) :: Point(nd,np)
268 logical , intent(in) :: isBounding
269 logical , intent(out) :: failed
270 type(Ellipsoid_type) :: Ellipsoid
271
272 real(RK) :: scaleFactorSqInverse
273 real(RK) :: NormedPoint(nd,np)
274 integer(IK) :: id, ip
275
276 ! Compute the mean.
277
278 Ellipsoid%Center = sum(Point, dim = 2) / np
279
280 do concurrent(ip = 1:np)
281 NormedPoint(1:nd,ip) = Point(1:nd,ip) - Ellipsoid%Center
282 end do
283
284 ! Compute the upper representative matrix of the cluster representative matrices.
285
286 allocate(Ellipsoid%ChoLowCovUpp(1:nd,1:nd), Ellipsoid%ChoDia(1:nd), source = 0._RK)
287 do ip = 1, np
288 do id = 1, nd
289 Ellipsoid%ChoLowCovUpp(1:id,id) = Ellipsoid%ChoLowCovUpp(1:id,id) + NormedPoint(1:id,ip) * NormedPoint(id,ip)
290 end do
291 end do
292 Ellipsoid%ChoLowCovUpp(1:nd,1:nd) = Ellipsoid%ChoLowCovUpp(1:nd,1:nd) / real(np - 1, RK)
293
294 ! Compute the Cholesky Factor of the cluster representative matrices.
295
296 call getChoLow(nd, Ellipsoid%ChoLowCovUpp(1:nd,1:nd), Ellipsoid%ChoDia(1:nd))
297 failed = Ellipsoid%ChoDia(1) < 0._RK
298 if (failed) return
299
300 ! Compute the inverse of the cluster representative matrices.
301
302 Ellipsoid%InvCovMat = genInvMatFromChoLow(nd, Ellipsoid%ChoLowCovUpp, Ellipsoid%ChoDia)
303
304 ! Compute the MahalSq of the points.
305
306 allocate(Ellipsoid%MahalSq(np))
307
308! if (isBounding) then
309
310 ! Compute the scaleFcator of the bounding region.
311
312 Ellipsoid%scaleFactorSq = NEGINF_RK
313 do ip = 1, np
314 Ellipsoid%MahalSq(ip) = dot_product( NormedPoint(1:nd,ip) , matmul(Ellipsoid%InvCovMat, NormedPoint(1:nd,ip)) )
315 if (Ellipsoid%scaleFactorSq < Ellipsoid%MahalSq(ip)) Ellipsoid%scaleFactorSq = Ellipsoid%MahalSq(ip)
316 end do
317
318 Ellipsoid%scaleFactor = sqrt(Ellipsoid%scaleFactorSq)
319 scaleFactorSqInverse = 1._RK / Ellipsoid%scaleFactorSq
320
321 Ellipsoid%ChoDia(1:nd) = Ellipsoid%ChoDia * Ellipsoid%scaleFactor
322 Ellipsoid%InvCovMat(1:nd,1:nd) = Ellipsoid%InvCovMat * scaleFactorSqInverse
323 do concurrent(id = 1:nd); Ellipsoid%ChoLowCovUpp(id+1:nd,id) = Ellipsoid%ChoLowCovUpp(id+1:nd,id) * Ellipsoid%scaleFactor; end do
324
325! else
326!
327! do concurrent(ip = 1:np)
328! Ellipsoid%MahalSq(ip) = dot_product( NormedPoint(1:nd,ip) , matmul(Ellipsoid%InvCovMat, NormedPoint(1:nd,ip)) )
329! end do
330!
331! Ellipsoid%scaleFactor = 1._RK
332! Ellipsoid%scaleFactorSq = 1._RK
333! Ellipsoid%logVolNormed = sum(log(Ellipsoid%ChoDia(1:nd)))
334!
335! end if
336
337 Ellipsoid%logVolNormed = sum(log(Ellipsoid%ChoDia(1:nd)))
338
339 end function constructEllipsoid

357 pure function genEffectiveSize(nd, np, Point, Center, InvCovMat) result(effectiveSize)
358#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
359 !DEC$ ATTRIBUTES DLLEXPORT :: genEffectiveSize
360#endif
361 use Constants_mod, only: IK, RK, RK32, NEGINF_RK
362 implicit none
363 integer(IK) , intent(in) :: nd, np
364 real(RK) , intent(in) :: Point(nd,np)
365 real(RK) , intent(in) :: Center(nd)
366 real(RK) , intent(in) :: InvCovMat(nd,nd)
367 integer(IK) :: effectiveSize
368 real(RK) :: NormedPoint(nd)
369 integer(IK) :: ip
370 effectiveSize = 0_IK
371 do ip = 1, np
372 NormedPoint(1:nd) = Point(1:nd,ip) - Center

374 end do
375 end function genEffectiveSize

379 ! see for example: http://math.stackexchange.com/questions/606184/volume-of-n-dimensional-ellipsoid
380 ! see for example: https://en.wikipedia.org/wiki/Volume_of_an_n-ball
381 ! see for example: https://en.wikipedia.org/wiki/Particular_values_of_the_Gamma_function
382 ! getEllVolCoef = PI^(nd/2) / gamma(nd/2+1) where n is a positive integer.
383 !
394 pure function getEllVolCoef(nd) result(ellVolCoef)
395#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
396 !DEC$ ATTRIBUTES DLLEXPORT :: getEllVolCoef
397#endif
398 use Constants_mod, only: IK, RK, PI
399 implicit none
400 integer(IK), intent(in) :: nd
401 integer(IK) :: i,k
402 real(RK), parameter :: FOUR_PI = PI * 4._RK
403 real(RK) :: ellVolCoef
404 if (mod(nd,2_IK)==0_IK) then ! nd is even
405
406 ellVolCoef = PI
407 do i = 2_IK, nd / 2_IK
408 ellVolCoef = ellVolCoef * PI / i ! nd = 2k ; ellVolCoef = PI^(k) / Factorial(k)
409 end do
410
411 else ! nd is an odd integer
412
413 ! nd = 2k-1 ; gamma(nd/2 + 1) = gamma(k + 1/2) ; gamma(k+1/2) = sqrt(PI) * (2k)! / (4^k * k!)
414 k = (nd + 1_IK) / 2_IK
415
416 ! This is to avoid an extra unnecessary division of ellVolCoef by PI
417 ellVolCoef = 4._RK / (k + 1_IK)
418
419 do i = k+2_IK, 2_IK*k
420 ! ellVolCoef = PI^(k-1/2) / gamma(k+1/2) = PI^(k+1) * 4^k * k! / (2k)!
421 ellVolCoef = ellVolCoef * FOUR_PI / i
422 end do
423
424 end if
425 end function getEllVolCoef

482! !> Return the logarithm of the volume of an `nd`-dimensional ball of unit-radius using the Fortran intrinsic `log_gamma()`.
483! !>
484! !> param[in] nd : The number of dimensions.
485! !>
486! !> \return
487! !> `logVolUnitBall` : The logarithm of the volume of an `nd`-dimensional ball of unit-radius.
488! !>
489! !> \brief
490! !> This routine is an exact replacement for [getLogEllVolCoef](@ref getLogEllVolCoef).
491! !> However, it is, on average, 10%-15% faster than [getLogEllVolCoef](@ref getLogEllVolCoef) based
492! !> on benchmarks with Intel ifort 19.4 with all optimization flags switched on.
493! !
494! ! see for example: http://math.stackexchange.com/questions/606184/volume-of-n-dimensional-ellipsoid
495! ! see for example: https://en.wikipedia.org/wiki/Volume_of_an_n-ball
496! ! see for example: https://en.wikipedia.org/wiki/Particular_values_of_the_Gamma_function
497! ! getEllVolCoef = PI^(nd/2) / gamma(nd/2+1) where n is just a positive integer
498! pure function genLogVolUnitBall(nd) result(logVolUnitBall)

500! !DEC$ ATTRIBUTES DLLEXPORT :: genLogVolUnitBall

503! use Constants_mod, only: IK, RK, PI
504! implicit none
505! integer(IK) , intent(in) :: nd
506! real(RK) , parameter :: LOG_PI = log(PI)
507! integer(IK) :: ndHalfInteger
508! real(RK) :: logVolUnitBall
509! real(RK) :: ndHalfReal
510! if (mod(nd,2_IK) == 0_IK) then ! nd is even
511! ndHalfInteger = nd / 2_IK
512! call getLogFactorial(ndHalfInteger,logVolUnitBall)
513! logVolUnitBall = ndHalfInteger*LOG_PI - logVolUnitBall ! nd = 2k ; logVolUnitBall = PI^(k) / Factorial(k)
514! else ! nd is an odd integer
515! ndHalfReal = 0.5_RK * real(nd,kind=RK)
516! logVolUnitBall = ndHalfReal * LOG_PI - log_gamma(ndHalfReal+1._RK)
517! end if

529 ! see for example: https://en.wikipedia.org/wiki/N-sphere#Volume_and_surface_area
530 pure function getLogSurfUnitBall(nd) result(logSurfUnitBall)
531#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
532 !DEC$ ATTRIBUTES DLLEXPORT :: getLogSurfUnitBall
533#endif
534 use Constants_mod, only: IK, RK, PI, LOG_TWO, NEGINF_RK
535 use SpecFuncGamma_mod, only: getLogFactorial
536 implicit none
537 integer(IK) , intent(in) :: nd
538 real(RK) , parameter :: LOG_PI = log(PI)
539 real(RK) :: logSurfUnitBall
540 real(RK) :: ndHalf_RK
541 integer(IK) :: ndHalf_IK
542 if (mod(nd,2_IK) == 0_IK) then ! nd is even
543 if (nd > 0_IK) then
544 ndHalf_IK = nd / 2_IK
545 call getLogFactorial(ndHalf_IK-1_IK,logSurfUnitBall)
546 logSurfUnitBall = LOG_TWO + ndHalf_IK * LOG_PI - logSurfUnitBall ! nd = 2k ; logSurfUnitBall = PI^(k) / Factorial(k)
547 else
548 logSurfUnitBall = NEGINF_RK
549 end if
550 else ! nd is an odd integer
551 ndHalf_RK = 0.5_RK * real(nd,kind=RK)
552 logSurfUnitBall = LOG_TWO + ndHalf_RK * LOG_PI - log_gamma(ndHalf_RK)
553 end if
554 end function getLogSurfUnitBall

640 function getLogVolChoLowEllipsoidFromRepMatUpp(ChoLowRepMatUpp,ChoDia,failed,logVolUnitBall) result(logVolEllipsoid)
641#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
642 !DEC$ ATTRIBUTES DLLEXPORT :: getLogVolChoLowEllipsoidFromRepMatUpp
643#endif
644 use MatrixCholesky_mod, only: getChoLow
645 use Matrix_mod, only: genUpperFromUpper
646 use Constants_mod, only: IK, RK
647 implicit none
648 real(RK) , intent(inout) , contiguous :: ChoLowRepMatUpp(:,:)
649 real(RK) , intent(out) , optional :: ChoDia(size(ChoLowRepMatUpp,dim=1))
650 logical , intent(out) , optional :: failed
651 real(RK) , intent(in) , optional :: logVolUnitBall
652 real(RK) :: logVolEllipsoid
653 real(RK) :: logVolUnitBallDef
654 real(RK) , allocatable :: ChoDiaDef(:), ChoLowRepMatUppDef(:,:)
655 integer(IK) :: nd
656 nd = size(ChoLowRepMatUpp,dim=1,kind=IK)
657 if (present(logVolUnitBall)) then
658 logVolUnitBallDef = logVolUnitBall
659 else
660 logVolUnitBallDef = genLogVolUnitBall(nd)
661 end if
662 if (present(ChoDia)) then
663 call getChoLow(nd,ChoLowRepMatUpp,ChoDia)
664 if (ChoDia(1) > 0._RK) then
665 logVolEllipsoid = genLogVolUnitBall(nd) + sum(log(ChoDia))
666 return
667 end if
668 else
669 allocate(ChoDiaDef(nd))
670 ChoLowRepMatUppDef = genUpperFromUpper(nd, ChoLowRepMatUpp, diagEnabled = .true.)
671 call getChoLow(nd,ChoLowRepMatUppDef,ChoDiaDef)
672 if (ChoDiaDef(1) > 0._RK) then
673 logVolEllipsoid = genLogVolUnitBall(nd) + sum(log(ChoDiaDef))
674 deallocate(ChoDiaDef,ChoLowRepMatUppDef)
675 return
676 end if
677 deallocate(ChoDiaDef)
678 end if
679 if (present(failed)) then
680 failed = .true.
681 return
682 else
683 write(*,"(A)") MODULE_NAME//"@getLogVolChoLowEllipsoidFromRepMatUpp(): Cholesky factorization failed."
684 error stop
685 end if
686 end function getLogVolChoLowEllipsoidFromRepMatUpp

753#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
754 !DEC$ ATTRIBUTES DLLEXPORT :: genRandPointOnEllipsoid
755#endif

757 implicit none
758 integer(IK), intent(in) :: nd
759 real(RK) , intent(in) :: Mean(nd)
760 real(RK) , intent(in) :: ChoLow(nd,nd) ! Cholesky lower triangle, calculated from the MVN CovMat.
761 real(RK) , intent(in) :: ChoDia(nd) ! Cholesky diagonal terms, calculated from the MVN CovMat.
762 real(RK) :: RandPointOnEllipsoid(nd), DummyVec(nd), sumSqDummyVec
763 integer(IK) :: i,j
764 sumSqDummyVec = 0._RK
765 do j=1,nd
766 call getNormRand(DummyVec(j))
767 sumSqDummyVec = sumSqDummyVec + DummyVec(j)**2
768 end do
769 DummyVec = DummyVec / sqrt(sumSqDummyVec) ! DummyVec is a random point on the surface of nd-sphere.
770 RandPointOnEllipsoid = 0._RK
771 do j = 1,nd
772 RandPointOnEllipsoid(j) = RandPointOnEllipsoid(j) + ChoDia(j) * DummyVec(j)
773 do i = j+1,nd
774 RandPointOnEllipsoid(i) = RandPointOnEllipsoid(i) + ChoLow(i,j) * DummyVec(j)
775 end do
776 end do
777 RandPointOnEllipsoid = RandPointOnEllipsoid + Mean
778 end function genRandPointOnEllipsoid
