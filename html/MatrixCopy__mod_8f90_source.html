343 pure function genLowerFromUpper(nd,Upper,diagEnabled) result(Lower)
344#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
345 !DEC$ ATTRIBUTES DLLEXPORT :: genLowerFromUpper
346#endif
347 use Constants_mod, only: RK, IK
348 implicit none
349 integer(IK) , intent(in) :: nd
350 real(RK) , intent(in) :: Upper(nd,nd)
351 logical , intent(in), optional :: diagEnabled
352 real(RK) :: Lower(nd,nd)
353 logical :: diagEnabledDef
354 integer(IK) :: i
355 if (present(diagEnabled)) then
356 diagEnabledDef = diagEnabled
357 else
358 diagEnabledDef = .false.
359 end if
360 if (diagEnabledDef) then
361 do concurrent(i = 1:nd)
362 Lower(i:nd,i) = Upper(i,i:nd)
363 end do
364 else
365 do concurrent(i = 1:nd-1)
366 Lower(i+1:nd,i) = Upper(i,i+1:nd)
367 end do
368 end if
369 end function genLowerFromUpper
