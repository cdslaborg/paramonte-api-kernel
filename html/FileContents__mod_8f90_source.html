104 subroutine getFileContents(path, Contents, numRecord, Err, delEnabled)
105#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
106 !DEC$ ATTRIBUTES DLLEXPORT :: getFileContents
107#endif
108 use String_mod, only: Str_type
109 use String_mod, only: val2char
110 use Err_mod, only: Err_type
111 implicit none
112 character(len=*), intent(in) :: path
113 type(Str_type), allocatable, intent(out) :: Contents(:)
114 logical, intent(in), optional :: delEnabled
115 type(Err_type), intent(out) :: Err
116 integer, intent(out) :: numRecord
117
118 character(99999) :: record
119 integer :: fileUnit, irecord, i
120 logical :: delEnabledDefault, isOpen, fileExists
121
122 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@getFileContents()"
123
124 Err%occurred = .false.
125 Err%msg = ""
126
127 delEnabledDefault = .false.
128 if (present(delEnabled)) then
129 delEnabledDefault = delEnabled
130 end if
131
132 call getNumRecordInFile(path,numRecord,Err)
133 if (Err%occurred) then
134 ! LCOV_EXCL_START
135 Err%msg = PROCEDURE_NAME // Err%msg
136 return
137 end if
138 ! LCOV_EXCL_STOP
139
140 allocate(Contents(numRecord))
141
142 open( newunit = fileUnit & ! LCOV_EXCL_LINE
143 , status = "old" & ! LCOV_EXCL_LINE
144 , file = path & ! LCOV_EXCL_LINE
145#if INTEL_COMPILER_ENABLED && WINDOWS_ENABLED
146 , SHARED & ! LCOV_EXCL_LINE
147#endif
148 )
149 do irecord = 1,numRecord
150 read(fileUnit,"(A)", iostat=Err%stat) record
151 if (Err%stat==0) then
152 Contents(irecord)%value = trim(adjustl(record))
153 ! LCOV_EXCL_START
154 elseif (is_iostat_end(Err%stat)) then
155 Err%occurred = .true.
156 Err%msg = PROCEDURE_NAME // ": End-of-file error occurred while expecting " // &
157 val2char(numRecord-irecord+1) // " records in file='" // path // "'."
158 return
159 elseif (is_iostat_eor(Err%stat)) then
160 Err%occurred = .true.
161 Err%msg = PROCEDURE_NAME // ": End-of-record error occurred while reading line number " // &
162 val2char(irecord) // " from file='" // path // "'."
163 return
164 else
165 Err%occurred = .true.
166 Err%msg = PROCEDURE_NAME // ": Unknown error occurred while reading line number " // &
167 val2char(irecord) // " from file='" // path // "'."
168 return
169 end if
170 ! LCOV_EXCL_STOP
171 end do
172
173 ! attempt to delete the file repeatedly. This is important on windows systems as the file often remains locked.
174
175 blockDeleteFile: if (delEnabledDefault) then
176
177 ! Attempt to delete the file repeatedly
178
179 loopDeleteFile: do i = 1, 100
180 inquire(file=path, opened=isOpen, exist=fileExists, iostat = Err%stat)
181 if (Err%stat==0) then
182 if (fileExists) then
183 if (.not. isOpen) then
184 open( newunit = fileUnit & ! LCOV_EXCL_LINE
185 , iostat = Err%stat & ! LCOV_EXCL_LINE
186 , status = "replace" & ! LCOV_EXCL_LINE
187 , file = path & ! LCOV_EXCL_LINE
188#if INTEL_COMPILER_ENABLED && WINDOWS_ENABLED
189 , SHARED & ! LCOV_EXCL_LINE
190#endif
191 )
192 end if
193 close(fileUnit, status="delete", iostat = Err%stat) ! parallel processes cannot delete the same file. Err%stat is required to handle exceptions.
194 else
195 return
196 end if
197 end if
198 end do loopDeleteFile
199
200 end if blockDeleteFile
201
202 ! If file deletion fails or if it should not be deleted, then simply close the file and return.
203
204 close(fileUnit, status = "keep", iostat = Err%stat) ! parallel processes cannot delete the same file. Err%stat is required to handle exceptions.
205
206 !if (Err%stat>0) then
207 !! LCOV_EXCL_START
208 ! Err%occurred = .true.
209 ! Err%msg = PROCEDURE_NAME // "Error occurred while attempting to close or delete the open file='" // path // "'."
210 ! return
211 !end if
212 !! LCOV_EXCL_STOP
213
214 end subroutine getFileContents

225 subroutine getNumRecordInFile(filePath,numRecord,Err,exclude)
226#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
227 !DEC$ ATTRIBUTES DLLEXPORT :: getNumRecordInFile
228#endif
229 use Constants_mod, only: IK
230 implicit none
231 character(len=*), intent(in) :: filePath
232 integer(IK) , intent(out) :: numRecord
233 type(Err_type) , intent(out) :: Err
234 character(*) , intent(in) , optional :: exclude
235 character(len=1) :: record
236 integer :: fileUnit
237 logical :: fileExists, fileIsOpen, excludeIsPresent
238 integer :: iostat
239
240 character(*), parameter :: PROCEDURE_NAME = "@getNumRecordInFile()"
241
242 Err%occurred = .false.
243 Err%msg = ""
244 excludeIsPresent = present(exclude)
245
246 ! Check if file exists
247 ! GFortran 7.3 bug: If file is not open, compiler assumes internal file if `number` is specified, causing runtime error.
248
249 inquire( file=filePath, exist=fileExists, opened=fileIsOpen, number=fileUnit, iostat=Err%stat )
250 if (Err%stat/=0) then
251 ! LCOV_EXCL_START
252 Err%occurred = .true.
253 Err%msg = PROCEDURE_NAME // ": Error occurred while inquiring the status of file='" // filePath // "'."
254 return
255 end if
256 ! LCOV_EXCL_STOP
257
258 if (.not.fileExists) then
259 ! LCOV_EXCL_START
260 Err%occurred = .true.
261 Err%msg = PROCEDURE_NAME // ": The input file='" // filePath // "' does not exist."
262 return
263 end if
264 ! LCOV_EXCL_STOP
265
266 if (fileIsOpen) close(unit=fileUnit,iostat=Err%stat)
267 if (Err%stat>0) then
268 ! LCOV_EXCL_START
269 Err%occurred = .true.
270 Err%msg = PROCEDURE_NAME // ": Error occurred while attempting to close the open input file='" // filePath // "'."
271 return
272 end if
273 ! LCOV_EXCL_STOP
274
275 open( newunit = fileUnit & ! LCOV_EXCL_LINE
276 , iostat = Err%stat & ! LCOV_EXCL_LINE
277 , file = filePath & ! LCOV_EXCL_LINE
278 , status = "old" & ! LCOV_EXCL_LINE
279#if INTEL_COMPILER_ENABLED && WINDOWS_ENABLED
280 , SHARED & ! LCOV_EXCL_LINE
281#endif
282 )
283 if (Err%stat>0) then
284 ! LCOV_EXCL_START
285 Err%occurred = .true.
286 Err%msg = PROCEDURE_NAME // ": Error occurred while opening input file='" // filePath // "'."
287 return
288 end if
289 ! LCOV_EXCL_STOP
290
291 numRecord = 0_IK
292 do
293 read(fileUnit,'(A)',iostat=iostat) record
294 if(iostat==0) then
295 if (excludeIsPresent) then
296 if (trim(adjustl(record))/=exclude) numRecord = numRecord + 1_IK
297 else
298 numRecord = numRecord + 1_IK
299 end if
300 cycle
301 elseif(is_iostat_end(iostat) .or. is_iostat_eor(iostat)) then
302 exit
303 ! LCOV_EXCL_START
304 else
305 Err%occurred = .true.
306 Err%stat = iostat
307 Err%msg = PROCEDURE_NAME // ": Error occurred while reading input file='" // filePath // "'."
308 return
309 end if
310 ! LCOV_EXCL_STOP
311 end do
312
313 close(fileUnit,iostat=Err%stat)
314 if (Err%stat>0) then
315 ! LCOV_EXCL_START
316 Err%occurred = .true.
317 Err%msg = PROCEDURE_NAME // ": Error occurred while attempting to close the open input file='" // &
318 filePath // "' after counting the number of records in file."
319 return
320 end if
321 ! LCOV_EXCL_STOP
322
323 end subroutine getNumRecordInFile
