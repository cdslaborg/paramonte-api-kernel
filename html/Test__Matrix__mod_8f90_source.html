291 function test_sortPosDefMat_1() result(assertion)
292
293 use Constants_mod, only: RK, IK
294
295 implicit none
296 logical :: assertion
297 integer(IK), parameter :: ColIndx(*) = [4]
298 integer(IK), parameter :: ColIndxMap(*) = [5]
299 integer(IK) :: rank
300 real(RK), allocatable :: PosDefMat(:,:), OutPosDefMat(:,:), OutPosDefMat_ref(:,:)
301 integer(IK) :: i, j
302
303 assertion = .true.
304
305 rank = 5
306 allocate(PosDefMat(rank,rank))
307 do j = 1,rank
308 do i = 1,j
309 PosDefMat(i,j) = i*10 + j
310 end do
311 end do
312
313 ! switch the variable 4 with 5, such that the output remains a positive-definite matrix.
314
315 OutPosDefMat = genSortedPosDefMat(rank, PosDefMat, 1_IK, ColIndx, ColIndxMap)
316
317 OutPosDefMat_ref = reshape( [ 11._RK, 12._RK, 13._RK, 15._RK, 14._RK &
318 , 0._RK, 22._RK, 23._RK, 25._RK, 24._RK &
319 , 0._RK, 0._RK, 33._RK, 35._RK, 34._RK &
320 , 0._RK, 0._RK, 0._RK, 55._RK, 45._RK &
321 , 0._RK, 0._RK, 0._RK, 0._RK, 44._RK ] &
322 , shape = [rank,rank] )
323 OutPosDefMat_ref = transpose(OutPosDefMat_ref)
324 do j = 1, rank
325 do i = 1, j
326 assertion = assertion .and. OutPosDefMat_ref(i,j) == OutPosDefMat(i,j)
327 end do
328 end do
329
330 if (Test%isVerboseMode .and. .not. assertion) then
331 ! LCOV_EXCL_START
332
333 write(Test%outputUnit,"(*(g0))")
334 write(Test%outputUnit,"(*(g0))") "OutPosDefMat_ref:"
335 do i = 1,rank
336 write(Test%outputUnit,"(*(F7.1))") (OutPosDefMat_ref(i,j),j=1,rank)
337 end do
338 write(Test%outputUnit,"(*(g0))")
339
340 write(Test%outputUnit,"(*(g0))")
341 write(Test%outputUnit,"(*(g0))") "Output Positive-Definite Matrix with variables 4 and 5 swapped:"
342 do i = 1,rank
343 write(Test%outputUnit,"(*(F7.1))") (OutPosDefMat(i,j),j=1,rank)
344 end do
345 write(Test%outputUnit,"(*(g0))")
346
347 write(Test%outputUnit,"(*(g0))")
348 write(Test%outputUnit,"(*(g0))") "Original Matrix:"
349 do i = 1,rank
350 write(Test%outputUnit,"(*(F7.1))") (PosDefMat(i,j),j=1,rank)
351 end do
352
353 end if
354 ! LCOV_EXCL_STOP
355
356 end function test_sortPosDefMat_1
