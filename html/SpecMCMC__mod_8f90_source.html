182 subroutine setFromInputArgs( SpecMCMC &
183 ! ParaMCMC variables
184 , outputChainSize &
185 , proposal &
186 , proposalStartPoint &
187 , proposalScaleFactor &
188 , proposalStartStdVec &
189 , proposalStartCorMat &
190 , proposalStartCovMat &
191 , outputSampleRefinementCount &
192 , outputSampleRefinementMethod &
193 , proposalStartPointRandomized &
194 , proposalStartPointDomainCubeLimitLower &
195 , proposalStartPointDomainCubeLimitUpper &
196 )
197#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
198 !DEC$ ATTRIBUTES DLLEXPORT :: setFromInputArgs
199#endif
200
201 use SpecBase_mod, only: SpecBase_type
202 use Constants_mod, only: IK, RK
203 implicit none
204 class(SpecMCMC_type), intent(inout) :: SpecMCMC
205
206 ! ParaMCMC variables
207 integer , intent(in), optional :: outputChainSize
208 character(*), intent(in), optional :: proposalScaleFactor
209 real(RK) , intent(in), optional :: proposalStartPoint(:)
210 character(*), intent(in), optional :: proposal
211 real(RK) , intent(in), optional :: proposalStartStdVec(:)
212 real(RK) , intent(in), optional :: proposalStartCorMat(:,:)
213 real(RK) , intent(in), optional :: proposalStartCovMat(:,:)
214 integer , intent(in), optional :: outputSampleRefinementCount
215 character(*), intent(in), optional :: outputSampleRefinementMethod
216 logical , intent(in), optional :: proposalStartPointRandomized
217 real(RK) , intent(in), optional :: proposalStartPointDomainCubeLimitLower(:)
218 real(RK) , intent(in), optional :: proposalStartPointDomainCubeLimitUpper(:)
219
220 logical :: proposalStartStdVecIsPresent
221 logical :: proposalStartCorMatIsPresent
222 logical :: proposalStartCovMatIsPresent
223
224 proposalStartStdVecIsPresent = present(proposalStartStdVec)
225 proposalStartCorMatIsPresent = present(proposalStartCorMat)
226 proposalStartCovMatIsPresent = present(proposalStartCovMat) .or. proposalStartCorMatIsPresent .or. proposalStartStdVecIsPresent
227
228 if (present(outputChainSize)) call SpecMCMC%OutputChainSize %set(outputChainSize)
229 if (present(proposal)) call SpecMCMC%Proposal %set(trim(adjustl(proposal)))
230 if (present(proposalScaleFactor)) call SpecMCMC%ProposalScaleFactor %set(proposalScaleFactor)
231 if (proposalStartStdVecIsPresent) call SpecMCMC%proposalStartStdVec %set(proposalStartStdVec)
232 if (proposalStartCorMatIsPresent) call SpecMCMC%proposalStartCorMat %set(proposalStartCorMat)
233 if (proposalStartCovMatIsPresent) call SpecMCMC%proposalStartCovMat %set(SpecMCMC%proposalStartStdVec%val, SpecMCMC%proposalStartCorMat%val, proposalStartCovMat)
234 if (present(outputSampleRefinementCount)) call SpecMCMC%OutputSampleRefinementCount %set(outputSampleRefinementCount)
235 if (present(outputSampleRefinementMethod)) call SpecMCMC%OutputSampleRefinementMethod %set(outputSampleRefinementMethod)
236 if (present(proposalStartPointRandomized)) call SpecMCMC%ProposalStartPointRandomized %set(proposalStartPointRandomized)
237
238 call SpecMCMC%ProposalStartPointDomainCubeLimitLower%set(proposalStartPointDomainCubeLimitLower)
239 call SpecMCMC%ProposalStartPointDomainCubeLimitUpper%set(proposalStartPointDomainCubeLimitUpper)
240 call SpecMCMC%proposalStartPoint %set(proposalStartPoint)
241
242 end subroutine setFromInputArgs
