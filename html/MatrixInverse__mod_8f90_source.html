146 PURE subroutine getInvPosDefMatSqrtDet(nd,MatInvMat,sqrtDetInvPosDefMat)
147#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
148 !DEC$ ATTRIBUTES DLLEXPORT :: getInvPosDefMatSqrtDet
149#endif
150 use MatrixCholesky_mod, only: getChoLow
151 use Constants_mod, only: RK, IK ! LCOV_EXCL_LINE
152 implicit none
153 integer(IK), intent(in) :: nd
154 real(RK) , intent(inout) :: MatInvMat(nd,nd) ! input: upper half is covariance matrix, output: inverse matrix
155 real(RK) , intent(out) :: sqrtDetInvPosDefMat !
156 real(RK) :: ChoLow(nd,nd) ! The lower triangle of the Cholesky factorization.
157 real(RK) :: ChoDia(nd) ! The diagonal elements of the inverse matrix
158 real(RK) :: summ
159 integer(IK) :: i,j,k
160 if (nd==1_IK) then
161 MatInvMat(1,1) = 1._RK / MatInvMat(1,1)
162 sqrtDetInvPosDefMat = MatInvMat(1,1)
163 return
164 end if
165 do j=1,nd
166 do i=1,j
167 ChoLow(i,j) = MatInvMat(i,j)
168 end do
169 end do
170 call getChoLow(nd,ChoLow,ChoDia)
171 if (ChoDia(1)<0._RK) then
172 sqrtDetInvPosDefMat = -1._RK
173 return
174 end if
175 sqrtDetInvPosDefMat = 1._RK / product(ChoDia)
176 do i = 1,nd
177 ChoLow(i,i) = 1._RK / ChoDia(i)
178 do j = i+1,nd
179 summ = 0._RK
180 do k = i,j-1
181 summ = summ - ChoLow(j,k) * ChoLow(k,i)
182 end do
183 ChoLow(j,i) = summ / ChoDia(j)
184 end do
185 end do
186 do i = 1,nd
187 do j = i,nd
188 MatInvMat(j,i) = dot_product(ChoLow(j:nd,j),ChoLow(j:nd,i))
189 end do
190 MatInvMat(i,i:nd) = MatInvMat(i:nd,i)
191 end do
192 end subroutine getInvPosDefMatSqrtDet

423#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
424 !DEC$ ATTRIBUTES DLLEXPORT :: getInvMatDet
425#endif
426 use Constants_mod, only: RK, IK
427 implicit none
428 integer(IK), intent(in) :: nd
429 real(RK) , intent(inout) :: MatrixLUP(nd,nd) ! On input it is the matrix, on output it is the LU decomposition.
430 real(RK) , intent(out) :: InverseMatrix(nd,nd)
431 real(RK) , intent(out) :: detInvMat
432 logical , intent(out) :: failed
433 integer(IK) :: i,j,RowPerm(nd) ! LCOV_EXCL_LINE

435 InverseMatrix(1,1) = 1._RK / MatrixLUP(1,1)

439 do i = 1,nd

442 end do

444 end do

446 if (failed) return
447 do j = 1,nd
448 detInvMat = detInvMat * MatrixLUP(j,j)
449 call solveLinSysLUP(InverseMatrix(1:nd,j), MatrixLUP, RowPerm)
450 end do
451 detInvMat = 1._RK/detInvMat

528 pure function genInvMatLow(nd,MatLow) result(InvMatLow)
529#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
530 !DEC$ ATTRIBUTES DLLEXPORT :: genInvMatLow
531#endif
532 use Constants_mod, only: RK, IK
533 implicit none
534 integer(IK), intent(in) :: nd
535 real(RK) , intent(in) :: MatLow(nd,nd)
536 real(RK) :: InvMatLow(nd,nd)
537 real(RK) :: summ
538 integer(IK) :: i, j, k
539 !InvMatLow = 0._RK
540 do i = 1, nd
541 InvMatLow(i,i) = 1._RK / MatLow(i,i);
542 do j = 1, i - 1
543 summ = 0._RK
544 do k = j, i - 1
545 summ = summ + MatLow(i,k) * InvMatLow(k,j);
546 end do
547 InvMatLow(i,j) = -summ * InvMatLow(i,i);
548 end do
549 end do
550 end function genInvMatLow
