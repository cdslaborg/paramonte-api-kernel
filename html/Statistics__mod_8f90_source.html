144 function getLogProbMixNormSP_RK(nmode,LogAmplitude,Mean,InvCovMat,LogSqrtDetInvCovMat,point) result(logProbMixNorm)
145#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
146 !DEC$ ATTRIBUTES DLLEXPORT :: getLogProbMixNormSP_RK
147#endif
148 use Constants_mod, only: IK, RK, LOGTINY_RK
149 implicit none
150 integer(IK), intent(in) :: nmode
151 real(RK) , intent(in) :: LogAmplitude(nmode),Mean(nmode)
152 real(RK) , intent(in) :: InvCovMat(nmode),LogSqrtDetInvCovMat(nmode)
153 real(RK) , intent(in) :: point
154 real(RK) :: logProbMixNorm
155 real(RK) :: normFac,LogProb(nmode)
156 integer(IK) :: imode
157 do imode = 1, nmode
158 LogProb(imode) = LogAmplitude(imode) + getLogProbNormSP_RK(Mean(imode),InvCovMat(imode),LogSqrtDetInvCovMat(imode),point)
159 end do
160 normFac = maxval(LogProb)
161 LogProb = LogProb - normFac
162 where(LogProb<LOGTINY_RK)
163 LogProb = 0._RK
164 elsewhere
165 LogProb = exp(LogProb)
166 end where
167 logProbMixNorm = normFac + log(sum(LogProb))
168 end function getLogProbMixNormSP_RK

203#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
204 !DEC$ ATTRIBUTES DLLEXPORT :: getLogProbMixNormMP_RK
205#endif
206 use Constants_mod, only: IK, RK, LOGTINY_RK
207 implicit none
208 integer(IK), intent(in) :: nmode,np
209 real(RK) , intent(in) :: LogAmplitude(nmode),Mean(nmode)
210 real(RK) , intent(in) :: InvCovMat(nmode),LogSqrtDetInvCovMat(nmode)
211 real(RK) , intent(in) :: Point(np)
212 real(RK) :: logProbMixNorm(np)
213 real(RK) :: NormFac(np),LogProb(nmode,np)
214 integer(IK) :: imode, ip

217 end do
218 NormFac = maxval(LogProb,dim=1)
219 do ip = 1,np
220 LogProb(1:nmode,ip) = LogProb(1:nmode,ip) - NormFac(ip)
221 do imode = 1,nmode
222 if ( LogProb(imode,ip) < LOGTINY_RK ) then
223 LogProb(imode,ip) = 0._RK
224 else
225 LogProb(imode,ip) = exp( LogProb(imode,ip) )
226 end if
227 end do
228 logProbMixNorm(ip) = NormFac(ip) + log(sum(LogProb(1:nmode,ip)))
229 end do
230 end function getLogProbMixNormMP_RK
