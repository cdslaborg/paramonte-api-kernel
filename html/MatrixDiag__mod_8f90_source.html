733 pure function genEyeMat(n,m,diag,offdiag) result(eye)
734#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
735 !DEC$ ATTRIBUTES DLLEXPORT :: genEyeMat
736#endif
737 use Constants_mod, only: RK, IK
738 implicit none
739 integer(IK), intent(in) :: n, m
740 real(RK) , intent(in), optional :: diag
741 real(RK) , intent(in), optional :: offdiag
742 real(RK) :: eye(n,m)
743 real(RK) :: diagonal
744 real(RK) :: offdiagonal
745 integer(IK) :: i, j
746 if (present(diag)) then
747 diagonal = diag
748 else
749 diagonal = 1._RK
750 end if
751 if (present(offdiag)) then
752 offdiagonal = offdiag
753 else
754 offdiagonal = 0._RK
755 end if
756 do concurrent(i=1:n, j=1:m)
757 if (i==j) then
758 eye(i,j) = diagonal
759 else
760 eye(i,j) = offdiagonal
761 end if
762 end do
763 end function genEyeMat
