496 PURE function getVariance_1D(np,mean,Sample,Weight,sumWeight) result(variance)
497#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
498 !DEC$ ATTRIBUTES DLLEXPORT :: getVariance_1D
499#endif
500 implicit none
501 integer(IK), intent(in) :: np ! np is the number of observations (points) whose variance is to be computed
502 real(RK) , intent(in) :: mean ! the mean of the sample.
503 real(RK) , intent(in) :: Sample(np) ! dataset.
504 integer(IK), intent(in), optional :: Weight(np), sumWeight ! sample weight
505 real(RK) :: variance ! output mean vector
506 integer(IK) :: ip
507 variance = 0._RK
508 if (present(Weight)) then
509 do ip = 1,np
510 variance = variance + Weight(ip) * ( Sample(ip) - mean )**2
511 end do
512 variance = variance / real(sumWeight-1_IK,kind=RK)
513 else
514 do ip = 1,np
515 variance = variance + ( Sample(ip) - mean )**2
516 end do
517 variance = variance / real(np-1_IK,kind=RK)
518 end if
519 end function getVariance_1D

780 pure subroutine getChoLowCovUpp(nd,np,Mean,Sample,ChoLowCovUpp,ChoDia)
781#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
782 !DEC$ ATTRIBUTES DLLEXPORT :: getChoLowCovUpp
783#endif
784 use Matrix_mod, only: getChoLow
785 implicit none
786 integer(IK), intent(in) :: nd,np ! np is the number of observations, nd is the number of parameters for each observation
787 real(RK) , intent(in) :: Mean(nd) ! The mean of the sample.
788 real(RK) , intent(in) :: Sample(nd,np) ! Sample is the data matrix, CovMat contains the elements of the sample covariance matrix
789 real(RK) , intent(out) :: ChoLowCovUpp(nd,nd) ! The lower Cholesky factor and the upper covariance matrix.
790 real(RK) , intent(out) :: ChoDia(nd) ! The diagonal elements of the Cholesky lower triangle.
791 real(RK) :: SampleShifted(np,nd), npMinusOneInverse
792 integer(IK) :: i,j
793
794 do i = 1,np
795 SampleShifted(i,1:nd) = Sample(1:nd,i) - Mean
796 end do
797
798 ! Only upper half of CovMat is needed
799 npMinusOneInverse = 1._RK / real(np-1,kind=RK)
800 do j = 1,nd
801 do i = 1,j
802 ! Get the covariance matrix elements: only the upper half of CovMat is needed
803 ChoLowCovUpp(i,j) = dot_product( SampleShifted(1:np,i) , SampleShifted(1:np,j) ) * npMinusOneInverse
804 end do
805 end do
806
807 ! @todo: The efficiency can be improved by merging it with the above loops
808 call getChoLow(nd, ChoLowCovUpp, ChoDia)
809
810 end subroutine getChoLowCovUpp

833 subroutine getChoLowCovUppHighDim(nd,np,Mean,Sample,ChoLowCovUpp,ChoDia)
834#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
835 !DEC$ ATTRIBUTES DLLEXPORT :: getChoLowCovUppHighDim
836#endif

838 implicit none
839 integer(IK), intent(in) :: nd,np ! np is the number of observations, nd is the number of parameters for each observation
840 real(RK) , intent(in) :: Mean(nd) ! Mean vector
841 real(RK) , intent(in) :: Sample(nd,np) ! Sample is the matrix of the data, CovMat contains the elements of the sample covariance matrix
842 real(RK) , intent(out) :: ChoLowCovUpp(nd,nd) ! The lower Cholesky factor of the covariance matrix
843 real(RK) , intent(out) :: ChoDia(nd) ! The diagonal elements of the Cholesky factorization
844 real(RK) :: SampleShifted(nd)

852 SampleShifted(j) = Sample(j,ip) - Mean(j)
853 ChoLowCovUpp(1:j,j) = ChoLowCovUpp(1:j,j) + SampleShifted(1:j) * SampleShifted(j)
854 end do
855 end do

860 call getChoLow(nd, ChoLowCovUpp, ChoDia)

894 subroutine getMeanCov(np,nd,Sample,CovMat,Mean,MahalSq,InvCovMat,sqrtDetInvCovMat)
895#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
896 !DEC$ ATTRIBUTES DLLEXPORT :: getMeanCov
897#endif
898 use Matrix_mod, only: getInvPosDefMatSqrtDet
899 implicit none
900 integer(IK), intent(in) :: np,nd ! np is the number of observations, nd is the number of parameters for each observation
901 real(RK) , intent(in) :: Sample(np,nd) ! Sample is the matrix of the data, CovMat contains the elements of the sample covariance matrix
902 real(RK) , intent(out) :: CovMat(nd,nd) ! Covariance matrix of the input data
903 real(RK) , intent(out) :: Mean(nd) ! Mean vector
904 real(RK) , intent(out), optional :: MahalSq(np) ! Vector of Mahalanobis Distances Squared, with respect to the mean position of the sample
905 real(RK) , intent(out), optional :: InvCovMat(nd,nd) ! Inverse Covariance matrix of the input data
906 real(RK) , intent(out), optional :: sqrtDetInvCovMat ! sqrt determinant of the inverse covariance matrix
907 real(RK) :: SampleShifted(np,nd)
908 real(RK) :: DummyVec(nd)
909 real(RK) :: npMinusOneInverse
910 integer(IK) :: i,j
911
912 ! Compute the mean and shift data.
913
914 do j = 1,nd
915 Mean(j) = sum(Sample(1:np,j)) / real(np,kind=RK)
916 SampleShifted(1:np,j) = Sample(1:np,j) - Mean(j)
917 end do
918
919 npMinusOneInverse = 1._RK / real(np-1,RK)
920 do i = 1,nd
921 do j = 1,nd
922 CovMat(i,j) = dot_product(SampleShifted(1:np,i),SampleShifted(1:np,j)) * npMinusOneInverse
923 end do
924 end do
925
926 ! Calculate InCovMat, MahalSq, and sqrt Determinant of InCovMat
927
928 if (present(sqrtDetInvCovMat)) then
929 do j = 1,nd
930 do i = 1,j
931 InvCovMat(i,j) = CovMat(i,j) ! Only the upper half of CovMat is needed
932 end do
933 end do

1181 ! This subroutine uses a recursion equation similar to http://stats.stackexchange.com/questions/97642/how-to-combine-sample-means-and-sample-variances
1182 ! See also: O’Neill, (2014), "Some Useful Moment Results in Sampling Problems".
1183 ! See also: https://en.wikipedia.org/wiki/Algorithms_for_calculating_variance#Covariance
1184 ! See also: https://stats.stackexchange.com/questions/43159/how-to-calculate-pooled-variance-of-two-groups-given-known-group-variances-mean
1185 subroutine mergeMeanCov(nd,npA,MeanA,CovMatA,npB,MeanB,CovMatB,MeanAB,CovMatAB)
1186#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
1187 !DEC$ ATTRIBUTES DLLEXPORT :: mergeMeanCov
1188#endif
1189 implicit none
1190 integer(IK), intent(in) :: nd
1191 integer(IK), intent(in) :: npA,npB
1192 real(RK) , intent(in) :: MeanA(nd),CovMatA(nd,nd)
1193 real(RK) , intent(in) :: MeanB(nd),CovMatB(nd,nd)
1194 real(RK) , intent(out) :: MeanAB(nd),CovMatAB(nd,nd)
1195 real(RK) , dimension(nd,1) :: MeanMatA,MeanMatB,MeanMat
1196 real(RK) :: DistanceSq(nd,nd)
1197 real(RK) :: npABinverse
1198 integer(IK) :: npAB
1199
1200 npAB = npA + npB
1201 npABinverse = 1._RK / real(npAB, kind=RK)
1202 MeanMatA(1:nd,1) = MeanA
1203 MeanMatB(1:nd,1) = MeanB
1204
1205 ! Compute the new Mean
1206
1207 MeanAB = ( npA * MeanA + npB * MeanB ) * npABinverse
1208 MeanMat(1:nd,1) = MeanAB
1209
1210 ! Compute the new Covariance matrix
1211
1212 DistanceSq = matmul( (MeanMatA-MeanMatB), transpose((MeanMatA-MeanMatB)) ) * npA * npB * npABinverse
1213 CovMatAB = ( (npA-1) * CovMatA + (npB-1) * CovMatB + DistanceSq ) / real(npAB-1, kind=RK)
1214
1215 end subroutine mergeMeanCov
