108 function genInvCovMat(ShiftedSample,Weight,failed) result(InvCovMat)
109#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
110 !DEC$ ATTRIBUTES DLLEXPORT :: genInvCovMat
111#endif
112 use Constants_mod, only: IK, RK
113 use Matrix_mod, only: genInvMatFromChoLow
114 use MatrixCholesky_mod, only: getChoLow
115 real(RK) , intent(in), contiguous :: ShiftedSample(:,:)
116 real(RK) , intent(in), contiguous, optional :: Weight(:)
117 logical , intent(out) , optional :: failed
118 real(RK) , allocatable :: InvCovMat(:,:)
119 real(RK) , allocatable :: ChoLowCovUpp(:,:)
120 real(RK) , allocatable :: ChoDia(:)
121 integer(IK) :: nd
122 ChoLowCovUpp = genCovUpp(ShiftedSample, Weight, biased = .false.)

134 end if

207 PURE subroutine getChoLowCovUpp(nd,np,Mean,Sample,ChoLowCovUpp,ChoDia)
208#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
209 !DEC$ ATTRIBUTES DLLEXPORT :: getChoLowCovUpp
210#endif
211 use Constants_mod, only: IK, RK
212 use MatrixCholesky_mod, only: getChoLow
213 implicit none
214 integer(IK), intent(in) :: nd,np ! np is the number of observations, nd is the number of parameters for each observation
215 real(RK) , intent(in) :: Mean(nd) ! The mean of the sample.
216 real(RK) , intent(in) :: Sample(nd,np) ! Sample is the data matrix, CovMat contains the elements of the sample covariance matrix
217 real(RK) , intent(out) :: ChoLowCovUpp(nd,nd) ! The lower Cholesky factor and the upper covariance matrix.
218 real(RK) , intent(out) :: ChoDia(nd) ! The diagonal elements of the Cholesky lower triangle.
219 real(RK) :: ShiftedSample(np,nd), npMinusOneInverse
220 integer(IK) :: i,j
221
222 do i = 1,np
223 ShiftedSample(i,1:nd) = Sample(1:nd,i) - Mean
224 end do
225
226 ! Only upper half of CovMat is needed
227 npMinusOneInverse = 1._RK / real(np-1,kind=RK)
228 do j = 1,nd
229 do i = 1,j
230 ! Get the covariance matrix elements: only the upper half of CovMat is needed
231 ChoLowCovUpp(i,j) = dot_product( ShiftedSample(1:np,i) , ShiftedSample(1:np,j) ) * npMinusOneInverse
232 end do
233 end do
234
235 ! @todo: The efficiency can be improved by merging it with the above loops
236 call getChoLow(nd, ChoLowCovUpp, ChoDia)
237
238 end subroutine getChoLowCovUpp

264 subroutine getChoLowCovUppHighDim(nd,np,Mean,Sample,ChoLowCovUpp,ChoDia)
265#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
266 !DEC$ ATTRIBUTES DLLEXPORT :: getChoLowCovUppHighDim
267#endif
268 use Constants_mod, only: IK, RK
269 use MatrixCholesky_mod, only: getChoLow
270 implicit none
271 integer(IK), intent(in) :: nd,np ! np is the number of observations, nd is the number of parameters for each observation
272 real(RK) , intent(in) :: Mean(nd) ! Mean vector
273 real(RK) , intent(in) :: Sample(nd,np) ! Sample is the matrix of the data, CovMat contains the elements of the sample covariance matrix
274 real(RK) , intent(out) :: ChoLowCovUpp(nd,nd) ! The lower Cholesky factor of the covariance matrix
275 real(RK) , intent(out) :: ChoDia(nd) ! The diagonal elements of the Cholesky factorization
276 real(RK) :: ShiftedSample(nd)
277 integer(IK) :: j, ip
278
279 ! Compute the covariance matrix upper.
280
281 ChoLowCovUpp = 0._RK
282 do ip = 1, np
283 do j = 1, nd
284 ShiftedSample(j) = Sample(j,ip) - Mean(j)
285 ChoLowCovUpp(1:j,j) = ChoLowCovUpp(1:j,j) + ShiftedSample(1:j) * ShiftedSample(j)
286 end do
287 end do
288 ChoLowCovUpp = ChoLowCovUpp / real(np-1,kind=RK)
289
290 ! Compute the Cholesky factor lower.
291
292 call getChoLow(nd, ChoLowCovUpp, ChoDia)
293
294 end subroutine getChoLowCovUppHighDim

328 subroutine getMeanCov(np,nd,Sample,CovMat,Mean,MahalSq,InvCovMat,sqrtDetInvCovMat)
329#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
330 !DEC$ ATTRIBUTES DLLEXPORT :: getMeanCov
331#endif
332 use Constants_mod, only: IK, RK
333 use Matrix_mod, only: getInvPosDefMatSqrtDet
334 implicit none
335 integer(IK), intent(in) :: np,nd ! np is the number of observations, nd is the number of parameters for each observation
336 real(RK) , intent(in) :: Sample(np,nd) ! Sample is the matrix of the data, CovMat contains the elements of the sample covariance matrix
337 real(RK) , intent(out) :: CovMat(nd,nd) ! Covariance matrix of the input data
338 real(RK) , intent(out) :: Mean(nd) ! Mean vector
339 real(RK) , intent(out), optional :: MahalSq(np) ! Vector of Mahalanobis Distances Squared, with respect to the mean position of the sample
340 real(RK) , intent(out), optional :: InvCovMat(nd,nd) ! Inverse Covariance matrix of the input data
341 real(RK) , intent(out), optional :: sqrtDetInvCovMat ! sqrt determinant of the inverse covariance matrix
342 real(RK) :: ShiftedSample(np,nd)
343 real(RK) :: DummyVec(nd)
344 real(RK) :: npMinusOneInverse
345 integer(IK) :: i,j
346
347 ! Compute the mean and shift data.
348
349 do j = 1,nd
350 Mean(j) = sum(Sample(1:np,j)) / real(np,kind=RK)
351 ShiftedSample(1:np,j) = Sample(1:np,j) - Mean(j)
352 end do
353
354 npMinusOneInverse = 1._RK / real(np-1,RK)
355 do i = 1,nd
356 do j = 1,nd
357 CovMat(i,j) = dot_product(ShiftedSample(1:np,i),ShiftedSample(1:np,j)) * npMinusOneInverse
358 end do
359 end do
360
361 ! Calculate InCovMat, MahalSq, and sqrt Determinant of InCovMat
362
363 if (present(sqrtDetInvCovMat)) then
364 do j = 1,nd
365 do i = 1,j
366 InvCovMat(i,j) = CovMat(i,j) ! Only the upper half of CovMat is needed
367 end do
368 end do
369 call getInvPosDefMatSqrtDet(nd,InvCovMat,sqrtDetInvCovMat)
370 do i = 1,np
371 do j = 1,nd
372 DummyVec(j) = dot_product(InvCovMat(1:nd,j),ShiftedSample(i,1:nd))
373 end do
374 MahalSq(i) = dot_product(ShiftedSample(i,1:nd),DummyVec)
375 end do
376 end if
377
378 end subroutine getMeanCov

489 subroutine getMeanCovUppTrans(nd,np,Sample,CovUpp,Mean)
490#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
491 !DEC$ ATTRIBUTES DLLEXPORT :: getMeanCovUppTrans
492#endif
493 use Constants_mod, only: IK, RK
494 use Matrix_mod, only: getInvPosDefMatSqrtDet
495 implicit none
496 integer(IK), intent(in) :: nd,np ! np is the number of observations, nd is the number of parameters for each observation
497 real(RK) , intent(in) :: Sample(nd,np) ! Sample is the matrix of the data, CovUpp contains the elements of the sample covariance matrix
498 real(RK) , intent(out) :: CovUpp(nd,nd) ! The covariance matrix of the input data
499 real(RK) , intent(out) :: Mean(nd) ! The mean vector
500 real(RK) :: npMinusOneInvReal, ShiftedSample(nd,np)

502

512 ShiftedSample(1:nd,i) = Sample(1:nd,i) - Mean
513 end do

516 do j = 1,nd
517 do i = 1,j
518 CovUpp(i,j) = dot_product(ShiftedSample(i,1:np),ShiftedSample(j,1:np)) * npMinusOneInvReal
519 end do
520 end do

522 end subroutine getMeanCovUppTrans

553 PURE subroutine getMeanCovUppTransWeighted(nd,np,Sample,CovUpp,Mean,Weight,sumWeight)
554#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
555 !DEC$ ATTRIBUTES DLLEXPORT :: getMeanCovUppTransWeighted
556#endif
557 use Constants_mod, only: IK, RK
558 use Matrix_mod, only: getInvPosDefMatSqrtDet
559 implicit none
560 integer(IK), intent(in) :: nd,np,sumWeight ! np is the number of observations, nd is the number of parameters for each observation
561 integer(IK), intent(in) :: Weight(np) ! weights of the points
562 real(RK) , intent(in) :: Sample(nd,np) ! Sample is the matrix of the data, CovUpp contains the elements of the sample covariance matrix
563 real(RK) , intent(out) :: CovUpp(nd,nd) ! Covariance matrix of the input data
564 real(RK) , intent(out) :: Mean(nd) ! Mean vector
565 real(RK) , allocatable :: ShiftedSample(:,:)
566 real(RK) :: sumWeightMinusOneInvReal
567 integer(IK) :: i,j,ip
568
569 Mean = 0._RK
570 do i = 1,np
571 do j = 1,nd
572 Mean(j) = Mean(j) + Weight(i)*Sample(j,i)
573 end do
574 end do
575 Mean = Mean / real(sumWeight,RK)
576
577 allocate(ShiftedSample(nd,np))
578 do i = 1,np
579 ShiftedSample(1:nd,i) = Sample(1:nd,i) - Mean
580 end do
581
582 sumWeightMinusOneInvReal = 1._RK / real(sumWeight-1,RK)
583 do j = 1,nd
584 do i = 1,j
585 CovUpp(i,j) = 0._RK
586 do ip = 1,np
587 CovUpp(i,j) = CovUpp(i,j) + Weight(ip)*ShiftedSample(i,ip)*ShiftedSample(j,ip)
588 end do
589 CovUpp(i,j) = CovUpp(i,j) * sumWeightMinusOneInvReal
590 end do
591 end do
592 deallocate(ShiftedSample)
593
594 end subroutine getMeanCovUppTransWeighted

619 ! This subroutine uses a recursion equation similar to http://stats.stackexchange.com/questions/97642/how-to-combine-sample-means-and-sample-variances
620 ! See also: ONeill, (2014), "Some Useful Moment Results in Sampling Problems".
621 ! See also: https://en.wikipedia.org/wiki/Algorithms_for_calculating_variance#Covariance
622 ! See also: https://stats.stackexchange.com/questions/43159/how-to-calculate-pooled-variance-of-two-groups-given-known-group-variances-mean
623 subroutine mergeMeanCov(nd,npA,MeanA,CovMatA,npB,MeanB,CovMatB,MeanAB,CovMatAB)
624#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
625 !DEC$ ATTRIBUTES DLLEXPORT :: mergeMeanCov
626#endif
627 use Constants_mod, only: IK, RK
628 integer(IK), intent(in) :: nd
629 integer(IK), intent(in) :: npA,npB
630 real(RK) , intent(in) :: MeanA(nd),CovMatA(nd,nd)
631 real(RK) , intent(in) :: MeanB(nd),CovMatB(nd,nd)
632 real(RK) , intent(out) :: MeanAB(nd),CovMatAB(nd,nd)
633 real(RK) , dimension(nd,1) :: MeanMatA,MeanMatB,MeanMat
634 real(RK) :: DistanceSq(nd,nd)
635 real(RK) :: npABinverse
636 integer(IK) :: npAB
637
638 npAB = npA + npB
639 npABinverse = 1._RK / real(npAB, kind=RK)
640 MeanMatA(1:nd,1) = MeanA
641 MeanMatB(1:nd,1) = MeanB
642
643 ! Compute the new Mean
644
645 MeanAB = ( npA * MeanA + npB * MeanB ) * npABinverse
646 MeanMat(1:nd,1) = MeanAB
647
648 ! Compute the new Covariance matrix
649
650 DistanceSq = matmul( (MeanMatA-MeanMatB), transpose((MeanMatA-MeanMatB)) ) * npA * npB * npABinverse
651 CovMatAB = ( (npA-1) * CovMatA + (npB-1) * CovMatB + DistanceSq ) / real(npAB-1, kind=RK)
652
653 end subroutine mergeMeanCov

682 subroutine mergeMeanCovUpp(nd,npA,MeanA,CovUppA,npB,MeanB,CovUppB,MeanAB,CovUppAB)
683#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
684 !DEC$ ATTRIBUTES DLLEXPORT :: mergeMeanCovUpp
685#endif
686 use Constants_mod, only: IK, RK
687 integer(IK), intent(in) :: nd
688 integer(IK), intent(in) :: npA,npB
689 real(RK) , intent(in) :: MeanA(nd),CovUppA(nd,nd)
690 real(RK) , intent(in) :: MeanB(nd),CovUppB(nd,nd)
691 real(RK) , intent(out) :: MeanAB(nd),CovUppAB(nd,nd)
692 real(RK) :: MeanDiffAB(nd)
693 real(RK) :: npAnpB2npAB2npABMinusOne
694 real(RK) :: npAMinusOne2npABMinusOne
695 real(RK) :: npBMinusOne2npABMinusOne
696 real(RK) :: npABMinusOneInverse
697 real(RK) :: npABinverse
698 real(RK) :: npA2npAB
699 real(RK) :: npB2npAB
700 integer(IK) :: npAB, i, j
701
702 npAB = npA + npB
703 npABinverse = 1._RK / real(npAB, kind=RK)
704 npABMinusOneInverse = 1._RK / real(npAB-1, kind=RK)
705 npAnpB2npAB2npABMinusOne = npA * npB * npABinverse * npABMinusOneInverse
706 npA2npAB = npA * npABinverse
707 npB2npAB = npB * npABinverse
708 npAMinusOne2npABMinusOne = (npA - 1_IK) * npABMinusOneInverse
709 npBMinusOne2npABMinusOne = (npB - 1_IK) * npABMinusOneInverse
710
711 ! Compute the new Mean and Covariance matrix
712
713 do j = 1, nd
714 MeanDiffAB(j) = MeanA(j) - MeanB(j)
715 MeanAB(j) = npA2npAB * MeanA(j) + npB2npAB * MeanB(j)
716 do i = 1, j
717 CovUppAB(i,j) = npAMinusOne2npABMinusOne * CovUppA(i,j) &
718 + npBMinusOne2npABMinusOne * CovUppB(i,j) &
719 + npAnpB2npAB2npABMinusOne * MeanDiffAB(i) * MeanDiffAB(j)
720 end do
721 end do
722
723
724 end subroutine mergeMeanCovUpp

750 subroutine mergeMeanCovUppDense(nd,npA,MeanA,CovUppA,npB,MeanB,CovUppB)
751#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
752 !DEC$ ATTRIBUTES DLLEXPORT :: mergeMeanCovUppDense
753#endif
754 use Constants_mod, only: IK, RK
755 implicit none
756 integer(IK), intent(in) :: nd
757 integer(IK), intent(in) :: npA,npB
758 real(RK) , intent(in) :: MeanA(nd),CovUppA(nd,nd)
759 real(RK) , intent(inout) :: MeanB(nd),CovUppB(nd,nd)
760 real(RK) :: MeanDiffAB(nd)
761 real(RK) :: npAnpB2npAB2npABMinusOne
762 real(RK) :: npAMinusOne2npABMinusOne
763 real(RK) :: npBMinusOne2npABMinusOne
764 real(RK) :: npABMinusOneInverse
765 real(RK) :: npABinverse
766 real(RK) :: npA2npAB
767 real(RK) :: npB2npAB
768 integer(IK) :: npAB, i, j
769
770 npAB = npA + npB
771 npABinverse = 1._RK / real(npAB, kind=RK)
772 npABMinusOneInverse = 1._RK / real(npAB-1, kind=RK)
773 npAnpB2npAB2npABMinusOne = npA * npB * npABinverse * npABMinusOneInverse
774 npA2npAB = npA * npABinverse
775 npB2npAB = npB * npABinverse
776 npAMinusOne2npABMinusOne = (npA - 1_IK) * npABMinusOneInverse
777 npBMinusOne2npABMinusOne = (npB - 1_IK) * npABMinusOneInverse
778
779 ! Compute the new Mean and Covariance matrix
780
781 do j = 1, nd
782 MeanDiffAB(j) = MeanA(j) - MeanB(j)
783 MeanB(j) = npA2npAB * MeanA(j) + npB2npAB * MeanB(j)
784 do i = 1, j
785 CovUppB(i,j) = npAMinusOne2npABMinusOne * CovUppA(i,j) &
786 + npBMinusOne2npABMinusOne * CovUppB(i,j) &
787 + npAnpB2npAB2npABMinusOne * MeanDiffAB(i) * MeanDiffAB(j)
788 end do
789 end do
790
791 end subroutine mergeMeanCovUppDense
