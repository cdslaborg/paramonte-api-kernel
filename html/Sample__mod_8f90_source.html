474 PURE function getVariance_1D(np,mean,Sample,Weight,sumWeight) result(variance)
475#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
476 !DEC$ ATTRIBUTES DLLEXPORT :: getVariance_1D
477#endif
478 implicit none
479 integer(IK), intent(in) :: np ! np is the number of observations (points) whose variance is to be computed
480 real(RK) , intent(in) :: mean ! the mean of the sample.
481 real(RK) , intent(in) :: Sample(np) ! dataset.
482 integer(IK), intent(in), optional :: Weight(np), sumWeight ! sample weight
483 real(RK) :: variance ! output mean vector
484 integer(IK) :: ip
485 variance = 0._RK
486 if (present(Weight)) then
487 do ip = 1,np
488 variance = variance + Weight(ip) * ( Sample(ip) - mean )**2
489 end do
490 variance = variance / real(sumWeight-1_IK,kind=RK)
491 else
492 do ip = 1,np
493 variance = variance + ( Sample(ip) - mean )**2
494 end do
495 variance = variance / real(np-1_IK,kind=RK)
496 end if
497 end function getVariance_1D

758 pure subroutine getChoLowCovUpp(nd,np,Mean,Sample,ChoLowCovUpp,ChoDia)
759#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
760 !DEC$ ATTRIBUTES DLLEXPORT :: getChoLowCovUpp
761#endif
762 use Matrix_mod, only: getChoLow
763 implicit none
764 integer(IK), intent(in) :: nd,np ! np is the number of observations, nd is the number of parameters for each observation
765 real(RK) , intent(in) :: Mean(nd) ! The mean of the sample.
766 real(RK) , intent(in) :: Sample(nd,np) ! Sample is the data matrix, CovMat contains the elements of the sample covariance matrix
767 real(RK) , intent(out) :: ChoLowCovUpp(nd,nd) ! The lower Cholesky factor and the upper covariance matrix.
768 real(RK) , intent(out) :: ChoDia(nd) ! The diagonal elements of the Cholesky lower triangle.
769 real(RK) :: NormedSample(np,nd), npMinusOneInverse
770 integer(IK) :: i,j
771
772 do i = 1,np
773 NormedSample(i,1:nd) = Sample(1:nd,i) - Mean
774 end do
775
776 ! Only upper half of CovMat is needed
777 npMinusOneInverse = 1._RK / real(np-1,kind=RK)
778 do j = 1,nd
779 do i = 1,j
780 ! Get the covariance matrix elements: only the upper half of CovMat is needed
781 ChoLowCovUpp(i,j) = dot_product( NormedSample(1:np,i) , NormedSample(1:np,j) ) * npMinusOneInverse
782 end do
783 end do
784
785 ! @todo: The efficiency can be improved by merging it with the above loops
786 call getChoLow(nd, ChoLowCovUpp, ChoDia)
787
788 end subroutine getChoLowCovUpp

812#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)

816 implicit none

819 real(RK) , intent(in) :: Sample(nd,np) ! Sample is the matrix of the data, CovMat contains the elements of the sample covariance matrix
820 real(RK) , intent(out) :: ChoLowCovUpp(nd,nd) ! The lower Cholesky factor of the covariance matrix
821 real(RK) , intent(out) :: ChoDia(nd) ! The diagonal elements of the Cholesky factorization

832 end do
833 end do
834 ChoLowCovUpp = ChoLowCovUpp / real(np-1,kind=RK)

837

872 subroutine getMeanCov(np,nd,Sample,CovMat,Mean,MahalSq,InvCovMat,sqrtDetInvCovMat)
873#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
874 !DEC$ ATTRIBUTES DLLEXPORT :: getMeanCov
875#endif
876 use Matrix_mod, only: getInvPosDefMatSqrtDet
877 implicit none
878 integer(IK), intent(in) :: np,nd ! np is the number of observations, nd is the number of parameters for each observation
879 real(RK) , intent(in) :: Sample(np,nd) ! Sample is the matrix of the data, CovMat contains the elements of the sample covariance matrix
880 real(RK) , intent(out) :: CovMat(nd,nd) ! Covariance matrix of the input data
881 real(RK) , intent(out) :: Mean(nd) ! Mean vector
882 real(RK) , intent(out), optional :: MahalSq(np) ! Vector of Mahalanobis Distances Squared, with respect to the mean position of the sample
883 real(RK) , intent(out), optional :: InvCovMat(nd,nd) ! Inverse Covariance matrix of the input data
884 real(RK) , intent(out), optional :: sqrtDetInvCovMat ! sqrt determinant of the inverse covariance matrix
885 real(RK) :: NormedSample(np,nd)
886 real(RK) :: DummyVec(nd)
887 real(RK) :: npMinusOneInverse
888 integer(IK) :: i,j
889
890 ! Compute the mean and normalize data.
891
892 do j = 1,nd
893 Mean(j) = sum(Sample(1:np,j)) / real(np,kind=RK)
894 NormedSample(1:np,j) = Sample(1:np,j) - Mean(j)
895 end do
896
897 npMinusOneInverse = 1._RK / real(np-1,RK)
898 do i = 1,nd
899 do j = 1,nd
900 CovMat(i,j) = dot_product(NormedSample(1:np,i),NormedSample(1:np,j)) * npMinusOneInverse
901 end do
902 end do
903
904 ! Calculate InCovMat, MahalSq, and sqrt Determinant of InCovMat
905
906 if (present(sqrtDetInvCovMat)) then
907 do j = 1,nd
908 do i = 1,j
909 InvCovMat(i,j) = CovMat(i,j) ! Only the upper half of CovMat is needed
910 end do
911 end do
912 call getInvPosDefMatSqrtDet(nd,InvCovMat,sqrtDetInvCovMat)

1159 ! This subroutine uses a recursion equation similar to http://stats.stackexchange.com/questions/97642/how-to-combine-sample-means-and-sample-variances
1160 ! See also: ONeill, (2014), "Some Useful Moment Results in Sampling Problems".
1161 ! See also: https://en.wikipedia.org/wiki/Algorithms_for_calculating_variance#Covariance
1162 ! See also: https://stats.stackexchange.com/questions/43159/how-to-calculate-pooled-variance-of-two-groups-given-known-group-variances-mean
1163 subroutine mergeMeanCov(nd,npA,MeanA,CovMatA,npB,MeanB,CovMatB,MeanAB,CovMatAB)
1164#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
1165 !DEC$ ATTRIBUTES DLLEXPORT :: mergeMeanCov
1166#endif
1167 implicit none
1168 integer(IK), intent(in) :: nd
1169 integer(IK), intent(in) :: npA,npB
1170 real(RK) , intent(in) :: MeanA(nd),CovMatA(nd,nd)
1171 real(RK) , intent(in) :: MeanB(nd),CovMatB(nd,nd)
1172 real(RK) , intent(out) :: MeanAB(nd),CovMatAB(nd,nd)
1173 real(RK) , dimension(nd,1) :: MeanMatA,MeanMatB,MeanMat
1174 real(RK) :: DistanceSq(nd,nd)
1175 real(RK) :: npABinverse
1176 integer(IK) :: npAB
1177
1178 npAB = npA + npB
1179 npABinverse = 1._RK / real(npAB, kind=RK)
1180 MeanMatA(1:nd,1) = MeanA
1181 MeanMatB(1:nd,1) = MeanB
1182
1183 ! Compute the new Mean
1184
1185 MeanAB = ( npA * MeanA + npB * MeanB ) * npABinverse
1186 MeanMat(1:nd,1) = MeanAB
1187
1188 ! Compute the new Covariance matrix
1189
1190 DistanceSq = matmul( (MeanMatA-MeanMatB), transpose((MeanMatA-MeanMatB)) ) * npA * npB * npABinverse
1191 CovMatAB = ( (npA-1) * CovMatA + (npB-1) * CovMatB + DistanceSq ) / real(npAB-1, kind=RK)
1192
1193 end subroutine mergeMeanCov
