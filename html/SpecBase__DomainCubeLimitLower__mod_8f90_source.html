78 function construct(methodName) result(self)
79#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
80 !DEC$ ATTRIBUTES DLLEXPORT :: construct
81#endif
82 use Decoration_mod, only: TAB
83 use Constants_mod, only: NULL_RK, NEGINF_RK, PMSM
84 use String_mod, only: val2char
85 implicit none
86 type(DomainCubeLimitLower_type) :: self
87 character(*), intent(in) :: methodName
88 if (methodName == PMSM%ParaDRAM .or. methodName == PMSM%ParaDISE) then
89 self%def = NEGINF_RK
90 elseif (methodName == PMSM%ParaNest) then
91 self%def = 0._RK
92 end if
93 self%null = NULL_RK
94 self%desc = &
95 "domainCubeLimitLower represents the lower boundaries of the cubical domain of the objective function to be sampled. &
96 &It is an ndim-dimensional vector of 64-bit real numbers, where ndim is the number of variables of the objective function. &
97 &It is also possible to assign only select values of domainCubeLimitLower and leave the rest of the components to be assigned &
98 &the default value. This is POSSIBLE ONLY when domainCubeLimitLower is defined inside the input file to "//methodName//". &
99 &For example, having the following inside the input file, \n\n&
100 & domainCubeLimitLower(3:5) = -100\n\n&
101 & will only set the lower limits of the third, fourth, and the fifth dimensions to -100, or,\n\n&
102 & domainCubeLimitLower(1) = -100, domainCubeLimitLower(2) = -1.e6 \n\n&
103 & will set the lower limit on the first dimension to -100, and 1.e6 on the second dimension, or,\n\n&
104 & domainCubeLimitLower = 3*-2.5e100\n\n&
105 & will only set the lower limits on the first, second, and the third dimensions to -2.5*10^100, while the rest of &
106 &the lower limits for the missing dimensions will be automatically set to the default value.\n\n&
107 &The default value for all elements of domainCubeLimitLower is: " // val2char(self%def) // "."
108 end function construct

143 subroutine checkForSanity(self,Err)
144#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
145 !DEC$ ATTRIBUTES DLLEXPORT :: checkForSanity
146#endif
147 use Err_mod, only: Err_type
148 use Constants_mod, only: IK, NLC, NEGINF_RK
149 use String_mod, only: val2char
150 implicit none
151 class(DomainCubeLimitLower_type), intent(in) :: self
152 type(Err_type), intent(inout) :: Err
153 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME//"@checkForSanity()"
154 integer(IK) :: i
155 do i = 1, size(self%Val)
156 if ( self%Val(i) < NEGINF_RK ) then
157 Err%occurred = .true.
158 Err%msg = Err%msg // &
159 PROCEDURE_NAME // ": Error occurred. The component " // val2char(i) // &
160 " of the variable domainCubeLimitLower (" // val2char(self%Val(i)) // ") &
161 &cannot be smaller than the smallest positive real number representable in the simulation (" // &
162 val2char(NEGINF_RK) // ")."//NLC//NLC
163 end if
164 end do
165 end subroutine checkForSanity
