112 pure function getLogLumDisWicMpc(zplus1) result(logLumDisWicMpc)
113#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
114 !DEC$ ATTRIBUTES DLLEXPORT :: getLogLumDisWicMpc
115#endif
116 use Constants_mod, only: RK
117 implicit none
118 real(RK), intent(in) :: zplus1 ! redshift + 1
119 real(RK) :: logLumDisWicMpc
120 real(RK) :: alpha1, x1, x1Sq, psix1
121 real(RK), parameter :: TWICE_OMEGA_DE_OVER_OMEGA_DM = 2._RK * OMEGA_DE / OMEGA_DM
122 real(RK), parameter :: ALPHA0 = 1._RK + TWICE_OMEGA_DE_OVER_OMEGA_DM
123 real(RK), parameter :: X0 = log( ALPHA0 + sqrt(ALPHA0**2-1._RK) )
124 real(RK), parameter :: X0Sq = X0**2
125 real(RK), parameter :: PSI_COEF1 = 2._RK**(2._RK/3._RK)
126 real(RK), parameter :: PSI_COEF2 = -PSI_COEF1 / 252._RK
127 real(RK), parameter :: PSI_COEF3 = +PSI_COEF1 / 21060._RK
128 real(RK), parameter :: PSIX0 = X0**0.333333333333333_RK * ( PSI_COEF1 + X0Sq * (PSI_COEF2 + X0Sq*PSI_COEF3) )
129 real(RK), parameter :: LOG_COEF = log(LS2HC / (OMEGA_DE**0.1666666666666667_RK*OMEGA_DM**0.3333333333333333_RK))
130 alpha1 = 1._RK + TWICE_OMEGA_DE_OVER_OMEGA_DM / zplus1**3
131 x1 = log( alpha1 + sqrt( alpha1**2 - 1._RK ) )
132 x1Sq = x1**2
133 psix1 = x1**0.333333333333333_RK * ( PSI_COEF1 + x1Sq * (PSI_COEF2 + x1Sq*PSI_COEF3) )
134 logLumDisWicMpc = LOG_COEF + log(zplus1*(PSIX0-psix1))
135 end function getLogLumDisWicMpc

153 pure function ldiswickram(zplus1)
154#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
155 !DEC$ ATTRIBUTES DLLEXPORT :: ldiswickram
156#endif
157 use Constants_mod, only: RK
158 implicit none
159 real(RK), intent(in) :: zplus1
160 real(RK) :: ldiswickram,alpha,alpha0,x,x0,psi,psi0
161 alpha=1._RK+2*OMEGA_DE/(OMEGA_DM*zplus1**3)
162 alpha0=1._RK+2*OMEGA_DE/OMEGA_DM
163 x=log(alpha+sqrt(alpha*alpha-1._RK))
164 x0=log(alpha0+sqrt(alpha0*alpha0-1._RK))
165 psi=x**0.333333333333333_RK*(1.5874010519682_RK-6.2992105236833e-3_RK*x*x+7.5375168659459e-5_RK*x**4)
166 psi0=x0**0.333333333333333_RK*(1.5874010519682_RK-6.2992105236833e-3_RK*x0*x0+7.5375168659459e-5_RK*x0**4)
167 ldiswickram=LS2HC*zplus1*(psi0-psi)/(OMEGA_DE**0.1666666666666667_RK*OMEGA_DM**0.333333333333333_RK)
168 end function ldiswickram

190 function getLookBackTime(zplus1,maxRelativeError,nRefinement) result(lookBackTime)
191#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
192 !DEC$ ATTRIBUTES DLLEXPORT :: getLookBackTime
193#endif
194 use, intrinsic :: iso_fortran_env, only: output_unit
195 use Integration_mod, only: doQuadRombClosed, ErrorMessage
196 use Constants_mod, only: RK, IK
197 implicit none
198 real(RK) , intent(in) :: zplus1
199 real(RK) , intent(in), optional :: maxRelativeError
200 integer(IK) , intent(in), optional :: nRefinement
201 real(RK) , parameter :: ZPLUS1_MIN = 1._RK
202 real(RK) :: lookBackTime
203 real(RK) :: maxRelativeErrorDefault, relerr
204 integer(IK) :: neval, ierr, nRefinementDefault
205
206 maxRelativeErrorDefault = 1.e-6_RK; if (present(maxRelativeError)) maxRelativeErrorDefault = maxRelativeError
207 nRefinementDefault = 7_IK; if (present(nRefinement)) nRefinementDefault = nRefinement
208
209 call doQuadRombClosed( getFunc = getLookBackTimeDensity &
210 , lowerLim = ZPLUS1_MIN &
211 , upperLim = zplus1 &
212 , maxRelativeError = maxRelativeErrorDefault &
213 , nRefinement = nRefinementDefault &
214 , integral = lookBackTime &
215 , relativeError = relerr &
216 , numFuncEval = neval &
217 , ierr = ierr &
218 )
219 if (ierr/=0_IK) then
220 ! LCOV_EXCL_START
221 write(output_unit,"(A)") ErrorMessage(ierr)
222 error stop
223 ! LCOV_EXCL_STOP
224 end if
225 lookBackTime = HUBBLE_TIME_GYRS * lookBackTime
226
227 end function getLookBackTime
