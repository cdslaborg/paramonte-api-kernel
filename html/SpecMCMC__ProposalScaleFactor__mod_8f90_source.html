77 function construct(nd,methodName) result(self)
78#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
79 !DEC$ ATTRIBUTES DLLEXPORT :: construct
80#endif
81 use Constants_mod, only: RK, IK, NULL_SK
82 use String_mod, only: val2str
83 use Decoration_mod, only: TAB
84 implicit none
85 integer(IK), intent(in) :: nd
86 character(*), intent(in) :: methodName
87 type(ProposalScaleFactor_type) :: self
88
89 self%strdef = "gelman"
90 self%valdef = 2.38_RK/sqrt(real(nd,kind=RK)) ! Gelman, Roberts, Gilks (1996): Efficient Metropolis Jumping Rules
91 self%null = repeat(NULL_SK, MAX_LEN_STRING_PROPOSAL_SCALE_FACTOR)
92 self%desc = &
93 "proposalScaleFactor is a real-valued positive number (which must be given as string), by the square of which the &
94 &covariance matrix of the proposal distribution of the MCMC sampler is scaled. In other words, &
95 &the proposal distribution will be scaled in every direction by the value of proposalScaleFactor. &
96 &It can also be given in units of the string keyword 'gelman' (which is case-INsensitive) after the paper:\n\n" &
97 // TAB // "Gelman, Roberts, and Gilks (1996): 'Efficient Metropolis Jumping Rules'.\n\n&
98 &The paper finds that the optimal scaling factor for a Multivariate Gaussian proposal distribution for the &
99 &Metropolis-Hastings Markov Chain Monte Carlo sampling of a target Multivariate Normal Distribution &
100 &of dimension ndim is given by:\n\n&
101 & proposalScaleFactor = 2.38/sqrt(ndim) , in the limit of ndim -> Infinity.\n\n&
102 &Multiples of the gelman scale factors are also acceptable as input and can be specified like the following examples:\n\n&
103 & proposalScaleFactor = '1'\n\n&
104 & multiplies the ndim-dimensional proposal covariance matrix by 1, essentially no change occurs to &
105 &the covariance matrix.\n\n" // &
106 ' proposalScaleFactor = "1"\n\n' // &
107 " same as the previous example. The double-quotation marks act the same way as single-quotation marks.\n\n&
108 & proposalScaleFactor = '2.5'\n\n&
109 & multiplies the ndim-dimensional proposal covariance matrix by 2.5.\n\n&
110 & proposalScaleFactor = '2.5*Gelman'\n\n&
111 & multiplies the ndim-dimensional proposal covariance matrix by 2.5 * 2.38/sqrt(ndim).\n\n" // &
112 ' proposalScaleFactor = "2.5 * gelman"\n\n' // &
113 " same as the previous example, but with double-quotation marks. space characters are ignored.\n\n" // &
114 ' proposalScaleFactor = "2.5 * gelman*gelman*2"\n\n' // &
115 " equivalent to gelmanFactor-squared multiplied by 5.\n\n&
116 &Note, however, that the result of Gelman et al. paper applies only to multivariate normal proposal distributions, in &
117 &the limit of infinite dimensions. Therefore, care must be taken when using Gelman's scaling factor with non-Gaussian &
118 &proposals and target objective functions. Note that only the product symbol (*) can be parsed &
119 &in the string value of proposalScaleFactor. The presence of other mathematical symbols or multiple appearances of the product &
120 &symbol will lead to a simulation crash. Also, note that the prescription of an acceptance range specified by the input &
121 &variable 'targetAcceptanceRate' will lead to dynamic modification of the initial input value of proposalScaleFactor throughout sampling &
122 &for proposalAdaptationCount times. &
123 &The default proposalScaleFactor string-value is 'gelman' (for all proposals), which is subsequently converted to 2.38/sqrt(ndim)."
124 end function construct

155 ! ATTN: This subroutine also assigns the value of ProposalScaleFactor. It MUST be executed by all images.
156 subroutine checkForSanity(self,Err,methodName)
157#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
158 !DEC$ ATTRIBUTES DLLEXPORT :: checkForSanity
159#endif
160 use Constants_mod, only: IK, RK, NLC
161 use String_mod, only: val2str, StrSplit_type
162 use Err_mod, only: Err_type
163 implicit none
164 class(ProposalScaleFactor_type), intent(inout) :: self
165 character(*), intent(in) :: methodName
166 type(Err_type), intent(inout) :: Err
167 character(*), parameter :: PROCEDURE_NAME = "@checkForSanity()"
168 type(StrSplit_type) :: String
169 integer(IK) :: i
170 real(RK) :: temp
171
172 ! First convert the proposalScaleFactor string to real value:
173
174 String%value = String%replace(self%str," ","") ! remove the white spaces
175 if (len_trim(adjustl(String%value))==0) then
176 Err%occurred = .true.
177 Err%msg = Err%msg // &
178 MODULE_NAME // PROCEDURE_NAME // ": Error occurred. &
179 &The input string value (" // self%str // ") for the variable proposalScaleFactor &
180 &is empty. Make sure the input string follows the syntax rules of " &
181 // methodName // " for this variable. Otherwise drop it from the input list. " &
182 // methodName // " will automatically assign an appropriate value to it."//NLC//NLC
183 return
184 end if
185
186 ! Now split the string by "*" to real coefficient and character (gelman) parts for further evaluations
187
188 String%Parts = String%split(value = String%value, delim = "*", nPart = String%nPart)
189 self%val = 1._RK
190 do i = 1, String%nPart
191 if ( String%genLowerCase( String%Parts(i)%value ) == "gelman" ) then
192 self%val = self%val * self%valdef
193 else
194 temp = String%real(value = String%Parts(i)%value, iostat = Err%stat)
195 if ( Err%stat/=0 ) then
196 Err%occurred = .true.
197 Err%msg = Err%msg // & ! LCOV_EXCL_LINE
198 MODULE_NAME // PROCEDURE_NAME // ": Error occurred while reading real number."//NLC// & ! LCOV_EXCL_LINE
199 "The input string value for the variable proposalScaleFactor (" // self%str // ") does not appear to follow "//& ! LCOV_EXCL_LINE
200 "the standard syntax rules of "// methodName // " for this variable. '" // String%Parts(i)%value // & ! LCOV_EXCL_LINE
201 "' cannot be parsed into any meaningful token. Please correct the input value, or drop it from the input list, "//& ! LCOV_EXCL_LINE
202 "in which case, " // methodName // " will automatically assign an appropriate value to it."//NLC//NLC
203 return
204 else
205 self%val = self%val * temp
206 end if
207 end if
208 end do
209
210 ! Now check if the real value is positive
211
212 if (self%val<=0) then
213 Err%occurred = .true.
214 Err%msg = Err%msg // &
215 MODULE_NAME // PROCEDURE_NAME // ": Error occurred. &
216 &The input string value (" // self%str // ") translates to a negative real value: " // &
217 val2str(self%val) // ". &
218 &Make sure the input string follows the syntax rules of " // methodName // " for this variable. &
219 &Otherwise drop it from the input list. " // methodName // &
220 " will automatically assign an appropriate value to it.\n\n"
221 return
222 end if
223
224 end subroutine checkForSanity
