72 pure function getLogProbKnnPoisson(nd, k, lambda, distance) result(logProbKnnPoisson)
73#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
74 !DEC$ ATTRIBUTES DLLEXPORT :: getLogProbKnnPoisson
75#endif
76 use Constants_mod, only: IK, RK
77 use GeometryEllipsoid_mod, only: genLogVolUnitBall
78 use SpecFuncGamma_mod, only: getLogFactorial
79 implicit none
80 integer(IK) , intent(in) :: nd, k
81 real(RK) , intent(in) :: lambda, distance
82 real(RK) :: logProbKnnPoisson
83 real(RK) :: logVolUnitBall
84 real(RK) :: logDistance
85 logDistance = log(distance)
86 logVolUnitBall = genLogVolUnitBall(nd)
87 logProbKnnPoisson = log(real(nd,RK)) + k * (log(lambda) + logVolUnitBall + nd * logDistance) &
88 - lambda * exp(logVolUnitBall) * distance**nd &
89 - logDistance - getLogFactorial(k-1_IK)
90 end function getLogProbKnnPoisson

223 subroutine getDistSortedExpDiffWeightKS2(SortedIndex, DistSortedExpDiff, logVolUnitBall, ProbKS, StatKS, Lambda, alpha, beta, Err)
224#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
225 !DEC$ ATTRIBUTES DLLEXPORT :: getDistSortedExpDiffWeightKS2
226#endif
227 use InterpLinear_mod, only: InterpLinear_type ! LCOV_EXCL_LINE
228 use HypoTest_mod, only: getSampleDisparity ! LCOV_EXCL_LINE
229 use SetTheory_mod, only: complementOrdered ! LCOV_EXCL_LINE
230 use Constants_mod, only: IK, RK ! LCOV_EXCL_LINE
231 use Sort_mod, only: indexArray ! LCOV_EXCL_LINE
232 use Err_mod, only: Err_type ! LCOV_EXCL_LINE
233 implicit none
234 integer(IK) , intent(in) , contiguous :: SortedIndex(:)
235 real(RK) , intent(in) , contiguous :: DistSortedExpDiff(:)
236 real(RK) , intent(in) , optional :: logVolUnitBall
237 real(RK) , intent(out) , optional :: ProbKS(size(DistSortedExpDiff))
238 real(RK) , intent(out) , optional :: StatKS(size(DistSortedExpDiff))
239 real(RK) , intent(out) , optional :: Lambda(size(DistSortedExpDiff))
240 real(RK) , intent(out) , optional :: alpha, beta
241 type(Err_type) , intent(out) , optional :: Err
242 real(RK) :: SortedIndexProbKS(size(SortedIndex))
243 real(RK) :: SortedIndexStatKS(size(SortedIndex))
244 real(RK) :: SortedIndexLambda(size(SortedIndex))
245 integer(IK) :: IndexMap(size(DistSortedExpDiff))
246 real(RK) :: IndexAll(size(DistSortedExpDiff))
247 real(RK) , allocatable :: InterpolationPrediction(:)
248 integer(IK) , allocatable :: InterpolationIndex(:)
249 type(InterpLinear_type) :: Interp
250 logical :: failed
251 if (present(ProbKS) .or. present(StatKS) .or. present(Lambda) .or. present(alpha) .or. present(beta)) then
252 call getSampleDisparity( Index = SortedIndex & ! LCOV_EXCL_LINE
253 , SortedSample = DistSortedExpDiff & ! LCOV_EXCL_LINE
254 , ProbKS = SortedIndexProbKS & ! LCOV_EXCL_LINE
255 , StatKS = SortedIndexStatKS & ! LCOV_EXCL_LINE
256 , Lambda = SortedIndexLambda & ! LCOV_EXCL_LINE
257 , Err = Err & ! LCOV_EXCL_LINE
258 )
259 if (Err%occurred) return
260 if (present(alpha)) alpha = sum(SortedIndexProbKS)
261 if (present(beta)) beta = sum(SortedIndexProbKS * DistSortedExpDiff(SortedIndex))
262 blockFullSortedIndex: if (size(SortedIndex) < size(DistSortedExpDiff)) then ! Interpolate the missing weights.
263 InterpolationIndex = complementOrdered(SortedIndex, lower = 1_IK, upper = size(DistSortedExpDiff))
264 Interp = InterpLinear_type(SortedX = real(SortedIndex,RK), SortedY = SortedIndexProbKS, Err = Err)
265 if (present(Err)) then; if (Err%occurred) return; end if ! LCOV_EXCL_LINE
266 InterpolationPrediction = Interp%predict(QueryX = real(InterpolationIndex,RK))
267 if (present(beta)) beta = beta + sum(InterpolationPrediction * DistSortedExpDiff(InterpolationIndex))
268 if (present(alpha)) alpha = alpha + sum(InterpolationPrediction)
269 if (present(ProbKS) .or. present(StatKS) .or. present(Lambda)) then
270 IndexAll = [SortedIndex, InterpolationIndex]
271 call indexArray(IndexAll, IndexMap, failed)
272 if (failed) then; if (present(Err)) Err%occurred = .true.; return; end if ! LCOV_EXCL_LINE
273 IndexAll = IndexAll(IndexMap)
274 if (present(ProbKS)) then
275 ProbKS = [SortedIndexProbKS, InterpolationPrediction]
276 ProbKS = ProbKS(IndexMap)
277 end if
278 if (present(StatKS)) then
279 Interp = InterpLinear_type(SortedX = real(SortedIndex,RK), SortedY = SortedIndexStatKS, Err = Err)
280 if (present(Err)) then; if (Err%occurred) return; end if ! LCOV_EXCL_LINE
281 InterpolationPrediction = Interp%predict(QueryX = real(InterpolationIndex,RK))
282 StatKS = [SortedIndexStatKS, InterpolationPrediction]
283 StatKS = StatKS(IndexMap)
284 end if
285 if (present(Lambda)) then
286 Interp = InterpLinear_type(SortedX = real(SortedIndex,RK), SortedY = SortedIndexLambda, Err = Err)
287 if (present(Err)) then; if (Err%occurred) return; end if ! LCOV_EXCL_LINE
288 InterpolationPrediction = Interp%predict(QueryX = real(InterpolationIndex,RK))
289 Lambda = [SortedIndexLambda, InterpolationPrediction]
290 Lambda = Lambda(IndexMap)
291 end if
292 end if
293#if (DEBUG_ENABLED || TESTING_ENABLED || CODECOV_ENABLED || BASIC_TEST_ENABLED)
294 block
295 use iso_fortran_env, only: output_unit
296 integer(IK) :: ip, npMissing, counterPredictionIndex
297 ! Check the consistency of the indices
298 do ip = 1, size(SortedIndex)
299 if (any(InterpolationIndex == SortedIndex(ip))) then
300 write(output_unit,"(*(g0:,', '))") "InterpolationIndex == SortedIndex for ip, SortedIndex(ip)", ip, SortedIndex(ip)
301 error stop
302 end if
303 end do
304 npMissing = size(DistSortedExpDiff) - size(SortedIndex)
305 counterPredictionIndex = size(InterpolationIndex, kind = IK)
306 if (counterPredictionIndex /= npMissing) then
307 write(output_unit,"(*(g0:,', '))") "counterPredictionIndex /= npMissing", counterPredictionIndex, npMissing
308 error stop
309 end if
310 end block
311#endif
312 deallocate(InterpolationIndex, InterpolationPrediction)
313 else blockFullSortedIndex
314 if (present(ProbKS)) ProbKS = SortedIndexProbKS
315 if (present(StatKS)) StatKS = SortedIndexStatKS
316 if (present(Lambda)) Lambda = SortedIndexLambda
317 end if blockFullSortedIndex
318 if (present(logVolUnitBall) .and. present(beta)) beta = exp(log(beta) + logVolUnitBall)
319 else
320 if (present(Err)) Err%occurred = .false.
321 end if
322 end subroutine getDistSortedExpDiffWeightKS2
