220 subroutine setFromInputFile( SpecBase, Err )
221#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
222 !DEC$ ATTRIBUTES DLLEXPORT :: setFromInputFile
223#endif
224
225 use Constants_mod, only: IK, RK
226 use Err_mod, only: Err_type
227
228 implicit none
229
230 class(SpecBase_type), intent(inout) :: SpecBase
231 type(Err_type), intent(inout) :: Err
232
233 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME // "@setFromInputFile()"
234
235 call SpecBase%OutputSampleSize %set(outputSampleSize)
236 call SpecBase%RandomSeed %set(randomSeed,Err)
237 call SpecBase%Description %set(description)
238 call SpecBase%OutputFileName %set(outputFileName)
239 call SpecBase%OutputChainFileFormat %set(outputChainFileFormat)
240 call SpecBase%DomainAxesNameList %set(DomainAxesNameList)
241 call SpecBase%OutputRestartFileFormat %set(outputRestartFileFormat)
242 call SpecBase%DomainCubeLimitLower %set(domainCubeLimitLower)
243 call SpecBase%DomainCubeLimitUpper %set(domainCubeLimitUpper)
244 call SpecBase%DomainEllipsoidCenter %set(domainEllipsoidCenter)
245 call SpecBase%DomainEllipsoidCorMat %set(domainEllipsoidCorMat) ! DO NOT CHANGE THE ORDER
246 call SpecBase%DomainEllipsoidStdVec %set(domainEllipsoidStdVec) ! DO NOT CHANGE THE ORDER
247 call SpecBase%DomainEllipsoidCovMat %set(SpecBase%DomainEllipsoidStdVec%Val, SpecBase%DomainEllipsoidCorMat%Val, domainEllipsoidCovMat) ! DO NOT CHANGE THE ORDER
248 call SpecBase%Domain %set(domain)
249
250 ! do not change the order with outputDelimiter
251 call SpecBase%OutputColumnWidth %set(outputColumnWidth)
252 call SpecBase%OutputDelimiter %set(SpecBase%outputColumnWidth%value, outputDelimiter)
253
254 call SpecBase%OutputOverwriteEnabled %set(outputOverwriteEnabled)
255 call SpecBase%OutputRealPrecision %set(outputRealPrecision)
256 call SpecBase%OutputSilentModeEnabled %set(outputSilentModeEnabled)
257 call SpecBase%OutputProgressReportPeriod %set(outputProgressReportPeriod)
258 call SpecBase%Parallelism %set(parallelism)
259 call SpecBase%InputFileHasPriority %set(inputFileHasPriority)
260 call SpecBase%TargetAcceptanceRate %set(TargetAcceptanceRate)
261 call SpecBase%ParallelismMpiFinalizeEnabled %set(parallelismMpiFinalizeEnabled)
262 call SpecBase%DomainMaxNumCheckToWarn %set(domainMaxNumCheckToWarn)
263 call SpecBase%DomainMaxNumCheckToStop %set(domainMaxNumCheckToStop)
264 call SpecBase%InterfaceType %set(interfaceType)
265 call SpecBase%SystemInfoFilePath %set(systemInfoFilePath)
266
267 if (Err%occurred) Err%msg = PROCEDURE_NAME // Err%msg
268
269 end subroutine setFromInputFile

274 subroutine setFromInputArgs( SpecBase, Err &
275 , outputSampleSize &
276 , randomSeed &
277 , description &
278 , outputFileName &
279 , outputDelimiter &
280 , outputChainFileFormat &
281 , domainAxesNameList &
282 , outputRestartFileFormat &
283 , outputColumnWidth &
284 , outputOverwriteEnabled &
285 , outputRealPrecision &
286 , outputSilentModeEnabled &
287 , domainCubeLimitLower &
288 , domainCubeLimitUpper &
289 , domainEllipsoidCenter &
290 , domainEllipsoidCovMat &
291 , domainEllipsoidCorMat &
292 , domainEllipsoidStdVec &
293 , domain &
294 , parallelism &
295 , outputProgressReportPeriod &
296 , TargetAcceptanceRate &
297 , parallelismMpiFinalizeEnabled &
298 , domainMaxNumCheckToWarn &
299 , domainMaxNumCheckToStop &
300 )
301#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
302 !DEC$ ATTRIBUTES DLLEXPORT :: setFromInputArgs
303#endif
304
305 use Constants_mod, only: IK, RK
306 use Err_mod, only: Err_type
307
308 implicit none
309
310 class(SpecBase_type), intent(inout) :: SpecBase
311 type(Err_type), intent(inout) :: Err
312
313 character(*), parameter :: PROCEDURE_NAME = MODULE_NAME // "@setFromInputArgs()"
314
315 ! ParaMonte variables
316 integer(IK) , intent(in), optional :: outputSampleSize
317 integer(IK) , intent(in), optional :: randomSeed
318 character(*), intent(in), optional :: description
319 character(*), intent(in), optional :: outputFileName
320 character(*), intent(in), optional :: outputDelimiter
321 character(*), intent(in), optional :: outputChainFileFormat
322 character(*), intent(in), optional :: domainAxesNameList(:)
323 character(*), intent(in), optional :: outputRestartFileFormat
324 integer(IK) , intent(in), optional :: outputColumnWidth
325 logical , intent(in), optional :: outputOverwriteEnabled
326 integer(IK) , intent(in), optional :: outputRealPrecision
327 logical , intent(in), optional :: outputSilentModeEnabled
328 real(RK) , intent(in), optional :: domainCubeLimitLower(:)
329 real(RK) , intent(in), optional :: domainCubeLimitUpper(:)
330 real(RK) , intent(in), optional :: domainEllipsoidCenter(:)
331 real(RK) , intent(in), optional :: domainEllipsoidStdVec(:)
332 real(RK) , intent(in), optional :: domainEllipsoidCovMat(:,:)
333 real(RK) , intent(in), optional :: domainEllipsoidCorMat(:,:)
334 character(*), intent(in), optional :: domain
335 character(*), intent(in), optional :: parallelism
336 integer(IK) , intent(in), optional :: outputProgressReportPeriod
337 real(RK) , intent(in), optional :: TargetAcceptanceRate(2)
338 logical , intent(in), optional :: parallelismMpiFinalizeEnabled
339 integer(IK) , intent(in), optional :: domainMaxNumCheckToWarn
340 integer(IK) , intent(in), optional :: domainMaxNumCheckToStop
341
342 if (present(outputSampleSize)) call SpecBase%OutputSampleSize %set(outputSampleSize)
343 if (present(randomSeed)) call SpecBase%RandomSeed %set(randomSeed,Err)
344 if (present(description)) call SpecBase%Description %set(description)
345 if (present(outputFileName)) call SpecBase%OutputFileName %set(outputFileName)
346 if (present(outputChainFileFormat)) call SpecBase%OutputChainFileFormat %set(outputChainFileFormat)
347 if (present(domainAxesNameList)) call SpecBase%DomainAxesNameList %set(domainAxesNameList)
348 if (present(outputRestartFileFormat)) call SpecBase%OutputRestartFileFormat %set(outputRestartFileFormat)
349 if (present(domainCubeLimitLower)) call SpecBase%DomainCubeLimitLower %set(domainCubeLimitLower)
350 if (present(domainCubeLimitUpper)) call SpecBase%DomainCubeLimitUpper %set(domainCubeLimitUpper)
351 if (present(DomainEllipsoidCenter)) call SpecBase%DomainEllipsoidCenter %set(domainEllipsoidCenter)
352 if (present(DomainEllipsoidCorMat)) call SpecBase%DomainEllipsoidCorMat %set(domainEllipsoidCorMat) ! DO NOT CHANGE THE ORDER
353 if (present(DomainEllipsoidStdVec)) call SpecBase%DomainEllipsoidStdVec %set(domainEllipsoidStdVec) ! DO NOT CHANGE THE ORDER
354 call SpecBase%DomainEllipsoidCovMat %set(SpecBase%DomainEllipsoidStdVec%Val, SpecBase%DomainEllipsoidCorMat%Val, domainEllipsoidCovMat) ! DO NOT CHANGE THE ORDER
355 if (present(Domain)) call SpecBase%Domain %set(domain)
356
357 ! do not change the order with outputDelimiter
358 if (present(outputColumnWidth)) call SpecBase%OutputColumnWidth %set(outputColumnWidth)
359 call SpecBase%OutputDelimiter%set(SpecBase%outputColumnWidth%value, outputDelimiter)
360
361 if (present(outputOverwriteEnabled)) call SpecBase%OutputOverwriteEnabled %set(outputOverwriteEnabled)
362 if (present(parallelism)) call SpecBase%Parallelism %set(parallelism)
363 if (present(outputRealPrecision)) call SpecBase%OutputRealPrecision %set(outputRealPrecision)
364 if (present(outputSilentModeEnabled)) call SpecBase%OutputSilentModeEnabled %set(outputSilentModeEnabled)
365 if (present(outputProgressReportPeriod)) call SpecBase%OutputProgressReportPeriod %set(outputProgressReportPeriod)
366 if (present(TargetAcceptanceRate)) call SpecBase%TargetAcceptanceRate %set(TargetAcceptanceRate)
367 if (present(parallelismMpiFinalizeEnabled)) call SpecBase%ParallelismMpiFinalizeEnabled %set(parallelismMpiFinalizeEnabled)
368 if (present(domainMaxNumCheckToWarn)) call SpecBase%DomainMaxNumCheckToWarn %set(domainMaxNumCheckToWarn)
369 if (present(domainMaxNumCheckToStop)) call SpecBase%DomainMaxNumCheckToStop %set(domainMaxNumCheckToStop)
370
371 if (Err%occurred) Err%msg = PROCEDURE_NAME // Err%msg
372
373 end subroutine setFromInputArgs

377 subroutine reportValues(SpecBase,prefix,outputUnit,isLeaderImage)
378#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
379 !DEC$ ATTRIBUTES DLLEXPORT :: reportValues
380#endif
381 use Decoration_mod, only: GENERIC_OUTPUT_FORMAT
382 use Decoration_mod, only: GENERIC_TABBED_FORMAT
383 use Decoration_mod, only: TAB
384 use Constants_mod, only: IK, UNDEFINED
385 use Err_mod, only: note, informUser
386 implicit none
387 class(SpecBase_type), intent(in) :: SpecBase
388 character(*), intent(in) :: prefix
389 integer(IK) , intent(in) :: outputUnit
390 logical , intent(in) :: isLeaderImage
391 integer(IK) :: i, ndim
392 character(:), allocatable :: msg
393
394 if (isLeaderImage) then
395
396 ndim = size(SpecBase%DomainCubeLimitLower%Val(:))
397 msg = "ndim is a 32-bit positive integer, representing the number of dimensions of the domain of the objective function. &
398 &It is the only simulation specification variable that the user must always provide along with the objective function, &
399 &separately from the rest of the simulation specifications. The variable ndim must be always provided directly to the ParaMonte &
400 &routines, along with the objective function. If specified within an input file, its value will be ignored and not used. &
401 &The variable ndim has no default value as it is the only mandatory piece of information that must be provided by the user."
402 write(outputUnit,GENERIC_OUTPUT_FORMAT)
403 write(outputUnit,GENERIC_OUTPUT_FORMAT) "ndim"
404 write(outputUnit,GENERIC_OUTPUT_FORMAT)
405 write(outputUnit,GENERIC_TABBED_FORMAT) ndim
406 if (SpecBase%OutputSilentModeEnabled%isFalse) call note( prefix = prefix, outputUnit = outputUnit, newline = "\n", msg = msg )
407
408
409 write(outputUnit,GENERIC_OUTPUT_FORMAT)
410 write(outputUnit,GENERIC_OUTPUT_FORMAT) "description"
411 write(outputUnit,GENERIC_OUTPUT_FORMAT)
412 call informUser( outputUnit = outputUnit, newline = "\n", wrapWidth = 125_IK, prefix = TAB//TAB, marginTop = 0_IK, marginBot = 0_IK, msg = SpecBase%Description%val )
413 if (SpecBase%OutputSilentModeEnabled%isFalse) call note( prefix = prefix, outputUnit = outputUnit, newline = "\n", msg = SpecBase%Description%desc )
414
415
416 write(outputUnit,GENERIC_OUTPUT_FORMAT)
417 write(outputUnit,GENERIC_OUTPUT_FORMAT) "inputFileHasPriority"
418 write(outputUnit,GENERIC_OUTPUT_FORMAT)
419 write(outputUnit,GENERIC_TABBED_FORMAT) SpecBase%InputFileHasPriority%val
420 if (SpecBase%OutputSilentModeEnabled%isFalse) call note( prefix = prefix, outputUnit = outputUnit, newline = "\n", msg = SpecBase%InputFileHasPriority%desc )
421
422
423 write(outputUnit,GENERIC_OUTPUT_FORMAT)
424 write(outputUnit,GENERIC_OUTPUT_FORMAT) "outputSilentModeEnabled"
425 write(outputUnit,GENERIC_OUTPUT_FORMAT)
426 write(outputUnit,GENERIC_TABBED_FORMAT) SpecBase%OutputSilentModeEnabled%val
427 if (SpecBase%OutputSilentModeEnabled%isFalse) call note( prefix = prefix, outputUnit = outputUnit, newline = "\n", msg = SpecBase%OutputSilentModeEnabled%desc )
428
429
430 write(outputUnit,GENERIC_OUTPUT_FORMAT)
431 write(outputUnit,GENERIC_OUTPUT_FORMAT) "domain"
432 write(outputUnit,GENERIC_OUTPUT_FORMAT)
433 write(outputUnit,GENERIC_TABBED_FORMAT) SpecBase%Domain%val
434 if (SpecBase%OutputSilentModeEnabled%isFalse) call note( prefix = prefix, outputUnit = outputUnit, newline = "\n", msg = SpecBase%Domain%desc )
435
436
437 write(outputUnit,GENERIC_OUTPUT_FORMAT)
438 write(outputUnit,GENERIC_OUTPUT_FORMAT) "domainCubeLimitLower"
439 write(outputUnit,GENERIC_OUTPUT_FORMAT)
440 do i = 1, ndim
441 write(outputUnit,GENERIC_TABBED_FORMAT) SpecBase%DomainCubeLimitLower%Val(i)
442 end do
443 if (SpecBase%OutputSilentModeEnabled%isFalse) call note( prefix = prefix, outputUnit = outputUnit, newline = "\n", msg = SpecBase%DomainCubeLimitLower%desc )
444
445
446 write(outputUnit,GENERIC_OUTPUT_FORMAT)
447 write(outputUnit,GENERIC_OUTPUT_FORMAT) "domainCubeLimitUpper"
448 write(outputUnit,GENERIC_OUTPUT_FORMAT)
449 do i = 1, ndim
450 write(outputUnit,GENERIC_TABBED_FORMAT) SpecBase%DomainCubeLimitUpper%Val(i)
451 end do
452 if (SpecBase%OutputSilentModeEnabled%isFalse) call note( prefix = prefix, outputUnit = outputUnit, newline = "\n", msg = SpecBase%DomainCubeLimitUpper%desc )
453
454
455 write(outputUnit,GENERIC_OUTPUT_FORMAT)
456 write(outputUnit,GENERIC_OUTPUT_FORMAT) "domainEllipsoidCenter"
457 write(outputUnit,GENERIC_OUTPUT_FORMAT)
458 do i = 1, ndim
459 write(outputUnit,GENERIC_TABBED_FORMAT) SpecBase%DomainEllipsoidCenter%Val(i)
460 end do
461 if (SpecBase%OutputSilentModeEnabled%isFalse) call note( prefix = prefix, outputUnit = outputUnit, newline = "\n", msg = SpecBase%DomainEllipsoidCenter%desc )
462
463
464 write(outputUnit,GENERIC_OUTPUT_FORMAT)
465 write(outputUnit,GENERIC_OUTPUT_FORMAT) "domainEllipsoidCovMat"
466 write(outputUnit,GENERIC_OUTPUT_FORMAT)
467 do i = 1, ndim
468 write(outputUnit,GENERIC_TABBED_FORMAT) SpecBase%DomainEllipsoidCovMat%Val(:,i)
469 end do
470 if (SpecBase%OutputSilentModeEnabled%isFalse) call note( prefix = prefix, outputUnit = outputUnit, newline = "\n", msg = SpecBase%DomainEllipsoidCovMat%desc )
471
472
473 write(outputUnit,GENERIC_OUTPUT_FORMAT)
474 write(outputUnit,GENERIC_OUTPUT_FORMAT) "domainEllipsoidCorMat"
