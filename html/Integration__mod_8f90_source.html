108 recursive subroutine doQuadRombClosed( getFunc &
109 , lowerLim &
110 , upperLim &
111 , maxRelativeError &
112 , nRefinement &
113 , integral &
114 , relativeError &
115 , numFuncEval &
116 , ierr &
117 )
118 !checked by Joshua Osborne on 5/28/2020 at 9:06pm
119#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
120 !DEC$ ATTRIBUTES DLLEXPORT :: doQuadRombClosed
121#endif
122 integer(IK) , intent(in) :: nRefinement
123 real(RK) , intent(in) :: lowerLim,upperLim,maxRelativeError
124 real(RK) , intent(out) :: integral, relativeError
125 integer(IK) , intent(out) :: ierr, numFuncEval
126 integer(IK) :: refinementStage,km,numFuncEvalNew
127 integer(IK), parameter :: NSTEP = 31_IK
128 real(RK) :: h(NSTEP+1),s(NSTEP+1)
129 procedure(integrand_proc) :: getFunc
130 ierr = 0_IK
131 km = nRefinement - 1_IK
132 h(1) = 1._RK
133 numFuncEval = 0_IK
134 do refinementStage = 1, NSTEP
135 call doQuadTrap(getFunc,lowerLim,upperLim,s(refinementStage),refinementStage,numFuncEvalNew)
136 numFuncEval = numFuncEval + numFuncEvalNew
137 if (refinementStage>=nRefinement) then
138 call doPolInt(h(refinementStage-km),s(refinementStage-km),nRefinement,0._RK,integral,relativeError,ierr)
139 if ( abs(relativeError)<=maxRelativeError*abs(integral) .or. ierr/=0_IK ) return
140 end if
141 s(refinementStage+1)=s(refinementStage)
142 h(refinementStage+1)=0.25_RK*h(refinementStage)
143 end do
144 ierr = 1_IK ! too many steps in doQuadRombClosed()
145 end subroutine doQuadRombClosed

170 , integrate &
171 , lowerLim &
172 , upperLim &
173 , maxRelativeError &
174 , nRefinement &
175 , integral &
176 , relativeError &
177 , numFuncEval &
178 , ierr &
179 )
180#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
181 !DEC$ ATTRIBUTES DLLEXPORT :: doQuadRombOpen
182#endif
183 integer(IK) , intent(in) :: nRefinement
184 real(RK) , intent(in) :: lowerLim,upperLim,maxRelativeError
185 real(RK) , intent(out) :: integral, relativeError
186 integer(IK) , intent(out) :: numFuncEval, ierr
187 integer(IK) :: refinementStage,km,numFuncEvalNew
188 integer(IK) , parameter :: NSTEP = 20_IK
189 real(RK) , parameter :: ONE_OVER_NINE = 1._RK / 9._RK
190 real(RK) :: h(NSTEP+1), s(NSTEP+1)
191 procedure(integrand_proc) :: getFunc
192 procedure(integrator_proc) :: integrate
193 ierr = 0_IK
194 km = nRefinement - 1_IK
195 h(1) = 1._RK
196 numFuncEval = 0_IK
197 do refinementStage = 1, NSTEP
198 call integrate(getFunc,lowerLim,upperLim,s(refinementStage),refinementStage,numFuncEvalNew)
199 numFuncEval = numFuncEval + numFuncEvalNew
200 if (refinementStage>=nRefinement) then
201 call doPolInt(h(refinementStage-km),s(refinementStage-km),nRefinement,0._RK,integral,relativeError,ierr)
202 if ( abs(relativeError) <= maxRelativeError * abs(integral) .or. ierr/=0_IK ) return
203 end if
204 s(refinementStage+1)=s(refinementStage)
205 h(refinementStage+1)=h(refinementStage)*ONE_OVER_NINE
206 end do
207 ierr = 2_IK ! too many steps in doQuadRombOpen()
208 end subroutine doQuadRombOpen

212 recursive subroutine doPolInt(xa,ya,ndata,x,y,dy,ierr)
213#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
214 !DEC$ ATTRIBUTES DLLEXPORT :: doPolInt
215#endif
216 implicit none
217 integer , intent(in) :: ndata
218 real(RK), intent(in) :: x,xa(ndata),ya(ndata)
219 real(RK), intent(out) :: dy,y
220 integer , intent(out) :: ierr
221 real(RK) :: den,dif,dift,ho,hp,w,c(ndata),d(ndata)
222 integer :: i,m,ns
223 ierr = 0
224 ns=1
225 dif=abs(x-xa(1))
226 do i=1,ndata
227 dift=abs(x-xa(i))
228 if (dift<dif) then
229 ns=i
230 dif=dift
231 endif
232 c(i)=ya(i)
233 d(i)=ya(i)
234 end do
235 y=ya(ns)
236 ns=ns-1
237 do m=1,ndata-1
238 do i=1,ndata-m
239 ho=xa(i)-x
240 hp=xa(i+m)-x
241 w=c(i+1)-d(i)
242 den=ho-hp
243 if(den==0._RK) then
244 ierr = 3 ! failure in doPolInt()
245 return
246 end if
247 den=w/den
248 d(i)=hp*den
249 c(i)=ho*den
250 end do
251 if (2*ns<ndata-m)then
252 dy=c(ns+1)
253 else
254 dy=d(ns)
255 ns=ns-1
256 endif
257 y=y+dy
258 end do
259 end subroutine doPolInt

263 recursive subroutine doQuadTrap(getFunc,lowerLim,upperLim,integral,refinementStage,numFuncEval)
264#if INTEL_COMPILER_ENABLED && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
265 !DEC$ ATTRIBUTES DLLEXPORT :: doQuadTrap
266#endif
267 implicit none
268 integer(IK), intent(in) :: refinementStage
269 real(RK), intent(in) :: lowerLim,upperLim
270 real(RK), intent(inout) :: integral
271 integer(IK), intent(out) :: numFuncEval
272 integer(IK) :: iFuncEval
273 real(RK) :: del,sum,tnm,x
274 procedure(integrand_proc) :: getFunc
275 if (refinementStage==1) then
276 numFuncEval = 2_IK
277 integral = 0.5_RK * (upperLim - lowerLim) * ( getFunc(lowerLim) + getFunc(upperLim) )
278 else
279 numFuncEval = 2**(refinementStage-2)
280 tnm = real(numFuncEval,kind=RK)
281 del = (upperLim-lowerLim) / tnm
282 x = lowerLim + 0.5_RK * del
283 sum = 0._RK
284 do iFuncEval = 1, numFuncEval
285 sum = sum + getFunc(x)
286 x = x + del
287 end do
288 integral = 0.5_RK * (integral + (upperLim-lowerLim) * sum / tnm)
289 endif
290 end subroutine doQuadTrap

318 real(RK) , intent(in) :: lowerLim,upperLim
319 integer(IK) , intent(out) :: numFuncEval
320 real(RK) , intent(inout) :: integral
321 procedure(integrand_proc) :: getFunc
322 real(RK) :: ddel,del,summ,x,lowerLimTrans,upperLimTrans
323 real(RK) :: inverseThreeNumFuncEval
324 integer(IK) :: iFuncEval
325 upperLimTrans = exp(-lowerLim)
326 lowerLimTrans = 0._RK; if (upperLim<LOGHUGE_RK) lowerLimTrans = exp(-upperLim)
327 if (refinementStage==1_IK) then
328 numFuncEval = 1_IK
329 integral = (upperLimTrans-lowerLimTrans)*getTransFunc(0.5_RK*(lowerLimTrans+upperLimTrans))
330 else
331 numFuncEval = 3**(refinementStage-2)
332 inverseThreeNumFuncEval = ONE_THIRD / numFuncEval
333 del = (upperLimTrans-lowerLimTrans) * inverseThreeNumFuncEval
334 ddel = del + del
335 x = lowerLimTrans + 0.5_RK*del
336 summ = 0._RK
337 do iFuncEval = 1,numFuncEval
338 summ = summ + getTransFunc(x)
339 x = x + ddel
340 summ = summ + getTransFunc(x)
341 x = x + del
342 end do
343 integral = ONE_THIRD * integral + (upperLimTrans-lowerLimTrans) * summ * inverseThreeNumFuncEval
344 numFuncEval = 2_IK * numFuncEval
345 end if
